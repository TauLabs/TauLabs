79 bool ClassNameValidatingLineEdit::validate(const QString &value, QString *errorMessage) const
80 {
81  if (!m_d->m_namespacesEnabled && value.contains(QLatin1Char(':'))) {
82  if (errorMessage)
83  *errorMessage = tr("The class name must not contain namespace delimiters.");
84  return false;
85  } else if (value.isEmpty()) {
86  if (errorMessage)
87  *errorMessage = tr("Please enter a class name.");
88  return false;
89  } else if (!m_d->m_nameRegexp.exactMatch(value)) {
90  if (errorMessage)
91  *errorMessage = tr("The class name contains invalid characters.");
92  return false;
93  }
94  return true;
95 }

112 QString ClassNameValidatingLineEdit::createClassName(const QString &name)
113 {
114  // Remove spaces and convert the adjacent characters to uppercase
115  QString className = name;
116  QRegExp spaceMatcher(QLatin1String(" +(\\w)"), Qt::CaseSensitive, QRegExp::RegExp2);
117  QTC_ASSERT(spaceMatcher.isValid(), );
118  int pos;
119  while ((pos = spaceMatcher.indexIn(className)) != -1) {
120  className.replace(pos, spaceMatcher.matchedLength(),
121  spaceMatcher.cap(1).toUpper());
122  }
123 
124  // Filter out any remaining invalid characters
125  className.remove(QRegExp(QLatin1String("[^a-zA-Z0-9_]")));
126 
127  // If the first character is numeric, prefix the name with a "_"
128  if (className.at(0).isNumber()) {
129  className.prepend(QLatin1Char('_'));
130  } else {
131  // Convert the first character to uppercase
132  className.replace(0, 1, className.left(1).toUpper());
133  }
134 
135  return className;
136 }
