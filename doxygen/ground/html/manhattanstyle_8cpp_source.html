140 void drawCornerImage(const QImage &img, QPainter *painter, QRect rect,
141  int left = 0, int top = 0, int right = 0,
142  int bottom = 0)
143 {
144  QSize size = img.size();
145  if (top > 0) { //top
146  painter->drawImage(QRect(rect.left() + left, rect.top(), rect.width() -right - left, top), img,
147  QRect(left, 0, size.width() -right - left, top));
148  if (left > 0) //top-left
149  painter->drawImage(QRect(rect.left(), rect.top(), left, top), img,
150  QRect(0, 0, left, top));
151  if (right > 0) //top-right
152  painter->drawImage(QRect(rect.left() + rect.width() - right, rect.top(), right, top), img,
153  QRect(size.width() - right, 0, right, top));
154  }
155  //left
156  if (left > 0)
157  painter->drawImage(QRect(rect.left(), rect.top()+top, left, rect.height() - top - bottom), img,
158  QRect(0, top, left, size.height() - bottom - top));
159  //center
160  painter->drawImage(QRect(rect.left() + left, rect.top()+top, rect.width() -right - left,
161  rect.height() - bottom - top), img,
162  QRect(left, top, size.width() -right -left,
163  size.height() - bottom - top));
164  if (right > 0) //right
165  painter->drawImage(QRect(rect.left() +rect.width() - right, rect.top()+top, right, rect.height() - top - bottom), img,
166  QRect(size.width() - right, top, right, size.height() - bottom - top));
167  if (bottom > 0) { //bottom
168  painter->drawImage(QRect(rect.left() +left, rect.top() + rect.height() - bottom,
169  rect.width() - right - left, bottom), img,
170  QRect(left, size.height() - bottom,
171  size.width() - right - left, bottom));
172  if (left > 0) //bottom-left
173  painter->drawImage(QRect(rect.left(), rect.top() + rect.height() - bottom, left, bottom), img,
174  QRect(0, size.height() - bottom, left, bottom));
175  if (right > 0) //bottom-right
176  painter->drawImage(QRect(rect.left() + rect.width() - right, rect.top() + rect.height() - bottom, right, bottom), img,
177  QRect(size.width() - right, size.height() - bottom, right, bottom));
178  }

440 void ManhattanStyle::drawPrimitive(PrimitiveElement element, const QStyleOption *option,
441  QPainter *painter, const QWidget *widget) const
442 {
443  if (!panelWidget(widget))
444  return d->style->drawPrimitive(element, option, painter, widget);
445 
446  bool animating = (option->state & State_Animating);
447  int state = option->state;
448  QRect rect = option->rect;
449  QRect oldRect;
450  QRect newRect;
451  if (widget && (element == PE_PanelButtonTool) && !animating) {
452  QWidget *w = const_cast<QWidget *> (widget);
453  int oldState = w->property("_q_stylestate").toInt();
454  oldRect = w->property("_q_stylerect").toRect();
455  newRect = w->rect();
456  w->setProperty("_q_stylestate", (int)option->state);
457  w->setProperty("_q_stylerect", w->rect());
458 
459  // Determine the animated transition
460  bool doTransition = ((state & State_On) != (oldState & State_On) ||
461  (state & State_MouseOver) != (oldState & State_MouseOver));
462  if (oldRect != newRect)
463  {
464  doTransition = false;
465  d->animator.stopAnimation(widget);
466  }
467 
468  if (doTransition) {
469  QImage startImage(option->rect.size(), QImage::Format_ARGB32_Premultiplied);
470  QImage endImage(option->rect.size(), QImage::Format_ARGB32_Premultiplied);
471  Animation *anim = d->animator.widgetAnimation(widget);
472  QStyleOption opt = *option;
473  opt.state = (QStyle::State)oldState;
474  opt.state |= (State)State_Animating;
475  startImage.fill(0);
476  Transition *t = new Transition;
477  t->setWidget(w);
478  QPainter startPainter(&startImage);
479  if (!anim) {
480  drawPrimitive(element, &opt, &startPainter, widget);
481  } else {
482  anim->paint(&startPainter, &opt);
483  d->animator.stopAnimation(widget);
484  }
485  QStyleOption endOpt = *option;
486  endOpt.state |= (State)State_Animating;
487  t->setStartImage(startImage);
488  d->animator.startAnimation(t);
489  endImage.fill(0);
490  QPainter endPainter(&endImage);
491  drawPrimitive(element, &endOpt, &endPainter, widget);
492  t->setEndImage(endImage);
493  if (oldState & State_MouseOver)
494  t->setDuration(150);
495  else
496  t->setDuration(75);
497  t->setStartTime(QTime::currentTime());
498  }
499  }
500 
501  switch (element) {
502  case PE_PanelLineEdit:
503  {
504  painter->save();
505 
506  // Fill the line edit background
507  QRect filledRect = option->rect.adjusted(1, 1, -1, -1);
508  painter->setBrushOrigin(filledRect.topLeft());
509  painter->fillRect(filledRect, option->palette.base());
510 
511  if (option->state & State_Enabled)
512  drawCornerImage(d->lineeditImage, painter, option->rect, 2, 2, 2, 2);
513  else
514  drawCornerImage(d->lineeditImage_disabled, painter, option->rect, 2, 2, 2, 2);
515 
516  if (option->state & State_HasFocus || option->state & State_MouseOver) {
517  QColor hover = Utils::StyleHelper::baseColor();
518  if (state & State_HasFocus)
519  hover.setAlpha(100);
520  else
521  hover.setAlpha(50);
522 
523  painter->setPen(QPen(hover, 1));
524  painter->drawRect(option->rect.adjusted(1, 1, -2 ,-2));
525  }
526  painter->restore();
527  }
528  break;
529 
530  case PE_FrameStatusBarItem:
531  break;
532 
533  case PE_PanelButtonTool: {
534  Animation *anim = d->animator.widgetAnimation(widget);
535  if (!animating && anim) {
536  anim->paint(painter, option);
537  } else {
538  bool pressed = option->state & State_Sunken || option->state & State_On;
539  QColor shadow(0, 0, 0, 30);
540  painter->setPen(shadow);
541  if (pressed) {
542  QColor shade(0, 0, 0, 40);
543  painter->fillRect(rect, shade);
544  painter->drawLine(rect.topLeft() + QPoint(1, 0), rect.topRight() - QPoint(1, 0));
545  painter->drawLine(rect.topLeft(), rect.bottomLeft());
546  painter->drawLine(rect.topRight(), rect.bottomRight());
547  // painter->drawLine(rect.bottomLeft() + QPoint(1, 0), rect.bottomRight() - QPoint(1, 0));
548  QColor highlight(255, 255, 255, 30);
549  painter->setPen(highlight);
550  }
551  else if (option->state & State_Enabled &&
552  option->state & State_MouseOver) {
553  QColor lighter(255, 255, 255, 37);
554  painter->fillRect(rect, lighter);
555  }
556  }
557  }
558  break;
559 
560  case PE_PanelStatusBar:
561  {
562  painter->save();
563  QLinearGradient grad(option->rect.topLeft(), QPoint(rect.center().x(), rect.bottom()));
564  QColor startColor = Utils::StyleHelper::shadowColor().darker(164);
565  QColor endColor = Utils::StyleHelper::baseColor().darker(130);
566  grad.setColorAt(0, startColor);
567  grad.setColorAt(1, endColor);
568  painter->fillRect(option->rect, grad);
569  painter->setPen(QColor(255, 255, 255, 60));
570  painter->drawLine(rect.topLeft() + QPoint(0,1),
571  rect.topRight()+ QPoint(0,1));
572  painter->setPen(Utils::StyleHelper::borderColor().darker(110));
573  painter->drawLine(rect.topLeft(), rect.topRight());
574  painter->restore();
575  }
576  break;
577 
578  case PE_IndicatorToolBarSeparator:
579  {
580  QColor separatorColor = Utils::StyleHelper::borderColor();
581  separatorColor.setAlpha(100);
582  painter->setPen(separatorColor);
583  const int margin = 6;
584  if (option->state & State_Horizontal) {
585  const int offset = rect.width()/2;
586  painter->drawLine(rect.bottomLeft().x() + offset,
587  rect.bottomLeft().y() - margin,
588  rect.topLeft().x() + offset,
589  rect.topLeft().y() + margin);
590  } else { //Draw vertical separator
591  const int offset = rect.height()/2;
592  painter->setPen(QPen(option->palette.background().color().darker(110)));
593  painter->drawLine(rect.topLeft().x() + margin ,
594  rect.topLeft().y() + offset,
595  rect.topRight().x() - margin,
596  rect.topRight().y() + offset);
597  }
598  }
599  break;
600 
601  case PE_IndicatorToolBarHandle:
602  {
603  bool horizontal = option->state & State_Horizontal;
604  painter->save();
605  QPainterPath path;
606  int x = option->rect.x() + horizontal ? 2 : 6;
607  int y = option->rect.y() + horizontal ? 6 : 2;
608  static const int RectHeight = 2;
609  if (horizontal) {
610  while (y < option->rect.height() - RectHeight - 6) {
611  path.moveTo(x, y);
612  path.addRect(x, y, RectHeight, RectHeight);
613  y += 6;
614  }
615  } else {
616  while (x < option->rect.width() - RectHeight - 6) {
617  path.moveTo(x, y);
618  path.addRect(x, y, RectHeight, RectHeight);
619  x += 6;
620  }
621  }
622 
623  painter->setPen(Qt::NoPen);
624  QColor dark = Utils::StyleHelper::borderColor();
625  dark.setAlphaF(0.4);
626 
627  QColor light = Utils::StyleHelper::baseColor();
628  light.setAlphaF(0.4);
629 
630  painter->fillPath(path, light);
631  painter->save();
632  painter->translate(1, 1);
633  painter->fillPath(path, dark);
634  painter->restore();
635  painter->translate(3, 3);
636  painter->fillPath(path, light);
637  painter->translate(1, 1);
638  painter->fillPath(path, dark);
639  painter->restore();
640  }
641  break;
642  case PE_IndicatorArrowUp:
643  case PE_IndicatorArrowDown:
644  case PE_IndicatorArrowRight:
645  case PE_IndicatorArrowLeft:
646  {
647  // From windowsstyle but modified to enable AA
648  if (option->rect.width() <= 1 || option->rect.height() <= 1)
649  break;
650 
651  QRect r = option->rect;
652  int size = qMin(r.height(), r.width());
653  QPixmap pixmap;
654  QString pixmapName;
655  QTextStream(&pixmapName) << "$qt_ia" << "-" << metaObject()->className() <<
656  "-" << uint(option->state) << "-" << element << "-" << size <<
