193 bool ShortcutSettings::filter(const QString &f, const QTreeWidgetItem *item)
194 {
195  if (item->childCount() == 0) {
196  if (f.isEmpty())
197  return false;
198  for (int i = 0; i < item->columnCount(); ++i) {
199  if (item->text(i).contains(f, Qt::CaseInsensitive))
200  return false;
201  }
202  return true;
203  }
204 
205  bool found = false;
206  for (int i = 0; i < item->childCount(); ++i) {
207  QTreeWidgetItem *citem = item->child(i);
208  if (filter(f, citem)) {
209  citem->setHidden(true);
210  } else {
211  citem->setHidden(false);
212  found = true;
213  }
214  }
215  return !found;
216 }

281 void ShortcutSettings::initialize()
282 {
283  m_am = ActionManagerPrivate::instance();
284  UniqueIDManager *uidm = UniqueIDManager::instance();
285 
286  foreach (Command *c, m_am->commands()) {
287  if (c->hasAttribute(Command::CA_NonConfigureable))
288  continue;
289  if (c->action() && c->action()->isSeparator())
290  continue;
291 
292  QTreeWidgetItem *item = 0;
293  ShortcutItem *s = new ShortcutItem;
294  m_scitems << s;
295  item = new QTreeWidgetItem(m_page->commandList);
296  s->m_cmd = c;
297  s->m_item = item;
298 
299  item->setText(0, uidm->stringForUniqueIdentifier(c->id()));
300 
301  if (c->action()) {
302  QString text = c->hasAttribute(Command::CA_UpdateText) && !c->defaultText().isNull() ? c->defaultText() : c->action()->text();
303  text.remove(QRegExp("&(?!&)"));
304  s->m_key = c->action()->shortcut();
305  item->setText(1, text);
306  } else {
307  s->m_key = c->shortcut()->key();
308  item->setText(1, c->shortcut()->whatsThis());
309  }
310 
311  item->setText(2, s->m_key);
312  item->setData(0, Qt::UserRole, qVariantFromValue(s));
313  }

316 void ShortcutSettings::handleKeyEvent(QKeyEvent *e)
317 {
318  int nextKey = e->key();
319  if ( m_keyNum > 3 ||
320  nextKey == Qt::Key_Control ||
321  nextKey == Qt::Key_Shift ||
322  nextKey == Qt::Key_Meta ||
323  nextKey == Qt::Key_Alt )
324  return;
325 
326  nextKey |= translateModifiers(e->modifiers(), e->text());
327  switch (m_keyNum) {
328  case 0:
329  m_key[0] = nextKey;
330  break;
331  case 1:
332  m_key[1] = nextKey;
333  break;
334  case 2:
335  m_key[2] = nextKey;
336  break;
337  case 3:
338  m_key[3] = nextKey;
339  break;
340  default:
341  break;
342  }
343  m_keyNum++;
344  QKeySequence ks(m_key[0], m_key[1], m_key[2], m_key[3]);
345  m_page->shortcutEdit->setText(ks);
346  e->accept();
347 }

349 int ShortcutSettings::translateModifiers(Qt::KeyboardModifiers state,
350  const QString &text)
351 {
352  int result = 0;
353  // The shift modifier only counts when it is not used to type a symbol
354  // that is only reachable using the shift key anyway
355  if ((state & Qt::ShiftModifier) && (text.size() == 0
356  || !text.at(0).isPrint()
357  || text.at(0).isLetter()
358  || text.at(0).isSpace()))
359  result |= Qt::SHIFT;
360  if (state & Qt::ControlModifier)
361  result |= Qt::CTRL;
362  if (state & Qt::MetaModifier)
363  result |= Qt::META;
364  if (state & Qt::AltModifier)
365  result |= Qt::ALT;
366  return result;
367 }
