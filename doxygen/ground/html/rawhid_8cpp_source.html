00231         {
00232             //wait on new data to write condition, the timeout
00233             //enable the thread to shutdown properly
00234             m_newDataToWrite.wait(&m_writeBufMtx, 200);
00235             if(!m_running)
00236                 return;

00238             size = m_writeBuffer.size();

00241         //NOTE: data size is limited to 2 bytes less than the
00242         //usb packet size (64 bytes for interrupt) to make room
00243         //for the reportID and valid data length
00244         size = qMin(WRITE_SIZE-2, m_writeBuffer.size());
00245         memcpy(&buffer[2], m_writeBuffer.constData(), size);

00249 
00250         // must hold lock through the send to know how much was sent
00251         int ret = hiddev->send(m_hid->m_deviceNo, buffer, WRITE_SIZE, WRITE_TIMEOUT);

00254         {
00255             //only remove the size actually written to the device
