348 void MainWindow::loadStyleSheet(QString name) {
349  /* Let's use QFile and point to a resource... */
350  QDir directory(QCoreApplication::applicationDirPath());
351 #ifdef Q_OS_MAC
352  directory.cdUp();
353  directory.cd("Resources");
354 #else
355  directory.cdUp();
356  directory.cd("share");
357  directory.cd("taulabs");
358 #endif
359  directory.cd("stylesheets");
360 #ifdef Q_OS_MAC
361  QFile data(directory.absolutePath()+QDir::separator()+name+"_macos.qss");
362 #elif defined(Q_OS_LINUX)
363  QFile data(directory.absolutePath()+QDir::separator()+name+"_linux.qss");
364 #else
365  QFile data(directory.absolutePath()+QDir::separator()+name+"_windows.qss");
366 #endif
367  QString style;
368  /* ...to open the file */
369  if(data.open(QFile::ReadOnly)) {
370  /* QTextStream... */
371  QTextStream styleIn(&data);
372  /* ...read file to a string. */
373  style = styleIn.readAll();
374  data.close();
375  /* We'll use qApp macro to get the QApplication pointer
376  * and set the style sheet application wide. */
377  qApp->setStyleSheet(style);
378  qDebug()<<"Loaded stylesheet:"<<style;
379  }
380  else
381  qDebug()<<"Failed to openstylesheet file"<<directory.absolutePath()<<name;
382 }

413 static bool isDesktopFileManagerDrop(const QMimeData *d, QStringList *files = 0)
414 {
415  if (files)
416  files->clear();
417  // Extract dropped files from Mime data.
418  if (!d->hasFormat(QLatin1String(uriListMimeFormatC)))
419  return false;
420  const QList<QUrl> urls = d->urls();
421  if (urls.empty())
422  return false;
423  // Try to find local files
424  bool hasFiles = false;
425  const QList<QUrl>::const_iterator cend = urls.constEnd();
426  for (QList<QUrl>::const_iterator it = urls.constBegin(); it != cend; ++it) {
427  const QString fileName = it->toLocalFile();
428  if (!fileName.isEmpty()) {
429  hasFiles = true;
430  if (files) {
431  files->push_back(fileName);
432  } else {
433  break; // No result list, sufficient for checking
434  }
435  }
436  }
437  return hasFiles;
438 }

1150 void MainWindow::createWorkspaces(QSettings* qs, bool diffOnly) {
1151 
1152  ExtensionSystem::PluginManager *pm = ExtensionSystem::PluginManager::instance();
1153 
1154  Core::UAVGadgetManager *uavGadgetManager;
1155 
1156  // If diffOnly is true, we only add/remove the number of workspaces
1157  // that has changed,
1158  // otherwise a complete reload of workspaces is done
1159  int toRemoveFirst = m_uavGadgetManagers.count();
1160  int newWorkspacesNo = m_workspaceSettings->numberOfWorkspaces();
1161  if (diffOnly && m_uavGadgetManagers.count() > newWorkspacesNo)
1162  toRemoveFirst = m_uavGadgetManagers.count() - newWorkspacesNo;
1163  else
1164  toRemoveFirst = 0;
1165 
1166  int removed = 0;
1167 
1168  while (!m_uavGadgetManagers.isEmpty() && (toRemoveFirst > removed)) {
1169  int index = takeLeastPriorityUavGadgetManager(m_uavGadgetManagers);
1170  uavGadgetManager = m_uavGadgetManagers.takeAt(index);
1171  uavGadgetManager->removeAllSplits();
1172  pm->removeObject(uavGadgetManager);
1173  delete uavGadgetManager;
1174  removed++;
1175  }
1176 
1177  int start = 0;
1178  if (diffOnly) {
1179  start = m_uavGadgetManagers.count();
1180  } else {
1181  m_uavGadgetManagers.clear();
1182  }
1183  for (int i = start; i < newWorkspacesNo; ++i) {
1184 
1185  const QString name = m_workspaceSettings->name(i);
1186  const QString iconName = m_workspaceSettings->iconName(i);
1187  const QString modeName = m_workspaceSettings->modeName(i);
1188  uavGadgetManager = new Core::UAVGadgetManager(CoreImpl::instance(), name,
1189  QIcon(iconName), 90-i+1, modeName, this);
1190 
1191  connect(uavGadgetManager, SIGNAL(showUavGadgetMenus(bool, bool)), this, SLOT(showUavGadgetMenus(bool, bool)));
1192 
1193  connect(m_showToolbarsAction, SIGNAL(triggered(bool)), uavGadgetManager, SLOT(showToolbars(bool)));
1194  connect(m_splitAction, SIGNAL(triggered()), uavGadgetManager, SLOT(split()));
1195  connect(m_splitSideBySideAction, SIGNAL(triggered()), uavGadgetManager, SLOT(splitSideBySide()));
1196  connect(m_removeCurrentSplitAction, SIGNAL(triggered()), uavGadgetManager, SLOT(removeCurrentSplit()));
1197  connect(m_removeAllSplitsAction, SIGNAL(triggered()), uavGadgetManager, SLOT(removeAllSplits()));
1198  connect(m_gotoOtherSplitAction, SIGNAL(triggered()), uavGadgetManager, SLOT(gotoOtherSplit()));
1199 
1200  pm->addObject(uavGadgetManager);
1201  m_uavGadgetManagers.append(uavGadgetManager);
1202  uavGadgetManager->readSettings(qs);
1203  }
1204 }
