29 /*
30  * Description of DeviceLink Protocol:
31  * A Request is initiated with R/ followed by id's of to be requested settings
32  * even id's indicate read only values, odd are write only
33  * (usually id =get value id+1= set - for same setting)
34  * id's are separated by /
35  * requests can contain values to set, or to select a subsystem
36  * values are separated by \
37  * example: R/30/48/64\0/64\1/
38  * request read only settings 30,48 and 64 with parameters 0 and 1
39  * the answer consists of an A followed by id value pairs in the same format
40  * example: A/30\0/48\0/64\0\22/64\1\102/
41  *
42  * A full protocol description as well as a list of ID's and their meanings
43  * can be found shipped with IL2 in the file DeviceLink.txt
44  *
45  * id's used in this file:
46  * 30: CAS in km/h (float)
47  * 32: vario in m/s (float)
48  * 38: angular speed 째/s (float) (which direction? azimuth?)
49  * 40: barometric alt in m (float)
50  * 42: flight course in 째 (0-360) (float)
51  * 46: roll angle in 째 (-180 - 180) (floatniguration)
52  * 48: pitch angle in 째 (-90 - 90) (float)
53  * 80/81: engine power (-1.0 (0%) - 1.0 (100%)) (float)
54  * 84/85: aileron servo (-1.0 - 1.0) (float)
55  * 86/87: elevator servo (-1.0 - 1.0) (float)
56  * 88/89: rudder servo (-1.0 - 1.0) (float)
57  *
58  * IL2 currently offers no useful way of providing GPS data
59  * therefore fake GPS data will be calculated using IMS
60  *
61  * unfortunately angular acceleration provided is very limited, too
62  */

117 void IL2Simulator::processUpdate(const QByteArray& inp)
118 {
119  // save old flight data to calculate delta's later
120  old=current;
121  QString data(inp);
122  // Split
123  QStringList fields = data.split("/");
124 
125  // split up response string
126  int t;
127  for (t=0; t<fields.length(); t++) {
128  QStringList values = fields[t].split("\\");
129  // parse values
130  if (values.length()>=2) {
131  int id = values[0].toInt();
132  float value = values[1].toFloat();
133  switch (id) {
134  case 30:
135  current.cas=value * KM_PER_HOUR2METERS_PER_SECOND;
136  break;
137  case 32:
138  current.dZ=value;
139  break;
140  case 40:
141  current.Z=value;
142  break;
143  case 42:
144  current.azimuth=value;
145  break;
146  case 46:
147  current.roll=-value;
148  break;
149  case 48:
150  current.pitch=value;
151  break;
152  }
153  }
154  }
155 
156  // measure time
157  current.dT = ((float)time->restart()) / 1000.0;
158  if (current.dT<0.001) current.dT=0.001;
159  current.T = old.T+current.dT;
160  current.i = old.i+1;
161  if (current.i==1) {
162  old.dRoll=0;
163  old.dPitch=0;
164  old.dAzimuth=0;
165  old.ddX=0;
166  old.ddX=0;
167  old.ddX=0;
168  }
169 
170  // calculate TAS from alt and CAS
171  current.tas = cas2tas(current.cas, current.Z, airParameters, GRAVITY);
172 
173  // assume the plane actually flies straight and no wind
174  // groundspeed is horizontal vector of TAS
175  current.groundspeed = current.tas*cos(current.pitch*DEG2RAD);
176  // x and y vector components
177  current.dX = current.groundspeed*sin(current.azimuth*DEG2RAD);
178  current.dY = current.groundspeed*cos(current.azimuth*DEG2RAD);
179 
180  // simple IMS - integration over time the easy way...
181  current.X = old.X + (current.dX*current.dT);
182  current.Y = old.Y + (current.dY*current.dT);
183 
184  // accelerations (filtered)
185  if (isnan(old.ddX) || isinf(old.ddX)) old.ddX=0;
186  if (isnan(old.ddY) || isinf(old.ddY)) old.ddY=0;
187  if (isnan(old.ddZ) || isinf(old.ddZ)) old.ddZ=0;
188 #define SPEED_FILTER 10
189  current.ddX = ((current.dX-old.dX)/current.dT + SPEED_FILTER * (old.ddX)) / (SPEED_FILTER+1);
190  current.ddY = ((current.dY-old.dY)/current.dT + SPEED_FILTER * (old.ddY)) / (SPEED_FILTER+1);
191  current.ddZ = ((current.dZ-old.dZ)/current.dT + SPEED_FILTER * (old.ddZ)) / (SPEED_FILTER+1);
192 
193 #define TURN_FILTER 10
194  // turn speeds (filtered)
195  if (isnan(old.dAzimuth) || isinf(old.dAzimuth)) old.dAzimuth=0;
196  if (isnan(old.dPitch) || isinf(old.dPitch)) old.dPitch=0;
197  if (isnan(old.dRoll) || isinf(old.dRoll)) old.dRoll=0;
198  current.dAzimuth = (angleDifference(current.azimuth,old.azimuth)/current.dT + TURN_FILTER * (old.dAzimuth)) / (TURN_FILTER+1);
199  current.dPitch = (angleDifference(current.pitch,old.pitch)/current.dT + TURN_FILTER * (old.dPitch)) / (TURN_FILTER+1);
200  current.dRoll = (angleDifference(current.roll,old.roll)/current.dT + TURN_FILTER * (old.dRoll)) / (TURN_FILTER+1);
201 
203  // Output formatting
205  Output2Hardware out;
206  memset(&out, 0, sizeof(Output2Hardware));
207 
208  // Compute rotation matrix, for later calculations
209  float Rbe[3][3];
210  float rpy[3];
211  float quat[4];
212  rpy[0]=current.roll;
213  rpy[1]=current.pitch;
214  rpy[2]=current.azimuth;
215  Utils::CoordinateConversions().RPY2Quaternion(rpy,quat);
216  Utils::CoordinateConversions().Quaternion2R(quat,Rbe);
217 
218  // Update GPS Position objects, convertin from the current NED position
219  // to LLA
220  double HomeLLA[3];
221  double LLA[3];
222  double NED[3];
223  HomeLLA[0]=settings.latitude.toFloat();
224  HomeLLA[1]=settings.longitude.toFloat();
225  HomeLLA[2]=0;
226  NED[0] = current.Y;
227  NED[1] = current.X;
228  NED[2] = -current.Z;
229  Utils::CoordinateConversions().NED2LLA_HomeLLA(HomeLLA, NED, LLA);
230  out.latitude = LLA[0];
231  out.longitude = LLA[1];
232  out.groundspeed = current.groundspeed;
233 
234  out.calibratedAirspeed = current.cas;
235  out.trueAirspeed=cas2tas(current.cas, current.Z, airParameters, GRAVITY);
236 
237  out.dstN=current.Y;
238  out.dstE=current.X;
239  out.dstD=-current.Z;
240 
241  // Update BaroAltitude object
242  out.altitude = current.Z;
243  out.agl = current.Z;
244  out.temperature = airParameters.groundTemp + (current.Z * airParameters.tempLapseRate) - 273.0;
245  out.pressure = airPressureFromAltitude(current.Z, airParameters, GRAVITY) ; // kpa
246 
247 
248  // Update attActual object
249  out.roll = current.roll; //roll;
250  out.pitch = current.pitch; // pitch
251  out.heading = current.azimuth; // yaw
252 
253 
254  // Update VelocityActual.{North,East,Down}
255  out.velNorth = current.dY;
256  out.velEast = current.dX;
257  out.velDown = -current.dZ;
258 
259  // rotate turn rates and accelerations into body frame
260  // (note: rotation deltas are NOT in NED frame but in RPY - manual conversion!)
261  out.rollRate = current.dRoll;
262  out.pitchRate = cos(DEG2RAD * current.roll) * current.dPitch + sin(DEG2RAD * current.roll) * current.dAzimuth;
263  out.yawRate = cos(DEG2RAD * current.roll) * current.dAzimuth - sin(DEG2RAD * current.roll) * current.dPitch;
264 
265  //Update accelerometer sensor data
266  out.accX = current.ddX*Rbe[0][0]
267  +current.ddY*Rbe[0][1]
268  +(current.ddZ+GRAVITY)*Rbe[0][2];
269  out.accY = current.ddX*Rbe[1][0]
270  +current.ddY*Rbe[1][1]
271  +(current.ddZ+GRAVITY)*Rbe[1][2];
272  out.accZ = - (current.ddX*Rbe[2][0]
273  +current.ddY*Rbe[2][1]
274  +(current.ddZ+GRAVITY)*Rbe[2][2]);
275 
276  updateUAVOs(out);
277 }
