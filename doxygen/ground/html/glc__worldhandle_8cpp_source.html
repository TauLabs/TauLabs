94 // An Occurence has been added
95 void GLC_WorldHandle::addOccurence(GLC_StructOccurence* pOccurence, bool isSelected, GLuint shaderId)
96 {
97  Q_ASSERT(!m_OccurenceHash.contains(pOccurence->id()));
98  m_OccurenceHash.insert(pOccurence->id(), pOccurence);
99  GLC_StructReference* pRef= pOccurence->structReference();
100  Q_ASSERT(NULL != pRef);
101 
102  // Add instance representation in the collection
103  if (pOccurence->useAutomatic3DViewInstanceCreation() && pRef->representationIsLoaded())
104  {
105  //qDebug() << "GLC_WorldHandle::addOccurence with rep";
106  GLC_3DRep* p3DRep= dynamic_cast<GLC_3DRep*>(pRef->representationHandle());
107  GLC_3DViewInstance representation(*p3DRep);
108  // Force instance representation id
109  representation.setId(pOccurence->id());
110  // Force instance representation name
111  representation.setName(pOccurence->name());
112  if (0 != shaderId) m_Collection.bindShader(shaderId);
113  m_Collection.add(representation, shaderId);
114  if (isSelected)
115  {
116  //qDebug() << pOccurence->name() << "selected";
117  m_Collection.select(pOccurence->id());
118  }
119  }
120 }

134 void GLC_WorldHandle::select(GLC_uint occurenceId)
135 {
136  Q_ASSERT(m_OccurenceHash.contains(occurenceId));
137  m_SelectionSet.insert(occurenceId);
138  m_Collection.select(occurenceId);
139 
140  const GLC_StructOccurence* pSelectedOccurence= m_OccurenceHash.value(occurenceId);
141  if (pSelectedOccurence->hasChild())
142  {
143  QList<GLC_StructOccurence*> subOccurenceList= pSelectedOccurence->subOccurenceList();
144  const int subOccurenceCount= subOccurenceList.size();
145  for (int i= 0; i < subOccurenceCount; ++i)
146  {
147  const GLC_uint currentOccurenceId= subOccurenceList.at(i)->id();
148  if (m_Collection.contains(currentOccurenceId))
149  {
150  m_Collection.select(currentOccurenceId);
151  }
152  }
153  }
154 }
