100 void TileMatrix::ClearPointsNotIn(QList<Point>list)
101 {
102  removals.clear();
103  mutex.lock();
104  foreach(Point p, matrix.keys())
105  {
106  if(!list.contains(p))
107  {
108  removals.append(p);
109  }
110  }
111  mutex.unlock();
112  foreach(Point p,removals)
113  {
114  Tile* t=TileAt(p);
115  if(t!=0)
116  {
117  mutex.lock();
118  delete t;
119  t=0;
120  matrix.remove(p);
121  mutex.unlock();
122  }
123 
124  }
125  removals.clear();
126 }
