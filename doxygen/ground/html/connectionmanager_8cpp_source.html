107 bool ConnectionManager::connectDevice(DevListItem device)
108 {
109  Q_UNUSED(device);
110 
111  DevListItem connection_device = findDevice(m_availableDevList->itemData(m_availableDevList->currentIndex(),Qt::ToolTipRole).toString());
112  if (!connection_device.connection)
113  return false;
114 
115  QIODevice *io_dev = connection_device.connection->openDevice(connection_device.device);
116  if (!io_dev)
117  return false;
118 
119  io_dev->open(QIODevice::ReadWrite);
120 
121  // check if opening the device worked
122  if (!io_dev->isOpen()) {
123  return false;
124  }
125 
126  // we appear to have connected to the device OK
127  // remember the connection/device details
128  m_connectionDevice = connection_device;
129  m_ioDev = io_dev;
130 
131  connect(m_connectionDevice.connection, SIGNAL(destroyed(QObject *)), this, SLOT(onConnectionDestroyed(QObject *)), Qt::QueuedConnection);
132 
133  // signal interested plugins that we connected to the device
134  emit deviceConnected(io_dev);
135  m_connectBtn->setText("Disconnect");
136  m_availableDevList->setEnabled(false);
137 
138  // tell the monitorwidget we're conneced
139  m_monitorWidget->connected();
140 
141  return true;
142 }

148 bool ConnectionManager::disconnectDevice()
149 {
150  // tell the monitor widget we're disconnected
151  m_monitorWidget->disconnect();
152 
153  if (!m_ioDev) {
154  // apparently we are already disconnected: this can
155  // happen if a plugin tries to force a disconnect whereas
156  // we are not connected. Just return.
157  return false;
158  }
159 
160  // We are connected - disconnect from the device
161 
162  // stop our timers
163  if(reconnect->isActive())
164  reconnect->stop();
165  if(reconnectCheck->isActive())
166  reconnectCheck->stop();
167 
168  // signal interested plugins that user is disconnecting his device
169  emit deviceAboutToDisconnect();
170 
171  try {
172  if (m_connectionDevice.connection) {
173  m_connectionDevice.connection->closeDevice(m_connectionDevice.getConName());
174  }
175  } catch (...) { // handle exception
176  qDebug() << "Exception: m_connectionDevice.connection->closeDevice(" << m_connectionDevice.getConName() << ")";
177  }
178 
179  m_connectionDevice.connection = NULL;
180  m_ioDev = NULL;
181 
182  emit deviceDisconnected();
183  m_connectBtn->setText("Connect");
184  m_availableDevList->setEnabled(true);
185 
186  return true;
187 }

208 void ConnectionManager::aboutToRemoveObject(QObject *obj)
209 {
210  //Check if a plugin added a connection object to the pool
211  IConnection *connection = Aggregation::query<IConnection>(obj);
212  if (!connection) return;
213 
214  if (m_connectionDevice.connection && m_connectionDevice.connection == connection)
215  { // we are currently using the one that is about to be removed
216  disconnectDevice();
217  m_connectionDevice.connection = NULL;
218  m_ioDev = NULL;
219  }
220 
221  if (m_connectionsList.contains(connection))
222  m_connectionsList.removeAt(m_connectionsList.indexOf(connection));

368 void ConnectionManager::updateConnectionList(IConnection *connection)
369 {
370  // Get the updated list of devices
371  QList <IDevice*> availableDev = connection->availableDevices();
372 
373  // Go through the list of connections of that type. If they are not in the
374  // available device list then remove them. If they are connected, then
375  // disconnect them.
376  for (QLinkedList<DevListItem>::iterator iter = m_devList.begin(); iter != m_devList.end(); )
377  {
378  if (iter->connection != connection) {
379  ++iter;
380  continue;
381  }
382 
383  // See if device exists in the updated availability list
384  bool found = availableDev.contains(iter->device);
385  if (!found) {
386  // we are currently using the one we are about to erase
387  if (m_connectionDevice.connection && m_connectionDevice.connection == connection && m_connectionDevice.device == iter->device) {
388  disconnectDevice();
389  }
390 
391  // We have to delete the IDevice in that DevListItem before getting rid
392  // of the iter itself.
393  delete(iter->device);
394 
395  iter = m_devList.erase(iter);
396  } else
397  ++iter;
398  }
399 
400  // Add any back to list that don't exist
401  foreach (IDevice *dev, availableDev)
402  {
403  bool found = m_devList.contains(DevListItem(connection, dev));
404  if (!found) {
405  registerDevice(connection,dev);
406  }
407  }
408 }
