00373     // Go through the list of connections of that type.  If they are not in the

00377     {
00378         if (iter->connection != connection) {
00379             ++iter;
00380             continue;
00381         }
00382 
00383         // See if device exists in the updated availability list
00384         bool found = availableDev.contains(iter->device);
00385         if (!found) {
00386             // we are currently using the one we are about to erase
00387             if (m_connectionDevice.connection && m_connectionDevice.connection == connection && m_connectionDevice.device == iter->device) {
00388                 disconnectDevice();

00392             // of the iter itself.
00393             delete(iter->device);

00395             iter = m_devList.erase(iter);

00397             ++iter;
