118 void PathCompiler::doAddWaypoint(waypoint newWaypointInternal, int /*position*/)
119 {
120  /* TODO: If a waypoint is inserted not at the end shift them all by one and */
121  /* add the data there */
122 
123  UAVObjectManager *objManager = getObjectManager();
124 
125  // Format the data from the map into a UAVO
126  Waypoint::DataFields newWaypoint = InternalToUavo(newWaypointInternal);
127 
128  // Search for any waypoints set to stop, because if they exist we
129  // should add the waypoint immediately after that one
130  int numWaypoints = objManager->getNumInstances(Waypoint::OBJID);
131  int i;
132  for (i = 0; i < numWaypoints; i++) {
133  Waypoint *waypoint = Waypoint::GetInstance(objManager, i);
134  Q_ASSERT(waypoint);
135  if(waypoint == NULL)
136  return;
137 
138  Waypoint::DataFields waypointData = waypoint->getData();
139  if(waypointData.Action == Waypoint::ACTION_STOP) {
140  waypointData.Action = Waypoint::ACTION_PATHTONEXT;
141  waypoint->setData(waypointData);
142  waypoint->updated();
143  break;
144  }
145  }
146 
147  if (i >= numWaypoints - 1) {
148  // We reached end of list so new waypoint needs to be registered
149 
150  Waypoint *waypoint = new Waypoint();
151  Q_ASSERT(waypoint);
152  if (waypoint) {
153  // Register a new waypoint instance
154  quint32 newInstId = objManager->getNumInstances(waypoint->getObjID());
155  waypoint->initialize(newInstId,waypoint->getMetaObject());
156  objManager->registerObject(waypoint);
157 
158  // Set the data in the new object
159  waypoint->setData(newWaypoint);
160  waypoint->updated();
161  }
162  } else {
163  Waypoint *waypoint = Waypoint::GetInstance(objManager, i + 1);
164  Q_ASSERT(waypoint);
165  if (waypoint) {
166  waypoint->setData(newWaypoint);
167  waypoint->updated();
168  }
169  }
170 }

175 void PathCompiler::doUpdateWaypoints(PathCompiler::waypoint changedWaypoint, int position)
176 {
177  int numWaypoints = getObjectManager()->getNumInstances(Waypoint::OBJID);
178  Q_ASSERT(position < numWaypoints);
179  if (position >= numWaypoints)
180  return;
181 
182  Waypoint *waypointInst = Waypoint::GetInstance(getObjectManager(), position);
183  Q_ASSERT(waypointInst);
184 
185  // Mirror over the updated position. We don't just use the changedWaypoint
186  // because things like action might need to be preserved
187  Waypoint::DataFields changedWaypointUAVO = InternalToUavo(changedWaypoint);
188  Waypoint::DataFields oldWaypointUAVO = waypointInst->getData();
189  oldWaypointUAVO.Position[0] = changedWaypointUAVO.Position[0];
190  oldWaypointUAVO.Position[1] = changedWaypointUAVO.Position[1];
191  // Don't take the altitude from the map for now
192 
193  waypointInst->setData(oldWaypointUAVO);
194  waypointInst->updated();
195 }

201 void PathCompiler::doDelWaypoint(int index)
202 {
203  // This method is awkward because there is no support
204  // on the FC for actually deleting a waypoint. We need
205  // to shift them all by one and set the new "last" waypoint
206  // to a stop action
207 
208  UAVObjectManager *objManager = getObjectManager();
209  Waypoint *waypoint = Waypoint::GetInstance(objManager);
210  Q_ASSERT(waypoint);
211  if (!waypoint)
212  return;
213 
214  int numWaypoints = objManager->getNumInstances(waypoint->getObjID());
215  for (int i = index; i < numWaypoints - 1; i++) {
216  Waypoint *waypointDest = Waypoint::GetInstance(objManager, i);
217  Q_ASSERT(waypointDest);
218 
219  Waypoint *waypointSrc = Waypoint::GetInstance(objManager, i + 1);
220  Q_ASSERT(waypointSrc);
221 
222  if (!waypointDest || !waypointSrc)
223  return;
224 
225  // Copy the data down an index
226  Waypoint::DataFields waypoint = waypointSrc->getData();
227  waypointDest->setData(waypoint);
228  waypointDest->updated();
229  }
230 
231  // Set the second to last waypoint to stop (and last for safety)
232  // the functional equivalent to deleting
233  for (int i = numWaypoints - 2; i < numWaypoints; i++) {
234  waypoint = Waypoint::GetInstance(objManager, i);
235  Q_ASSERT(waypoint);
236  if (waypoint) {
237  Waypoint::DataFields waypointData = waypoint->getData();
238  waypointData.Action = Waypoint::ACTION_STOP;
239  waypoint->setData(waypointData);
240  waypoint->updated();
241  }
242  }
243 }

273 void PathCompiler::doUpdateFromUAV(UAVObject *obj)
274 {
275  UAVObjectManager *objManager = getObjectManager();
276  if (!objManager)
277  return;
278 
279  Waypoint *waypointObj = Waypoint::GetInstance(getObjectManager());
280  Q_ASSERT(waypointObj);
281  if (waypointObj == NULL)
282  return;
283 
284  /* Get all the waypoints from the UAVO and create a representation for the visualization */
285  QList <waypoint> waypoints;
286  waypoints.clear();
287  int numWaypoints = objManager->getNumInstances(waypointObj->getObjID());
288  bool stopped = false;
289  for (int i = 0; i < numWaypoints && !stopped; i++) {
290  Waypoint *waypoint = Waypoint::GetInstance(objManager, i);
291  Q_ASSERT(waypoint);
292  if(waypoint == NULL)
293  return;
294 
295  Waypoint::DataFields waypointData = waypoint->getData();
296  waypoints.append(UavoToInternal(waypointData));
297  stopped = waypointData.Action == Waypoint::ACTION_STOP;
298  }
299 
300  if (previousWaypoints != waypoints) {
301  /* Because the waypoints have to update periodically (or we miss new ones on the FC */
302  /* side - needs telem fix) we want to filter updates to map that are simply periodic */
303  previousWaypoints = waypoints;
304 
305  /* Inform visualization */
306  emit visualizationChanged(waypoints);
307  }
308 }

315 struct PathCompiler::waypoint PathCompiler::UavoToInternal(Waypoint::DataFields uavo)
316 {
317  double homeLLA[3];
318  double LLA[3];
319  double NED[3];
320  waypoint internalWaypoint;
321 
322  HomeLocation *homeLocation = HomeLocation::GetInstance(getObjectManager());
323  Q_ASSERT(homeLocation);
324  if (homeLocation == NULL)
325  return internalWaypoint;
326  HomeLocation::DataFields homeLocationData = homeLocation->getData();
327  homeLLA[0] = homeLocationData.Latitude / 10e6;
328  homeLLA[1] = homeLocationData.Longitude / 10e6;
329  homeLLA[2] = homeLocationData.Altitude;
330 
331  NED[0] = uavo.Position[Waypoint::POSITION_NORTH];
332  NED[1] = uavo.Position[Waypoint::POSITION_EAST];
333  NED[2] = uavo.Position[Waypoint::POSITION_DOWN];
334  Utils::CoordinateConversions().GetLLA(homeLLA, NED, LLA);
335 
336  internalWaypoint.latitude = LLA[0];
337  internalWaypoint.longitude = LLA[1];
338  internalWaypoint.altitude = LLA[2];
339  return internalWaypoint;
340 }

347 Waypoint::DataFields PathCompiler::InternalToUavo(waypoint internal)
348 {
349  Waypoint::DataFields uavo;

352  double LLA[3];
353  double NED[3];
354 
355  HomeLocation *homeLocation = HomeLocation::GetInstance(getObjectManager());
356  Q_ASSERT(homeLocation);
357  if (homeLocation == NULL)
358  return uavo;
359  HomeLocation::DataFields homeLocationData = homeLocation->getData();
360  homeLLA[0] = homeLocationData.Latitude / 10e6;
361  homeLLA[1] = homeLocationData.Longitude / 10e6;
362  homeLLA[2] = homeLocationData.Altitude;
363 
364  // TODO: Give the point a concept of altitude
365  LLA[0] = internal.latitude;
366  LLA[1] = internal.longitude;
367  LLA[2] = internal.altitude;
368 
369  Utils::CoordinateConversions().GetNED(homeLLA, LLA, NED);
370 
371  uavo.Position[Waypoint::POSITION_NORTH] = NED[0];
372  uavo.Position[Waypoint::POSITION_EAST] = NED[1];
373  uavo.Position[Waypoint::POSITION_DOWN] = NED[2];
374 
375  uavo.Action = Waypoint::ACTION_PATHTONEXT;
376 
377  uavo.Velocity[Waypoint::VELOCITY_NORTH] = 5;
378  uavo.Velocity[Waypoint::VELOCITY_EAST] = 0;
379  uavo.Velocity[Waypoint::VELOCITY_DOWN] = 0;
380 
381  return uavo;
