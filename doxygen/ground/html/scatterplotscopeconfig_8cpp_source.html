219 void Scatterplot2dScopeConfig::loadConfiguration(ScopeGadgetWidget *scopeGadgetWidget)
220 {
221  preparePlot(scopeGadgetWidget);
222  scopeGadgetWidget->setScope(this);
223  scopeGadgetWidget->startTimer(m_refreshInterval);
224 
225 
226  // Configure each data source
227  foreach (Plot2dCurveConfiguration* plotCurveConfig, m_scatterplotSourceConfigs)
228  {
229  QString uavObjectName = plotCurveConfig->uavObjectName;
230  QString uavFieldName = plotCurveConfig->uavFieldName;
231  QRgb color = plotCurveConfig->color;
232 
233  ScatterplotData* scatterplotData;
234 
235  switch(scatterplot2dType){
236  case SERIES2D:
237  scatterplotData = new SeriesPlotData(uavObjectName, uavFieldName);
238  break;
239  case TIMESERIES2D:
240  scatterplotData = new TimeSeriesPlotData(uavObjectName, uavFieldName);
241  break;
242  }
243 
244  scatterplotData->setXWindowSize(timeHorizon);
245  scatterplotData->setScalePower(plotCurveConfig->yScalePower);
246  scatterplotData->setMeanSamples(plotCurveConfig->yMeanSamples);
247  scatterplotData->setMathFunction(plotCurveConfig->mathFunction);
248 
249  //Generate the curve name
250  QString curveName = (scatterplotData->getUavoName()) + "." + (scatterplotData->getUavoFieldName());
251  if(scatterplotData->getHaveSubFieldFlag())
252  curveName = curveName.append("." + scatterplotData->getUavoSubFieldName());
253 
254  //Get the uav object
255  ExtensionSystem::PluginManager *pm = ExtensionSystem::PluginManager::instance();
256  UAVObjectManager *objManager = pm->getObject<UAVObjectManager>();
257  UAVDataObject* obj = dynamic_cast<UAVDataObject*>(objManager->getObject((scatterplotData->getUavoName())));
258  if(!obj) {
259  qDebug() << "Object " << scatterplotData->getUavoName() << " is missing";
260  return;
261  }
262 
263  //Get the units
264  QString units = getUavObjectFieldUnits(scatterplotData->getUavoName(), scatterplotData->getUavoFieldName());
265 
266  //Generate name with scaling factor appeneded
267  QString curveNameScaled;
268  if(plotCurveConfig->yScalePower == 0)
269  curveNameScaled = curveName + "(" + units + ")";
270  else
271  curveNameScaled = curveName + "(x10^" + QString::number(plotCurveConfig->yScalePower) + " " + units + ")";
272 
273  QString curveNameScaledMath;
274  if (plotCurveConfig->mathFunction == "None")
275  curveNameScaledMath = curveNameScaled;
276  else if (plotCurveConfig->mathFunction == "Boxcar average"){
277  curveNameScaledMath = curveNameScaled + " (avg)";
278  }
279  else if (plotCurveConfig->mathFunction == "Standard deviation"){
280  curveNameScaledMath = curveNameScaled + " (std)";
281  }
282  else
283  {
284  //Shouldn't be able to get here. Perhaps a new math function was added without
285  // updating this list?
286  Q_ASSERT(0);
287  }
288 
289  while(scopeGadgetWidget->getDataSources().keys().contains(curveNameScaledMath))
290  curveNameScaledMath=curveNameScaledMath+"*";
291 
292  //Create the curve plot
293  QwtPlotCurve* plotCurve = new QwtPlotCurve(curveNameScaledMath);
294  plotCurve->setPen(QPen(QBrush(QColor(color), Qt::SolidPattern), (qreal)1, Qt::SolidLine, Qt::SquareCap, Qt::BevelJoin));
295  plotCurve->setSamples(*(scatterplotData->getXData()), *(scatterplotData->getYData()));
296  plotCurve->attach(scopeGadgetWidget);
297  scatterplotData->setCurve(plotCurve);
298 
299  //Keep the curve details for later
300  scopeGadgetWidget->insertDataSources(curveNameScaledMath, scatterplotData);
301 
302  // Connect the UAVO
303  scopeGadgetWidget->connectUAVO(obj);
304  }
305  mutex.lock();
306  scopeGadgetWidget->replot();
307  mutex.unlock();
308 }

316 void Scatterplot2dScopeConfig::setGuiConfiguration(Ui::ScopeGadgetOptionsPage *options_page)
317 {
318  //Set the tab widget to 2D
319  options_page->tabWidget2d3d->setCurrentWidget(options_page->tabPlot2d);
320 
321  //Set the plot type
322  options_page->cmb2dPlotType->setCurrentIndex(options_page->cmb2dPlotType->findData(SCATTERPLOT2D));
323 
324  //add the configured 2D curves
325  options_page->lst2dCurves->clear(); //Clear list first
326 
327  foreach (Plot2dCurveConfiguration* plotData, m_scatterplotSourceConfigs) {

339  options_page->lst2dCurves->addItem(listItemDisplayText); // Add the name to the list
340  int itemIdx = options_page->lst2dCurves->count() - 1; // Get the index number for the new value
341  QListWidgetItem *listWidgetItem = options_page->lst2dCurves->item(itemIdx); //Find the widget item
342 
343  bool parseOK = false;
344  QRgb rgbColor;
345 
346  if(uavObjectName!="")
347  {
348  //Set the properties of the newly added list item
349  listItemDisplayText = uavObjectName + "." + uavFieldName;
350  rgbColor = (QRgb)varColor.toInt(&parseOK);
351  if(!parseOK)
352  rgbColor = qRgb(255,0,0);
353  }
354  else{
355  listItemDisplayText = "New graph";
356  rgbColor = qRgb(255,0,0);
357  }
358 
359  QColor color = QColor( rgbColor );
360  listWidgetItem->setText(listItemDisplayText);
361  listWidgetItem->setTextColor( color );
362 
363  //Store some additional data for the plot curve on the list item
364  listWidgetItem->setData(Qt::UserRole + ScopeGadgetOptionsPage::UR_UAVOBJECT, QVariant(uavObjectName));
365  listWidgetItem->setData(Qt::UserRole + ScopeGadgetOptionsPage::UR_UAVFIELD, QVariant(uavFieldName));
366  listWidgetItem->setData(Qt::UserRole + ScopeGadgetOptionsPage::UR_SCALE, QVariant(scale));
367  listWidgetItem->setData(Qt::UserRole + ScopeGadgetOptionsPage::UR_COLOR, varColor);
368  listWidgetItem->setData(Qt::UserRole + ScopeGadgetOptionsPage::UR_MEAN, QVariant(mean));
369  listWidgetItem->setData(Qt::UserRole + ScopeGadgetOptionsPage::UR_MATHFUNCTION, QVariant(mathFunction));
370 
371  //Select the row with the new name
372  options_page->lst2dCurves->setCurrentRow(itemIdx);
373  }
374 
375  //Select row 1st row in list

414 void Scatterplot2dScopeConfig::configureAxes(ScopeGadgetWidget *scopeGadgetWidget)
415 {
416  switch (scatterplot2dType)
417  {
418  case TIMESERIES2D: {
419  // Configure axes
420  scopeGadgetWidget->setAxisScaleDraw(QwtPlot::xBottom, new TimeScaleDraw());
421  uint NOW = QDateTime::currentDateTime().toTime_t();
422  scopeGadgetWidget->setAxisScale(QwtPlot::xBottom, NOW - timeHorizon / 1000, NOW);
423  break;
424  }
425  case SERIES2D:
426  default:
427  scopeGadgetWidget->setAxisScaleDraw(QwtPlot::xBottom, new QwtScaleDraw());
428  scopeGadgetWidget->setAxisScale(QwtPlot::xBottom, 0, timeHorizon);

431 
432  scopeGadgetWidget->setAxisAutoScale(QwtPlot::yLeft, true);
433  scopeGadgetWidget->setAxisLabelRotation(QwtPlot::xBottom, 0.0);
434  scopeGadgetWidget->setAxisLabelAlignment(QwtPlot::xBottom, Qt::AlignLeft | Qt::AlignBottom);
435  scopeGadgetWidget->axisWidget( QwtPlot::yRight )->setColorBarEnabled( false );
436  scopeGadgetWidget->enableAxis( QwtPlot::yRight, false );
437 
438 
439 
440  // Reduce the gap between the scope canvas and the axis scale
