00089     proto_tree *op_uavtalk_tree = NULL;

00092

00094     ti = proto_tree_add_item(tree, proto_op_uavtalk, tvb, 0, -1, ENC_NA);
00095

00097     op_uavtalk_tree = proto_item_add_subtree(ti, ett_op_uavtalk);

00106     ptvcursor_add(cursor, hf_op_uavtalk_len, 2, ENC_LITTLE_ENDIAN);
00107     ptvcursor_add(cursor, hf_op_uavtalk_objid, 4, ENC_LITTLE_ENDIAN);
00108 
00109     offset = ptvcursor_current_offset(cursor);
00110 
00111     ptvcursor_free(cursor);

00113     proto_tree_add_item(op_uavtalk_tree, hf_op_uavtalk_crc8, tvb, reported_length - UAVTALK_TRAILER_SIZE, UAVTALK_TRAILER_SIZE, ENC_LITTLE_ENDIAN);

00115     offset = UAVTALK_HEADER_SIZE;
00116   }
