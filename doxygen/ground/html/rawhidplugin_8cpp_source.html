00085 QList < Core::IDevice*> RawHIDConnection::availableDevices()
00086 {
00087     QList < Core::IDevice*> devices;
00088     QList<USBPortInfo> portsList;
00089 
00090     // Loop for all vendorIDs known by the board manager
00091     Core::BoardManager* brdMgr = Core::ICore::instance()->boardManager();
00092     QList<int> brdVID = brdMgr->getKnownVendorIDs();
00093     foreach(int vendorID, brdVID) {
00094         qDebug() << "[rawhidplugin] VendorID type known: " << vendorID;
00095         portsList = m_usbMonitor->availableDevices(vendorID, -1, -1,USBMonitor::Running);
00096         // We currently list devices by their serial number        
00097         USBDevice* dev = new USBDevice();
00098         foreach(USBPortInfo prt, portsList) {
00099             dev->setName(prt.serialNumber);
00100             dev->setDisplayName(prt.product);
00101             dev->setVendorID(prt.vendorID);
00102             dev->setProductID(prt.productID);
00103             devices.append(dev);
00104         }
00105     }
00106     return devices;
00107 }
