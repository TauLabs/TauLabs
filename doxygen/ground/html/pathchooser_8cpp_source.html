155 void PathChooser::slotBrowse()
156 {
157  emit beforeBrowsing();
158 
159  QString predefined = path();
160  if ((predefined.isEmpty() || !QFileInfo(predefined).isDir())
161  && !m_d->m_initialBrowsePathOverride.isNull()) {
162  predefined = m_d->m_initialBrowsePathOverride;
163  if (!QFileInfo(predefined).isDir())
164  predefined.clear();
165  }
166 
167  if (predefined.startsWith(":"))
168  predefined.clear();
169 
170  // Prompt for a file/dir
171  QString dialogTitle;
172  QString newPath;
173  switch (m_d->m_acceptingKind) {
174  case PathChooser::Directory:
175  newPath = QFileDialog::getExistingDirectory(this,
176  makeDialogTitle(tr("Choose a directory")), predefined);
177  break;
178 
179  case PathChooser::File: // fall through
180  case PathChooser::Command:
181  newPath = QFileDialog::getOpenFileName(this,
182  makeDialogTitle(tr("Choose a file")), predefined,
183  m_d->m_dialogFilter);
184  break;
185 
186  default:
187  ;
188  }
189 
190  // Delete trailing slashes unless it is "/"|"\\", only
191  if (!newPath.isEmpty()) {
192  newPath = QDir::toNativeSeparators(newPath);
193  if (newPath.size() > 1 && newPath.endsWith(QDir::separator()))
194  newPath.truncate(newPath.size() - 1);
195  setPath(newPath);
196  }
197 
198  emit browsingFinished();
199 }

211 bool PathChooser::validatePath(const QString &path, QString *errorMessage)
212 {
213  if (path.isEmpty()) {
214  if (errorMessage)
215  *errorMessage = tr("The path must not be empty.");
216  return false;
217  }
218 
219  const QFileInfo fi(path);
220  const bool isDir = fi.isDir();
221 
222  // Check if existing
223  switch (m_d->m_acceptingKind) {
224  case PathChooser::Directory: // fall through
225  case PathChooser::File:
226  if (!fi.exists()) {
227  if (errorMessage)
228  *errorMessage = tr("The path '%1' does not exist.").arg(path);
229  return false;
230  }
231  break;
232 
233  case PathChooser::Command: // fall through
234  default:
235  ;
236  }
237 
238  // Check expected kind
239  switch (m_d->m_acceptingKind) {
240  case PathChooser::Directory:
241  if (!isDir) {
242  if (errorMessage)
243  *errorMessage = tr("The path '%1' is not a directory.").arg(path);
244  return false;
245  }
246  break;
247 
248  case PathChooser::File:
249  if (isDir) {
250  if (errorMessage)
251  *errorMessage = tr("The path '%1' is not a file.").arg(path);
252  return false;
253  }
254  break;
255 
256  case PathChooser::Command:
257  // TODO do proper command validation
258  // i.e. search $PATH for a matching file
259  break;
260 
261  default:
262  ;
263  }
264 
265  return true;
266 }
