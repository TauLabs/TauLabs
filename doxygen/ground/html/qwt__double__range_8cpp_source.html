79 void QwtDoubleRange::setNewValue( double value, bool align )
80 {
81  d_data->prevValue = d_data->value;
82 
83  const double vmin = qMin( d_data->minValue, d_data->maxValue );
84  const double vmax = qMax( d_data->minValue, d_data->maxValue );
85 
86  if ( value < vmin )
87  {
88  if ( d_data->periodic && vmin != vmax )
89  {
90  d_data->value = value +
91  qwtCeilF( ( vmin - value ) / ( vmax - vmin ) ) * ( vmax - vmin );
92  }
93  else
94  d_data->value = vmin;
95  }
96  else if ( value > vmax )
97  {
98  if ( ( d_data->periodic ) && ( vmin != vmax ) )
99  {
100  d_data->value = value -
101  qwtCeilF( ( value - vmax ) / ( vmax - vmin ) ) * ( vmax - vmin );
102  }
103  else
104  d_data->value = vmax;
105  }
106  else
107  {
108  d_data->value = value;
109  }
110 
111  d_data->exactPrevValue = d_data->exactValue;
112  d_data->exactValue = d_data->value;
113 
114  if ( align )
115  {
116  if ( d_data->step != 0.0 )
117  {
118  d_data->value = d_data->minValue +
119  qRound( ( d_data->value - d_data->minValue ) / d_data->step ) * d_data->step;
120  }
121  else
122  d_data->value = d_data->minValue;
123 
124  const double minEps = 1.0e-10;
125  // correct rounding error at the border
126  if ( qFabs( d_data->value - d_data->maxValue ) < minEps * qAbs( d_data->step ) )
127  d_data->value = d_data->maxValue;
128 
129  // correct rounding error if value = 0
130  if ( qFabs( d_data->value ) < minEps * qAbs( d_data->step ) )
131  d_data->value = 0.0;
132  }
133 
134  if ( !d_data->isValid || d_data->prevValue != d_data->value )
135  {
136  d_data->isValid = true;
137  valueChange();
138  }
139 }

187 void QwtDoubleRange::setRange(
188  double vmin, double vmax, double vstep, int pageSize )
189 {
190  const bool rchg = ( d_data->maxValue != vmax || d_data->minValue != vmin );
191 
192  if ( rchg )
193  {
194  d_data->minValue = vmin;
195  d_data->maxValue = vmax;
196  }
197 
198  // look if the step width has an acceptable
199  // value or otherwise change it.
200  setStep( vstep );
201 
202  // limit page size
203  const int max =
204  int( qAbs( ( d_data->maxValue - d_data->minValue ) / d_data->step ) );
205  d_data->pageSize = qBound( 0, pageSize, max );
206 
207  // If the value lies out of the range, it
208  // will be changed. Note that it will not be adjusted to
209  // the new step width.
210  setNewValue( d_data->value, false );
211 
212  // call notifier after the step width has been
213  // adjusted.
214  if ( rchg )
215  rangeChange();
216 }

223 void QwtDoubleRange::setStep( double vstep )
224 {
225  const double intv = d_data->maxValue - d_data->minValue;
226 
227  double newStep;
228  if ( vstep == 0.0 )
229  {
230  const double defaultRelStep = 1.0e-2;
231  newStep = intv * defaultRelStep;
232  }
233  else
234  {
235  if ( ( intv > 0.0 && vstep < 0.0 ) || ( intv < 0.0 && vstep > 0.0 ) )
236  newStep = -vstep;
237  else
238  newStep = vstep;
239 
240  const double minRelStep = 1.0e-10;
241  if ( qFabs( newStep ) < qFabs( minRelStep * intv ) )
242  newStep = minRelStep * intv;
243  }
244 
245  if ( newStep != d_data->step )
246  {
247  d_data->step = newStep;
248  stepChange();
249  }
250 }
