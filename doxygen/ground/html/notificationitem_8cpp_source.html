272 QStringList valueToSoundList(QString value)
273 {
274  qNotifyDebug()<<"notificationItem valueToSoundList input param"<<value;
275  // replace point chr if exists
276  value = value.replace(',', '.');
277  QStringList numberParts = value.trimmed().split(".");
278  QStringList digitWavs;
279  bool negative=false;
280  if(numberParts.at(0).toInt()<0)
281  {
282  negative=true;
283  digitWavs.append("minus");
284  numberParts[0]=QString::number(numberParts.at(0).toInt()*-1);
285  }
286  if ( (numberParts.at(0).size() == 1) || (numberParts.at(0).toInt() < 20) ) {
287  // [1] check, is this number < 20, these numbers played by one wav file
288  digitWavs.append(numberParts.at(0));
289  } else {
290  int i=0;
291  // [2] store two lowest digits of number
292  int num = numberParts.at(0).right(2).toInt();
293  if (num < 20 && num != 0) {
294  // store eighter number in range [0...10) or in range [10...20)
295  digitWavs.append(numberParts.at(0).right(1 + num/11));
296  i=2;
297  }
298  // [3] prepend 100 and 1000 digits of number
299  for (;i<numberParts.at(0).size();i++)
300  {
301  int offset=0;
302  if(negative)
303  offset=1;
304  digitWavs.insert(offset,numberParts.at(0).at(numberParts.at(0).size()-i-1));
305  if(digitWavs.at(offset)==QString("0")) {
306  digitWavs.removeAt(offset);
307  continue;
308  }
309  if (i==1)
310  digitWavs.replace(0+offset,digitWavs.at(offset)+'0');
311  if (i==2)
312  digitWavs.insert(1+offset,"100");
313  if (i==3)
314  digitWavs.insert(1+offset,"1000");
315  }
316  }
317  // check, is there fractional part of number?
318  if (1 < numberParts.size()) {
319  digitWavs.append("point");
320  if (numberParts.at(1).size()==1) {
321  // this mean -> number < 1
322  digitWavs.append(numberParts.at(1));
323  } else {
324  // append fractional part of number
325  QString left = numberParts.at(1).left(1);
326  (left == "0") ? digitWavs.append(left) : digitWavs.append(left + '0');
327  digitWavs.append(numberParts.at(1).right(1));
328  }
329  }
330  qNotifyDebug()<<"notificationItem valueToSoundList return value"<<digitWavs;
331  return digitWavs;
332 }

351 QString NotificationItem::toString()
352 {
353  QString str;
354  UAVObjectField* field = getUAVObjectField();
355  QString value = stringFromValue(singleValue(), field);
356 
357  int pos = getSayOrder()-1;
358  QStringList lst;
359  lst.append(getSoundCaption(getSound1()));
360  lst.append(getSoundCaption(getSound2()));
361  lst.append(getSoundCaption(getSound3()));
362  QStringList valueSounds = valueToSoundList(value);
363  bool missed = false;
364  foreach(QString sound, valueSounds) {
365  if(checkSoundExists(sound).isEmpty()) {
366  missed = true;
367  break;
368  }
369  }
370 
371  // if not "Never" case
372  if(-1 != pos) {
373  if(missed)
374  lst.insert(pos, "[missed]" + value);
375  else
376  lst.insert(pos, value);
377  }
378  str = lst.join(" ");
379  return str;

382 QStringList& NotificationItem::toSoundList()
383 {
384  // tips:
385  // check of *.wav files exist needed for playing phonon queues;
386  // if phonon player don't find next file in queue, it buzz
387  UAVObjectField* field = getUAVObjectField();
388  QString value = stringFromValue(singleValue(), field);
389 
390  // generate queue of sound files to play
391  _messageSequence.clear();
392  int pos = getSayOrder()-1;
393  QStringList lst;
394  if(!getSound1().isEmpty())
395  lst.append(getSound1());
396  if(!getSound2().isEmpty())
397  lst.append(getSound2());
398  if(!getSound3().isEmpty())
399  lst.append(getSound3());
400 
401  // if not "Never" case
402  if(-1 != pos) {
403  QStringList valueSounds = valueToSoundList(value);
404  foreach(QString sound, valueSounds)
405  lst.insert(pos++, sound);
406  }
407 
408  foreach(QString sound, lst) {
409  QString path = checkSoundExists(sound);
410  if (!path.isEmpty()) {
411  _messageSequence.append(path);
412  } else {
413  _messageSequence.clear();
414  break;
415  }
416  }
417  return _messageSequence;
