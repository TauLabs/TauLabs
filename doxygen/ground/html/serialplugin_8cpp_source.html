00055 void SerialEnumerationThread::run()
00056 {
00057     QList <Core::IDevice*> devices = m_serial->availableDevices();
00058 
00059     while(m_running)
00060     {
00061         if(!m_serial->deviceOpened())
00062         {
00063             QList <Core::IDevice*> newDev = m_serial->availableDevices();
00064             // Note: if(devices != newDev) does not work here (QList of pointers)...
00065             bool different = false;
00066             if (newDev.length()!= devices.length()) {
00067                 different = true;
00068             } else {
00069                 for (int i= 0; i< newDev.length(); i++) {
00070                     Core::IDevice* oldd = devices.at(i);
00071                     Core::IDevice* newd = newDev.at(i);
00072                     different |= !(oldd->equals(newd));
00073                 }
00074             }
00075             if (different) {
00076                 devices = newDev;
00077                 emit enumerationChanged();
00078             }
00079         }
00080         msleep(2000); //update available devices every two seconds (doesn't need more)
00081     }
