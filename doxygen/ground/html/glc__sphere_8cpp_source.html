105 void GLC_Sphere::createMesh()
106 {
107 
108  Q_ASSERT(GLC_Mesh::isEmpty());
109 
110  GLfloatVector verticeFloat;
111  GLfloatVector normalsFloat;
112  GLfloatVector texelVector;
113 
114  int currentIndex=0;
115 
116  float wishedThetaStep= glc::PI / m_Discret;
117  float thetaRange= m_ThetaMax-m_ThetaMin;
118  int nbThetaSteps= (int) (thetaRange / wishedThetaStep) + 1 ;
119  float thetaStep= thetaRange / nbThetaSteps;
120 
121  float wishedPhiStep= wishedThetaStep;
122  float phiRange= m_PhiMax-m_PhiMin;
123  int nbPhiSteps= (int) (phiRange / wishedPhiStep) + 1 ;
124  float phiStep= phiRange / nbPhiSteps;
125 
126  float cost, sint, cosp, sinp, cospp, sinpp;
127  float xi, yi, zi, xf, yf, zf;
128  float theta= m_ThetaMin;
129  float phi= m_PhiMin;
130 
131  GLfloatVector thetaMinWire;
132  GLfloatVector thetaMaxWire;
133  GLfloatVector phiMinWire;
134  GLfloatVector phiMaxWire;
135 
136  GLC_Material* pMaterial;
137  if (hasMaterial())
138  pMaterial= this->firstMaterial();
139  else
140  pMaterial= new GLC_Material();
141 
142  // shaded face
143  for (int p= 0; p < nbPhiSteps; ++p)
144  {
145  cosp= cos (phi);
146  sinp= sin (phi);
147  cospp= cos (phi + phiStep);
148  sinpp= sin (phi + phiStep);
149 
150  zi = m_Radius * sinp;
151  zf = m_Radius * sinpp;
152 
153  IndexList indexFace;
154 
155  theta = m_ThetaMin;
156  int t;
157  for (t= 0; t <= nbThetaSteps; ++t)
158  {
159  cost= cos( theta );
160  sint= sin( theta );
161 
162  xi= m_Radius * cost * cosp;
163  yi= m_Radius * sint * cosp;
164  xf= m_Radius * cost * cospp;
165  yf= m_Radius * sint * cospp;
166 
167  verticeFloat << xf << yf << zf << xi << yi << zi;
168  normalsFloat << cost * cospp << sint * cospp << sinpp << cost * cosp << sint * cosp << sinp;
169  texelVector << static_cast<double>(t) * 1.0 / static_cast<double>(nbThetaSteps)
170  << static_cast<double>(p) * 1.0 / static_cast<double>(nbPhiSteps)
171  << static_cast<double>(t) * 1.0 / static_cast<double>(nbThetaSteps)
172  << static_cast<double>(p+1) * 1.0 / static_cast<double>(nbPhiSteps);
173 
174  indexFace << currentIndex + 2 * t << currentIndex + 2 * t + 1 ;
175  theta+= thetaStep;
176 
177  }
178 
179  currentIndex+= 2 * t;
180  addTrianglesStrip(pMaterial, indexFace);
181  phi+= phiStep;
182  }
183 
184  addVertice(verticeFloat);
185  addNormals(normalsFloat);
186  addTexels(texelVector);
187 
188  finish();
189 }
