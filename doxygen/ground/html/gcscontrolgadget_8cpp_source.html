107 void GCSControlGadget::manualControlCommandUpdated(UAVObject * obj) {
108 
109  // In GCS Receiver mode do not show the updates from ManualControl
110  if (gcsReceiverMode)
111  return;
112 
113  double roll = obj->getField("Roll")->getDouble();
114  double pitch = obj->getField("Pitch")->getDouble();
115  double yaw = obj->getField("Yaw")->getDouble();
116  double throttle = obj->getField("Throttle")->getDouble();
117  // Remap RPYT to left X/Y and right X/Y depending on mode
118  switch (controlsMode) {
119  case 1:
120  // Mode 1: LeftX = Yaw, LeftY = Pitch, RightX = Roll, RightY = Throttle
121  emit sticksChangedRemotely(yaw,-pitch,roll,throttle);
122  break;
123  case 2:
124  // Mode 2: LeftX = Yaw, LeftY = Throttle, RightX = Roll, RightY = Pitch
125  emit sticksChangedRemotely(yaw,throttle,roll,-pitch);
126  break;
127  case 3:
128  // Mode 3: LeftX = Roll, LeftY = Pitch, RightX = Yaw, RightY = Throttle
129  emit sticksChangedRemotely(roll,-pitch,yaw,throttle);
130  break;
131  case 4:
132  // Mode 4: LeftX = Roll, LeftY = Throttle, RightX = Yaw, RightY = Pitch;
133  emit sticksChangedRemotely(roll,throttle,yaw,-pitch);
134  break;
135  }

150 void GCSControlGadget::setGcsReceiver(double leftX, double leftY, double rightX, double rightY)
151 {
152  GCSReceiver *obj = getGcsReceiver();
153  Q_ASSERT(obj);
154  if (obj == NULL)
155  return;
156 
157  double newRoll;
158  double newPitch;
159  double newYaw;
160  double newThrottle;
161 
162  // Remap left X/Y and right X/Y to RPYT depending on mode
163  switch (controlsMode) {
164  case 1:
165  // Mode 1: LeftX = Yaw, LeftY = Pitch, RightX = Roll, RightY = Throttle
166  newRoll = rightX;
167  newPitch = -leftY;
168  newYaw = leftX;
169  newThrottle = rightY;
170  break;
171  case 2:
172  // Mode 2: LeftX = Yaw, LeftY = Throttle, RightX = Roll, RightY = Pitch
173  newRoll = rightX;
174  newPitch = -rightY;
175  newYaw = leftX;
176  newThrottle = leftY;
177  break;
178  case 3:
179  // Mode 3: LeftX = Roll, LeftY = Pitch, RightX = Yaw, RightY = Throttle
180  newRoll = leftX;
181  newPitch = -leftY;
182  newYaw = rightX;
183  newThrottle = rightY;
184  break;
185  case 4:
186  // Mode 4: LeftX = Roll, LeftY = Throttle, RightX = Yaw, RightY = Pitch;
187  newRoll = leftX;
188  newPitch = -rightY;
189  newYaw = rightX;
190  newThrottle = leftY;
191  break;
192  }
193 
194  GCSReceiver::DataFields gcsReceiverData = obj->getData();
195  gcsReceiverData.Channel[0] = 1500 + 500 * newThrottle;
196  gcsReceiverData.Channel[1] = 1500 + 500 * newRoll;
197  gcsReceiverData.Channel[2] = 1500 + 500 * newPitch;
198  gcsReceiverData.Channel[3] = 1500 + 500 * newYaw;
199  obj->setData(gcsReceiverData);
200  obj->updated();
201 
202  switch (controlsMode) {
203  case 1:
204  // Mode 1: LeftX = Yaw, LeftY = Pitch, RightX = Roll, RightY = Throttle
205  emit sticksChangedRemotely(newYaw,-newPitch,newRoll,newThrottle);
206  break;
207  case 2:
208  // Mode 2: LeftX = Yaw, LeftY = Throttle, RightX = Roll, RightY = Pitch
209  emit sticksChangedRemotely(newYaw,newThrottle,newRoll,-newPitch);
210  break;
211  case 3:
212  // Mode 3: LeftX = Roll, LeftY = Pitch, RightX = Yaw, RightY = Throttle
213  emit sticksChangedRemotely(newRoll,-newPitch,newYaw,newThrottle);
214  break;
215  case 4:
216  // Mode 4: LeftX = Roll, LeftY = Throttle, RightX = Yaw, RightY = Pitch;
217  emit sticksChangedRemotely(newRoll,newThrottle,newYaw,-newPitch);

219  }

299 void GCSControlGadget::readUDPCommand()
300 {
301  double pitch, yaw, roll, throttle;
302  while (control_sock->hasPendingDatagrams()) {
303  QByteArray datagram;
304  datagram.resize(control_sock->pendingDatagramSize());
305  control_sock->readDatagram(datagram.data(), datagram.size());
306  QDataStream readData(datagram);
307  bool badPack = false;
308  int state = 0;
309  while(!readData.atEnd() && !badPack)
310  {
311  double buffer;
312  readData >> buffer;
313  switch(state)
314  {
315  case 0:
316  if(buffer == 42){
317  state = 1;
318  }else{
319  state = 0;
320  badPack = true;
321  }
322  break;
323  case 1:
324  pitch = buffer;
325  state = 2;
326  break;
327  case 2:
328  yaw = buffer;
329  state = 3;
330  break;
331  case 3:
332  roll = buffer;
333  state = 4;
334  break;
335  case 4:
336  throttle = buffer;
337  state = 5;
338  break;
339  case 5:
340  if(buffer != 36 || !readData.atEnd())
341  badPack=true;
342  break;
343  }
344 
345  }
346  if(!badPack && ((GCSControlGadgetWidget *)m_widget)->getUDPControl())
347  {
348  ManualControlCommand * obj = getManualControlCommand();
349  bool update = false;
350 
351  if(pitch != obj->getField("Pitch")->getDouble()){
352  obj->getField("Pitch")->setDouble(constrain(pitch));
353  update = true;
354  }
355  if(yaw != obj->getField("Yaw")->getDouble()){
356  obj->getField("Yaw")->setDouble(constrain(yaw));
357  update = true;
358  }
359  if(roll != obj->getField("Roll")->getDouble()){
360  obj->getField("Roll")->setDouble(constrain(roll));
361  update = true;
362  }
363  if(throttle != obj->getField("Throttle")->getDouble()){
364  obj->getField("Throttle")->setDouble(constrain(throttle));
365  update = true;
366  }
367  if(update)
368  obj->updated();
369  }
370  }
371 
372  qDebug() << "Pitch: " << pitch << " Yaw: " << yaw << " Roll: " << roll << " Throttle: " << throttle;

481 void GCSControlGadget::axesValues(QListInt16 values)
482 {
483  int chMax = values.length();
484  if (rollChannel >= chMax || pitchChannel >= chMax ||
485  yawChannel >= chMax || throttleChannel >= chMax ) {
486  qDebug() << "GCSControl: configuration is inconsistent with current joystick! Aborting update.";
487  return;
488  }
489 
490  double rValue = (rollChannel > -1) ? values[rollChannel] : 0;
491  double pValue = (pitchChannel > -1) ? values[pitchChannel] : 0;
492  double yValue = (yawChannel > -1) ? values[yawChannel] : 0;
493  double tValue = (throttleChannel > -1) ? values[throttleChannel] : 0;
494  double max = 32767;
495 
496  if (rollChannel > -1) if(channelReverse[rollChannel]==true)rValue = -rValue;
497  if (pitchChannel > -1) if(channelReverse[pitchChannel]==true)pValue = -pValue;
498  if (yawChannel > -1) if(channelReverse[yawChannel]==true)yValue = -yValue;
499  if (throttleChannel > -1) if(channelReverse[throttleChannel]==true)tValue = -tValue;
500 
501 
502  if(joystickTime.elapsed() > JOYSTICK_UPDATE_RATE) {
503  joystickTime.restart();
504  // Remap RPYT to left X/Y and right X/Y depending on mode
505  // Mode 1: LeftX = Yaw, LeftY = Pitch, RightX = Roll, RightY = Throttle
506  // Mode 2: LeftX = Yaw, LeftY = THrottle, RightX = Roll, RightY = Pitch
507  // Mode 3: LeftX = Roll, LeftY = Pitch, RightX = Yaw, RightY = Throttle
508  // Mode 4: LeftX = Roll, LeftY = Throttle, RightX = Yaw, RightY = Pitch;
509  switch (controlsMode) {
510  case 1:
511  sticksChangedLocally(yValue/max,-pValue/max,rValue/max,-tValue/max);
512  break;
513  case 2:
514  sticksChangedLocally(yValue/max,-tValue/max,rValue/max,-pValue/max);
515  break;
516  case 3:
517  sticksChangedLocally(rValue/max,-pValue/max,yValue/max,-tValue/max);
518  break;
519  case 4:
520  sticksChangedLocally(rValue/max,-tValue/max,yValue/max,-pValue/max);
521  break;
522  }
523  }
