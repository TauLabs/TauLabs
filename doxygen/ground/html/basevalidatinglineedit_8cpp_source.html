00122 void BaseValidatingLineEdit::slotChanged(const QString &t)
00123 {
00124     m_bd->m_errorMessage.clear();
00125     // Are we displaying the initial text?
00126     const bool isDisplayingInitialText = !m_bd->m_initialText.isEmpty() && t == m_bd->m_initialText;
00127     const State newState = isDisplayingInitialText ?
00128                                DisplayingInitialText :
00129                                (validate(t, &m_bd->m_errorMessage) ? Valid : Invalid);
00130     setToolTip(m_bd->m_errorMessage);
00131     if (debug)
00132         qDebug() << Q_FUNC_INFO << t << "State" <<  m_bd->m_state << "->" << newState << m_bd->m_errorMessage;
00133     // Changed..figure out if valid changed. DisplayingInitialText is not valid,
00134     // but should not show error color. Also trigger on the first change.
00135     if (newState != m_bd->m_state || m_bd->m_firstChange) {
00136         const bool validHasChanged = (m_bd->m_state == Valid) != (newState == Valid);
00137         m_bd->m_state = newState;
00138         m_bd->m_firstChange = false;
00139         setTextColor(this, newState == Invalid ? m_bd->m_errorTextColor : m_bd->m_okTextColor);
00140         if (validHasChanged) {
00141             emit validChanged(newState == Valid);
00142             emit validChanged();
00143         }
00144     }
00145 }
