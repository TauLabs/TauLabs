00099 /* Check for complete lines read from the device and return them, moving the
00100  * buffer position. This is based on the assumption that '\n' is the new line
00101  * marker in any sane codec. */
00102 QByteArray ChannelBuffer::linesRead()
00103 {
00104     // Any new lines?
00105     const int lastLineIndex = data.lastIndexOf('\n');
00106     if (lastLineIndex == -1 || lastLineIndex <= bufferPos)
00107         return QByteArray();
00108     const int nextBufferPos = lastLineIndex + 1;
00109     const QByteArray lines = data.mid(bufferPos, nextBufferPos - bufferPos);
00110     bufferPos = nextBufferPos;
00111     return lines;
00112 }

00405 // Locate a binary in a directory, applying all kinds of
00406 // extensions the operating system supports.
00407 static QString checkBinary(const QDir &dir, const QString &binary)
00408 {
00409     // naive UNIX approach
00410     const QFileInfo info(dir.filePath(binary));
00411     if (info.isFile() && info.isExecutable())
00412         return info.absoluteFilePath();
00413 
00414     // Does the OS have some weird extension concept or does the
00415     // binary have a 3 letter extension?
00416     if (pathOS == OS_Unix)
00417         return QString();
00418     const int dotIndex = binary.lastIndexOf(QLatin1Char('.'));
00419     if (dotIndex != -1 && dotIndex == binary.size() - 4)
00420         return  QString();
00421 
00422     switch (pathOS) {
00423     case OS_Unix:
00424         break;
00425     case OS_Windows: {
00426             static const char *windowsExtensions[] = {".cmd", ".bat", ".exe", ".com" };
00427             // Check the Windows extensions using the order
00428             const int windowsExtensionCount = sizeof(windowsExtensions)/sizeof(const char*);
00429             for (int e = 0; e < windowsExtensionCount; e ++) {
00430                 const QFileInfo windowsBinary(dir.filePath(binary + QLatin1String(windowsExtensions[e])));
00431                 if (windowsBinary.isFile() && windowsBinary.isExecutable())
00432                     return windowsBinary.absoluteFilePath();
00433             }
00434         }
00435         break;
00436     case OS_Mac: {
00437             // Check for Mac app folders
00438             const QFileInfo appFolder(dir.filePath(binary + QLatin1String(".app")));
00439             if (appFolder.isDir()) {
00440                 QString macBinaryPath = appFolder.absoluteFilePath();
00441                 macBinaryPath += QLatin1String("/Contents/MacOS/");
00442                 macBinaryPath += binary;
00443                 const QFileInfo macBinary(macBinaryPath);
00444                 if (macBinary.isFile() && macBinary.isExecutable())
00445                     return macBinary.absoluteFilePath();
00446             }
00447         }
00448         break;
00449     }
00450     return QString();
00451 }
