295 double ConfigTaskWidget::listVar(QList<double> list)
296 {
297  double mean_accum = 0;
298  double var_accum = 0;
299  double mean;
300 
301  for(int i = 0; i < list.size(); i++)
302  mean_accum += list[i];
303  mean = mean_accum / list.size();
304 
305  for(int i = 0; i < list.size(); i++)
306  var_accum += (list[i] - mean) * (list[i] - mean);
307 
308  // Use unbiased estimator
309  return var_accum / (list.size() - 1);
310 }
