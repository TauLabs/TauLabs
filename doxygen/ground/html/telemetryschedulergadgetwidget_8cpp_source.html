00708 void QFrozenTableViewWithCopyPaste::copy()
00709 {
00710     QItemSelectionModel *selection = selectionModel();
00711     QModelIndexList indexes = selection->selectedIndexes();
00712 
00713     if(indexes.size() < 1)
00714       return;
00715 
00716     // QModelIndex::operator < sorts first by row, then by column.
00717     // this is what we need
00718     std::sort(indexes.begin(), indexes.end());
00719 
00720     // You need a pair of indexes to find the row changes
00721     QModelIndex previous = indexes.first();
00722     indexes.removeFirst();
00723     QString selected_text;
00724     QModelIndex current;
00725     foreach(current, indexes)
00726     {
00727         QVariant data = model()->data(previous);
00728         QString text = data.toString();
00729         // At this point `text` contains the text in one cell
00730         selected_text.append(text);
00731         // If you are at the start of the row the row number of the previous index
00732         // isn't the same.  Text is followed by a row separator, which is a newline.
00733         if (current.row() != previous.row())
00734         {
00735             selected_text.append(QLatin1Char('\n'));
00736         }
00737         // Otherwise it's the same row, so append a column separator, which is a tab.
00738         else
00739         {
00740             selected_text.append(QLatin1Char('\t'));
00741         }
00742         previous = current;
00743     }
00744 
00745     // add last element
00746     selected_text.append(model()->data(current).toString());
00747     selected_text.append(QLatin1Char('\n'));
00748     qApp->clipboard()->setText(selected_text);
00749 }

00751 void QFrozenTableViewWithCopyPaste::paste()
00752 {
00753     QItemSelectionModel *selection = selectionModel();
00754     QModelIndexList indexes = selection->selectedIndexes();
00755 
00756     QString selected_text = qApp->clipboard()->text();
00757     QStringList cells = selected_text.split(QRegExp(QLatin1String("\\n|\\t")));
00758     while(!cells.empty() && cells.back().size() == 0)
00759     {
00760         cells.pop_back(); // strip empty trailing tokens
00761     }
00762     int rows = selected_text.count(QLatin1Char('\n'));
00763     int cols = cells.size() / rows;
00764     if(cells.size() % rows != 0)
00765     {
00766         // error, uneven number of columns, probably bad data
00767         QMessageBox::critical(this, tr("Error"),
00768                               tr("Invalid clipboard data, unable to perform paste operation."));
00769         return;
00770     }
00771 
00772     // Give an error if there are too few rows. A better solution would be to expand the view size
00773     if(indexes.front().row() + rows > model()->rowCount())
00774     {
00775         // error, clipboard does not match current number of rows
00776         QMessageBox::critical(this, tr("Error"),
00777                               tr("Invalid operation, pasting would exceed the number of rows."));
00778         return;
00779     }
00780 
00781     // Give an error if there are too few columns. A better solution would be to expand the view size
00782     if(indexes.front().column() + cols > model()->columnCount())
00783     {
00784         // error, clipboard does not match current number of columns
00785         QMessageBox::critical(this, tr("Error"),
00786                               tr("Invalid operation, pasting would exceed the number of columns."));
00787         return;
00788     }
00789 
00790     // Paste the results into the appropriate cells
00791     int cell = 0;
00792     for(int row=0; row < rows; ++row)
00793     {
00794         for(int col=0; col < cols; ++col, ++cell)
00795         {
00796             QModelIndex index = model()->index(indexes.front().row() + row, indexes.front().column() + col, QModelIndex());
00797             model()->setData(index, cells[cell]);
00798         }
00799     }
00800 }
