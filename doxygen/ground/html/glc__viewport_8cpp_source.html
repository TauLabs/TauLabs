173 void GLC_Viewport::updateProjectionMat(void)
174 {
175  GLC_Context::current()->glcMatrixMode(GL_PROJECTION); // select The Projection Matrix
176  GLC_Context::current()->glcLoadIdentity(); // Reset The Projection Matrix
177 
178  if (m_UseParallelProjection)
179  {
180  const double ChampsVision = m_pViewCam->distEyeTarget() * m_ViewTangent;
181  const double height= ChampsVision;
182  const double with= ChampsVision * m_AspectRatio;
183  const double left= -with * 0.5;
184  const double right= -left;
185  const double bottom= - height * 0.5;
186  const double top= -bottom;
187  GLC_Context::current()->glcOrtho(left, right, bottom, top, m_dDistanceMini, m_DistanceMax);
188  }
189  else
190  {
191  const double yMax= m_dDistanceMini * tan(m_ViewAngle * glc::PI / 360.0);
192  const double yMin= -yMax;
193  const double xMax= yMax * m_AspectRatio;
194  const double xMin= -xMax;
195  GLC_Context::current()->glcFrustum(xMin, xMax, yMin, yMax, m_dDistanceMini, m_DistanceMax);
196  }
197 
198  // Save the projection matrix
199  m_ProjectionMatrix= GLC_Context::current()->projectionMatrix();
200 
201  GLC_Context::current()->glcMatrixMode(GL_MODELVIEW); // select The Modelview Matrix
202 }

216 {
217  const int halfSize= m_SelectionSquareSize / 2;
218  // Calculate the 4 points of the selection
219  //p1->p2
220  //
221  //p0 p3
222  QList<int> coordinates;
223  // Point 0
224  coordinates << (x - halfSize) << (y + halfSize);
225  // Point 1
226  coordinates << (x - halfSize) << (y - halfSize);
227  // Point 2
228  coordinates << (x + halfSize) << (y - halfSize);
229  // Point 3
230  coordinates << (x + halfSize) << (y + halfSize);
231 
232  // Unproject the 4 point
233  QList<GLC_Point3d> listOfPoint= unproject(coordinates);

235  Q_ASSERT(4 == listOfPoint.size());

239  const GLC_Plane rightPlane(listOfPoint.at(3), eye , listOfPoint.at(2));
240  const GLC_Plane upPlane(listOfPoint.at(2), eye, listOfPoint.at(1));
241  const GLC_Plane bottomPlane(listOfPoint.at(0), eye, listOfPoint.at(3));
242 
243  GLC_Frustum selectionFrustum(m_Frustum);
244  selectionFrustum.setLeftClippingPlane(leftPlane);
245  selectionFrustum.setRightClippingPlane(rightPlane);
246  selectionFrustum.setTopClippingPlane(upPlane);
247  selectionFrustum.setBottomClippingPlane(bottomPlane);

252 GLC_Point3d GLC_Viewport::unProject(int x, int y) const
253 {
254  // Z Buffer component of the given coordinate is between 0 and 1
255  GLfloat Depth;
256  // read selected point
257  glReadPixels(x, m_WindowVSize - y , 1, 1, GL_DEPTH_COMPONENT, GL_FLOAT, &Depth);
258 
259  // The current viewport opengl definition
260  GLint Viewport[4];
261  glGetIntegerv(GL_VIEWPORT, Viewport);
262 
263  // OpenGL ccordinate of selected point
264  GLdouble pX, pY, pZ;
265  glc::gluUnProject((GLdouble) x, (GLdouble) (m_WindowVSize - y) , Depth
266  , m_pViewCam->modelViewMatrix().getData(), m_ProjectionMatrix.getData(), Viewport, &pX, &pY, &pZ);
267 
268  return GLC_Point3d(pX, pY, pZ);
269 }

271 QList<GLC_Point3d> GLC_Viewport::unproject(const QList<int>& list)const
272 {
273  const int size= list.size();
274  Q_ASSERT((size % 2) == 0);
275 
276  // The current viewport opengl definition
277  GLint Viewport[4];
278  glGetIntegerv(GL_VIEWPORT, Viewport);
279 
280  // Z Buffer component of the given coordinate is between 0 and 1
281  GLfloat Depth;
282 
283  // Coordinate of readed points
284  GLdouble pX, pY, pZ;
285  QList<GLC_Point3d> unprojectedPoints;
286  for (int i= 0; i < size; i+= 2)
287  {
288  const int x= list.at(i);
289  const int y= m_WindowVSize - list.at(i + 1);
290  glReadPixels(x, y , 1, 1, GL_DEPTH_COMPONENT, GL_FLOAT, &Depth);
291 
292  glc::gluUnProject(static_cast<GLdouble>(x), static_cast<GLdouble>(y) , Depth , m_pViewCam->modelViewMatrix().getData()
293  , m_ProjectionMatrix.getData(), Viewport, &pX, &pY, &pZ);
294  unprojectedPoints.append(GLC_Point3d(pX, pY, pZ));
295  }
296 
297  return unprojectedPoints;
298 }

468 GLC_uint GLC_Viewport::meaningfulIdInsideSquare(GLint x, GLint y, GLsizei width, GLsizei height)
469 {
470  const int squareSize= width * height;
471  const GLsizei arraySize= squareSize * 4; // 4 -> R G B A
472  QVector<GLubyte> colorId(arraySize);
473 
474  // Get the array of pixels
475  glReadPixels(x, y, width, height, GL_RGBA, GL_UNSIGNED_BYTE, colorId.data());
476 
477  // Restore Background color
478  m_pQGLWidget->qglClearColor(m_BackgroundColor);
479 
480  QHash<GLC_uint, int> idHash;
481  QList<int> idWeight;
482 
483  // Find the most meaningful color
484  GLC_uint returnId= 0;
485  // There is nothing at the center
486  int maxWeight= 0;
487  int currentIndex= 0;
488  for (int i= 0; i < squareSize; ++i)
489  {
490  GLC_uint id= glc::decodeRgbId(&colorId[i * 4]);
491  if (idHash.contains(id))
492  {
493  const int currentWeight= ++(idWeight[idHash.value(id)]);
494  if (maxWeight < currentWeight)
495  {
496  returnId= id;
497  maxWeight= currentWeight;
498  }
499  }
500  else if (id != 0)
501  {
502  idHash.insert(id, currentIndex++);
503  idWeight.append(1);
504  if (maxWeight < 1)
505  {
506  returnId= id;
507  maxWeight= 1;
508  }
509  }
510  }
511 
512  return returnId;
513 }

630 void GLC_Viewport::setDistMinAndMax(const GLC_BoundingBox& bBox)
631 {
632  if(!bBox.isEmpty())
633  {
634  // The scene is not empty
635  GLC_Matrix4x4 matComp(m_pViewCam->modelViewMatrix());
636 
637  // The bounding Box in Camera coordinate
638  GLC_BoundingBox boundingBox(bBox);
639  boundingBox.transform(matComp);
640 
641  // Increase size of the bounding box
642  const double increaseFactor= 1.1;
643  // Convert box distance in sphere distance
644  const double center= fabs(boundingBox.center().z());
645  const double radius= boundingBox.boundingSphereRadius();
646  const double min= center - radius * increaseFactor;
647  const double max= center + radius * increaseFactor;
648 
649  GLC_Point3d camEye(m_pViewCam->eye());
650  camEye= matComp * camEye;
651 
652  if (min > 0.0)
653  {
654  // Outside bounding Sphere
655  m_dDistanceMini= min;
656  m_DistanceMax= max;
657  //qDebug() << "distmin" << m_dCamDistMin;
658  //qDebug() << "distmax" << m_dCamDistMax;
659  }
660  else
661  {
662  // Inside bounding Sphere
663  m_dDistanceMini= qMin(0.01 * radius, m_pViewCam->distEyeTarget() / 4.0);
664  m_DistanceMax= max;
665  //qDebug() << "inside distmin" << m_dCamDistMin;
666  //qDebug() << "inside distmax" << m_dCamDistMax;
667  }
668  }
