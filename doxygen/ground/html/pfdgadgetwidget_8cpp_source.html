00250 void PFDGadgetWidget::updateAttitude(UAVObject *object1) {
00251     setToolTipPrivate();
00252     UAVObjectField * rollField = object1->getField(QString("Roll"));
00253     UAVObjectField * yawField = object1->getField(QString("Yaw"));
00254     UAVObjectField * pitchField = object1->getField(QString("Pitch"));
00255     if(rollField && yawField && pitchField) {
00256         // These factors assume some things about the PFD SVG, namely:
00257         // - Roll, Pitch and Heading value in degrees
00258         // - Pitch lines are 300px high for a +20/-20 range, which means
00259         //   7.5 pixels per pitch degree.
00260         // TODO: loosen this constraint and only require a +/- 20 deg range,
00261         //       and compute the height from the SVG element.
00262         // Also: keep the integer value only, to avoid unnecessary redraws
00263         rollTarget = -floor(rollField->getDouble()*10)/10;
00264         if ((rollTarget - rollValue) > 180) {
00265             rollValue += 360;
00266         } else if (((rollTarget - rollValue) < -180)) {
00267             rollValue -= 360;
00268         }
00269         pitchTarget = floor(pitchField->getDouble()*7.5);
00270 
00271         // These factors assume some things about the PFD SVG, namely:
00272         // - Heading value in degrees
00273         // - "Scale" element is 540 degrees wide
00274 
00275         // Corvus Corax: "If you want a smooth transition between two angles, It is usually solved that by substracting
00276         // one from another, and if the result is >180 or <-180 I substract (respectively add) 360 degrees
00277         // to it. That way you always get the "shorter difference" to turn in."
00278         double fac = compassBandWidth/540;
00279         headingTarget = yawField->getDouble()*(-fac);
00280         if (headingTarget != headingTarget)
00281             headingTarget = headingValue; // NaN checking.
00282         if ((headingValue - headingTarget)/fac > 180) {
00283             headingTarget += 360*fac;
00284         } else if (((headingValue - headingTarget)/fac < -180)) {
00285             headingTarget -= 360*fac;
00286         }
00287         headingTarget = floor(headingTarget*10)/10; // Avoid stupid redraws
00288 
00289         if (!dialTimer.isActive())
00290             dialTimer.start(); // Rearm the dial Timer which might be stopped.
00291 
00292     } else {
00293         qDebug() << "Unable to get one of the fields for attitude update";
00294     }
00295 }
