249 void PFDGadgetWidget::updateAttitude(UAVObject *object1) {
250  setToolTipPrivate();
251  UAVObjectField * rollField = object1->getField(QString("Roll"));
252  UAVObjectField * yawField = object1->getField(QString("Yaw"));
253  UAVObjectField * pitchField = object1->getField(QString("Pitch"));
254  if(rollField && yawField && pitchField) {
255  // These factors assume some things about the PFD SVG, namely:
256  // - Roll, Pitch and Heading value in degrees
257  // - Pitch lines are 300px high for a +20/-20 range, which means
258  // 7.5 pixels per pitch degree.
259  // TODO: loosen this constraint and only require a +/- 20 deg range,
260  // and compute the height from the SVG element.
261  // Also: keep the integer value only, to avoid unnecessary redraws
262  rollTarget = -floor(rollField->getDouble()*10)/10;
263  if ((rollTarget - rollValue) > 180) {
264  rollValue += 360;
265  } else if (((rollTarget - rollValue) < -180)) {
266  rollValue -= 360;
267  }
268  pitchTarget = floor(pitchField->getDouble()*7.5);
269 
270  // These factors assume some things about the PFD SVG, namely:
271  // - Heading value in degrees
272  // - "Scale" element is 540 degrees wide
273 
274  // Corvus Corax: "If you want a smooth transition between two angles, It is usually solved that by substracting
275  // one from another, and if the result is >180 or <-180 I substract (respectively add) 360 degrees
276  // to it. That way you always get the "shorter difference" to turn in."
277  double fac = compassBandWidth/540;
278  headingTarget = yawField->getDouble()*(-fac);
279  if (headingTarget != headingTarget)
280  headingTarget = headingValue; // NaN checking.
281  if ((headingValue - headingTarget)/fac > 180) {
282  headingTarget += 360*fac;
283  } else if (((headingValue - headingTarget)/fac < -180)) {
284  headingTarget -= 360*fac;
285  }
286  headingTarget = floor(headingTarget*10)/10; // Avoid stupid redraws
287 
288  if (!dialTimer.isActive())
289  dialTimer.start(); // Rearm the dial Timer which might be stopped.
290 
291  } else {
292  qDebug() << "Unable to get one of the fields for attitude update";
293  }

394 void PFDGadgetWidget::setDialFile(QString dfn)
395 {
396  QGraphicsScene *l_scene = scene();
397  setBackgroundBrush(QBrush(Utils::StyleHelper::baseColor()));
398  if (QFile::exists(dfn) && m_renderer->load(dfn) && m_renderer->isValid())
399  {
400 /* The PFD element IDs are fixed, not like with the analog dial.
401  - Background: background
402  - Foreground: foreground (contains all fixed elements, including plane)
403  - earth/sky : world
404  - Roll scale: rollscale
405  - compass frame: compass (part of the foreground)
406  - compass band : compass-band
407  - Home point: homewaypoint
408  - Next point: nextwaypoint
409  - Home point bearing: homewaypoint-bearing
410  - Next point bearing: nextwaypoint-bearing
411  - Speed rectangle (left side): speed-bg
412  - Speed scale: speed-scale.
413  - Black speed window: speed-window.
414  - Altitude rectangle (right site): altitude-bg.
415  - Altitude scale: altitude-scale.
416  - Black altitude window: altitude-window.
417  - GCS Telemetry status arrow: gcstelemetry-XXXX
418  - Telemetry link rate: linkrate
419  - GPS status text: gps-txt
420  - Battery stats: battery-txt
421  */
422  l_scene->clear(); // Deletes all items contained in the scene as well.
423  m_background = new CachedSvgItem();
424  // All other items will be clipped to the shape of the background
425  m_background->setFlags(QGraphicsItem::ItemClipsChildrenToShape|
426  QGraphicsItem::ItemClipsToShape);
427  m_background->setSharedRenderer(m_renderer);
428  m_background->setElementId("background");
429  l_scene->addItem(m_background);
430 
431  m_world = new CachedSvgItem();
432  m_world->setParentItem(m_background);
433  m_world->setSharedRenderer(m_renderer);
434  m_world->setElementId("world");
435  l_scene->addItem(m_world);
436 
437  // red Roll scale: rollscale

915 // Movement is smooth, starts fast and slows down when

918 void PFDGadgetWidget::moveNeedles()
919 {
920  int dialCount = 3; // Gets decreased by one for each element

926  if (pfdError) {

932  // Heading
933  //
934  // If you want a smooth transition between two angles, It is usually solved that by substracting
935  // one from another, and if the result is >180 or <-180 I substract (respectively add) 360 degrees
936  // to it. That way you always get the "shorter difference" to turn in.
938  if (headingValue != headingTarget) {
939  double headingDiff;
940  if ((abs((headingValue-headingTarget)*10) > 5) && beSmooth ) {
941  headingDiff = (headingTarget - headingValue)/5;
942  } else {
943  headingDiff = headingTarget-headingValue;
944  dialCount--;
945  }
946  double threshold = 180*compassBandWidth/540;
947  // Note: rendering can jump oh so very slightly when crossing the 180 degree
948  // boundary, should not impact actual useability of the display.
949  if ((headingValue+headingDiff)>=threshold) {
950  // We went over 180: activate a -360 degree offset
951  headingDiff -= 2*threshold;
952  headingTarget -= 2*threshold;
953  } else if ((headingValue+headingDiff)<-threshold) {
954  // We went under -180: remove the -360 degree offset
955  headingDiff += 2*threshold;
956  headingTarget += 2*threshold;
957  }
958  QPointF opd = QPointF(headingDiff,0);
959  m_compassband->setTransform(QTransform::fromTranslate(opd.x(),opd.y()), true);
960  headingValue += headingDiff;

963  }
