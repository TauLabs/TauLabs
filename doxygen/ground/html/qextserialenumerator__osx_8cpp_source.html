26  io_iterator_t serialPortIterator = 0;
27  kern_return_t kernResult = KERN_FAILURE;
28  CFMutableDictionaryRef matchingDictionary;
29 
30  // first try to get any serialbsd devices, then try any USBCDC devices
31  if( !(matchingDictionary = IOServiceMatching(kIOSerialBSDServiceValue) ) ) {
32  qWarning("IOServiceMatching returned a NULL dictionary.");

63 bool QextSerialEnumerator::getServiceDetailsOSX( io_object_t service, QextPortInfo* portInfo )
64 {
65  bool retval = true;
66  CFTypeRef bsdPathAsCFString = NULL;
67  CFTypeRef productNameAsCFString = NULL;
68  CFTypeRef vendorIdAsCFNumber = NULL;
69  CFTypeRef productIdAsCFNumber = NULL;
70  // check the name of the modem's callout device
71  bsdPathAsCFString = IORegistryEntryCreateCFProperty(service, CFSTR(kIOCalloutDeviceKey),
72  kCFAllocatorDefault, 0);
73 
74  // wander up the hierarchy until we find the level that can give us the
75  // vendor/product IDs and the product name, if available
76  io_registry_entry_t parent;
77  kern_return_t kernResult = IORegistryEntryGetParentEntry(service, kIOServicePlane, &parent);
78  while( kernResult == KERN_SUCCESS && !vendorIdAsCFNumber && !productIdAsCFNumber )
79  {
80  if(!productNameAsCFString)
81  productNameAsCFString = IORegistryEntrySearchCFProperty(parent,
82  kIOServicePlane,
83  CFSTR("Product Name"),
84  kCFAllocatorDefault, 0);
85  vendorIdAsCFNumber = IORegistryEntrySearchCFProperty(parent,
86  kIOServicePlane,
87  CFSTR(kUSBVendorID),
88  kCFAllocatorDefault, 0);
89  productIdAsCFNumber = IORegistryEntrySearchCFProperty(parent,
90  kIOServicePlane,
91  CFSTR(kUSBProductID),
92  kCFAllocatorDefault, 0);
93  io_registry_entry_t oldparent = parent;
94  kernResult = IORegistryEntryGetParentEntry(parent, kIOServicePlane, &parent);
95  IOObjectRelease(oldparent);
96  }
97 
98  io_string_t ioPathName;
99  IORegistryEntryGetPath( service, kIOServicePlane, ioPathName );
100  portInfo->portName = ioPathName;
101 
102 
103  if( bsdPathAsCFString )
104  {
105  char path[MAXPATHLEN];
106  if( CFStringGetCString((CFStringRef)bsdPathAsCFString, path,
107  PATH_MAX, kCFStringEncodingUTF8) ) {
108  portInfo->physName = path;
109  portInfo->friendName = path;
110  }
111  CFRelease(bsdPathAsCFString);
112  }
113 
114  if(productNameAsCFString)
115  {
116  char productName[MAXPATHLEN];
117  if( CFStringGetCString((CFStringRef)productNameAsCFString, productName,
118  PATH_MAX, kCFStringEncodingUTF8) )
119  portInfo->friendName = productName;
120  CFRelease(productNameAsCFString);
121  }
122 
123  if(vendorIdAsCFNumber)
124  {
125  SInt32 vID;
126  if(CFNumberGetValue((CFNumberRef)vendorIdAsCFNumber, kCFNumberSInt32Type, &vID))
127  portInfo->vendorID = vID;
128  CFRelease(vendorIdAsCFNumber);
129  }
130 
131  if(productIdAsCFNumber)
132  {
133  SInt32 pID;
134  if(CFNumberGetValue((CFNumberRef)productIdAsCFNumber, kCFNumberSInt32Type, &pID))
135  portInfo->productID = pID;
136  CFRelease(productIdAsCFNumber);
137  }
138  IOObjectRelease(service);
139  return retval;
140 }

188 /*
189  Create matching dictionaries for the devices we want to get notifications for,
190  and add them to the current run loop. Invoke the callbacks that will be responding
191  to these notifications once to arm them, and discover any devices that
192  are currently connected at the time notifications are setup.
193 */
194 void QextSerialEnumerator::setUpNotifications( )
195 {
196  kern_return_t kernResult;
197  mach_port_t masterPort;
198  CFRunLoopSourceRef notificationRunLoopSource;
199  CFMutableDictionaryRef classesToMatch;
200  CFMutableDictionaryRef cdcClassesToMatch;
201  io_iterator_t portIterator;
202 
203  kernResult = IOMasterPort(MACH_PORT_NULL, &masterPort);
204  if (KERN_SUCCESS != kernResult) {
205  qDebug() << "IOMasterPort returned:" << kernResult;
206  return;
207  }
208 
209  classesToMatch = IOServiceMatching(kIOSerialBSDServiceValue);
210  if (classesToMatch == NULL)
211  qDebug("IOServiceMatching returned a NULL dictionary.");
212  else
213  CFDictionarySetValue(classesToMatch, CFSTR(kIOSerialBSDTypeKey), CFSTR(kIOSerialBSDAllTypes));
214 
215  if( !(cdcClassesToMatch = IOServiceNameMatching("AppleUSBCDC") ) ) {
216  qWarning("couldn't create cdc matching dict");
217  return;
218  }
219 
220  // Retain an additional reference since each call to IOServiceAddMatchingNotification consumes one.
221  classesToMatch = (CFMutableDictionaryRef) CFRetain(classesToMatch);
222  cdcClassesToMatch = (CFMutableDictionaryRef) CFRetain(cdcClassesToMatch);
223 
224  notificationPortRef = IONotificationPortCreate(masterPort);
225  if(notificationPortRef == NULL) {
226  qDebug("IONotificationPortCreate return a NULL IONotificationPortRef.");
227  return;
228  }
229 
230  notificationRunLoopSource = IONotificationPortGetRunLoopSource(notificationPortRef);
231  if (notificationRunLoopSource == NULL) {
232  qDebug("IONotificationPortGetRunLoopSource returned NULL CFRunLoopSourceRef.");
233  return;
234  }
235 
236  CFRunLoopAddSource(CFRunLoopGetCurrent(), notificationRunLoopSource, kCFRunLoopDefaultMode);
237 
238  kernResult = IOServiceAddMatchingNotification(notificationPortRef, kIOMatchedNotification, classesToMatch,
239  deviceDiscoveredCallbackOSX, this, &portIterator);
240  if (kernResult != KERN_SUCCESS) {
241  qDebug() << "IOServiceAddMatchingNotification return:" << kernResult;
242  return;
243  }
244 
245  // arm the callback, and grab any devices that are already connected
246  deviceDiscoveredCallbackOSX( this, portIterator );
247 
248  kernResult = IOServiceAddMatchingNotification(notificationPortRef, kIOMatchedNotification, cdcClassesToMatch,
249  deviceDiscoveredCallbackOSX, this, &portIterator);
250  if (kernResult != KERN_SUCCESS) {
251  qDebug() << "IOServiceAddMatchingNotification return:" << kernResult;
252  return;
253  }
254 
255  // arm the callback, and grab any devices that are already connected
256  deviceDiscoveredCallbackOSX( this, portIterator );
257 
258  kernResult = IOServiceAddMatchingNotification(notificationPortRef, kIOTerminatedNotification, classesToMatch,
259  deviceTerminatedCallbackOSX, this, &portIterator);
260  if (kernResult != KERN_SUCCESS) {
261  qDebug() << "IOServiceAddMatchingNotification return:" << kernResult;
262  return;
263  }
264 
265  // arm the callback, and clear any devices that are terminated
266  deviceTerminatedCallbackOSX( this, portIterator );
267 
268  kernResult = IOServiceAddMatchingNotification(notificationPortRef, kIOTerminatedNotification, cdcClassesToMatch,
269  deviceTerminatedCallbackOSX, this, &portIterator);

271  qDebug() << "IOServiceAddMatchingNotification return:" << kernResult;
272  return;
273  }
274 
275  // arm the callback, and clear any devices that are terminated

277 }
