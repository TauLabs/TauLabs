83 QImage SvgImageProvider::requestImage(const QString &id, QSize *size, const QSize &requestedSize)
84 {
85  QString svgFile = id;
86  QString element;
87  QString parameters;
88 
89  int separatorPos = id.indexOf('!');
90  if (separatorPos != -1) {
91  svgFile = id.left(separatorPos);
92  element = id.mid(separatorPos+1);
93  }
94 
95  int parametersPos = element.indexOf('?');
96  if (parametersPos != -1) {
97  parameters = element.mid(parametersPos+1);
98  element = element.left(parametersPos);
99  }
100 
101  int hSlicesCount = 0;
102  int hSlice = 0;
103  int vSlicesCount = 0;
104  int vSlice = 0;
105  int border = 0;
106  if (!parameters.isEmpty()) {
107  QRegExp hSliceRx("hslice=(\\d+):(\\d+)");
108  if (hSliceRx.indexIn(parameters) != -1) {
109  hSlice = hSliceRx.cap(1).toInt();
110  hSlicesCount = hSliceRx.cap(2).toInt();
111  }
112  QRegExp vSliceRx("vslice=(\\d+):(\\d+)");
113  if (vSliceRx.indexIn(parameters) != -1) {
114  vSlice = vSliceRx.cap(1).toInt();
115  vSlicesCount = vSliceRx.cap(2).toInt();
116  }
117  QRegExp borderRx("border=(\\d+)");
118  if (borderRx.indexIn(parameters) != -1) {
119  border = borderRx.cap(1).toInt();
120  }
121  }
122 
123  if (size)
124  *size = QSize();
125 
126  QSvgRenderer *renderer = loadRenderer(svgFile);
127  if (!renderer)
128  return QImage();
129 
130  qreal xScale = 1.0;
131  qreal yScale = 1.0;
132 
133  QSize docSize = renderer->defaultSize();
134 
135  if (!requestedSize.isEmpty()) {
136  if (!element.isEmpty()) {
137  QRectF elementBounds = renderer->boundsOnElement(element);
138  xScale = qreal(requestedSize.width())/elementBounds.width();
139  yScale = qreal(requestedSize.height())/elementBounds.height();
140  } else if (!docSize.isEmpty()) {
141  xScale = qreal(requestedSize.width())/docSize.width();
142  yScale = qreal(requestedSize.height())/docSize.height();
143  }
144  }
145 
146  //keep the aspect ratio
147  //TODO: how to configure it? as a part of image path?
148  xScale = yScale = qMin(xScale, yScale);
149 
150  if (!element.isEmpty()) {
151  if (!renderer->elementExists(element)) {
152  qWarning() << "invalid element:" << element << "of" << svgFile;
153  return QImage();
154  }
155 
156  QRectF elementBounds = renderer->boundsOnElement(element);
157  int elementWidth = qRound(elementBounds.width() * xScale);
158  int elementHeigh = qRound(elementBounds.height() * yScale);
159  int w = elementWidth;
160  int h = elementHeigh;
161  int x = 0;
162  int y = 0;
163 
164  if (hSlicesCount > 1) {
165  x = (w*hSlice)/hSlicesCount;
166  w = (w*(hSlice+1))/hSlicesCount - x;
167  }
168 
169  if (vSlicesCount > 1) {
170  y = (h*(vSlice))/vSlicesCount;
171  h = (h*(vSlice+1))/vSlicesCount - y;
172  }
173 
174  QImage img(w+border*2, h+border*2, QImage::Format_ARGB32_Premultiplied);
175  img.fill(0);
176  QPainter p(&img);
177  p.setRenderHints(QPainter::TextAntialiasing |
178  QPainter::Antialiasing |
179  QPainter::SmoothPixmapTransform);
180 
181  p.translate(-x+border,-y+border);
182  renderer->render(&p, element, QRectF(0, 0, elementWidth, elementHeigh));
183 
184  if (size)
185  *size = QSize(w, h);
186 
187  //img.save(element+parameters+".png");
188  return img;
189  } else {
190  //render the whole svg file
191  int w = qRound(docSize.width() * xScale);
192  int h = qRound(docSize.height() * yScale);
193 
194  QImage img(w, h, QImage::Format_ARGB32_Premultiplied);
195  img.fill(0);
196  QPainter p(&img);
197  p.setRenderHints(QPainter::TextAntialiasing |
198  QPainter::Antialiasing |
199  QPainter::SmoothPixmapTransform);
200 
201  p.scale(xScale, yScale);
202  renderer->render(&p, QRectF(QPointF(), QSizeF(docSize)));
203 
204  if (size)
205  *size = QSize(w, h);
206  return img;
207  }
208 }

222 {
