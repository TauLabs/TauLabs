237 void SplitterOrView::split(Qt::Orientation orientation)
238 {
239  Q_ASSERT(m_view);

244  m_layout->addWidget(m_splitter);

247  if (ourGadget) {
248  // Give our gadget to the new left or top SplitterOrView.
249  m_view->removeGadget();
250  m_splitter->addWidget(new SplitterOrView(m_uavGadgetManager, ourGadget));
251  m_splitter->addWidget(new SplitterOrView(m_uavGadgetManager));
252  } else {
253  m_splitter->addWidget(new SplitterOrView(m_uavGadgetManager));
254  m_splitter->addWidget(new SplitterOrView(m_uavGadgetManager));
255  }
256 
257  m_layout->setCurrentWidget(m_splitter);
258 
259  if (m_view) {
260  m_uavGadgetManager->emptyView(m_view);
261  delete m_view;
262  m_view = 0;

301 void SplitterOrView::unsplit()
302 {
303  if (!m_splitter)
304  return;
305  Q_ASSERT(m_splitter->count() == 1);
306  SplitterOrView *childSplitterOrView = qobject_cast<SplitterOrView*>(m_splitter->widget(0));
307  QSplitter *oldSplitter = m_splitter;
308  m_splitter = 0;
309 
310  if (childSplitterOrView->isSplitter()) {
311  Q_ASSERT(childSplitterOrView->view() == 0);
312  m_splitter = childSplitterOrView->takeSplitter();
313  m_layout->addWidget(m_splitter);
314  m_layout->setCurrentWidget(m_splitter);
315  } else {
316  UAVGadgetView *childView = childSplitterOrView->view();
317  Q_ASSERT(childView);
318  if (m_view) {
319  if (IUAVGadget *e = childView->gadget()) {
320  childView->removeGadget();
321  m_view->setGadget(e);
322  }
323  m_uavGadgetManager->emptyView(childView);
324  } else {
325  m_view = childSplitterOrView->takeView();
326  m_layout->addWidget(m_view);
327  }
328  m_layout->setCurrentWidget(m_view);
329  }
330  delete oldSplitter;
331  m_uavGadgetManager->setCurrentGadget(findFirstView()->gadget());
332 }

358 void SplitterOrView::restoreState(QSettings* qSettings)
359 {
360  QString mode = qSettings->value("type").toString();
361  if (mode == "splitter") {
362  qint32 orientation = qSettings->value("splitterOrientation").toInt();
363  QList<QVariant> sizesQVariant = qSettings->value("splitterSizes").toList();
364  m_sizes.clear();
365  foreach (QVariant value, sizesQVariant) {
366  m_sizes.append(value.toInt());
367  }
368  split((Qt::Orientation)orientation);
369  m_splitter->setSizes(m_sizes);
370  qSettings->beginGroup("side0");
371  static_cast<SplitterOrView*>(m_splitter->widget(0))->restoreState(qSettings);
372  qSettings->endGroup();
373  qSettings->beginGroup("side1");
374  static_cast<SplitterOrView*>(m_splitter->widget(1))->restoreState(qSettings);
375  qSettings->endGroup();
376  } else if (mode == "uavGadget") {
377  QString classId = qSettings->value("classId").toString();
378  int index = m_view->indexOfClassId(classId);
379  m_view->listSelectionActivated(index);
380  if(qSettings->childGroups().contains("gadget")) {
381  qSettings->beginGroup("gadget");
382  gadget()->restoreState(qSettings);
383  qSettings->endGroup();
384  }
385  }
