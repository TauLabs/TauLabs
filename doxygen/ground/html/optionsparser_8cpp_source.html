00064     while (!m_hasError) {
00065         if (!nextToken()) // move forward
00066             break;
00067         if (checkForEndOfOptions())
00068             break;
00069         if (checkForNoLoadOption())

00071         if (checkForTestOption())
00072             continue;
00073         if (checkForAppOption())
00074             continue;
00075         if (checkForPluginOption())
00076             continue;

00078             break;

00080         m_pmPrivate->arguments << m_currentArg;

00082     if (m_isDependencyRefreshNeeded)
00083         m_pmPrivate->resolveDependencies();
00084     return !m_hasError;
