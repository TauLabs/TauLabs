91  connect(message,SIGNAL(changed(GlobalMessage*)),this,SIGNAL(changedMessage(GlobalMessage*)));
92  connect(message,SIGNAL(changed(GlobalMessage*)),this,SIGNAL(changedInfo(GlobalMessage*)));

159 void GlobalMessaging::messageDeleted()
160 {
161  GlobalMessage * message=dynamic_cast<GlobalMessage *>(sender());
162  if(!message)
163  return;
164  switch(message->getType())
165  {
166  case ERROR:
167  errorList.removeAll(message);
168  emit deletedError();
169  break;
170  case WARNING:
171  warningList.removeAll(message);
172  emit deletedWarning();
173  break;
174  case INFO:
175  infoList.removeAll(message);
176  emit deletedInfo();
177  break;
178  default:
179  break;
180  }
181  emit deletedMessage();
182 }
