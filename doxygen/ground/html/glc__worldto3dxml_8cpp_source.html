72 bool GLC_WorldTo3dxml::exportTo3dxml(const QString& filename, GLC_WorldTo3dxml::ExportType exportType, bool exportMaterial)
73 {
74  m_3dxmlFileSet.clear();
75  m_ListOfOverLoadedOccurence.clear();
76  m_FileNameIncrement= 0;
77  m_ExportMaterial= exportMaterial;
78  m_FileName= filename;
79  m_ExportType= exportType;
80  bool isExported= false;
81  if (m_ExportType == Compressed3dxml)
82  {
83  m_p3dxmlArchive= new QuaZip(m_FileName);
84  isExported= m_p3dxmlArchive->open(QuaZip::mdCreate);
85  // Add the manifest
86  addManifest();
87 
88  }
89  else
90  {
91  m_AbsolutePath= QFileInfo(m_FileName).absolutePath() + QDir::separator();
92  QFile exportFile(m_FileName);
93  isExported= exportFile.open(QIODevice::WriteOnly);
94  exportFile.close();
95  }
96  if (isExported)
97  {
98  if (m_ExportMaterial && (m_ExportType != StructureOnly))
99  {
100  writeAllMaterialRelatedFilesIn3dxml();
101  }
102 
103  // Export the assembly structure from the list of structure reference
104  exportAssemblyStructure();
105 
106  if (m_ExportType != StructureOnly)
107  {
108  int previousQuantumValue= 0;
109  int currentQuantumValue= 0;
110  emit currentQuantum(currentQuantumValue);
111 
112  int currentRepIndex= 0;
113  const int size= m_ReferenceRepTo3dxmlFileName.size();
114  // Export the representation
115  QHash<const GLC_3DRep*, QString>::const_iterator iRep= m_ReferenceRepTo3dxmlFileName.constBegin();
116  while ((m_ReferenceRepTo3dxmlFileName.constEnd() != iRep) && continu())
117  {
118  write3DRep(iRep.key(), iRep.value());
119  ++iRep;
120 
121  // Progrees bar indicator
122  ++currentRepIndex;
123  currentQuantumValue = static_cast<int>((static_cast<double>(currentRepIndex) / size) * 100);
124  if (currentQuantumValue > previousQuantumValue)
125  {
126  emit currentQuantum(currentQuantumValue);
127  }
128  previousQuantumValue= currentQuantumValue;
129  if (!m_IsThreaded)
130  {
131  QCoreApplication::processEvents(QEventLoop::ExcludeUserInputEvents);
132  }
133  }
134  }
135  }
136 
137  emit currentQuantum(100);
138  return isExported;

351 void GLC_WorldTo3dxml::exportAssemblyFromOccurence(const GLC_StructOccurence* pOccurence)
352 {
353  if (pOccurence->isOrphan())
354  {
355  writeReference3D(pOccurence->structReference());
356  }
357  else
358  {
359  // Reference 3D
360  GLC_StructReference* pCurrentRef= pOccurence->structReference();
361  if (!m_ReferenceToIdHash.contains(pCurrentRef))
362  {
363  writeReference3D(pCurrentRef);
364  // Reference Rep
365  if (pCurrentRef->hasRepresentation())
366  {
367  GLC_3DRep* pCurrentRep= dynamic_cast<GLC_3DRep*>(pCurrentRef->representationHandle());
368  if (NULL != pCurrentRep && !m_ReferenceRepToIdHash.contains(pCurrentRep))
369  {
370  writeReferenceRep(pCurrentRep);
371  }
372  }
373  }
374  // Instance 3D and instance rep
375  GLC_StructInstance* pCurrentInstance= pOccurence->structInstance();
376  if (!m_InstanceToIdHash.contains(pCurrentInstance))
377  {
378  // Instance 3D
379  const unsigned int parentId= m_ReferenceToIdHash.value(pOccurence->parent()->structReference());
380  writeInstance3D(pCurrentInstance, parentId);
381 
382  // Instance Rep
383  if (pCurrentRef->hasRepresentation())
384  {
385  GLC_3DRep* pCurrentRep= dynamic_cast<GLC_3DRep*>(pCurrentRef->representationHandle());
386  const unsigned int parentId= m_ReferenceToIdHash.value(pCurrentRef);
387  if (NULL != pCurrentRep && !m_InstanceRep.contains(parentId))
388  {
389  writeInstanceRep(pCurrentRep, parentId);
390  }
391  }
392  }
393  }
394 
395  // Process children
396  const int childCount= pOccurence->childCount();
397  for (int i= 0; i < childCount; ++i)
398  {
399  exportAssemblyFromOccurence(pOccurence->child(i));
400  }
401 
402  // Add occurence with Overload properties to a list
403  if (m_World.collection()->contains(pOccurence->id()))
404  {
405  GLC_3DViewInstance* pInstance= m_World.collection()->instanceHandle(pOccurence->id());
406  Q_ASSERT(NULL != pInstance);
407  const bool isVisible= pInstance->isVisible();
408  const bool isOverload= !isVisible || !pInstance->renderPropertiesHandle()->isDefault() || pOccurence->isFlexible();
409  if (isOverload)
410  {
411  m_ListOfOverLoadedOccurence.append(pOccurence);
412  }
413  }

509 void GLC_WorldTo3dxml::writeGeometry(const GLC_Mesh* pMesh)
510 {
511  // Get the list of material id
512  QList<GLC_uint> materialList= pMesh->materialIds();
513  const int materialCount= materialList.size();
514 
515  m_pOutStream->writeStartElement("Rep");
516  m_pOutStream->writeAttribute("xsi:type", "PolygonalRepType");
517  m_pOutStream->writeAttribute("id", QString::number(++m_CurrentId));
518  const double masterAccuracy= pMesh->getLodAccuracy(0);
519  m_pOutStream->writeAttribute("accuracy", QString::number(masterAccuracy));
520  m_pOutStream->writeAttribute("solid", "1");
521  const int lodCount= pMesh->lodCount();
522  if (lodCount > 1)
523  {
524  // The mesh contains LOD
525  for (int i= 1; i < lodCount; ++i)
526  {
527  const double lodAccuracy= pMesh->getLodAccuracy(i);
528  m_pOutStream->writeStartElement("PolygonalLOD");
529  m_pOutStream->writeAttribute("accuracy", QString::number(lodAccuracy));
530  m_pOutStream->writeStartElement("Faces");
531  for (int matIndex= 0; matIndex < materialCount; ++matIndex)
532  {
533  const GLC_uint materialId= materialList.at(matIndex);
534  if (pMesh->lodContainsMaterial(i, materialId))
535  {
536  writeGeometryFace(pMesh, i, materialId);
537  }
538  }
539  m_pOutStream->writeEndElement(); // Faces
540  m_pOutStream->writeEndElement(); // PolygonalLOD
541  }
542  }
543 
544  // Master LOD
545  m_pOutStream->writeStartElement("Faces");
546  for (int matIndex= 0; matIndex < materialCount; ++matIndex)
547  {
548  const GLC_uint materialId= materialList.at(matIndex);
549  if (pMesh->lodContainsMaterial(0, materialId))
550  {
551  writeGeometryFace(pMesh, 0, materialId);
552  }
553  }
554  m_pOutStream->writeEndElement(); // Faces
555  if (!pMesh->wireDataIsEmpty())
556  {
557  writeEdges(pMesh);
558  }
559 
560  // Save Bulk data
561  m_pOutStream->writeStartElement("VertexBuffer");
562  {
563  // Get positions
564  GLfloatVector positionVector= pMesh->positionVector();
565  QString positions;
566  const int size= positionVector.size();
567  for (int i= 0; i < size; i+=3)
568  {
569  positions.append(QString::number(positionVector.at(i)));
570  positions.append(' ');
571  positions.append(QString::number(positionVector.at(i + 1)));
572  positions.append(' ');
573  positions.append(QString::number(positionVector.at(i + 2)));
574  positions.append(", ");
575  }
576  positions.remove(positions.size() - 2, 2);
577  m_pOutStream->writeTextElement("Positions", positions);
578  }
579  {
580  // Get normals
581  GLfloatVector normalVector= pMesh->normalVector();
582  QString normals;
583  const int size= normalVector.size();
584  for (int i= 0; i < size; i+=3)
585  {
586  normals.append(QString::number(normalVector.at(i)));
587  normals.append(' ');
588  normals.append(QString::number(normalVector.at(i + 1)));
589  normals.append(' ');
590  normals.append(QString::number(normalVector.at(i + 2)));
591  normals.append(", ");
592  }
593  normals.remove(normals.size() - 2, 2);
594  m_pOutStream->writeTextElement("Normals", normals);
595  }
596  {
597  // Get texture coordinates
598  GLfloatVector texelVector= pMesh->texelVector();
599  if (!texelVector.isEmpty())
600  {
601  QString texels;
602  const int size= texelVector.size();
603  for (int i= 0; i < size; i+=2)
604  {
605  texels.append(QString::number(texelVector.at(i)));
606  texels.append(' ');
607  texels.append(QString::number(texelVector.at(i + 1)));
608  texels.append(", ");
