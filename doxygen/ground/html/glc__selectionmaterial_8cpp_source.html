152 // Private services fonction
154 QGLContext* GLC_SelectionMaterial::sharingContext(const QGLContext* pContext)
155 {
156  QGLContext* pSharingContext= NULL;
157  QHash<const QGLContext*, GLC_Shader*>::const_iterator iContext= m_SelectionShaderHash.constBegin();
158 
159  while ((NULL == pSharingContext) && (iContext != m_SelectionShaderHash.constEnd()))
160  {
161  const QGLContext* pCurrentContext= iContext.key();
162  if (QGLContext::areSharing(pContext, pCurrentContext))
163  {
164  pSharingContext= const_cast<QGLContext*>(pCurrentContext);
165  }
166  ++iContext;
167  }
168 
169  return pSharingContext;
170 }
