140 void ModeManager::objectAdded(QObject *obj)
141 {
142  IMode *mode = Aggregation::query<IMode>(obj);
143  if (!mode)
144  return;
145 
146  m_mainWindow->addContextObject(mode);
147 
148  // Count the number of modes with a higher priority
149  int index = 0;
150  foreach (const IMode *m, m_modes)
151  if (m->priority() > mode->priority())
152  ++index;
153 
154  m_modes.insert(index, mode);
155  m_modeStack->insertTab(index, mode->widget(), mode->icon(), mode->name());
156 
157  // Register mode shortcut
158  ActionManager *am = m_mainWindow->actionManager();
159  const QString shortcutId = QLatin1String("GCS.Mode.") + mode->uniqueModeName();
160  QShortcut *shortcut = new QShortcut(m_mainWindow);
161  shortcut->setWhatsThis(tr("Switch to %1 mode").arg(mode->name()));
162  Command *cmd = am->registerShortcut(shortcut, shortcutId, QList<int>() << Constants::C_GLOBAL_ID);
163 
164  m_modeShortcuts.insert(index, cmd);
165  connect(cmd, SIGNAL(keySequenceChanged()), this, SLOT(updateModeToolTip()));
166 
167  setDefaultKeyshortcuts();
168 
169  m_signalMapper->setMapping(shortcut, mode->uniqueModeName());
170  connect(shortcut, SIGNAL(activated()), m_signalMapper, SLOT(map()));
171 }

269 void ModeManager::tabMoved(int from, int to)
270 {
271  IMode *mode = m_modes.at(from);
272  m_modes.remove(from);
273  m_modes.insert(to, mode);
274  Command *cmd = m_modeShortcuts.at(from);
275  m_modeShortcuts.remove(from);
276  m_modeShortcuts.insert(to, cmd);
277  setDefaultKeyshortcuts();
278  // Reprioritize, high priority means show to the left
279  if (!m_isReprioritizing) {
280  for (int i = 0; i < m_modes.count(); ++i) {

282  }
283  emit newModeOrder(m_modes);
284  }

287 void ModeManager::reorderModes(QMap<QString, int> priorities)
288 {
289  foreach (IMode *mode, m_modes)
290  mode->setPriority(priorities.value(QString(QLatin1String(mode->uniqueModeName())), mode->priority()));
291 
292  m_isReprioritizing = true;
293  IMode *current = currentMode();
294  // Bubble sort
295  bool swapped = false;
296  do {
297  swapped = false;
298  for (int i = 0; i < m_modes.count()-1; ++i) {
299  IMode *mode1 = m_modes.at(i);
300  IMode *mode2 = m_modes.at(i+1);
301 // qDebug() << "Comparing " << i << " to " << i+1 << " p1 " << mode1->priority() << " p2 " << mode2->priority();
302  if (mode2->priority() > mode1->priority()) {
303  m_modeStack->moveTab(i, i+1);
304 // qDebug() << "Tab moved from " << i << " to " << i+1;
305  swapped = true;
306  }
307  }
308  } while (swapped);
309  m_isReprioritizing = false;
310  m_modeStack->setCurrentIndex(0);
311  activateMode(current->uniqueModeName());
312  emit newModeOrder(m_modes);
313 }
