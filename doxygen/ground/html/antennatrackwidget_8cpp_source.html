00131 void AntennaTrackWidget::calcAntennaPosition(void)
00132 {
00134     double lat1, lat2, lon1, lon2, a, c, d, x, y, brng;
00135     double azimuth, elevation;
00136     double gcsAlt=TrackData.HomeAltitude; // Home MSL altitude
00137     double uavAlt=TrackData.Altitude; // UAV MSL altitude
00138     double dAlt=uavAlt-gcsAlt; // Altitude difference
00139 
00140     // Convert to radians
00141     lat1 = TrackData.HomeLatitude*(M_PI/180); // Home lat
00142     lon1 = TrackData.HomeLongitude*(M_PI/180); // Home lon
00143     lat2 = TrackData.Latitude*(M_PI/180); // UAV lat
00144     lon2 = TrackData.Longitude*(M_PI/180); // UAV lon
00145 
00146     // Bearing
00153     y = sin(lon2-lon1) * cos(lat2);
00154     x = cos(lat1) * sin(lat2) - sin(lat1) * cos(lat2) * cos(lon2-lon1);
00155     brng = atan2((sin(lon2-lon1)*cos(lat2)),(cos(lat1)*sin(lat2)-sin(lat1)*cos(lat2)*cos(lon2-lon1)))*(180/M_PI);
00156     if(brng<0)
00157         brng+=360;
00158 
00159     // bearing to stepper
00160     azimuth = brng;
00161 
00162     // Haversine formula for distance
00173     a = sin((lat2-lat1)/2) * sin((lat2-lat1)/2) +
00174             cos(lat1) * cos(lat2) *
00175             sin((lon2-lon1)/2) * sin((lon2-lon1)/2);
00176     c = 2 * atan2(sqrt(a), sqrt(1-a));
00177     d = WGS84_RADIUS_EARTH_KM * 1000 * c;
00178 
00179     // Elevation  v depends servo direction
00180     if(d!=0)
00181         elevation = 90-(atan(dAlt/d)*(180/M_PI));
00182     else
00183         elevation = 0;
00185
