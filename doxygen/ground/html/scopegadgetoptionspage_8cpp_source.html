355 void ScopeGadgetOptionsPage::on_cmbUAVObjects_currentIndexChanged(QString val)
356 {
357  options_page->cmbUAVField->clear();
358 
359  ExtensionSystem::PluginManager *pm = ExtensionSystem::PluginManager::instance();
360  UAVObjectManager *objManager = pm->getObject<UAVObjectManager>();
361  UAVDataObject* objData = dynamic_cast<UAVDataObject*>( objManager->getObject(val) );
362 
363  if (objData == NULL)
364  return;
365 
366  QList<UAVObjectField*> fieldList = objData->getFields();
367  foreach (UAVObjectField* field, fieldList) {
368  if(field->getType() == UAVObjectField::STRING || field->getType() == UAVObjectField::ENUM )
369  continue;
370 
371  if(field->getElementNames().count() > 1)
372  {
373  foreach(QString elemName , field->getElementNames())
374  {
375  options_page->cmbUAVField->addItem(field->getName() + "-" + elemName);
376  }
377  }
378  else
379  options_page->cmbUAVField->addItem(field->getName());
380  }
