122 void GLC_Rectangle::createMeshAndWire()
123 {
124  Q_ASSERT(GLC_Mesh::isEmpty());
125 
126  const GLfloat lgX= static_cast<const GLfloat>(m_L1 / 2.0);
127  const GLfloat lgY= static_cast<const GLfloat>(m_L2 / 2.0);
128 
129  GLfloatVector verticeVector;
130  GLfloatVector normalsVector;
131  GLfloatVector texelVector;
132 
133  // Wire data
134  GLfloatVector wireData;
135 
136  // the unique face of this rectangle
137  verticeVector << -lgX; verticeVector << -lgY; verticeVector << 0.0f;
138  normalsVector << 0.0f; normalsVector << 0.0f; normalsVector << 1.0f;
139  texelVector << 0.0f; texelVector << 0.0f;
140 
141  verticeVector << lgX; verticeVector << -lgY; verticeVector << 0.0f;
142  normalsVector << 0.0f; normalsVector << 0.0f; normalsVector << 1.0f;
143  texelVector << 1.0f; texelVector << 0.0f;
144 
145  verticeVector << lgX; verticeVector << lgY; verticeVector << 0.0f;
146  normalsVector << 0.0f; normalsVector << 0.0f; normalsVector << 1.0f;
147  texelVector << 1.0f; texelVector << 1.0f;
148 
149  verticeVector << -lgX; verticeVector << lgY; verticeVector << 0.0f;
150  normalsVector << 0.0f; normalsVector << 0.0f; normalsVector << 1.0f;
151  texelVector << 0.0f; texelVector << 1.0f;
152 
153  // Add bulk data in to the mesh
154  GLC_Mesh::addVertice(verticeVector);
155  GLC_Mesh::addNormals(normalsVector);
156  GLC_Mesh::addTexels(texelVector);
157 
158  // Add the first point of the rectangle for wire
159  verticeVector << -lgX; verticeVector << -lgY; verticeVector << 0.0f;
160  GLC_Geometry::addVerticeGroup(verticeVector);
161 
162  // Set the material to use
163  GLC_Material* pMaterial;
164  if (hasMaterial())
165  {
166  pMaterial= this->firstMaterial();
167  }
168  else
169  {
170  pMaterial= new GLC_Material();
171  }
172 
173  IndexList index;
174  // Face 1
175  index << 0 << 1 << 3 << 2;
176  GLC_Mesh::addTrianglesStrip(pMaterial, index);
177 
178  GLC_Mesh::finish();
179 }
