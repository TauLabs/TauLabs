00343 // Handle change of a combo. Return "false" if the combo
00344 // is to be reset (refuse new field).
00345 bool SubmitFieldWidget::comboIndexChange(int pos, int index)
00346 {
00347     const QString newField = m_d->fieldEntries.at(pos).combo->itemText(index);
00348     // If the field is visible elsewhere: focus the existing one and refuse
00349     if (!m_d->allowDuplicateFields) {
00350         const int existingFieldIndex = m_d->findField(newField, pos);
00351         if (existingFieldIndex != -1) {
00352             m_d->focusField(existingFieldIndex);
00353             return false;
00354         }
00355     }
00356     // Empty value: just change the field
00357     if (m_d->fieldValue(pos).isEmpty())
00358         return true;
00359     // Non-empty: Create a new field and reset the triggering combo
00360     createField(newField);
00361     return false;
00362 }
