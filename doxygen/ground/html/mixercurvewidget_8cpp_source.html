363 void MixerCurveWidget::initNodes(int numPoints)
364 {
365  // First of all, clear any existing list
366  if (nodeList.count()) {
367  foreach (MixerNode *node, nodeList ) {
368  foreach(Edge *edge, node->edges()) {
369  if (edge->sourceNode() == node) {
370  scene()->removeItem(edge);
371  }
372  }
373  scene()->removeItem(node);
374  }
375 
376  nodeList.clear();
377  }
378 
379  // Create the nodes and edges
380  MixerNode* prevNode = 0;
381  for (int i=0; i<numPoints; i++) {
382 
383  MixerNode *node = getNode(i);
384 
385  nodeList.append(node);
386  scene()->addItem(node);
387 
388  node->setPos(0,0);
389 
390  if (prevNode) {
391  scene()->addItem(getEdge(i, prevNode, node));
392  }
393 
394  prevNode = node;
395  }
396 }

428 void MixerCurveWidget::setCurve(const QList<double>* points)
429 {
430  curveUpdating = true;
431 
432  int ptCnt = points->count();
433  if (nodeList.count() != ptCnt)
434  initNodes(ptCnt);
435 
436  double range = curveMax - curveMin;
437 
438  qreal w = plot->boundingRect().width()/(ptCnt-1);
439  qreal h = plot->boundingRect().height();
440  for (int i=0; i<ptCnt; i++) {
441 
442  double val = (points->at(i) < curveMin) ? curveMin : (points->at(i) > curveMax) ? curveMax : points->at(i);
443 
444  val += range;
445  val -= (curveMin + range);
446  val /= range;
447 
448  MixerNode* node = nodeList.at(i);
449  node->setPos(w*i, h - (val*h));
450  node->verticalMove(true);
451 
452  node->update();
453  }
454  curveUpdating = false;
455 
456  update();
457 
458  emit curveUpdated();
459 }
