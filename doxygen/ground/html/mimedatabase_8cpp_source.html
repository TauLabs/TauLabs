00655 // Evaluate a magic match rule like
00656 //  <match value="must be converted with BinHex" type="string" offset="11"/>
00657 //  <match value="0x9501" type="big16" offset="0:64"/>
00658 static bool addMagicMatchRule(const QXmlStreamAttributes &atts,
00659                               MagicRuleMatcher *ruleMatcher,
00660                               QString *errorMessage)
00661 {
00662     const QString type = atts.value(QLatin1String(matchTypeAttributeC)).toString();
00663     if (type != QLatin1String(matchStringTypeValueC)) {
00664         qWarning("%s: match type %s is not supported.", Q_FUNC_INFO, type.toUtf8().constData());
00665         return true;
00666     }
00667     const QString value = atts.value(QLatin1String(matchValueAttributeC)).toString();
00668     if (value.isEmpty()) {
00669         *errorMessage = QString::fromLatin1("Empty match value detected.");
00670         return false;
00671     }
00672     // Parse for offset as "1" or "1:10"
00673     int startPos, endPos;
00674     const QString offsetS = atts.value(QLatin1String(matchOffsetAttributeC)).toString();
00675     const int colonIndex = offsetS.indexOf(QLatin1Char(':'));
00676     const QString startPosS = colonIndex == -1 ? offsetS : offsetS.mid(0, colonIndex);
00677     const QString endPosS   = colonIndex == -1 ? offsetS : offsetS.mid(colonIndex + 1);
00678     if (!parseNumber(startPosS, &startPos, errorMessage) || !parseNumber(endPosS, &endPos, errorMessage))
00679         return false;
00680     if (debugMimeDB)
00681         qDebug() << Q_FUNC_INFO << value << startPos << endPos;
00682     ruleMatcher->add(QSharedPointer<MagicRule>(MagicRule::createStringRule(value, startPos, endPos)));
00683     return true;
00684 }

00695         case QXmlStreamReader::StartElement:

00714             }
00715                 break;

00717                 // comments have locale attributes. We want the default, English one

00733             case ParseMagic: {

00737                     if (!parseNumber(priorityS, &priority, errorMessage))

00742                 ruleMatcher->setPriority(priority);
00743             }
00744                 break;

00747                     return false;
00748                 break;

00750                 reader.raiseError(QString::fromLatin1("Unexpected element <%1>").arg(reader.name().toString()));
00751                 break;
00752             default:
00753                 break;
00754             } // switch nextStage
00755             break;
00756         // continue switch QXmlStreamReader::Token...
00757         case QXmlStreamReader::EndElement: // Finished element
00758             if (reader.name() == QLatin1String(mimeTypeTagC)) {
00759                 if (!process(MimeType(data), errorMessage))
00760                     return false;
00761                 data.clear();
00762             } else {
00763                 // Finished a match sequence
00764                 if (reader.name() == QLatin1String(QLatin1String(magicTagC))) {
00765                     data.magicMatchers.push_back(QSharedPointer<IMagicMatcher>(ruleMatcher));
00766                     ruleMatcher = 0;
00767                 }
00768             }
00769             break;
00770 
00771         default:
00772             break;
00773         } // switch reader.readNext()
00774     }
00775 
00776     if (reader.hasError()) {
00777         *errorMessage = QString::fromLatin1("An error has been encountered at line %1 of %2: %3:").arg(reader.lineNumber()).arg(fileName, reader.errorString());
00778         return false;
00779     }
00780     return true;

00802 /* MimeDatabasePrivate: Requirements for storage:
00803  * - Must be robust in case of incomplete hierachies, dangling entries
00804  * - Plugins will not load and register their mime types in order
00805  *   of inheritance.
00806  * - Multiple inheritance (several subClassesOf) can occur
00807  * - Provide quick lookup by name
00808  * - Provide quick lookup by file type.
00809  * This basically rules out some pointer-based tree, so the structure choosen
00810  * is:
00811  * - An alias map <QString->QString> for mapping aliases to types
00812  * - A Map <QString-MimeMapEntry> for the types (MimeMapEntry being a pair of
00813  *   MimeType and (hierarchy) level.
00814  * - A map  <QString->QString> representing parent->child relations (enabling
00815  *   recursing over children)
00816  * Using strings avoids dangling pointers.
00817  * The hierarchy level is used for mapping by file types. When findByFile()
00818  * is first called after addMimeType() it recurses over the hierarchy and sets
00819  * the hierarchy level of the entries accordingly (0 toplevel, 1 first
00820  * order...). It then does several passes over the type map, checking the
00821  * globs for maxLevel, maxLevel-1....until it finds a match (idea being to
00822  * to check the most specific types first). Starting a recursion from the
00823  * leaves is not suitable since it will hit parent nodes several times. */

00903 bool MimeDatabasePrivate::addMimeType(MimeType mt)
00904 {
00905     if (!mt)
00906         return false;
00907 
00908     const QString type = mt.type();
00909     // Hack: Add a magic text matcher to "text/plain" and the fallback matcher to
00910     // binary types "application/octet-stream"
00911     if (type == QLatin1String(textTypeC)) {
00912         mt.addMagicMatcher(QSharedPointer<IMagicMatcher>(new Internal::HeuristicTextMagicMatcher));
00913     } else {
00914         if (type == QLatin1String(binaryTypeC))
00915              mt.addMagicMatcher(QSharedPointer<IMagicMatcher>(new Internal::BinaryMatcher));
00916     }
00917     // insert the type.
00918     m_typeMimeTypeMap.insert(type, MimeMapEntry(mt));
00919     // Register the children, resolved via alias map. Note that it is still
00920     // possible that aliases end up in the map if the parent classes are not inserted
00921     // at this point (thus their aliases not known).
00922     const QStringList subClassesOf = mt.subClassesOf();
00923     if (!subClassesOf.empty()) {
00924         const QStringList::const_iterator socend = subClassesOf.constEnd();
00925         for (QStringList::const_iterator soit = subClassesOf.constBegin(); soit !=  socend; ++soit)
00926             m_parentChildrenMap.insert(resolveAlias(*soit), type);
00927     }
00928     // register aliasses
00929     const QStringList aliases = mt.aliases();
00930     if (!aliases.empty()) {
00931         const QStringList::const_iterator cend = aliases.constEnd();
00932         for (QStringList::const_iterator it = aliases.constBegin(); it !=  cend; ++it)
00933             m_aliasMap.insert(*it, type);
00934     }
00935     m_maxLevel = -1; // Mark as dirty
00936     return true;
00937 }

00945 void MimeDatabasePrivate::raiseLevelRecursion(MimeMapEntry &e, int level)
00946 {
00947     if (e.level == Dangling || e.level < level)
00948         e.level = level;
00949     if (m_maxLevel < level)
00950         m_maxLevel = level;
00951     // At all events recurse over children since nodes might have been
00952     // added.
00953     const QStringList childTypes = m_parentChildrenMap.values(e.type.type());
00954     if (childTypes.empty())
00955         return;
00956     // look them up in the type->mime type map
00957     const int nextLevel = level + 1;
00958     const TypeMimeTypeMap::iterator tm_end = m_typeMimeTypeMap.end();
00959     const QStringList::const_iterator cend = childTypes.constEnd();
00960     for (QStringList::const_iterator it = childTypes.constBegin(); it !=  cend; ++it) {
00961         const TypeMimeTypeMap::iterator tm_it = m_typeMimeTypeMap.find(resolveAlias(*it));
00962         if (tm_it == tm_end) {
00963             qWarning("%s: Inconsistent mime hierarchy detected, child %s of %s cannot be found.",
00964                      Q_FUNC_INFO, it->toUtf8().constData(), e.type.type().toUtf8().constData());
00965         } else {
00966             raiseLevelRecursion(*tm_it, nextLevel);
00967         }
00968     }

00971 void MimeDatabasePrivate::determineLevels()
00972 {
00973     // Loop over toplevels and recurse down their hierarchies.
00974     // Determine top levels by subtracting the children from the parent
00975     // set. Note that a toplevel at this point might have 'subclassesOf'
00976     // set to some class that is not in the DB, so, checking for an empty
00977     // 'subclassesOf' set is not sufficient to find the toplevels.
00978     // First, take the parent->child entries  whose parent exists and build
00979     // sets of parents/children
00980     QSet<QString> parentSet, childrenSet;
00981     const ParentChildrenMap::const_iterator pcend = m_parentChildrenMap.constEnd();
00982     for (ParentChildrenMap::const_iterator it =  m_parentChildrenMap.constBegin(); it !=  pcend; ++it)
00983         if (m_typeMimeTypeMap.contains(it.key())) {
00984             parentSet.insert(it.key());
00985             childrenSet.insert(it.value());
00986         }
00987     const QSet<QString> topLevels = parentSet.subtract(childrenSet);
00988     if (debugMimeDB)
00989         qDebug() << Q_FUNC_INFO << "top levels" << topLevels;
00990     const TypeMimeTypeMap::iterator tm_end = m_typeMimeTypeMap.end();
00991     const QSet<QString>::const_iterator tl_cend = topLevels.constEnd();
00992     for (QSet<QString>::const_iterator tl_it =  topLevels.constBegin(); tl_it !=  tl_cend; ++tl_it) {
00993         const TypeMimeTypeMap::iterator tm_it = m_typeMimeTypeMap.find(resolveAlias(*tl_it));
00994         if (tm_it == tm_end) {
00995             qWarning("%s: Inconsistent mime hierarchy detected, top level element %s cannot be found.",
00996                      Q_FUNC_INFO, tl_it->toUtf8().constData());
00997         } else {
00998             raiseLevelRecursion(tm_it.value(), 0);
00999         }
01000     }
01001 }

01037 // Returns a mime type or Null one if none found
01038 MimeType MimeDatabasePrivate::findByFile(const QFileInfo &f, unsigned *priorityPtr) const
01039 {
01040     typedef QList<MimeMapEntry> MimeMapEntryList;
01041 
01042     // Is the hierarchy set up in case we find several matches?
01043     if (m_maxLevel < 0) {
01044         MimeDatabasePrivate *db = const_cast<MimeDatabasePrivate *>(this);
01045         db->determineLevels();
01046     }
01047     // Starting from max level (most specific): Try to find a match of
01048     // best (max) priority. Return if a glob match triggers.
01049     *priorityPtr = 0;
01050     unsigned maxPriority = 0;
01051     MimeType rc;
01052     Internal::FileMatchContext context(f);
01053     const TypeMimeTypeMap::const_iterator cend = m_typeMimeTypeMap.constEnd();
01054     for (int level = m_maxLevel; level >= 0; level--)
01055         for (TypeMimeTypeMap::const_iterator it = m_typeMimeTypeMap.constBegin(); it != cend; ++it)
01056             if (it.value().level == level) {
01057                 const unsigned priority = it.value().type.matchesFile(context);
01058                 if (debugMimeDB > 1)
01059                     qDebug() <<  "pass" << level << it.value().type.type() << " matches " << priority;
01060                 if (priority)
01061                     if (priority > maxPriority) {

01063                         maxPriority = priority;
01064                         // Glob (exact) match?! We are done
01065                         if (maxPriority == MimeType::GlobMatchPriority) {
01066                             *priorityPtr = priority;
01067                             return rc;
01068                         }
01069                     }
01070             }
