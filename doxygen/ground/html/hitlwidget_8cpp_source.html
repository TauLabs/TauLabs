75 void HITLWidget::startButtonClicked()
76 {
77  QThread* mainThread = QThread::currentThread();
78  qDebug() << "Main Thread: "<< mainThread;
79 
80  //Allow only one instance per simulator
81  if(Simulator::Instances().indexOf(settings.simulatorId) != -1)
82  {
83  widget->textBrowser->append(settings.simulatorId + " alreary started!");
84  return;
85  }
86 
87  if(!HITLPlugin::typeSimulators.size())
88  {
89  qxtLog->info("There is no registered simulators, add through HITLPlugin::addSimulator");
90  return;
91  }
92 
93  // Stop running process if one is active
94  if(simulator)
95  {
96  QMetaObject::invokeMethod(simulator, "onDeleteSimulator",Qt::QueuedConnection);
97  simulator = NULL;
98  }
99 
100  if(settings.hostAddress == "" || settings.inPort == 0)
101  {
102  widget->textBrowser->append("Before start, set UDP parameters in options page!");
103  return;
104  }
105 
106  SimulatorCreator* creator = HITLPlugin::getSimulatorCreator(settings.simulatorId);
107  simulator = creator->createSimulator(settings);
108 
109  // move to thread <--[BCH]
110  simulator->setName(creator->Description());
111  simulator->setSimulatorId(creator->ClassId());
112 
113  connect(simulator, SIGNAL(processOutput(QString)), this, SLOT(onProcessOutput(QString)));
114 
115  // Setup process
116  widget->textBrowser->append(QString("[%1] Starting %2... ").arg(QTime::currentTime().toString("hh:mm:ss")).arg(creator->Description()));
117  qxtLog->info("HITL: Starting " + creator->Description());
118 
119  // Start bridge
120  bool ret = QMetaObject::invokeMethod(simulator, "setupProcess",Qt::QueuedConnection);
121  if(ret)
122  {
123  Simulator::setInstance(settings.simulatorId);
124 
125  widget->startButton->setEnabled(false);
126  widget->stopButton->setEnabled(true);
127  qxtLog->info("HITL: Starting bridge, initializing flight simulator and Autopilot connections");
128 
129  connect(simulator, SIGNAL(autopilotConnected()), this, SLOT(onAutopilotConnect()),Qt::QueuedConnection);
130  connect(simulator, SIGNAL(autopilotDisconnected()), this, SLOT(onAutopilotDisconnect()),Qt::QueuedConnection);
131  connect(simulator, SIGNAL(simulatorConnected()), this, SLOT(onSimulatorConnect()),Qt::QueuedConnection);
132  connect(simulator, SIGNAL(simulatorDisconnected()), this, SLOT(onSimulatorDisconnect()),Qt::QueuedConnection);
133 
134  // Initialize connection status
135  if ( simulator->isAutopilotConnected() )
136  {
137  onAutopilotConnect();
138  }
139  else
140  {
141  onAutopilotDisconnect();
142  }
143 
144  if ( simulator->isSimulatorConnected() )
145  {
146  onSimulatorConnect();
147  }
148  else
149  {
150  onSimulatorDisconnect();
151  }
152  }
153 }
