00039 // Naming a file like a device name will break on Windows, even if it is
00040 // "com1.txt". Since we are cross-platform, we generally disallow such file

00088 bool FileNameValidatingLineEdit::validateFileName(const QString &name,
00089                                                   bool allowDirectories,
00090                                                   QString *errorMessage /* = 0*/)
00091 {
00092     if (name.isEmpty()) {
00093         if (errorMessage)
00094             *errorMessage = tr("The name must not be empty");
00095         return false;
00096     }
00097     // Characters
00098     const char *notAllowedChars = allowDirectories ? notAllowedCharsSubDir : notAllowedCharsNoSubDir;
00099     for (const char *c = notAllowedChars; *c; c++)
00100         if (name.contains(QLatin1Char(*c))) {
00101             if (errorMessage)
00102                 *errorMessage = tr("The name must not contain any of the characters '%1'.").arg(QLatin1String(notAllowedChars));
00103             return false;
00104         }
00105     // Substrings
00106     const int notAllowedSubStringCount = sizeof(notAllowedSubStrings)/sizeof(const char *);
00107     for (int s = 0; s < notAllowedSubStringCount; s++) {
00108         const QLatin1String notAllowedSubString(notAllowedSubStrings[s]);
00109         if (name.contains(notAllowedSubString)) {
00110             if (errorMessage)
00111                 *errorMessage = tr("The name must not contain '%1'.").arg(QString(notAllowedSubString));
00112             return false;
00113         }
00114     }
00115     // Windows devices
00116     bool matchesWinDevice = windowsDeviceNoSubDirPattern().exactMatch(name);
00117     if (!matchesWinDevice && allowDirectories)
00118         matchesWinDevice = windowsDeviceSubDirPattern().exactMatch(name);
00119     if (matchesWinDevice) {
00120         if (errorMessage)
00121             *errorMessage = tr("The name must not match that of a MS Windows device. (%1).").
00122                             arg(windowsDeviceNoSubDirPattern().pattern().replace(QLatin1Char('|'), QLatin1Char(',')));
00123         return false;
00124     }
00125     return true;
00126 }
