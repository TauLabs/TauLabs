00137 QString ConsoleProcess::stubServerListen()
00138 {
00139     // We need to put the socket in a private directory, as some systems simply do not
00140     // check the file permissions of sockets.
00141     QString stubFifoDir;
00142     forever {
00143         {
00144             QTemporaryFile tf;
00145             if (!tf.open())
00146                 return msgCannotCreateTempFile(tf.errorString());
00147             stubFifoDir = QFile::encodeName(tf.fileName());
00148         }
00149         // By now the temp file was deleted again
00150         m_stubServerDir = QFile::encodeName(stubFifoDir);
00151         if (!::mkdir(m_stubServerDir.constData(), 0700))
00152             break;
00153         if (errno != EEXIST)
00154             return msgCannotCreateTempDir(stubFifoDir, QString::fromLocal8Bit(strerror(errno)));
00155     }
00156     const QString stubServer  = stubFifoDir + "/stub-socket";
00157     if (!m_stubServer.listen(stubServer)) {
00158         ::rmdir(m_stubServerDir.constData());
00159         return tr("Cannot create socket '%1': %2").arg(stubServer, m_stubServer.errorString());
00160     }
00161     return QString();
00162 }

00185 void ConsoleProcess::readStubOutput()
00186 {
00187     while (m_stubSocket->canReadLine()) {
00188         QByteArray out = m_stubSocket->readLine();
00189         out.chop(1); // \n
00190         if (out.startsWith("err:chdir ")) {
00191             emit processError(msgCannotChangeToWorkDir(workingDirectory(), errorMsg(out.mid(10).toInt())));
00192         } else if (out.startsWith("err:exec ")) {
00193             emit processError(msgCannotExecute(m_executable, errorMsg(out.mid(9).toInt())));
00194         } else if (out.startsWith("pid ")) {
00195             // Will not need it any more
00196             delete m_tempFile;
00197             m_tempFile = 0;
00198 
00199             m_appPid = out.mid(4).toInt();
00200             emit processStarted();
00201         } else if (out.startsWith("exit ")) {
00202             m_appStatus = QProcess::NormalExit;
00203             m_appCode = out.mid(5).toInt();
00204             m_appPid = 0;
00205             emit processStopped();
00206         } else if (out.startsWith("crash ")) {
00207             m_appStatus = QProcess::CrashExit;
00208             m_appCode = out.mid(6).toInt();
00209             m_appPid = 0;
00210             emit processStopped();
00211         } else {
00212             emit processError(msgUnexpectedOutput());
00213             m_process.terminate();
00214             break;
00215         }
00216     }
