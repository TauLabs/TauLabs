94  {
95  QNetworkReply *reply;
96  QNetworkRequest qheader;

99  QTimer tT;
100  tT.setSingleShot(true);
101  connect(&network, SIGNAL(finished(QNetworkReply*)),

103  connect(&tT, SIGNAL(timeout()), &q, SLOT(quit()));
104  network.setProxy(Proxy);

106  qDebug()<<"Correct GoogleVersion";

110 
111  qheader.setUrl(QUrl(url));
112  qheader.setRawHeader("User-Agent",UserAgent);
113  reply=network.get(qheader);
114  tT.start(Timeout);
115  q.exec();

117  return;
118  tT.stop();
119  if( (reply->error()!=QNetworkReply::NoError))
120  {
121 #ifdef DEBUG_URLFACTORY

124  return;

689  double UrlFactory::GetDistance(internals::PointLatLng p1, internals::PointLatLng p2)
690  {
691  double dLat1InRad = p1.Lat() * (M_PI / 180);
692  double dLong1InRad = p1.Lng() * (M_PI / 180);
693  double dLat2InRad = p2.Lat() * (M_PI / 180);
694  double dLong2InRad = p2.Lng() * (M_PI / 180);
695  double dLongitude = dLong2InRad - dLong1InRad;
696  double dLatitude = dLat2InRad - dLat1InRad;
697  double a = pow(sin(dLatitude / 2), 2) + cos(dLat1InRad) * cos(dLat2InRad) * pow(sin(dLongitude / 2), 2);
698  double c = 2 * atan2(sqrt(a), sqrt(1 - a));
699  double dDistance = EarthRadiusKm * c;
700  return dDistance;
701  }
