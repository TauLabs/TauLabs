00689     double UrlFactory::GetDistance(internals::PointLatLng p1, internals::PointLatLng p2)
00690     {

00692         double dLong1InRad = p1.Lng() * (M_PI / 180);
00693         double dLat2InRad = p2.Lat() * (M_PI / 180);
00694         double dLong2InRad = p2.Lng() * (M_PI / 180);
00695         double dLongitude = dLong2InRad - dLong1InRad;
00696         double dLatitude = dLat2InRad - dLat1InRad;
00697         double a = pow(sin(dLatitude / 2), 2) + cos(dLat1InRad) * cos(dLat2InRad) * pow(sin(dLongitude / 2), 2);
00698         double c = 2 * atan2(sqrt(a), sqrt(1 - a));
00699         double dDistance = EarthRadiusKm * c;
00700         return dDistance;
