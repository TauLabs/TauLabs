128 void ModelViewGadgetWidget::paintGL()
129 {
130  try
131  {
132  // Clear screen
133  glClear(GL_COLOR_BUFFER_BIT | GL_DEPTH_BUFFER_BIT);
134 
135  // OpenGL error handler
136  {
137  GLenum error= glGetError();
138  if (error != GL_NO_ERROR)
139  {
140  GLC_OpenGlException OpenGlException("ModelViewGadgetWidget::paintGL() ", error);
141  throw(OpenGlException);
142  }
143  }
144 
145  // Load identity matrix
146  glLoadIdentity();
147 
148  // Enable antialiasing
149  glEnable(GL_MULTISAMPLE);
150 
151  // Calculate camera depth of view
152  m_GlView.setDistMinAndMax(m_World.boundingBox());
153 
154  // define view matrix
155  m_Light.glExecute();
156  m_GlView.glExecuteCam();
157 
158  // Display the collection of GLC_Object
159  m_World.render(0, glc::ShadingFlag);
160  m_World.render(0, glc::TransparentRenderFlag);
161 
162  // Display UI Info (orbit circle)
163  m_MoverController.drawActiveMoverRep();
164  }

189 void ModelViewGadgetWidget::CreateScene()
190 {
191  // put a black background if the 3D model is invalid or if the background image is also invalid
192  if (acFilename == ":/modelview/models/warning_sign.obj" or !QFile::exists(bgFilename))
193  bgFilename= ":/modelview/models/black.jpg";
194 
195 
196  try
197  {
198  m_GlView.loadBackGroundImage(bgFilename);

202  qDebug("ModelView: background image file loading failed.");
203  }
204 
205  try
206  {
207  if(QFile::exists(acFilename))

209  QFile aircraft(acFilename);
210  m_World= GLC_Factory::instance()->createWorldFromFile(aircraft);

248  printf("Extensions - %s\n", (char*)glGetString(GL_EXTENSIONS));
249  break;
250  default:
251  break;
252  }

334 void ModelViewGadgetWidget::updateAttitude()
335 {
336  AttitudeActual::DataFields data = attActual->getData(); // get attitude data
337  GLC_StructOccurence* rootObject= m_World.rootOccurence(); // get the full 3D model
338  double x= data.q3;
339  double y= data.q2;
340  double z= data.q4;
341  double w= data.q1;
342  if (w == 0.0)
343  w = 1.0;
344  // create and gives the product of 2 4x4 matrices to get the rotation of the 3D model's matrix
345  QMatrix4x4 m1;
346  m1.setRow(0, QVector4D(w,z,-y,x));
347  m1.setRow(1, QVector4D(-z,w,x,y));
348  m1.setRow(2, QVector4D(y,-x,w,z));
349  m1.setRow(3, QVector4D(-x,-y,-z,w));
350  QMatrix4x4 m2;
351  m2.setRow(0, QVector4D(w,z,-y,-x));
352  m2.setRow(1, QVector4D(-z,w,x,-y));
353  m2.setRow(2, QVector4D(y,-x,w,-z));
354  m2.setRow(3, QVector4D(x,y,z,w));
355  QMatrix4x4 m0= m1 * m2;
356  // convert QMatrix4x4 to GLC_Matrix4x4
357  GLC_Matrix4x4 rootObjectRotation;
358  {
359  double* newMatrixData = rootObjectRotation.setData();
360  double* oldMatrixData = (double*) m0.data();
361  for (int i=0; i<16; i++){
362  newMatrixData[i]= oldMatrixData[i];
363  }
364  }
365  // sets and updates the 3D model's matrix
366  rootObject->structInstance()->setMatrix(rootObjectRotation);
367  rootObject->updateChildrenAbsoluteMatrix();
368  updateGL();
369 }
