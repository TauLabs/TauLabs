340 // Draw instances of a PointerViewInstanceHash
341 void GLC_3DViewCollection::glDrawInstancesOf(PointerViewInstanceHash* pHash, glc::RenderFlag renderFlag)
342 {
343  bool forceDisplay= false;
344  if (GLC_State::isInSelectionMode())
345  {
346  forceDisplay= true;
347  }
348 
349  PointerViewInstanceHash::iterator iEntry= pHash->begin();
350  // The current instance
351  GLC_3DViewInstance* pCurInstance;
352  if (forceDisplay)
353  {
354  while (iEntry != pHash->constEnd())
355  {
356  pCurInstance= iEntry.value();
357  if ((pCurInstance->viewableFlag() != GLC_3DViewInstance::NoViewable) && (pCurInstance->isVisible() == m_IsInShowSate))
358  {
359  pCurInstance->render(renderFlag, m_UseLod, m_pViewport);
360  }
361  ++iEntry;
362  }
363  }
364  else
365  {
366  if (!(renderFlag == glc::TransparentRenderFlag))
367  {
368  while (iEntry != pHash->constEnd())
369  {
370  pCurInstance= iEntry.value();
371  if ((pCurInstance->viewableFlag() != GLC_3DViewInstance::NoViewable) && (pCurInstance->isVisible() == m_IsInShowSate))
372  {
373  if (!pCurInstance->isTransparent() || pCurInstance->renderPropertiesHandle()->isSelected() || (renderFlag == glc::WireRenderFlag))
374  {
375  pCurInstance->render(renderFlag, m_UseLod, m_pViewport);
376  }
377  }
378 
379  ++iEntry;
380  }
381 
382  }
383  else
384  {
385  while (iEntry != pHash->constEnd())
386  {
387  pCurInstance= iEntry.value();
388  if ((pCurInstance->viewableFlag() != GLC_3DViewInstance::NoViewable) && (pCurInstance->isVisible() == m_IsInShowSate))
389  {
390  if (pCurInstance->hasTransparentMaterials())
391  {
392  pCurInstance->render(renderFlag, m_UseLod, m_pViewport);
393  }
394  }
395 
396  ++iEntry;
397  }
398  }
399 
400  }
401 }
