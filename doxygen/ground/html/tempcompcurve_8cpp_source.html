52 void TempCompCurve::plotData(QList<double> temp, QList<double> gyro, QList <double> coeff)
53 {
54  // TODO: Keep the curves and free them in the destructors
55  const int STEPS = 100;
56 
57  points.clear();
58  fit.clear();
59 
60  clearCurves();
61 
62  double min = temp[0];
63  double max = temp[0];
64  for (int i = 0; i < temp.size(); i++) {
65  points.append(QPointF(temp[i],gyro[i]));
66  min = qMin(min, temp[i]);
67  max = qMax(max, temp[i]);
68  }
69 
70  double step = (max - min) / STEPS;
71  for (int i = 0; i < STEPS; i++) {
72  double t = min + step * i;
73  double f = coeff[0] + coeff[1] * t + coeff[2] * pow(t,2) + coeff[3] * pow(t,3);
74  fit.append(QPointF(t, f));
75  }
76 
77  // Plot the raw data points
78  QPen pen = QPen(Qt::DotLine);
79  pen.setColor(QColor(244, 244, 244));
80 
81  dataCurve = new QwtPlotCurve("Gyro");

83  dataCurve->setSamples(this->points);
84  dataCurve->attach(this);
85 
86  // Plot the fit

90  fitCurve = new QwtPlotCurve("Fit");
