252 glc::WidgetEventFlag GLC_CuttingPlane::mouseMove(const GLC_Point3d& pos, Qt::MouseButtons button, GLC_uint)
253 {
254  glc::WidgetEventFlag returnFlag= glc::IgnoreEvent;
255  if (button & Qt::LeftButton)
256  {
257  if (NULL != m_pCurrentManipulator)
258  {
259  if (m_SelectionIndex != -1)
260  {
261  moveManipulatorRep(m_CurrentNavigatorPosition);
262  m_SelectionIndex= -1;
263  }
264  GLC_Matrix4x4 moveMatrix(m_pCurrentManipulator->manipulate(pos));
265 
266  // Update plane normal
267  if (m_CurrentManipulator == RotationManipulator)
268  {
269  m_Normal= moveMatrix.rotationMatrix() * m_Normal;
270  }
271  m_CompMatrix= moveMatrix * m_CompMatrix;
272  m_Center= moveMatrix * m_Center;
273  m_CurrentNavigatorPosition= moveMatrix * m_CurrentNavigatorPosition;
274 
275  // Update the instance
276  for (int i= 0; i < 5; ++i)
277  {
278  GLC_3DWidget::instanceHandle(i)->multMatrix(moveMatrix);
279  }
280 
281  // Plane throw intersection and plane normal and camera up vector
282  m_pCurrentManipulator->enterManipulateState(m_pCurrentManipulator->previousPosition());
283 
284  emit asChanged();
285  returnFlag= glc::AcceptEvent;
286  }
287  }
288 
289  return returnFlag;
290 }

363 void GLC_CuttingPlane::moveManipulatorRep(const GLC_Point3d& pos)
364 {
365  // Create the widget rotation matrix
366  const GLC_Matrix4x4 rotationMatrix(m_CompMatrix.rotationMatrix());
367 
368  const GLC_Matrix4x4 translationMatrix(pos);
369  const GLC_Matrix4x4 offsetMatrix(m_Normal * m_ManipulatorOffsetFactor);
370  GLC_Matrix4x4 scaleMatrix;
371  scaleMatrix.setMatScaling(m_ScaleFactor, m_ScaleFactor, m_ScaleFactor);
372  GLC_3DWidget::instanceHandle(1)->setMatrix(offsetMatrix * translationMatrix * rotationMatrix *scaleMatrix);
373 
374  // Rotation manipulator
375  QVector<GLC_Matrix4x4> rotations(3);
376  rotations[0].setMatRot(glc::Y_AXIS, glc::PI / 2.0); // X
377  rotations[0]= GLC_Matrix4x4(glc::X_AXIS, -glc::PI / 2.0) * rotations[0];
378  rotations[1].setMatRot(glc::X_AXIS, -glc::PI / 2.0); // Y
379  // Z
380  for (int i= 0; i < 3; ++i)
381  {
382  GLC_3DWidget::instanceHandle(2 + i)->setMatrix(offsetMatrix * translationMatrix * rotationMatrix * rotations.at(i) * scaleMatrix);
383  }
384 
385  GLC_Arrow* pArrow= dynamic_cast<GLC_Arrow*>(GLC_3DWidget::instanceHandle(1)->geomAt(0));
386  Q_ASSERT(NULL != pArrow);
387 
388  pArrow->setViewDir(rotationMatrix * GLC_3DWidget::widgetManagerHandle()->cameraHandle()->forward().normalize());
389 }

391 GLC_AbstractManipulator* GLC_CuttingPlane::rotationNavigator(int index)
392 {
393  index= index - 2;
394  Q_ASSERT((index > -1) && (index < 3));
395 
396  const GLC_Matrix4x4 rotationMatrix(m_CompMatrix.rotationMatrix());
397  GLC_Vector3d axis;
398  if (index == 0)
399  {
400  axis= rotationMatrix * glc::X_AXIS;
401  }
402  else if (index == 1)
403  {
404  axis= rotationMatrix * glc::Y_AXIS;
405  }
406  else
407  {
408  axis= rotationMatrix * glc::Z_AXIS;
409  }
410  GLC_AbstractManipulator* pManipulator= new GLC_RotationManipulator(GLC_3DWidget::widgetManagerHandle()->viewport(), GLC_Line3d(m_CurrentNavigatorPosition, axis));
411 
412  return pManipulator;
413 }
