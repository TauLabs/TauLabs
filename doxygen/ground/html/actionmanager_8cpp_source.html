332 Command *ActionManagerPrivate::registerOverridableAction(QAction *action, const QString &id, bool checkUnique)
333 {
334  OverrideableAction *a = 0;
335  const int uid = UniqueIDManager::instance()->uniqueIdentifier(id);
336  if (CommandPrivate *c = m_idCmdMap.value(uid, 0)) {
337  a = qobject_cast<OverrideableAction *>(c);
338  if (!a) {
339  qWarning() << "registerAction: id" << id << "is registered with a different command type.";
340  return c;
341  }
342  } else {
343  a = new OverrideableAction(uid);
344  m_idCmdMap.insert(uid, a);
345  }
346 
347  if (!a->action()) {
348  QAction *baseAction = new QAction(m_mainWnd);
349  baseAction->setObjectName(id);
350  baseAction->setCheckable(action->isCheckable());
351  baseAction->setIcon(action->icon());
352  baseAction->setIconText(action->iconText());
353  baseAction->setText(action->text());
354  baseAction->setToolTip(action->toolTip());
355  baseAction->setStatusTip(action->statusTip());
356  baseAction->setWhatsThis(action->whatsThis());
357  baseAction->setChecked(action->isChecked());
358  baseAction->setSeparator(action->isSeparator());
359  baseAction->setShortcutContext(Qt::ApplicationShortcut);
360  baseAction->setEnabled(false);
361  baseAction->setParent(m_mainWnd);
362 #ifdef Q_WS_MAC
363  baseAction->setIconVisibleInMenu(false);
364 #endif
365  a->setAction(baseAction);
366  m_mainWnd->addAction(baseAction);
367  a->setKeySequence(a->keySequence());
368  a->setDefaultKeySequence(QKeySequence());
369  } else if (checkUnique) {
370  qWarning() << "registerOverridableAction: id" << id << "is already registered.";
371  }
372 
373  return a;
374 }

376 Command *ActionManagerPrivate::registerShortcut(QShortcut *shortcut, const QString &id, const QList<int> &context)
377 {
378  Shortcut *sc = 0;
379  int uid = UniqueIDManager::instance()->uniqueIdentifier(id);
380  if (CommandPrivate *c = m_idCmdMap.value(uid, 0)) {
381  sc = qobject_cast<Shortcut *>(c);
382  if (!sc) {
383  qWarning() << "registerShortcut: id" << id << "is registered with a different command type.";
384  return c;
385  }
386  } else {
387  sc = new Shortcut(uid);
388  m_idCmdMap.insert(uid, sc);
389  }
390 
391  if (sc->shortcut()) {
392  qWarning() << "registerShortcut: action already registered (id" << id << ".";
393  return sc;
394  }
395 
396  if (!hasContext(context))
397  shortcut->setEnabled(false);
398  shortcut->setObjectName(id);
399  shortcut->setParent(m_mainWnd);
400  sc->setShortcut(shortcut);
401 
402  if (context.isEmpty())
403  sc->setContext(QList<int>() << 0);
404  else
405  sc->setContext(context);
406 
407  sc->setKeySequence(shortcut->key());
408  sc->setDefaultKeySequence(QKeySequence());
409 
410  return sc;
411 }

463 void ActionManagerPrivate::readSettings(QSettings *settings)
464 {
465  const int shortcuts = settings->beginReadArray(QLatin1String(settingsGroup));
466  for (int i=0; i<shortcuts; ++i) {
467  settings->setArrayIndex(i);
468  const QString sid = settings->value(QLatin1String(idKey)).toString();
469  const QKeySequence key(settings->value(QLatin1String(sequenceKey)).toString());
470  const int id = UniqueIDManager::instance()->uniqueIdentifier(sid);
471 
472  Command *cmd = command(id);
473  if (cmd)
474  cmd->setKeySequence(key);
475  }
476  settings->endArray();
477 }

479 void ActionManagerPrivate::saveSettings(QSettings *settings)
480 {
481  settings->beginWriteArray(QLatin1String(settingsGroup));
482  int count = 0;
483 
484  const IdCmdMap::const_iterator cmdcend = m_idCmdMap.constEnd();
485  for (IdCmdMap::const_iterator j = m_idCmdMap.constBegin(); j != cmdcend; ++j) {
486  const int id = j.key();
487  CommandPrivate *cmd = j.value();
488  QKeySequence key = cmd->keySequence();
489  if (key != cmd->defaultKeySequence()) {
490  const QString sid = UniqueIDManager::instance()->stringForUniqueIdentifier(id);
491  settings->setArrayIndex(count);
492  settings->setValue(QLatin1String(idKey), sid);
493  settings->setValue(QLatin1String(sequenceKey), key.toString());
494  count++;
495  }
496  }
497 
498  settings->endArray();
