00268 void PathListEditor::insertPathAtCursor(const QString &path)
00269 {
00270     // If the cursor is at an empty line or at end(),
00271     // just insert. Else insert line before
00272     QTextCursor cursor = m_d->edit->textCursor();
00273     QTextBlock block = cursor.block();
00274     const bool needNewLine = !block.text().isEmpty();
00275     if (needNewLine) {
00276         cursor.movePosition(QTextCursor::StartOfLine, QTextCursor::MoveAnchor);
00277         cursor.insertBlock();
00278         cursor.movePosition(QTextCursor::PreviousBlock, QTextCursor::MoveAnchor);
00279     }
00280     cursor.insertText(path);
00281     if (needNewLine) {
00282         cursor.movePosition(QTextCursor::StartOfLine, QTextCursor::MoveAnchor);
00283         m_d->edit->setTextCursor(cursor);
00284     }
00285 }
