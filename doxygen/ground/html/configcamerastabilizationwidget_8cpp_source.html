108  * This overridden function refreshes widgets which have no direct relation

113 void ConfigCameraStabilizationWidget::refreshWidgetsValues(UAVObject *obj)
114 {
115  bool dirty = isDirty();
116 
117  // Set module enable checkbox from OptionalModules UAVObject item.
118  // It needs special processing because ConfigTaskWidget uses TRUE/FALSE
119  // for QCheckBox, but OptionalModules uses Enabled/Disabled enum values.
120  ModuleSettings *moduleSettings = ModuleSettings::GetInstance(getObjectManager());
121  ModuleSettings::DataFields moduleSettingsData = moduleSettings->getData();
122 
123  m_camerastabilization->enableCameraStabilization->setChecked(
124  moduleSettingsData.AdminState[ModuleSettings::ADMINSTATE_CAMERASTAB] == ModuleSettings::ADMINSTATE_ENABLED);
125 
126  // Load mixer outputs which are mapped to camera controls
127  MixerSettings *mixerSettings = MixerSettings::GetInstance(getObjectManager());
128  MixerSettings::DataFields mixerSettingsData = mixerSettings->getData();
129 
130  // TODO: Need to reformat object so types are an
131  // array themselves. This gets really awkward
132  quint8 *mixerTypes[] = {
133  &mixerSettingsData.Mixer1Type,
134  &mixerSettingsData.Mixer2Type,
135  &mixerSettingsData.Mixer3Type,
136  &mixerSettingsData.Mixer4Type,
137  &mixerSettingsData.Mixer5Type,
138  &mixerSettingsData.Mixer6Type,
139  &mixerSettingsData.Mixer7Type,
140  &mixerSettingsData.Mixer8Type,
141  &mixerSettingsData.Mixer9Type,
142  &mixerSettingsData.Mixer10Type,
143  };
144  const int NUM_MIXERS = sizeof(mixerTypes) / sizeof(mixerTypes[0]);
145 
146  QComboBox *outputs[] = {
147  m_camerastabilization->rollChannel,
148  m_camerastabilization->pitchChannel,
149  m_camerastabilization->yawChannel
150  };
151  const int NUM_OUTPUTS = sizeof(outputs) / sizeof(outputs[0]);
152 
153  for (int i = 0; i < NUM_OUTPUTS; i++) {
154  // Default to none if not found.
155  // Then search for any mixer channels set to this
156  outputs[i]->setCurrentIndex(0);
157  for (int j = 0; j < NUM_MIXERS; j++)
158  if (*mixerTypes[j] == (MixerSettings::MIXER1TYPE_CAMERAROLL + i) &&
159  outputs[i]->currentIndex() != (j + 1))
160  outputs[i]->setCurrentIndex(j + 1);
161  }
162 
163  setDirty(dirty);
164 
165  ConfigTaskWidget::refreshWidgetsValues(obj);
166 }

169  * This overridden function updates UAVObjects which have no direct relation
170  * to any of widgets. Aftewards it calls base class function to take care of
171  * other object to widget relations which were dynamically added.
172  */
173 void ConfigCameraStabilizationWidget::updateObjectsFromWidgets()
174 {
175  // Save state of the module enable checkbox first.
176  // Do not use setData() member on whole object, if possible, since it triggers
177  // unnessesary UAVObect update.
178  quint8 enableModule = m_camerastabilization->enableCameraStabilization->isChecked() ?
179  (ModuleSettings::ADMINSTATE_ENABLED) : (ModuleSettings::ADMINSTATE_DISABLED);
180  ModuleSettings *moduleSettings = ModuleSettings::GetInstance(getObjectManager());
181  moduleSettings->setAdminState(ModuleSettings::ADMINSTATE_CAMERASTAB, enableModule);
182 
183  // Update mixer channels which were mapped to camera outputs in case they are
184  // not used for other function yet
185  MixerSettings *mixerSettings = MixerSettings::GetInstance(getObjectManager());
186  MixerSettings::DataFields mixerSettingsData = mixerSettings->getData();
187 
188  // TODO: Need to reformat object so types are an
189  // array themselves. This gets really awkward
190  quint8 *mixerTypes[] = {
191  &mixerSettingsData.Mixer1Type,
192  &mixerSettingsData.Mixer2Type,
193  &mixerSettingsData.Mixer3Type,
194  &mixerSettingsData.Mixer4Type,
195  &mixerSettingsData.Mixer5Type,
196  &mixerSettingsData.Mixer6Type,
197  &mixerSettingsData.Mixer7Type,
198  &mixerSettingsData.Mixer8Type,
199  &mixerSettingsData.Mixer9Type,
200  &mixerSettingsData.Mixer10Type,
201  };
202  const int NUM_MIXERS = sizeof(mixerTypes) / sizeof(mixerTypes[0]);
203 
204  QComboBox *outputs[] = {
205  m_camerastabilization->rollChannel,
206  m_camerastabilization->pitchChannel,
207  m_camerastabilization->yawChannel
208  };
209  const int NUM_OUTPUTS = sizeof(outputs) / sizeof(outputs[0]);
210 
211  m_camerastabilization->message->setText("");
212  bool widgetUpdated;
213  do {
214  widgetUpdated = false;
215 
216  for (int i = 0; i < NUM_OUTPUTS; i++) {
217  // Channel 1 is second entry, so becomes zero
218  int mixerNum = outputs[i]->currentIndex() - 1;
219 
220  if ((mixerNum >= 0) && // Short circuit in case of none
221  (*mixerTypes[mixerNum] != MixerSettings::MIXER1TYPE_DISABLED) &&
222  (*mixerTypes[mixerNum] != MixerSettings::MIXER1TYPE_CAMERAROLL + i) ) {
223  // If the mixer channel already mapped to something, it should not be
224  // used for camera output, we reset it to none
225  outputs[i]->setCurrentIndex(0);
226  m_camerastabilization->message->setText("One of the channels is already assigned, reverted to none");
227 
228  // Loop again or we may have inconsistent widget and UAVObject
229  widgetUpdated = true;
230  } else {
231  // Make sure no other channels have this output set
232  for (int j = 0; j < NUM_MIXERS; j++)
233  if (*mixerTypes[j] == (MixerSettings::MIXER1TYPE_CAMERAROLL + i))
234  *mixerTypes[j] = MixerSettings::MIXER1TYPE_DISABLED;
235 
236  // If this channel is assigned to one of the outputs that is not disabled
237  // set it
238  if ((mixerNum >= 0) && (mixerNum < NUM_MIXERS))
239  *mixerTypes[mixerNum] = MixerSettings::MIXER1TYPE_CAMERAROLL + i;
240  }
241  }
242  } while(widgetUpdated);
243 
244  // FIXME: Should not use setData() to prevent double updates.
245  // It should be refactored after the reformatting of MixerSettings UAVObject.
246  mixerSettings->setData(mixerSettingsData);
247 
248  ConfigTaskWidget::updateObjectsFromWidgets();
249 }
