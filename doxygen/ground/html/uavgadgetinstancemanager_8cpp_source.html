00095         // We are in trouble now. User wants us to quit the import, but when he saves
00096         // the GCS, his old config will be lost.

00098                 tr("You might want to save your old config NOW since it might be replaced by broken one when you exit the GCS!")
00099                 );
00100     }
00101     else{
00102         readConfigs_1_2_0(qs);
00103     }
