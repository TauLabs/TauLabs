75 void UAVGadgetInstanceManager::readSettings(QSettings *qs)
76 {
77  while ( !m_configurations.isEmpty() ){
78  emit configurationToBeDeleted(m_configurations.takeLast());
79  }
80  qs->beginGroup("UAVGadgetConfigurations");
81  UAVConfigInfo configInfo(qs);
82  configInfo.setNameOfConfigurable("UAVGadgetConfigurations");
83  if ( configInfo.version() == UAVConfigVersion() ){
84  // If version is not set, assume its a old version before readable config (1.0.0).
85  // however compatibility to 1.0.0 is broken.
86  configInfo.setVersion("1.0.0");
87  }
88 
89  if ( configInfo.version() == UAVConfigVersion("1.1.0") ){
90  configInfo.notify(tr("Migrating UAVGadgetConfigurations from version 1.1.0 to ")
91  + m_versionUAVGadgetConfigurations.toString());
92  readConfigs_1_1_0(qs); // this is fully compatible with 1.2.0
93  }
94  else if ( !configInfo.standardVersionHandlingOK(m_versionUAVGadgetConfigurations) ){
95  // We are in trouble now. User wants us to quit the import, but when he saves
96  // the GCS, his old config will be lost.
97  configInfo.notify(
98  tr("You might want to save your old config NOW since it might be replaced by broken one when you exit the GCS!")
99  );
100  }
101  else{
102  readConfigs_1_2_0(qs);
103  }
104 
105  qs->endGroup();
106  createOptionsPages();
107 }

109 void UAVGadgetInstanceManager::readConfigs_1_2_0(QSettings *qs)
110 {
111  UAVConfigInfo configInfo;
112 
113  foreach (QString classId, m_classIdNameMap.keys())
114  {
115  IUAVGadgetFactory *f = factory(classId);
116  qs->beginGroup(classId);
117 
118  QStringList configs = QStringList();
119 
120  configs = qs->childGroups();
121  foreach (QString configName, configs) {
122  qDebug() << "Loading config: " << classId << "," << configName;
123  qs->beginGroup(configName);
124  configInfo.read(qs);
125  configInfo.setNameOfConfigurable(classId+"-"+configName);
126  qs->beginGroup("data");
127  IUAVGadgetConfiguration *config = f->createConfiguration(qs, &configInfo);
128  if (config){
129  config->setName(configName);
130  config->setProvisionalName(configName);
131  config->setLocked(configInfo.locked());
132  int idx = indexForConfig(m_configurations, classId, configName);
133  if ( idx >= 0 ){
134  // We should replace the config, but it might be used, so just
135  // throw it out of the list. The GCS should be reinitialised soon.
136  m_configurations[idx] = config;
137  }
138  else{
139  m_configurations.append(config);
140  }
141  }
142  qs->endGroup();
143  qs->endGroup();
144  }
145 
146  if (configs.count() == 0) {
147  IUAVGadgetConfiguration *config = f->createConfiguration(0, 0);
148  // it is not mandatory for uavgadgets to have any configurations (settings)
149  // and therefore we have to check for that
150  if (config) {
151  config->setName(tr("default"));
152  config->setProvisionalName(tr("default"));
153  m_configurations.append(config);
154  }
155  }
156  qs->endGroup();
157  }

209 void UAVGadgetInstanceManager::saveSettings(QSettings *qs)
210 {
211  UAVConfigInfo *configInfo;
212  qs->beginGroup("UAVGadgetConfigurations");
213  qs->remove(""); // Remove existing configurations
214  configInfo = new UAVConfigInfo(m_versionUAVGadgetConfigurations, "UAVGadgetConfigurations");
215  configInfo->save(qs);
216  delete configInfo;
217  foreach (IUAVGadgetConfiguration *config, m_configurations)
218  {
219  configInfo = new UAVConfigInfo(config);
220  qs->beginGroup(config->classId());
221  qs->beginGroup(config->name());
222  qs->beginGroup("data");
223  config->saveConfig(qs, configInfo);
224  qs->endGroup();
225  configInfo->save(qs);
226  qs->endGroup();
227  qs->endGroup();
228  delete configInfo;
229  }
230  qs->endGroup();

233 void UAVGadgetInstanceManager::createOptionsPages()
234 {
235  // In case there are pages (import a configuration), remove them.
236  // Maybe they should be deleted as well (memory-leak),
237  // but this might lead to NULL-pointers?
238  while (!m_optionsPages.isEmpty()){
239  m_pm->removeObject(m_optionsPages.takeLast());
240  }
241 
242  QMutableListIterator<IUAVGadgetConfiguration*> ite(m_configurations);
243  while (ite.hasNext()) {
244  IUAVGadgetConfiguration *config = ite.next();
245  IUAVGadgetFactory *f = factory(config->classId());
246  IOptionsPage *p = f->createOptionsPage(config);
247  if (p) {
248  IOptionsPage *page = new UAVGadgetOptionsPageDecorator(p, config, f->isSingleConfigurationGadget());
249  page->setIcon(f->icon());
250  m_optionsPages.append(page);
251  m_pm->addObject(page);
252  }
253  else {
254  qWarning()
255  << "UAVGadgetInstanceManager::createOptionsPages - failed to create options page for configuration "
256  + config->classId() + ":" + config->name() + ", configuration will be removed.";
257  // The m_optionsPages list and m_configurations list must be in synch otherwise nasty issues happen later
258  // so if we fail to create an options page we must remove the associated configuration
259  ite.remove();
260  }
261  }

356 QString UAVGadgetInstanceManager::suggestName(QString classId, QString name)
357 {
358  QString suggestedName;
359 
360  QString last = name.split(" ").last();
361  bool ok;
362  int num = last.toInt(&ok);
363  int i = 1;
364  if (ok) {
365  QStringList n = name.split(" ");
366  n.removeLast();
367  name = n.join(" ");
368  i = num+1;
369  }
370  do {
371  suggestedName = name + " " + QString::number(i);
372  ++i;
373  } while (m_takenNames[classId].contains(suggestedName));
374 
375  m_takenNames[classId].append(suggestedName);
376  return suggestedName;
377 }

379 void UAVGadgetInstanceManager::applyChanges(IUAVGadgetConfiguration *config)
380 {
381  if (m_provisionalDeletes.contains(config)) {
382  m_provisionalDeletes.removeAt(m_provisionalDeletes.indexOf(config));
383  int i = m_configurations.indexOf(config);
384  if (i >= 0) {
385  emit configurationToBeDeleted(config);
386  int j = m_takenNames[config->classId()].indexOf(config->name());
387  m_takenNames[config->classId()].removeAt(j);
388  m_pm->removeObject(m_optionsPages.at(i));
389  m_configurations.removeAt(i);
390  m_optionsPages.removeAt(i);//TODO delete
391  }
392  return;
393  }
394  if (config->provisionalName() != config->name()) {
395  emit configurationNameChanged(config, config->name(), config->provisionalName());
396  config->setName(config->provisionalName());
397  }
398  if (m_configurations.contains(config)) {
399  emit configurationChanged(config);
400  } else if (m_provisionalConfigs.contains(config)) {
401  emit configurationAdded(config);
402  int i = m_provisionalConfigs.indexOf(config);
403  IOptionsPage *page = m_provisionalOptionsPages.at(i);
404  m_configurations.append(config);
405  m_optionsPages.append(page);
406  m_provisionalConfigs.removeAt(i);
407  m_provisionalOptionsPages.removeAt(i);
408  m_pm->addObject(page);
409  }
410 }
