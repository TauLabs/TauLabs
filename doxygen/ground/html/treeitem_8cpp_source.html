00077 /*
00078  * Callback called periodically by the timer.
00079  * This method checks for expired highlights and
00080  * removes them if they are expired.
00081  * Expired highlights are restored.
00082  */
00083 void HighLightManager::checkItemsExpired()
00084 {
00085     // Lock to ensure thread safety
00086     QMutexLocker locker(&m_listMutex);
00087 
00088     // Get a mutable iterator for the list
00089     QMutableLinkedListIterator<TreeItem*> iter(m_itemsList);
00090 
00091     // Loop over all items, check if they expired.
00092     while(iter.hasNext())
00093     {
00094         TreeItem* item = iter.next();
00095         if(item->getHiglightExpires() < *m_currentTime)
00096         {
00097             // If expired, call removeHighlight
00098             item->removeHighlight();
00099 
00100             // Remove from list since it is restored.
00101             iter.remove();
00102         }
00103     }
00104 }

00193 void TreeItem::setHighlight(bool highlight) {
00194     m_highlight = highlight;
00195     m_changed = false;
00196     if (highlight) {
00197         // Update the expires timestamp
00198         if (m_currentTime != NULL)
00199             m_highlightExpires = m_currentTime->addMSecs(m_highlightTimeMs);
00200         else
00201             m_highlightExpires = QTime::currentTime().addMSecs(m_highlightTimeMs);
00202 
00203         // Add to highlightmanager
00204         if(m_highlightManager->add(this))
00205         {
00206             // Only emit signal if it was added
00207             emit updateHighlight(this);
00208         }
00209     }
00210     else if(m_highlightManager->remove(this))
00211     {
00212         // Only emit signal if it was removed
00213         emit updateHighlight(this);
00214     }
00215 
00216     // If we have a parent, call recursively to update highlight status of parents.
00217     // This will ensure that the root of a leaf that is changed also is highlighted.
00218     // Only updates that really changes values will trigger highlight of parents.
00219     if(m_parent)
00220     {
00221         m_parent->setHighlight(highlight);
00222     }
00223 }
