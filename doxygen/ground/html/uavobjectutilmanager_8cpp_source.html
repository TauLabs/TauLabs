00162 void UAVObjectUtilManager::objectPersistenceTransactionCompleted(UAVObject* obj, bool success)
00163 {
00164     if(success) {
00165         Q_ASSERT(obj->getName().compare("ObjectPersistence") == 0);
00166         Q_ASSERT(saveState == AWAITING_ACK);
00167         // Two things can happen:
00168         // Either the Object Save Request did actually go through, and then we should get in
00169         // "AWAITING_COMPLETED" mode, or the Object Save Request did _not_ go through, for example
00170         // because the object does not exist and then we will never get a subsequent update.
00171         // For this reason, we will arm a 1 second timer to make provision for this and not block
00172         // the queue:
00173         saveState = AWAITING_COMPLETED;
00174         disconnect(obj, SIGNAL(transactionCompleted(UAVObject*,bool)), this, SLOT(objectPersistenceTransactionCompleted(UAVObject*,bool)));
00175         failureTimer.start(2000); // Create a timeout
00176     } else {
00177         // Can be caused by timeout errors on sending.  Forget it and send next.
00178         qDebug() << "objectPersistenceTranscationCompleted (error)";
00179         UAVObject *obj = getObjectManager()->getObject(ObjectPersistence::NAME);
00180         obj->disconnect(this);
00181         queue.dequeue(); // We can now remove the object, it failed anyway.
00182         saveState = IDLE;
00183         emit saveCompleted(obj->getField("ObjectID")->getValue().toInt(), false);

00611 bool UAVObjectUtilManager::descriptionToStructure(QByteArray desc, deviceDescriptorStruct & struc)
00612 {
00613    if (desc.startsWith("TlFw") || desc.startsWith("OpFw")) {
00614        /*
00615         * This looks like a binary with a description at the end
00616         *  4 bytes: header: "TlFw"
00617         *  4 bytes: git commit hash (short version of SHA1)
00618         *  4 bytes: Unix timestamp of last git commit
00619         *  2 bytes: target platform. Should follow same rule as BOARD_TYPE and BOARD_REVISION in board define files.
00620         *  26 bytes: commit tag if it is there, otherwise "Unreleased". Zero-padded
00621         *  20 bytes: SHA1 sum of the firmware.
00622         *  20 bytes: SHA1 sum of the UAVO definition files.
00623         *  20 bytes: free for now.
00624         */
00625 
00626        // Note: the ARM binary is big-endian:
00627        quint32 gitCommitHash = desc.at(7) & 0xFF;
00628        for (int i = 1; i < 4; i++) {
00629            gitCommitHash = gitCommitHash << 8;
00630            gitCommitHash += desc.at(7-i) & 0xFF;
00631        }
00632        struc.gitHash = QString("%1").arg(gitCommitHash, 8, 16, QChar('0'));
00633 
00634        quint32 gitDate = desc.at(11) & 0xFF;
00635        for (int i = 1; i < 4; i++) {
00636            gitDate = gitDate << 8;
00637            gitDate += desc.at(11-i) & 0xFF;
00638        }
00639        struc.gitDate = QDateTime::fromTime_t(gitDate).toUTC().toString("yyyyMMdd HH:mm");
00640 
00641        QString gitTag = QString(desc.mid(14,26));
00642        struc.gitTag = gitTag;
00643 
00644        // TODO: check platform compatibility
00645        QByteArray targetPlatform = desc.mid(12,2);
00646        struc.boardType = (int)targetPlatform.at(0);
00647        struc.boardRevision = (int)targetPlatform.at(1);
00648        struc.fwHash.clear();

00650        struc.uavoHash.clear();
