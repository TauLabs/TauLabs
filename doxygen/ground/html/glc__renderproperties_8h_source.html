281 // Select the instance
282 void GLC_RenderProperties::select(bool primitive)
283 {
284  m_IsSelected= true;
285  if (primitive && (m_RenderMode != glc::PrimitiveSelected))
286  {
287  m_SavedRenderMode= m_RenderMode;
288  m_RenderMode= glc::PrimitiveSelected;
289  }
290 }
291 
292 // Unselect the instance
293 void GLC_RenderProperties::unselect(void)
294 {
295  m_IsSelected= false;
296  if (m_RenderMode == glc::PrimitiveSelected)
297  {
298  m_RenderMode= m_SavedRenderMode;
299  }
300 }
301 // Used the specified material
302 void GLC_RenderProperties::useMaterial(GLC_Material* pMaterial)
303 {
304  if (m_MaterialsUsage.contains(pMaterial))
305  {
306  QHash<GLC_Material*, int>::iterator iMat= m_MaterialsUsage.find(pMaterial);
307  iMat.value()= iMat.value() + 1;
308  }
309  else
310  {
311  m_MaterialsUsage.insert(pMaterial, 1);
312  pMaterial->addUsage(m_Uid);
313  }
314 
315 }
316 
317 // Unused the specified material
318 void GLC_RenderProperties::unUseMaterial(GLC_Material* pMaterial)
319 {
320  Q_ASSERT(m_MaterialsUsage.contains(pMaterial));
321  QHash<GLC_Material*, int>::iterator iMat= m_MaterialsUsage.find(pMaterial);
322  iMat.value()= iMat.value() - 1;
323  if (iMat.value() == 0)
324  {
325  pMaterial->delUsage(m_Uid);
326  if (pMaterial->isUnused()) delete pMaterial;
327  m_MaterialsUsage.remove(pMaterial);
328  }
329 }
