00092 void LowPassFilter::updateValue()
00093 {
00094     //check if the update timer should be running
00095     bool reached = qFuzzyIsNull(qAbs(m_input)+qAbs(m_value)) ||
00096             (qAbs(m_input-m_value)/(qAbs(m_input)+qAbs(m_value)) < 1e-3);
00097 
00098     if (reached && m_updateTimer.isActive())
00099         m_updateTimer.stop();
00100     else if (!reached && !m_updateTimer.isActive())
00101         m_updateTimer.start(qRound(m_updatePeriod*1000), this);
00102 
00103     int dT = m_lastUpdateTime.elapsed();
00104     m_lastUpdateTime.restart();
00105     if (dT == 0)
00106         return;
00107 
00108     m_value += (m_input - m_value)*dT/(m_timeConstant*1000 + dT);
00109     emit valueChanged(m_value);
00110 }
