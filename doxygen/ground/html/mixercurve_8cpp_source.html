231 void MixerCurve::GenerateCurve()
232 {
233  double scale;
234  double newValue;
235 
236  //get the user settings
237  double value1 = getCurveMin();
238  double value2 = getCurveMax();
239  double value3 = getCurveStep();
240 
241  m_curve->setCommandText("StepValue", QString("%0").arg(value3));
242 
243  QString CurveType = m_mixerUI->CurveType->currentText();
244 
245  QList<double> points;
246 
247  for (int i=0; i<MixerCurveWidget::NODE_NUMELEM; i++)
248  {
249  scale =((double)i/(double)(MixerCurveWidget::NODE_NUMELEM - 1));
250 
251  if ( CurveType.compare("Flat")==0)
252  {
253  points.append(value3);
254  }
255  if ( CurveType.compare("Linear")==0)
256  {
257  newValue =value1 +(scale*(value2-value1));
258  points.append(newValue);
259  }
260  if ( CurveType.compare("Step")==0)
261  {
262  if (scale*100<value3)
263  {
264  points.append(value1);
265  }
266  else
267  {
268  points.append(value2);
269  }
270  }
271  if ( CurveType.compare("Exp")==0)
272  {
273  newValue =value1 +(((exp(scale*(value3/10))-1))/(exp((value3/10))-1)*(value2-value1));
274  points.append(newValue);
275  }
276  if ( CurveType.compare("Log")==0)
277  {
278  newValue = value1 +(((log(scale*(value3*2)+1))/(log(1+(value3*2))))*(value2-value1));
279  points.append(newValue);
280  }
281  }
282 
283  setCurve(&points);
284 }
