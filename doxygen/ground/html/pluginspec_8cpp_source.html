481 void PluginSpecPrivate::readPluginSpec(QXmlStreamReader &reader)
482 {
483  QString element = reader.name().toString();
484  if (element != QString(PLUGIN)) {
485  reader.raiseError(QCoreApplication::translate("PluginSpec", "Expected element '%1' as top level element").arg(PLUGIN));
486  return;
487  }
488  name = reader.attributes().value(PLUGIN_NAME).toString();
489  if (name.isEmpty()) {
490  reader.raiseError(msgAttributeMissing(PLUGIN, PLUGIN_NAME));
491  return;
492  }
493  version = reader.attributes().value(PLUGIN_VERSION).toString();
494  if (version.isEmpty()) {
495  reader.raiseError(msgAttributeMissing(PLUGIN, PLUGIN_VERSION));
496  return;
497  }
498  if (!isValidVersion(version)) {
499  reader.raiseError(msgInvalidFormat(PLUGIN_VERSION));
500  return;
501  }
502  compatVersion = reader.attributes().value(PLUGIN_COMPATVERSION).toString();
503  if (!compatVersion.isEmpty() && !isValidVersion(compatVersion)) {
504  reader.raiseError(msgInvalidFormat(PLUGIN_COMPATVERSION));
505  return;
506  } else if (compatVersion.isEmpty()) {
507  compatVersion = version;
508  }
509  while (!reader.atEnd()) {
510  reader.readNext();
511  switch (reader.tokenType()) {
512  case QXmlStreamReader::StartElement:
513  element = reader.name().toString();
514  if (element == VENDOR)
515  vendor = reader.readElementText().trimmed();
516  else if (element == COPYRIGHT)
517  copyright = reader.readElementText().trimmed();
518  else if (element == LICENSE)
519  license = reader.readElementText().trimmed();
520  else if (element == DESCRIPTION)
521  description = reader.readElementText().trimmed();
522  else if (element == URL)
523  url = reader.readElementText().trimmed();
524  else if (element == DEPENDENCYLIST)
525  readDependencies(reader);
526  else if (element == ARGUMENTLIST)
527  readArgumentDescriptions(reader);
528  else
529  reader.raiseError(msgInvalidElement(name));
530  break;
531  case QXmlStreamReader::EndDocument:
532  case QXmlStreamReader::Comment:
533  case QXmlStreamReader::EndElement:
534  case QXmlStreamReader::Characters:
535  break;
536  default:
537  reader.raiseError(msgUnexpectedToken());
538  break;
539  }
540  }
541 }

635 void PluginSpecPrivate::readDependencyEntry(QXmlStreamReader &reader)
636 {
637  PluginDependency dep;
638  dep.name = reader.attributes().value(DEPENDENCY_NAME).toString();
639  if (dep.name.isEmpty()) {
640  reader.raiseError(msgAttributeMissing(DEPENDENCY, DEPENDENCY_NAME));
641  return;
642  }
643  dep.version = reader.attributes().value(DEPENDENCY_VERSION).toString();
644  if (!dep.version.isEmpty() && !isValidVersion(dep.version)) {
645  reader.raiseError(msgInvalidFormat(DEPENDENCY_VERSION));
646  return;

649  reader.readNext();
650  if (reader.tokenType() != QXmlStreamReader::EndElement)
651  reader.raiseError(msgUnexpectedToken());

687 int PluginSpecPrivate::versionCompare(const QString &version1, const QString &version2)
688 {
689  QRegExp reg1 = versionRegExp();
690  QRegExp reg2 = versionRegExp();
691  if (!reg1.exactMatch(version1))
692  return 0;
693  if (!reg2.exactMatch(version2))
694  return 0;
695  int number1;
696  int number2;
697  for (int i = 0; i < 4; ++i) {
698  number1 = reg1.cap(i+1).toInt();
699  number2 = reg2.cap(i+1).toInt();
700  if (number1 < number2)
701  return -1;
702  if (number1 > number2)
703  return 1;
704  }
705  return 0;
706 }

712 bool PluginSpecPrivate::resolveDependencies(const QList<PluginSpec *> &specs)
713 {
714  if (hasError)
715  return false;
716  if (state == PluginSpec::Resolved)
717  state = PluginSpec::Read; // Go back, so we just re-resolve the dependencies.
718  if (state != PluginSpec::Read) {
719  errorString = QCoreApplication::translate("PluginSpec", "Resolving dependencies failed because state != Read");
720  hasError = true;
721  return false;
722  }
723  QList<PluginSpec *> resolvedDependencies;
724  foreach (const PluginDependency &dependency, dependencies) {
725  PluginSpec *found = 0;
726  foreach (PluginSpec *spec, specs) {
727  if (spec->provides(dependency.name, dependency.version)) {
728  found = spec;
729  break;
730  }
731  }
732  if (!found) {
733  hasError = true;
734  if (!errorString.isEmpty())
735  errorString.append("\n");
736  errorString.append(QCoreApplication::translate("PluginSpec", "Could not resolve dependency '%1(%2)'")
737  .arg(dependency.name).arg(dependency.version));
738  continue;
739  }
740  resolvedDependencies.append(found);
741  }
742  if (hasError)
743  return false;
744  dependencySpecs = resolvedDependencies;
745  state = PluginSpec::Resolved;
746  return true;
747 }

753 bool PluginSpecPrivate::loadLibrary()
754 {
755  if (hasError)
756  return false;
757  if (state != PluginSpec::Resolved) {
758  if (state == PluginSpec::Loaded)
759  return true;
760  errorString = QCoreApplication::translate("PluginSpec", "Loading the library failed because state != Resolved");
761  hasError = true;
762  return false;
763  }
764 #ifdef QT_NO_DEBUG
765 
766 #ifdef Q_OS_WIN
767  QString libName = QString("%1/%2.dll").arg(location).arg(name);
768 #elif defined(Q_OS_MAC)
769  QString libName = QString("%1/lib%2.dylib").arg(location).arg(name);
770 #else
771  QString libName = QString("%1/lib%2.so").arg(location).arg(name);
772 #endif
773 
774 #else //Q_NO_DEBUG
775 
776 #ifdef Q_OS_WIN
777  QString libName = QString("%1/%2d.dll").arg(location).arg(name);
778 #elif defined(Q_OS_MAC)
779  QString libName = QString("%1/lib%2_debug.dylib").arg(location).arg(name);
780 #else
781  QString libName = QString("%1/lib%2.so").arg(location).arg(name);
782 #endif
783 
784 #endif
785 
786  PluginLoader loader(libName);
787  if (!loader.load()) {
788  hasError = true;
789  errorString = libName + QString::fromLatin1(": ") + loader.errorString();
790  return false;
791  }
792  IPlugin *pluginObject = qobject_cast<IPlugin*>(loader.instance());
793  if (!pluginObject) {
794  hasError = true;
795  errorString = QCoreApplication::translate("PluginSpec", "Plugin is not valid (does not derive from IPlugin)");
796  loader.unload();
797  return false;
798  }
799  state = PluginSpec::Loaded;
800  plugin = pluginObject;
801  plugin->d->pluginSpec = q;
802  return true;
803 }
