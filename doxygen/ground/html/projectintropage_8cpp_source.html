00129 bool ProjectIntroPage::validate()
00130 {
00131     // Validate and display status
00132     if (!m_d->m_ui.pathChooser->isValid()) {
00133         displayStatusMessage(Error, m_d->m_ui.pathChooser->errorMessage());
00134         return false;
00135     }
00136 
00137     // Name valid? Ignore 'DisplayingInitialText' state.
00138     bool nameValid = false;
00139     switch (m_d->m_ui.nameLineEdit->state()) {
00140     case BaseValidatingLineEdit::Invalid:
00141         displayStatusMessage(Error, m_d->m_ui.nameLineEdit->errorMessage());
00142         return false;
00143     case BaseValidatingLineEdit::DisplayingInitialText:
00144         break;
00145     case BaseValidatingLineEdit::Valid:
00146         nameValid = true;
00147         break;
00148     }
00149 
00150     // Check existence of the directory
00151     QString projectDir = path();
00152     projectDir += QDir::separator();
00153     projectDir += m_d->m_ui.nameLineEdit->text();
00154     const QFileInfo projectDirFile(projectDir);
00155     if (!projectDirFile.exists()) { // All happy
00156         hideStatusLabel();
00157         return nameValid;
00158     }
00159 
00160     if (projectDirFile.isDir()) {
00161         displayStatusMessage(Warning, tr("The project already exists."));
00162         return nameValid;;
00163     }
00164     // Not a directory, but something else, likely causing directory creation to fail
00165     displayStatusMessage(Error, tr("A file with that name already exists."));
00166     return false;
00167 }
