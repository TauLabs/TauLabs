192 SystemSettings::AirframeTypeOptions ConfigGroundVehicleWidget::updateConfigObjectsFromWidgets()
193 {
194  SystemSettings::AirframeTypeOptions airframeType = SystemSettings::AIRFRAMETYPE_GROUNDVEHICLECAR;
195 
196  // Save the curve (common to all ground vehicle frames)
197  MixerSettings *mixerSettings = MixerSettings::GetInstance(getObjectManager());
198  Q_ASSERT(mixerSettings);
199 
200  // Remove Feed Forward, it is pointless on a ground vehicle:
201  setMixerValue(mixerSettings, "FeedForward", 0.0);
202 
203  // set the throttle curves
204  setThrottleCurve(mixerSettings, VehicleConfig::MIXER_THROTTLECURVE1, m_aircraft->groundVehicleThrottle1->getCurve() );
205  setThrottleCurve(mixerSettings, VehicleConfig::MIXER_THROTTLECURVE2, m_aircraft->groundVehicleThrottle2->getCurve() );
206 
207  //All airframe types must start with "GroundVehicle"
208  if (m_aircraft->groundVehicleType->currentText() == "Turnable (car)" ) {
209  airframeType = SystemSettings::AIRFRAMETYPE_GROUNDVEHICLECAR;
210  setupGroundVehicleCar(airframeType);
211  } else if (m_aircraft->groundVehicleType->currentText() == "Differential (tank)") {
212  airframeType = SystemSettings::AIRFRAMETYPE_GROUNDVEHICLEDIFFERENTIAL;
213  setupGroundVehicleDifferential(airframeType);
214  } else { // "Motorcycle"
215  airframeType = SystemSettings::AIRFRAMETYPE_GROUNDVEHICLEMOTORCYCLE;
216  setupGroundVehicleMotorcycle(airframeType);
217  }
218 
219  return airframeType;
