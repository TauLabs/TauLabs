49 bool UAVObjectManager::registerObject(UAVDataObject* obj)
50 {
51  QMutexLocker locker(mutex);
52  // Check if this object type is already in the list
53  quint32 objID = obj->getObjID();
54  for (int objidx = 0; objidx < objects.size(); ++objidx)
55  {
56  // Check if the object ID is in the list
57  if (objects[objidx].size() > 0 && objects[objidx][0]->getObjID() == objID)
58  {
59  // Check if this is a single instance object, if yes we can not add a new instance
60  if (obj->isSingleInstance())
61  {
62  return false;
63  }
64  // The object type has alredy been added, so now we need to initialize the new instance with the appropriate id
65  // There is a single metaobject for all object instances of this type, so no need to create a new one
66  // Get object type metaobject from existing instance
67  UAVDataObject* refObj = dynamic_cast<UAVDataObject*>(objects[objidx][0]);
68  if (refObj == NULL)
69  {
70  return false;
71  }
72  UAVMetaObject* mobj = refObj->getMetaObject();
73  // If the instance ID is specified and not at the default value (0) then we need to make sure
74  // that there are no gaps in the instance list. If gaps are found then then additional instances
75  // will be created.
76  if ( (obj->getInstID() > 0) && (obj->getInstID() < MAX_INSTANCES) )
77  {
78  for (int instidx = 0; instidx < objects[objidx].size(); ++instidx)
79  {
80  if ( objects[objidx][instidx]->getInstID() == obj->getInstID() )
81  {
82  // Instance conflict, do not add
83  return false;
84  }
85  }
86  // Check if there are any gaps between the requested instance ID and the ones in the list,
87  // if any then create the missing instances.
88  for (quint32 instidx = objects[objidx].size(); instidx < obj->getInstID(); ++instidx)
89  {
90  UAVDataObject* cobj = obj->clone(instidx);
91  cobj->initialize(mobj);
92  objects[objidx].append(cobj);
93  getObject(cobj->getObjID())->emitNewInstance(cobj);
94  emit newInstance(cobj);
95  }
96  // Finally, initialize the actual object instance
97  obj->initialize(mobj);
98  }
99  else if (obj->getInstID() == 0)
100  {
101  // Assign the next available ID and initialize the object instance
102  obj->initialize(objects[objidx].size(), mobj);
103  }
104  else
105  {
106  return false;
107  }
108  // Add the actual object instance in the list
109  objects[objidx].append(obj);
110  getObject(objID)->emitNewInstance(obj);
111  emit newInstance(obj);
112  return true;
113  }
114  }
115  // If this point is reached then this is the first time this object type (ID) is added in the list
116  // create a new list of the instances, add in the object collection and create the object's metaobject
117  // Create metaobject
118  QString mname = obj->getName();
119  mname.append("Meta");
120  UAVMetaObject* mobj = new UAVMetaObject(objID + 1, mname, obj);
121  // Initialize object
122  obj->initialize(0, mobj);
123  // Add to list
124  addObject(obj);
125  addObject(mobj);
126  return true;
127 }

151 QVector< QVector<UAVDataObject*> > UAVObjectManager::getDataObjects()
152 {
153  QMutexLocker locker(mutex);
154  QVector< QVector<UAVDataObject*> > dObjects;
155 
156  // Go through objects and copy to new list when types match
157  for (int objidx = 0; objidx < objects.size(); ++objidx)
158  {
159  if (objects[objidx].size() > 0)
160  {
161  // Check type
162  UAVDataObject* obj = dynamic_cast<UAVDataObject*>(objects[objidx][0]);
163  if (obj != NULL)
164  {
165  // Create instance list
166  QVector<UAVDataObject*> list;
167  // Go through instances and cast them to UAVDataObject, then add to list
168  for (int instidx = 0; instidx < objects[objidx].size(); ++instidx)
169  {
170  obj = dynamic_cast<UAVDataObject*>(objects[objidx][instidx]);
171  if (obj != NULL)
172  {
173  list.append(obj);
174  }
175  }
176  // Append to object list
177  dObjects.append(list);
178  }
179  }
180  }
181  // Done
182  return dObjects;

243 UAVObject* UAVObjectManager::getObject(const QString* name, quint32 objId, quint32 instId)
244 {
245  QMutexLocker locker(mutex);
246  // Check if this object type is already in the list
247  for (int objidx = 0; objidx < objects.size(); ++objidx)
248  {
249  // Check if the object ID is in the list
250  if (objects[objidx].size() > 0)
251  {
252  if ( (name != NULL && objects[objidx][0]->getName().compare(name) == 0) || (name == NULL && objects[objidx][0]->getObjID() == objId) )
253  {
254  // Look for the requested instance ID
255  for (int instidx = 0; instidx < objects[objidx].size(); ++instidx)
256  {
257  if (objects[objidx][instidx]->getInstID() == instId)
258  {
259  return objects[objidx][instidx];
260  }
261  }
262  }
263  }
264  }
265  //qWarning("UAVObjectManager::getObject: Object not found. Probably a bug or mismatched GCS/flight versions.");
266  // If this point is reached then the requested object could not be found
267  return NULL;
