00050 {
00051     QMutexLocker locker(mutex);
00052     // Check if this object type is already in the list
00053     quint32 objID = obj->getObjID();
00054     for (int objidx = 0; objidx < objects.size(); ++objidx)
00055     {
00056         // Check if the object ID is in the list
00057         if (objects[objidx].size() > 0 && objects[objidx][0]->getObjID() == objID)
00058         {
00059             // Check if this is a single instance object, if yes we can not add a new instance
00060             if (obj->isSingleInstance())
00061             {
00062                 return false;
00063             }
00064             // The object type has alredy been added, so now we need to initialize the new instance with the appropriate id
00065             // There is a single metaobject for all object instances of this type, so no need to create a new one
00066             // Get object type metaobject from existing instance
00067             UAVDataObject* refObj = dynamic_cast<UAVDataObject*>(objects[objidx][0]);
00068             if (refObj == NULL)
00069             {
00070                 return false;
00071             }
00072             UAVMetaObject* mobj = refObj->getMetaObject();
00073             // If the instance ID is specified and not at the default value (0) then we need to make sure
00074             // that there are no gaps in the instance list. If gaps are found then then additional instances
00075             // will be created.
00076             if ( (obj->getInstID() > 0) && (obj->getInstID() < MAX_INSTANCES) )
00077             {
00078                 for (int instidx = 0; instidx < objects[objidx].size(); ++instidx)
00079                 {
00080                     if ( objects[objidx][instidx]->getInstID() == obj->getInstID() )
00081                     {
00082                         // Instance conflict, do not add
00083                         return false;
00084                     }
00085                 }
00086                 // Check if there are any gaps between the requested instance ID and the ones in the list,
00087                 // if any then create the missing instances.
00088                 for (quint32 instidx = objects[objidx].size(); instidx < obj->getInstID(); ++instidx)
00089                 {
00090                     UAVDataObject* cobj = obj->clone(instidx);
00091                     cobj->initialize(mobj);
00092                     objects[objidx].append(cobj);
00093                     getObject(cobj->getObjID())->emitNewInstance(cobj);
00094                     emit newInstance(cobj);
00095                 }
00096                 // Finally, initialize the actual object instance
00097                 obj->initialize(mobj);
00098             }
00099             else if (obj->getInstID() == 0)
00100             {
00101                 // Assign the next available ID and initialize the object instance
00102                 obj->initialize(objects[objidx].size(), mobj);
00103             }
00104             else
00105             {
00106                 return false;
00107             }
00108             // Add the actual object instance in the list
00109             objects[objidx].append(obj);
00110             getObject(objID)->emitNewInstance(obj);
00111             emit newInstance(obj);
00112             return true;
00113         }
00114     }
00115     // If this point is reached then this is the first time this object type (ID) is added in the list
00116     // create a new list of the instances, add in the object collection and create the object's metaobject
00117     // Create metaobject
00118     QString mname = obj->getName();
00119     mname.append("Meta");
00120     UAVMetaObject* mobj = new UAVMetaObject(objID + 1, mname, obj);
00121     // Initialize object
00122     obj->initialize(0, mobj);
00123     // Add to list
00124     addObject(obj);
00125     addObject(mobj);
00126     return true;
