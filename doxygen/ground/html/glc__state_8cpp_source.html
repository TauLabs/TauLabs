147 void GLC_State::init()
148 {
149  if (!m_IsValid)
150  {
151  Q_ASSERT((NULL != QGLContext::currentContext()) && QGLContext::currentContext()->isValid());
152  setVboSupport();
153  setGlslSupport();
154  setPointSpriteSupport();
155  setFrameBufferSupport();
156  m_Version= (char *) glGetString(GL_VERSION);
157  m_Vendor= (char *) glGetString(GL_VENDOR);
158  m_Renderer= (char *) glGetString(GL_RENDERER);
159 
160  m_IsValid= true;
161  }
162 }
