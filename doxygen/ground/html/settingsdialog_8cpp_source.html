57 bool optionsPageLessThan(const IOptionsPage *p1, const IOptionsPage *p2)
58 {
59  const UAVGadgetOptionsPageDecorator *gp1 = qobject_cast<const UAVGadgetOptionsPageDecorator*>(p1);
60  const UAVGadgetOptionsPageDecorator *gp2 = qobject_cast<const UAVGadgetOptionsPageDecorator*>(p2);
61  if (gp1 && (gp2 == NULL))
62  return false;
63 
64  if (gp2 && (gp1 == NULL))
65  return true;
66 
67  if (const int cc = QString::localeAwareCompare(p1->trCategory(), p2->trCategory()))
68  return cc < 0;
69 
70  return QString::localeAwareCompare(p1->trName(), p2->trName()) < 0;
71 }

237  m_currentPage = data.id;
238  // check if we are looking at a place holder or not
239  QWidget *widget = dynamic_cast<QLabel*>(stackedPages->widget(index));
240  if (widget) {
241  // place holder found, get rid of it...
242  stackedPages->removeWidget(widget);
243  delete widget;
244  // and replace place holder with actual option page
245  IOptionsPage *page = m_pages.at(index);
246  stackedPages->insertWidget(index, page->createPage(stackedPages));
247  }
248  stackedPages->setCurrentIndex(index);
249  // If user selects a toplevel item, select the first child for them
250  // I.e. Top level items are not really selectable
251  if ((pageTree->indexOfTopLevelItem(item) >= 0) && (item->childCount() > 0)) {
252  emit categoryItemSelected();

285  QTreeWidgetItem *categoryItem = 0;
