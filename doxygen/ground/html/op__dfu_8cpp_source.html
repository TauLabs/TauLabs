00953 quint32 DFUObject::CRC32WideFast(quint32 Crc, quint32 Size, quint32 *Buffer)
00954 {
00955     //Size = Size >> 2; // /4  Size passed in as a byte count, assumed to be a multiple of 4
00956 
00957     while(Size--)
00958     {
00959         static const quint32 CrcTable[16] = { // Nibble lookup table for 0x04C11DB7 polynomial
00960             0x00000000,0x04C11DB7,0x09823B6E,0x0D4326D9,0x130476DC,0x17C56B6B,0x1A864DB2,0x1E475005,
00961             0x2608EDB8,0x22C9F00F,0x2F8AD6D6,0x2B4BCB61,0x350C9B64,0x31CD86D3,0x3C8EA00A,0x384FBDBD };
00962 
00963         Crc = Crc ^ *((quint32 *)Buffer); // Apply all 32-bits
00964 
00965         Buffer += 1;
00966 
00967         // Process 32-bits, 4 at a time, or 8 rounds
00968 
00969         Crc = (Crc << 4) ^ CrcTable[Crc >> 28]; // Assumes 32-bit reg, masking index to 4-bits
00970         Crc = (Crc << 4) ^ CrcTable[Crc >> 28]; //  0x04C11DB7 Polynomial used in STM32

00972         Crc = (Crc << 4) ^ CrcTable[Crc >> 28];
