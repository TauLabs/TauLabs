148 void QxtBasicSTDLoggerEngine::writeFormatted(QxtLogger::LogLevel level, const QList<QVariant> &msgs)
149 {
150  switch (level)
151  {
152  case QxtLogger::ErrorLevel:
153  writeToStdErr("Error", msgs);
154  break;
155  case QxtLogger::WarningLevel:
156  writeToStdOut("Warning", msgs);

158  case QxtLogger::CriticalLevel:
159  writeToStdErr("Critical", msgs);
160  break;
161  case QxtLogger::FatalLevel:
162  writeToStdErr("!!FATAL!!", msgs);
163  break;
164  case QxtLogger::TraceLevel:
165  writeToStdOut("Trace", msgs);
166  break;
167  case QxtLogger::DebugLevel:
168  writeToStdErr("DEBUG", msgs);
169  break;
170  case QxtLogger::InfoLevel:
171  writeToStdOut("INFO", msgs);
172  break;
173  default:
174  writeToStdOut("", msgs);
175  break;
176  }

184 void QxtBasicSTDLoggerEngine::writeToStdErr(const QString &level, const QList<QVariant> &msgs)
185 {
186  if (msgs.isEmpty()) return;
187  QString header = '[' + QTime::currentTime().toString("hh:mm:ss.zzz") + "] [" + level + "] ";
188  QString padding;
189  QTextStream* errstream = stdErrStream();
190  Q_ASSERT(errstream);
191  *errstream << header;
192  for (int i = 0; i < header.size(); i++) padding.append(" ");
193  int count = 0;
194  Q_FOREACH(const QVariant& out, msgs)
195  {
196  if (!out.isNull())
197  {
198  if (count != 0) *errstream << padding;
199  *errstream << out.toString();// << '\n';
200  }
201  count++;
202  }
203  *errstream << endl;
204 }

211 void QxtBasicSTDLoggerEngine::writeToStdOut(const QString& level, const QList<QVariant> &msgs)
212 {
213  /* Message format...
