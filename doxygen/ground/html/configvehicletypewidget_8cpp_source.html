417 void ConfigVehicleTypeWidget::enableFFTest()
418 {
419  // Role:
420  // - Check if all three checkboxes are checked
421  // - Every other timer event: toggle engine from 45% to 55%
422  // - Every other time event: send FF settings to flight FW
423  if (m_aircraft->ffTestBox1->isChecked() &&
424  m_aircraft->ffTestBox2->isChecked() &&
425  m_aircraft->ffTestBox3->isChecked()) {
426  if (!ffTuningInProgress)
427  {
428  // Initiate tuning:
429  UAVDataObject* obj = dynamic_cast<UAVDataObject*>(getObjectManager()->getObject(QString("ManualControlCommand")));
430  UAVObject::Metadata mdata = obj->getMetadata();
431  accInitialData = mdata;
432  UAVObject::SetFlightAccess(mdata, UAVObject::ACCESS_READONLY);
433  obj->setMetadata(mdata);
434  }
435  // Depending on phase, either move actuator or send FF settings:
436  if (ffTuningPhase) {
437  // Send FF settings to the board
438  MixerSettings *mixerSettings = MixerSettings::GetInstance(getObjectManager());
439  Q_ASSERT(mixerSettings);
440 
441  QPointer<VehicleConfig> vconfig = new VehicleConfig();
442 
443  // Update feed forward settings
444  vconfig->setMixerValue(mixerSettings, "FeedForward", m_aircraft->feedForwardSlider->value() / 100.0);
445  vconfig->setMixerValue(mixerSettings, "AccelTime", m_aircraft->accelTime->value());
446  vconfig->setMixerValue(mixerSettings, "DecelTime", m_aircraft->decelTime->value());
447  vconfig->setMixerValue(mixerSettings, "MaxAccel", m_aircraft->maxAccelSlider->value());
448  mixerSettings->updated();
449  } else {
450  // Toggle motor state
451  UAVDataObject* obj = dynamic_cast<UAVDataObject*>(getObjectManager()->getObject(QString("ManualControlCommand")));
452  double value = obj->getField("Throttle")->getDouble();
453  double target = (value < 0.5) ? 0.55 : 0.45;
454  obj->getField("Throttle")->setValue(target);
455  obj->updated();
456  }
457  ffTuningPhase = !ffTuningPhase;
458  ffTuningInProgress = true;
459  QTimer::singleShot(1000, this, SLOT(enableFFTest()));
460  } else {
461  // - If no: disarm timer, restore actuatorcommand metadata
462  // Disarm!
463  if (ffTuningInProgress) {
464  ffTuningInProgress = false;
465  UAVDataObject* obj = dynamic_cast<UAVDataObject*>(getObjectManager()->getObject(QString("ManualControlCommand")));
466  UAVObject::Metadata mdata = obj->getMetadata();
467  mdata = accInitialData; // Restore metadata
468  obj->setMetadata(mdata);
469  }
470  }
471 }

479 void ConfigVehicleTypeWidget::refreshWidgetsValues(UAVObject * obj)
480 {
481 
482  ConfigTaskWidget::refreshWidgetsValues(obj);
483 
484  if(!allObjectsUpdated())
485  return;
486 
487  bool dirty=isDirty();
488 
489  MixerSettings *mixerSettings = MixerSettings::GetInstance(getObjectManager());
490  Q_ASSERT(mixerSettings);
491 
492  SystemSettings *systemSettings = SystemSettings::GetInstance(getObjectManager());
493  Q_ASSERT(systemSettings);
494  SystemSettings::DataFields systemSettingsData = systemSettings->getData();
495 
496  // Get the Airframe type from the system settings:
497  // At this stage, we will need to have some hardcoded settings in this code, this
498  // is not ideal, but there you go.
499  frameType = (SystemSettings::AirframeTypeOptions) systemSettingsData.AirframeType;
500  setupAirframeUI(frameType);
501 
502  QPointer<VehicleConfig> vconfig = new VehicleConfig();
503 
504  QList<double> curveValues;
505  vconfig->getThrottleCurve(mixerSettings, VehicleConfig::MIXER_THROTTLECURVE1, &curveValues);
506 
507  // is at least one of the curve values != 0?
508  if (vconfig->isValidThrottleCurve(&curveValues)) {
509  // yes, use the curve we just read from mixersettings
510  m_aircraft->multiThrottleCurve->initCurve(&curveValues);
511  m_aircraft->fixedWingThrottle->initCurve(&curveValues);
512  m_aircraft->groundVehicleThrottle1->initCurve(&curveValues);
513  }
514  else {
515  // no, init a straight curve
516  m_aircraft->multiThrottleCurve->initLinearCurve(curveValues.count(), 0.9);
517  m_aircraft->fixedWingThrottle->initLinearCurve(curveValues.count(), 1.0);
518  m_aircraft->groundVehicleThrottle1->initLinearCurve(curveValues.count(), 1.0);
519  }
520 
521  // Setup all Throttle2 curves for all types of airframes //AT THIS MOMENT, THAT MEANS ONLY GROUND VEHICLES
522  vconfig->getThrottleCurve(mixerSettings, VehicleConfig::MIXER_THROTTLECURVE2, &curveValues);
523 
524  if (vconfig->isValidThrottleCurve(&curveValues)) {
525  m_aircraft->groundVehicleThrottle2->initCurve(&curveValues);
526  }
527  else {
528  m_aircraft->groundVehicleThrottle2->initLinearCurve(curveValues.count(), 1.0);
529  }
530 
531  // Load the Settings for vehicle frames:
532  switch(frameType) {
533  case SystemSettings::AIRFRAMETYPE_FIXEDWING:
534  case SystemSettings::AIRFRAMETYPE_FIXEDWINGELEVON:
535  case SystemSettings::AIRFRAMETYPE_FIXEDWINGVTAIL:
536  // Retrieve fixed wing settings
537  m_fixedwing->refreshAirframeWidgetsValues(frameType);
538  break;
539  case SystemSettings::AIRFRAMETYPE_TRI:
540  case SystemSettings::AIRFRAMETYPE_QUADX:
541  case SystemSettings::AIRFRAMETYPE_QUADP:
542  case SystemSettings::AIRFRAMETYPE_HEXA:
543  case SystemSettings::AIRFRAMETYPE_HEXAX:
544  case SystemSettings::AIRFRAMETYPE_HEXACOAX:
545  case SystemSettings::AIRFRAMETYPE_OCTO:
546  case SystemSettings::AIRFRAMETYPE_OCTOV:
547  case SystemSettings::AIRFRAMETYPE_OCTOCOAXP:
548  case SystemSettings::AIRFRAMETYPE_OCTOCOAXX:
549  // Retrieve multirotor settings
550  m_multirotor->refreshAirframeWidgetsValues(frameType);
551  break;
552  case SystemSettings::AIRFRAMETYPE_HELICP:
553  // Retrieve helicopter settings
554  setComboCurrentIndex(m_aircraft->aircraftType, m_aircraft->aircraftType->findText("Helicopter"));
555  m_heli->refreshAirframeWidgetsValues(frameType);
556  break;
557  case SystemSettings::AIRFRAMETYPE_GROUNDVEHICLECAR:
558  case SystemSettings::AIRFRAMETYPE_GROUNDVEHICLEDIFFERENTIAL:
559  case SystemSettings::AIRFRAMETYPE_GROUNDVEHICLEMOTORCYCLE:
560  // Retrieve ground vehicle settings
561  m_groundvehicle->refreshAirframeWidgetsValues(frameType);
562  break;
563  default:
564  // Retrieve custom settings
565  setComboCurrentIndex(m_aircraft->aircraftType, m_aircraft->aircraftType->findText("Custom"));
566  break;
567  }
568 
569  updateCustomAirframeUI();
570 
571  // Tell UI is has unsaved changes
572  setDirty(dirty);

668 void ConfigVehicleTypeWidget::updateCustomAirframeUI()
669 {
670  MixerSettings *mixerSettings = MixerSettings::GetInstance(getObjectManager());
671  Q_ASSERT(mixerSettings);
672 
673  QPointer<VehicleConfig> vconfig = new VehicleConfig();
674 
675  QList<double> curveValues;
676  vconfig->getThrottleCurve(mixerSettings, VehicleConfig::MIXER_THROTTLECURVE1, &curveValues);
677 
678  // is at least one of the curve values != 0?
679  if (vconfig->isValidThrottleCurve(&curveValues)) {
680  m_aircraft->customThrottle1Curve->initCurve(&curveValues);
681  }
682  else {
683  // no, init a straight curve
684  m_aircraft->customThrottle1Curve->initLinearCurve(curveValues.count(), 1.0);
685  }
686 
687  MixerSettings::DataFields mixerSettingsData = mixerSettings->getData();
688  if (mixerSettingsData.Curve2Source == MixerSettings::CURVE2SOURCE_THROTTLE)
689  m_aircraft->customThrottle2Curve->setMixerType(MixerCurve::MIXERCURVE_THROTTLE, false);
690  else {
691  m_aircraft->customThrottle2Curve->setMixerType(MixerCurve::MIXERCURVE_OTHER, false);
692  }
693 
694  // Setup all Throttle2 curves for all types of airframes
695  vconfig->getThrottleCurve(mixerSettings, VehicleConfig::MIXER_THROTTLECURVE2, &curveValues);
696 
697  if (vconfig->isValidThrottleCurve(&curveValues)) {
698  m_aircraft->customThrottle2Curve->initCurve(&curveValues);
699  }
700  else {
701  m_aircraft->customThrottle2Curve->initLinearCurve(curveValues.count(), 1.0, m_aircraft->customThrottle2Curve->getMin());
702  }
703 
704  // Update the mixer table:
705  for (int channel=0; channel<(int)(VehicleConfig::CHANNEL_NUMELEM); channel++) {
706  UAVObjectField* field = mixerSettings->getField(mixerTypes.at(channel));
707  if (field)
708  {
709  QComboBox* q = (QComboBox*)m_aircraft->customMixerTable->cellWidget(0,channel);
710  if (q)
711  {
712  QString s = field->getValue().toString();
713  setComboCurrentIndex(q, q->findText(s));
714  }
715 
716  m_aircraft->customMixerTable->item(1,channel)->setText(
717  QString::number(vconfig->getMixerVectorValue(mixerSettings,channel,VehicleConfig::MIXERVECTOR_THROTTLECURVE1)));
718  m_aircraft->customMixerTable->item(2,channel)->setText(
719  QString::number(vconfig->getMixerVectorValue(mixerSettings,channel,VehicleConfig::MIXERVECTOR_THROTTLECURVE2)));
720  m_aircraft->customMixerTable->item(3,channel)->setText(
721  QString::number(vconfig->getMixerVectorValue(mixerSettings,channel,VehicleConfig::MIXERVECTOR_ROLL)));
722  m_aircraft->customMixerTable->item(4,channel)->setText(
723  QString::number(vconfig->getMixerVectorValue(mixerSettings,channel,VehicleConfig::MIXERVECTOR_PITCH)));
724  m_aircraft->customMixerTable->item(5,channel)->setText(
725  QString::number(vconfig->getMixerVectorValue(mixerSettings,channel,VehicleConfig::MIXERVECTOR_YAW)));
726  }
727  }
728 
729  // Update feed forward settings
730  m_aircraft->feedForwardSlider->setValue(vconfig->getMixerValue(mixerSettings,"FeedForward") * 100);
731  m_aircraft->accelTime->setValue(vconfig->getMixerValue(mixerSettings,"AccelTime"));
732  m_aircraft->decelTime->setValue(vconfig->getMixerValue(mixerSettings,"DecelTime"));
733  m_aircraft->maxAccelSlider->setValue(vconfig->getMixerValue(mixerSettings,"MaxAccel"));
734 }
