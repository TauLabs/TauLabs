00419     // Role:
00420     // - Check if all three checkboxes are checked
00421     // - Every other timer event: toggle engine from 45% to 55%
00422     // - Every other time event: send FF settings to flight FW

00424         m_aircraft->ffTestBox2->isChecked() &&

00433             obj->setMetadata(mdata);
00434         }
00435         // Depending on phase, either move actuator or send FF settings:

00437             // Send FF settings to the board

00439             Q_ASSERT(mixerSettings);

00446             vconfig->setMixerValue(mixerSettings, "DecelTime", m_aircraft->decelTime->value());
