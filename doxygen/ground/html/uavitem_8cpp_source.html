164  //Add text for time rings.

167  temp.addText(-(groundspeed_mps_filt*ringTime*1*meters2pixels+10), 0, borderfont, QString("%1 s").arg(ringTime,0,'f',0));
168  temp.addText(-(groundspeed_mps_filt*ringTime*2*meters2pixels+10), 0, borderfont, QString("%1 s").arg(ringTime*2,0,'f',0));

172  if(groundspeed_mps_filt*ringTime*2*meters2pixels > 200){
173  if(groundspeed_mps_filt*ringTime*1*meters2pixels > 200){
174  temp.addText(groundspeed_mps_filt*ringTime*1*meters2pixels-8, 0, borderfont, QString("%1 s").arg(ringTime,0,'f',0));
175  }
176  temp.addText(groundspeed_mps_filt*ringTime*2*meters2pixels-8, 0, borderfont, QString("%1 s").arg(ringTime*2,0,'f',0));

206  void UAVItem::SetYawRate(double yawRate_dps){
207  this->yawRate_dps=yawRate_dps;
208 
209  if (fabs(this->yawRate_dps) < 5e-1){ //This number is really the smallest we can go. Any smaller, and it might have problems if we forecast a shorter distance into the future

224  void UAVItem::SetGroundspeed(double vNED[3], int m_maxUpdateRate_ms){
225  this->vNED[0] = vNED[0];
226  this->vNED[1] = vNED[1];
227  this->vNED[2] = vNED[2];
228  groundspeed_kph=sqrt(vNED[0]*vNED[0] + vNED[1]*vNED[1] + vNED[2]*vNED[2])*3.6;
229  groundspeed_mps=groundspeed_kph/3.6;
230  //On the first pass, set the filtered speed to the reported speed.
231  static bool firstGroundspeed=true;
232  if (firstGroundspeed){
233  groundspeed_mps_filt=groundspeed_kph/3.6;
234  firstGroundspeed=false;
235  }
236  else{
237  int riseTime_ms=1000;
238  double alpha= m_maxUpdateRate_ms/(double)(m_maxUpdateRate_ms+riseTime_ms);
239  groundspeed_mps_filt= alpha*groundspeed_mps_filt + (1-alpha)*(groundspeed_kph/3.6);
240  }
241  ringTime=10*pow(2,17-map->ZoomTotal()); //Basic ring is 10 seconds wide at zoom level 17
242  precalcRings=groundspeed_mps_filt*ringTime*meters2pixels;
243  boundingRectSize=groundspeed_mps_filt*ringTime*4*meters2pixels+20;
244  prepareGeometryChange();
245  }

248  void UAVItem::SetUAVPos(const internals::PointLatLng &position, const int &altitude)
249  {
250  if(coord.IsEmpty())
251  lastcoord=coord;
252  if(coord!=position)
253  {
254 
255  if(trailtype==UAVTrailType::ByTimeElapsed)
256  {
257  if(timer.elapsed()>trailtime*1000)
258  {
259  TrailItem * ob=new TrailItem(position,altitude,Qt::green,map);
260  trail->addToGroup(ob);
261  connect(this,SIGNAL(setChildPosition()),ob,SLOT(setPosSLOT()));
262  if(!lasttrailline.IsEmpty())
263  {
264  TrailLineItem * obj=new TrailLineItem(lasttrailline,position,Qt::red,map);
265  trailLine->addToGroup(obj);
266  connect(this,SIGNAL(setChildLine()),obj,SLOT(setLineSlot()));
267  }
268  lasttrailline=position;
269  timer.restart();
270  }
271 
272  }
273  else if(trailtype==UAVTrailType::ByDistance)
274  {
275  if(qAbs(internals::PureProjection::DistanceBetweenLatLng(lastcoord,position)*1000)>traildistance)
276  {
277  TrailItem * ob=new TrailItem(position,altitude,Qt::green,map);
278  trail->addToGroup(ob);
279  connect(this,SIGNAL(setChildPosition()),ob,SLOT(setPosSLOT()));
280  if(!lasttrailline.IsEmpty())
281  {
282  TrailLineItem * obj=new TrailLineItem(lasttrailline,position,Qt::red,map);
283  trailLine->addToGroup(obj);
284  connect(this,SIGNAL(setChildLine()),obj,SLOT(setLineSlot()));
285  }
286  lasttrailline=position;
287  lastcoord=position;
288  }
289  }
290  coord=position;
291  this->altitude=altitude;
292  RefreshPos();
293  if(mapfollowtype==UAVMapFollowType::CenterAndRotateMap||mapfollowtype==UAVMapFollowType::CenterMap)
294  {
295  mapwidget->SetCurrentPosition(coord);
296  }
297  if(autosetreached)
298  {
299  foreach(QGraphicsItem* i,map->childItems())
300  {
301  WayPointItem* wp=qgraphicsitem_cast<WayPointItem*>(i);
302  if(wp)
303  {
304  if(Distance3D(wp->Coord(),wp->Altitude())<autosetdistance)
305  {
306  wp->SetReached(true);
307  emit UAVReachedWayPoint(wp->Number(),wp);
308  }
309  }
310  }
311  }
312  if(mapwidget->Home!=0)
313  {
314  //verify if the UAV is inside the safety bouble
315  if(Distance3D(mapwidget->Home->Coord(),mapwidget->Home->Altitude())>mapwidget->Home->SafeArea())
316  {
317  if(mapwidget->Home->safe!=false)
318  {
319  mapwidget->Home->safe=false;
320  mapwidget->Home->update();
321  emit UAVLeftSafetyBouble(this->coord);
322  }
323  }
324  else
325  {
326  if(mapwidget->Home->safe!=true)
327  {
328  mapwidget->Home->safe=true;
329  mapwidget->Home->update();
330  }
331  }
332 
333  }
334  }

424  void UAVItem::generateArrowhead(){
425  qreal arrowSize = 10;
426 
427  //Create line from (0,0), to (1,1). Later, we'll scale and rotate it
428  arrowShaft=QLineF(0,0,1.0,1.0);
429 
430  //Set the starting point to (0,0)
431  arrowShaft.setP1(QPointF(0,0));
432 
433  //Set angle and length
434  arrowShaft.setLength(60.0);
435  arrowShaft.setAngle(90.0);
436 
437  //Form arrowhead
438  double angle = ::acos(arrowShaft.dx() / arrowShaft.length());
439  if (arrowShaft.dy() <= 0)
440  angle = (M_PI * 2) - angle;
441 
442  QPointF arrowP1 = arrowShaft.pointAt(1) + QPointF(sin(angle + M_PI / 3) * arrowSize,
443  cos(angle + M_PI / 3) * arrowSize);
444  QPointF arrowP2 = arrowShaft.pointAt(1) + QPointF(sin(angle + M_PI - M_PI / 3) * arrowSize,
445  cos(angle + M_PI - M_PI / 3) * arrowSize);
446 
447  //Assemble arrowhead
448  arrowHead.clear();
449  arrowHead << arrowShaft.pointAt(1) << arrowP1 << arrowP2;
450 
451  }
