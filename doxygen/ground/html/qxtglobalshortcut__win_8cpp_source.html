28 bool QxtGlobalShortcutPrivate::eventFilter(void* message)
29 {
30  MSG* msg = static_cast<MSG*>(message);
31  if (msg->message == WM_HOTKEY)
32  {
33  const quint32 keycode = HIWORD(msg->lParam);
34  const quint32 modifiers = LOWORD(msg->lParam);
35  activateShortcut(keycode, modifiers);
36  }
37  return false;
38 }
