65 void GLC_RepFlyMover::update()
66 {
67  Q_ASSERT(NULL != m_pRepMoverInfo);
68  Q_ASSERT(!m_pRepMoverInfo->m_VectorInfo.isEmpty());
69  Q_ASSERT(!m_pRepMoverInfo->m_DoubleInfo.isEmpty());
70 
71  GLC_Vector3d vector(m_pRepMoverInfo->m_VectorInfo.first());
72 
73  // Rotation
74  double deltaX= vector.x();
75  double angle= - deltaX;
76  GLC_Matrix4x4 rotation(glc::Z_AXIS, angle);
77 
78  // Translation
79  vector.setX(vector.x() * m_Radius * 4.0);
80  vector.setY(vector.y() * m_Radius * 4.0);
81  GLC_Matrix4x4 translation(vector);
82 
83  m_Plane.setMatrix(translation * rotation);
84 }

101 void GLC_RepFlyMover::glDraw()
102 {
103  Q_ASSERT(NULL != m_pRepMoverInfo);
104  Q_ASSERT(!m_pRepMoverInfo->m_DoubleInfo.isEmpty());
105 
106  // Get viewport informations
107  const double calibre= 800.0;
108  const double hRatio= static_cast<double>(m_pViewport->viewHSize()) / calibre;
109  const double vRatio= static_cast<double>(m_pViewport->viewVSize()) / calibre;
110 
111  glDisable(GL_TEXTURE_2D);
112  GLC_Context::current()->glcEnableLighting(false);
113  glDisable(GL_DEPTH_TEST);
114 
115  GLC_Context::current()->glcMatrixMode(GL_PROJECTION);
116  GLC_Context::current()->glcPushMatrix();
117  GLC_Context::current()->glcLoadIdentity();
118  GLC_Context::current()->glcOrtho(hRatio * -1.0 ,hRatio * 1.0 ,vRatio * -1.0 ,vRatio * 1.0 ,-1.0 ,1.0);
119  GLC_Context::current()->glcMatrixMode(GL_MODELVIEW);
120  GLC_Context::current()->glcPushMatrix();
121  GLC_Context::current()->glcLoadIdentity();
122 
123  m_CenterCircle.render(glc::WireRenderFlag);
124  m_Hud.render(glc::WireRenderFlag);
125  m_Plane.render(glc::WireRenderFlag);
126 
127  glEnable(GL_BLEND);
128  glBlendFunc(GL_SRC_ALPHA, GL_ONE_MINUS_SRC_ALPHA);
129 
130  m_CenterCircle.render(glc::TransparentRenderFlag);
131  m_Hud.render(glc::TransparentRenderFlag);
132  m_Plane.render(glc::TransparentRenderFlag);
133 
134  // Render velocity value + text
135  QString velocity(QChar(' ') + QString::number(static_cast<int>(100.0 * m_pRepMoverInfo->m_DoubleInfo.first())));
136  QFont myFont;
137  myFont.setBold(true);
138  QFontMetrics fontmetrics(myFont);
139  int txtHeight= fontmetrics.boundingRect(velocity).height();
140  double posy= 2.0 * static_cast<double>(txtHeight) / calibre;
141  m_pViewport->qGLWidgetHandle()->renderText(- m_HudOffset.getX(), m_HudOffset.getY() - posy, 0.0, velocity, myFont);
142 
143  GLC_Context::current()->glcPopMatrix();
144  GLC_Context::current()->glcMatrixMode(GL_PROJECTION);
145  GLC_Context::current()->glcPopMatrix();
146  GLC_Context::current()->glcMatrixMode(GL_MODELVIEW);
147 
148  glEnable(GL_DEPTH_TEST);
149 }

151 void GLC_RepFlyMover::createRepresentation()
152 {
153  // HUD creation
154  GLC_Circle* pCircle= new GLC_Circle(m_Radius);
155  pCircle->setWireColor(GLC_RepMover::m_MainColor);
156  pCircle->setLineWidth(GLC_RepMover::m_Thickness);
157  m_CenterCircle.addGeometry(pCircle);
158 
159  GLC_Polylines* pPolylines= new GLC_Polylines();
160  GLfloatVector points;
161  const double hudx= m_HudOffset.getX();
162  const double hudy= m_HudOffset.getY();
163  points << -hudx << -hudy << 0.0;
164  points << -hudx << hudy << 0.0;
165  pPolylines->addPolyline(points);
166  points.clear();
167  points << hudx << -hudy << 0.0;
168  points << hudx << hudy << 0.0;
169  pPolylines->addPolyline(points);
170  pPolylines->setWireColor(GLC_RepMover::m_MainColor);
171  pPolylines->setLineWidth(GLC_RepMover::m_Thickness);
172  m_Hud.addGeometry(pPolylines);
173 
174  // Plane creation
175  pPolylines= new GLC_Polylines();
176  points.clear();
177  const double l1= m_Radius * 1.5;
178  points << (-m_Radius - l1) << -m_Radius << 0.0;
179  points << -m_Radius << -m_Radius << 0.0;
180  points << 0.0 << 0.0 << 0.0;
181  points << m_Radius << -m_Radius << 0.0;
182  points << (m_Radius + l1) << -m_Radius << 0.0;
183  pPolylines->addPolyline(points);
184  pPolylines->setWireColor(GLC_RepMover::m_MainColor);
185  pPolylines->setLineWidth(GLC_RepMover::m_Thickness);
186  m_Plane.addGeometry(pPolylines);
187 }
