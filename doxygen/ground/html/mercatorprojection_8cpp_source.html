00037 Point MercatorProjection::FromLatLngToPixel(double lat, double lng, const int &zoom)
00038 {
00039     Point ret;// = Point.Empty;
00040 
00041     lat = bound(lat, MinLatitude, MaxLatitude);
00042     lng = bound(lng, MinLongitude, MaxLongitude);
00043 
00044     double x = (lng + 180) / 360;
00045     double sinLatitude = sin(lat * M_PI / 180);
00046     double y = 0.5 - log((1 + sinLatitude) / (1 - sinLatitude)) / (4 * M_PI);
00047 
00048     Size s = GetTileMatrixSizePixel(zoom);
00049     qint64 mapSizeX = s.Width();
00050     qint64 mapSizeY = s.Height();
00051 
00052     ret.SetX((qint64) round(bound(x * mapSizeX + 0.5, 0, mapSizeX - 1)));
00053     ret.SetY((qint64) round(bound(y * mapSizeY + 0.5, 0, mapSizeY - 1)));
00054 
00055     return ret;

00065 internals::PointLatLng MercatorProjection::FromPixelToLatLng(const qint64 &x,const qint64 &y,const int &zoom)
00066 {
00067     internals::PointLatLng ret;// = internals::PointLatLng.Empty;
00068 
00069     Size s = GetTileMatrixSizePixel(zoom);
00070     double mapSizeX = s.Width();
00071     double mapSizeY = s.Height();
00072 
00073     //Calculate the percentage distance between top and bottom, and left and right
00074     double xx = (bound(x, 0, mapSizeX - 1) / mapSizeX) - 0.5;
00075     double yy = 0.5 - (bound(y, 0, mapSizeY - 1) / mapSizeY);
00076 
00077     ret.SetLat(90 - 360 * atan(exp(-yy * 2 * M_PI)) / M_PI);
00078     ret.SetLng(360 * xx);
00079 
00080     return ret;
