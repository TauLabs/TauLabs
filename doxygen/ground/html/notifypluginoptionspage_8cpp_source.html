270 void NotifyPluginOptionsPage::addDynamicFieldWidget(UAVObjectField* objField)
271 {
272  if(!objField) {
273  qNotifyDebug() << "objField == NULL!";
274  return;
275  }
276  // check if dynamic fileld already settled,
277  // so if its exists remove it and install new field
278  if (_dynamicFieldWidget) {
279  _optionsPage->dynamicValueLayout->removeWidget(_dynamicFieldWidget);
280  delete _dynamicFieldWidget;
281  _dynamicFieldWidget = NULL;
282  }
283 
284  QSizePolicy sizePolicy(QSizePolicy::Fixed, QSizePolicy::Fixed);
285  sizePolicy.setHorizontalStretch(0);
286  sizePolicy.setVerticalStretch(0);
287 
288  _dynamicFieldType = objField->getType();
289  switch(_dynamicFieldType)
290  {
291  case UAVObjectField::ENUM:
292  {
293  _dynamicFieldWidget = new QComboBox(_form);
294  QStringList enumValues(objField->getOptions());
295  (dynamic_cast<QComboBox*>(_dynamicFieldWidget))->addItems(enumValues);
296  }
297  break;
298 
299  default:
300  Q_ASSERT(_dynamicFieldCondition);
301  if (NotifyPluginOptionsPage::conditionValues.indexOf(_dynamicFieldCondition->currentText()) == NotifyPluginOptionsPage::inrange) {
302  _dynamicFieldWidget = new QLineEdit(_form);
303 
304  (static_cast<QLineEdit*>(_dynamicFieldWidget))->setInputMask("#999.99 : #999.99;");
305  (static_cast<QLineEdit*>(_dynamicFieldWidget))->setText("0000000000");
306  (static_cast<QLineEdit*>(_dynamicFieldWidget))->setCursorPosition(0);
307  } else {
308  _dynamicFieldWidget = new QDoubleSpinBox(_form);
309  (dynamic_cast<QDoubleSpinBox*>(_dynamicFieldWidget))->setRange(-999.99, 999.99);
310  }
311  break;
312  };
313  enum { eDynamicFieldWidth = 100 };
314  _dynamicFieldWidget->setSizePolicy(sizePolicy);
315  _dynamicFieldWidget->setFixedWidth(eDynamicFieldWidth);
316  _optionsPage->dynamicValueLayout->addWidget(_dynamicFieldWidget);
317 }
