86 int pjrc_rawhid::open(int max, int vid, int pid, int usage_page, int usage)
87 {
88  Q_UNUSED(max);
89 
90  CFMutableDictionaryRef dict;
91  CFNumberRef num;
92  IOReturn ret;
93 
94  Q_ASSERT(hid_manager == NULL);
95  Q_ASSERT(device_open == false);
96 
97  attach_count = 0;
98 
99  // Start the HID Manager
100  hid_manager = IOHIDManagerCreate(kCFAllocatorDefault, kIOHIDOptionsTypeNone);
101  if (hid_manager == NULL || CFGetTypeID(hid_manager) != IOHIDManagerGetTypeID()) {
102  if (hid_manager) CFRelease(hid_manager);
103  return 0;
104  }
105 
106  if (vid > 0 || pid > 0 || usage_page > 0 || usage > 0) {
107  // Tell the HID Manager what type of devices we want
108  dict = CFDictionaryCreateMutable(kCFAllocatorDefault, 0,
109  &kCFTypeDictionaryKeyCallBacks, &kCFTypeDictionaryValueCallBacks);
110  if (!dict) return 0;
111  if (vid > 0) {
112  num = CFNumberCreate(kCFAllocatorDefault, kCFNumberIntType, &vid);
113  CFDictionarySetValue(dict, CFSTR(kIOHIDVendorIDKey), num);
114  CFRelease(num);
115  }
116  if (pid > 0) {
117  num = CFNumberCreate(kCFAllocatorDefault, kCFNumberIntType, &pid);
118  CFDictionarySetValue(dict, CFSTR(kIOHIDProductIDKey), num);
119  CFRelease(num);
120  }
121  if (usage_page > 0) {
122  num = CFNumberCreate(kCFAllocatorDefault, kCFNumberIntType, &usage_page);
123  CFDictionarySetValue(dict, CFSTR(kIOHIDPrimaryUsagePageKey), num);
124  CFRelease(num);
125  }
126  if (usage > 0) {
127  num = CFNumberCreate(kCFAllocatorDefault, kCFNumberIntType, &usage);
128  CFDictionarySetValue(dict, CFSTR(kIOHIDPrimaryUsageKey), num);
129  CFRelease(num);
130  }
131  IOHIDManagerSetDeviceMatching(hid_manager, dict);
132  CFRelease(dict);
133  } else {
134  IOHIDManagerSetDeviceMatching(hid_manager, NULL);
135  }
136 
137  // Set the run loop reference before configuring the attach callback
138  the_correct_runloop = CFRunLoopGetCurrent();
139 
140  // set up a callbacks for device attach & detach
141  IOHIDManagerScheduleWithRunLoop(hid_manager, CFRunLoopGetCurrent(),
142  kCFRunLoopDefaultMode);
143  IOHIDManagerRegisterDeviceMatchingCallback(hid_manager, pjrc_rawhid::attach_callback, this);
144  IOHIDManagerRegisterDeviceRemovalCallback(hid_manager, pjrc_rawhid::dettach_callback, this);
145  ret = IOHIDManagerOpen(hid_manager, kIOHIDOptionsTypeNone);
146  if (ret != kIOReturnSuccess) {
147  IOHIDManagerUnscheduleFromRunLoop(hid_manager,
148  CFRunLoopGetCurrent(), kCFRunLoopDefaultMode);
149  CFRelease(hid_manager);
150  return 0;
151  }
152 
153  // let it do the callback for all devices
154  while (CFRunLoopRunInMode(kCFRunLoopDefaultMode, 0, true) == kCFRunLoopRunHandledSource) ;
155 
156  // count up how many were added by the callback
157  return attach_count;
158 }

168 int pjrc_rawhid::receive(int, void *buf, int len, int timeout)
169 {
170  QMutexLocker locker(m_readMutex);
171  Q_UNUSED(locker);
172 
173  if (!device_open)
174  return -1;
175 
176  // Pass information to the callback to stop this run loop and signal if a timeout occurred
177  struct timeout_info info;
178  info.loopRef = CFRunLoopGetCurrent();;
179  info.timed_out = false;
180  CFRunLoopTimerContext context;
181  memset(&context, 0, sizeof(context));
182  context.info = &info;
183 
184  // Set up the timer for the timeout
185  CFRunLoopTimerRef timer;
186  timer = CFRunLoopTimerCreate(NULL, CFAbsoluteTimeGetCurrent() + (double)timeout / 1000.0, 0, 0, 0, timeout_callback, &context);
187  CFRunLoopAddTimer(CFRunLoopGetCurrent(), timer, kCFRunLoopDefaultMode);
188 
189  received_runloop = CFRunLoopGetCurrent();
190 
191  // Run the CFRunLoop until either a timeout or data is available
192  while(1) {
193  if (buffer_count != 0) {
194  if (len > buffer_count) len = buffer_count;
195  memcpy(buf, buffer, len);
196  buffer_count = 0;
197  break;
198  } else if (info.timed_out) {
199  len = 0;
200  break;
201  }
202  CFRunLoopRun(); // Wait for data
203  }
204 
205  CFRunLoopTimerInvalidate(timer);
206  CFRelease(timer);
207 
208  received_runloop = NULL;
209 
210  return len;
211 }

244 int pjrc_rawhid::send(int, void *buf, int len, int timeout)
245 {
246  // This lock ensures that when closing we don't do it until the
247  // write has terminated (and then the device_open flag is set to false)
248  QMutexLocker locker(m_writeMutex);
249  Q_UNUSED(locker);
250 
251  if(!device_open || unplugged) {
252  return -1;
253  }
254 
255  uint8_t *report_buf = (uint8_t *) malloc(len);
256  memcpy(&report_buf[0], buf,len);
257 
258  QEventLoop el;
259  Sender sender(dev, report_buf, len);
260  connect(&sender, SIGNAL(finished()), &el, SLOT(quit()));
261  sender.start();
262  QTimer::singleShot(timeout, &el, SLOT(quit()));
263  el.exec();
264 
265  return sender.result;
266 }

292 void pjrc_rawhid::close(int)
293 {
294  // Make sure any pending locks are done
295  QMutexLocker lock(m_writeMutex);
296 
297  if (device_open) {
298  device_open = false;
299  CFRunLoopStop(the_correct_runloop);
300 
301  if (!unplugged) {
302  IOHIDDeviceUnscheduleFromRunLoop(dev, the_correct_runloop, kCFRunLoopDefaultMode);
303  IOHIDDeviceRegisterInputReportCallback(dev, buffer, sizeof(buffer), NULL, NULL);
304  IOHIDDeviceClose(dev, kIOHIDOptionsTypeNone);
305  }
306 
307  IOHIDManagerRegisterDeviceRemovalCallback(hid_manager, NULL, NULL);
308  IOHIDManagerClose(hid_manager, 0);
309 
310  dev = NULL;
311  hid_manager = NULL;
312  }

375 void pjrc_rawhid::attach(IOHIDDeviceRef d)
376 {
377  // Store the device handle
378  dev = d;
379 
380  if (IOHIDDeviceOpen(dev, kIOHIDOptionsTypeNone) != kIOReturnSuccess) return;
381  // Disconnect the attach callback since we don't want to automatically reconnect
382  IOHIDManagerRegisterDeviceMatchingCallback(hid_manager, NULL, NULL);
383  IOHIDDeviceScheduleWithRunLoop(dev, the_correct_runloop, kCFRunLoopDefaultMode);
384  IOHIDDeviceRegisterInputReportCallback(dev, buffer, sizeof(buffer), pjrc_rawhid::input_callback, this);
385 
386  attach_count++;
387  device_open = true;
388  unplugged = false;
389 }
