00072 void IPConnection::onOpenDevice(QString HostName, int Port, bool UseTCP)
00073 {
00074     QAbstractSocket *ipSocket;
00075     const int Timeout = 5 * 1000;
00076 
00077     ipConMutex.lock();
00078     if (UseTCP) {
00079         ipSocket = new QTcpSocket(this);
00080     } else {
00081         ipSocket = new QUdpSocket(this);
00082     }
00083 
00084     //do sanity check on hostname and port...
00085     if((HostName.length()==0)||(Port<1)){
00086         errorMsg = "Please configure Host and Port options before opening the connection";
00087 
00088     }
00089     else {
00090         //try to connect...
00091         ipSocket->connectToHost(HostName, Port);
00092 
00093         //in blocking mode so we wait for the connection to succeed
00094         if (ipSocket->waitForConnected(Timeout)) {
00095             ret = ipSocket;
00096             openDeviceWait.wakeAll();
00097             ipConMutex.unlock();
00098             return;
00099         }
00100         //tell user something went wrong
00101         errorMsg = ipSocket->errorString ();
00102     }
00103     /* BUGBUG TODO - returning null here leads to segfault because some caller still calls disconnect without checking our return value properly
00104     * someone needs to debug this, I got lost in the calling chain.*/
00105     ret = NULL;
00106     openDeviceWait.wakeAll();
00107     ipConMutex.unlock();
00108 }
