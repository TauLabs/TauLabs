136 void MultiPage::updateImageAndDescription()
137 {
138  SetupWizard::VEHICLE_SUB_TYPE type = (SetupWizard::VEHICLE_SUB_TYPE) ui->typeCombo->itemData(ui->typeCombo->currentIndex()).toInt();
139  QString elementId = "";
140  QString description = m_descriptions.at(ui->typeCombo->currentIndex());
141  switch(type)
142  {
143  case SetupWizard::MULTI_ROTOR_TRI_Y:
144  elementId = "tri";
145  break;
146  case SetupWizard::MULTI_ROTOR_QUAD_X:
147  elementId = "quad-x";
148  break;
149  case SetupWizard::MULTI_ROTOR_QUAD_PLUS:
150  elementId = "quad-plus";
151  break;
152  case SetupWizard::MULTI_ROTOR_HEXA:
153  elementId = "quad-hexa";
154  break;
155  case SetupWizard::MULTI_ROTOR_HEXA_COAX_Y:
156  elementId = "hexa-coax";
157  break;
158  case SetupWizard::MULTI_ROTOR_HEXA_H:
159  elementId = "quad-hexa-H";
160  break;
161  case SetupWizard::MULTI_ROTOR_OCTO:
162  elementId = "quad-octo";
163  break;
164  case SetupWizard::MULTI_ROTOR_OCTO_COAX_X:
165  elementId = "octo-coax-X";
166  break;
167  case SetupWizard::MULTI_ROTOR_OCTO_COAX_PLUS:
168  elementId = "octo-coax-P";
169  break;
170  case SetupWizard::MULTI_ROTOR_OCTO_V:
171  elementId = "quad-octo-v";
172  break;
173  default:
174  elementId = "";
175  break;
176  }
177  m_multiPic->setElementId(elementId);
178  ui->typeGraphicsView->setSceneRect(m_multiPic->boundingRect());
179  ui->typeGraphicsView->fitInView(m_multiPic, Qt::KeepAspectRatio);
180 
181  ui->typeDescription->setText(description);
182 }
