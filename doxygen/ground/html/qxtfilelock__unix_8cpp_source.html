69 bool QxtFileLockRegistry::registerLock(QxtFileLock * lock)
70 {
71  QMutexLocker locker(&this->registryMutex);
72 
73  QFile *fileToLock = lock ->file();
74 
75  if (fileToLock)
76  {
77  struct stat fileInfo;
78  if (fstat(fileToLock->handle(), &fileInfo) < 0)
79  return false;
80 
81  int newLockStart = lock ->offset();
82  int newLockEnd = lock ->offset() + lock ->length();
83 
84  QMutableLinkedListIterator< QPointer<QxtFileLock> >iterator(this->procLocks);
85 
86  while (iterator.hasNext())
87  {
88  QPointer<QxtFileLock> currLock = iterator.next();
89  if (currLock && currLock->file() && currLock->file()->isOpen())
90  {
91  struct stat currFileInfo;
92 
93  /*first check if the current lock is on the same file*/
94  if (fstat(currLock->file()->handle(), &currFileInfo) < 0)
95  {
96  /*that should never happen because a closing file should remove all locks*/
97  Q_ASSERT(false);
98  continue;
99  }
100 
101  if (currFileInfo.st_dev == fileInfo.st_dev && currFileInfo.st_ino == fileInfo.st_ino)
102  {
103  /*same file, check if our locks are in conflict*/
104  int currLockStart = currLock->offset();
105  int currLockEnd = currLock->offset() + currLock->length();
106 
107  /*do we have to check for threads here?*/
108  if (newLockEnd >= currLockStart && newLockStart <= currLockEnd)
109  {
110  qDebug() << "we may have a collision";
111  qDebug() << newLockEnd << " >= " << currLockStart << " && " << newLockStart << " <= " << currLockEnd;
112 
113  /*same lock region if one of both locks are exclusive we have a collision*/
114  if (lock ->mode() == QxtFileLock::WriteLockWait || lock ->mode() == QxtFileLock::WriteLock ||
115  currLock->mode() == QxtFileLock::WriteLockWait || currLock->mode() == QxtFileLock::WriteLock)
116  {
117  qDebug() << "Okay if this is not the same thread using the same handle there is a collision";
118  /*the same thread can lock the same region with the same handle*/
119 
120  qDebug() << "! (" << lock ->thread() << " == " << currLock->thread() << " && " << lock ->file()->handle() << " == " << currLock->file()->handle() << ")";
121 
122  if (!(lock ->thread() == currLock->thread() && lock ->file()->handle() == currLock->file()->handle()))
123  {
124  qDebug() << "Collision";
125  return false;
126  }
127  }
128  }
129  }
130  }
131  else //remove dead locks
132  iterator.remove();
133  }
134  qDebug() << "The lock is okay";
135  /*here we can insert the lock into the list and return*/
136  procLocks.append(QPointer<QxtFileLock>(lock));
137  return true;
138 
139  }
140 
141  return false;
142 }

151 bool QxtFileLock::unlock()
152 {
153  if (file() && file()->isOpen() && isActive())
154  {
155  /*first remove real lock*/
156  int lockmode, locktype;
157  int result = -1;
158  struct flock lockDesc;
159 
160  lockmode = F_SETLK;
161  locktype = F_UNLCK;
162 
163  errno = 0;
164  do
165  {
166  lockDesc.l_type = locktype;
167  lockDesc.l_whence = SEEK_SET;
168  lockDesc.l_start = qxt_d().offset;
169  lockDesc.l_len = qxt_d().length;
170  lockDesc.l_pid = 0;
171  result = fcntl(this->file()->handle(), lockmode, &lockDesc);
172  }
173  while (result && errno == EINTR);
174 
175  QxtFileLockRegistry::instance().removeLock(this);
176  qxt_d().isLocked = false;
177  return true;
178  }
179  return false;
180 }

182 bool QxtFileLock::lock()
183 {
184  if (file() && file()->isOpen() && !isActive())
185  {
186  /*this has to block if we can get no lock*/
187 
188  bool locked = false;
189 
190  while (1)
191  {
192  locked = QxtFileLockRegistry::instance().registerLock(this);
193  if (locked)
194  break;
195  else
196  {
197  if (qxt_d().mode == ReadLockWait || qxt_d().mode == WriteLockWait)
198  usleep(1000 * 5);
199  else
200  return false;
201  }
202  }
203 
204  /*now get real lock*/
205  int lockmode,
206  locktype;
207 
208  int result = -1;
209 
210  struct flock lockDesc;
211 
212  switch (qxt_d().mode)
213  {
214  case ReadLock:
215  lockmode = F_SETLK;
216  locktype = F_RDLCK;
217  break;
218 
219  case ReadLockWait:
220  lockmode = F_SETLKW;
221  locktype = F_RDLCK;
222  break;
223 
224  case WriteLock:
225  lockmode = F_SETLK;
226  locktype = F_WRLCK;
227  break;
228 
229  case WriteLockWait:
230  lockmode = F_SETLKW;
231  locktype = F_WRLCK;
232  break;
233 
234  default:
235  QxtFileLockRegistry::instance().removeLock(this);
236  return (false);
237  break;
238  }
239 
240  errno = 0;
241  do
242  {
243  lockDesc.l_type = locktype;
244  lockDesc.l_whence = SEEK_SET;
245  lockDesc.l_start = qxt_d().offset;
246  lockDesc.l_len = qxt_d().length;
247  lockDesc.l_pid = 0;
248  result = fcntl(this->file()->handle(), lockmode, &lockDesc);
249  }
250  while (result && errno == EINTR);
251 
252  /*we dot get the lock unregister from lockregistry and return*/
253  if (result == -1)
254  {
255  QxtFileLockRegistry::instance().removeLock(this);
256  return false;
257  }
258 
259  qxt_d().isLocked = true;
260  return true;
261  }
262  return false;
263 }
