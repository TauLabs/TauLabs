207 bool UAVTalk::processInputByte(quint8 rxbyte)
208 {
209  // Update stats
210  stats.rxBytes++;
211 
212  rxPacketLength++; // update packet byte count
213 
214  if(useUDPMirror)
215  rxDataArray.append(rxbyte);
216 
217  // Receive state machine
218  switch (rxState)
219  {
220  case STATE_SYNC:
221 
222  if (rxbyte != SYNC_VAL)
223  {
224  UAVTALK_QXTLOG_DEBUG("UAVTalk: Sync->Sync (" + QString::number(rxbyte) + " " + QString("0x%1").arg(rxbyte,2,16) + ")");
225  break;
226  }
227 
228  // Initialize and update CRC
229  rxCS = updateCRC(0, rxbyte);
230 
231  rxPacketLength = 1;
232 
233  if(useUDPMirror)
234  {
235  rxDataArray.clear();
236  rxDataArray.append(rxbyte);
237  }
238 
239  rxState = STATE_TYPE;
240  UAVTALK_QXTLOG_DEBUG("UAVTalk: Sync->Type");
241  break;
242 
243  case STATE_TYPE:
244 
245  // Update CRC
246  rxCS = updateCRC(rxCS, rxbyte);
247 
248  if ((rxbyte & TYPE_MASK) != TYPE_VER)
249  {
250  rxState = STATE_SYNC;
251  UAVTALK_QXTLOG_DEBUG("UAVTalk: Type->Sync");
252  break;
253  }
254 
255  rxType = rxbyte;
256 
257  packetSize = 0;
258 
259  rxState = STATE_SIZE;
260  UAVTALK_QXTLOG_DEBUG("UAVTalk: Type->Size");
261  rxCount = 0;
262  break;
263 
264  case STATE_SIZE:
265 
266  // Update CRC
267  rxCS = updateCRC(rxCS, rxbyte);
268 
269  if (rxCount == 0)
270  {
271  packetSize += rxbyte;
272  rxCount++;
273  UAVTALK_QXTLOG_DEBUG("UAVTalk: Size->Size");
274  break;
275  }
276 
277  packetSize += (quint32)rxbyte << 8;
278 
279  if (packetSize < MIN_HEADER_LENGTH || packetSize > MAX_HEADER_LENGTH + MAX_PAYLOAD_LENGTH)
280  { // incorrect packet size
281  rxState = STATE_SYNC;
282  UAVTALK_QXTLOG_DEBUG("UAVTalk: Size->Sync");
283  break;
284  }
285 
286  rxCount = 0;
287  rxState = STATE_OBJID;
288  UAVTALK_QXTLOG_DEBUG("UAVTalk: Size->ObjID");
289  break;
290 
291  case STATE_OBJID:
292 
293  // Update CRC
294  rxCS = updateCRC(rxCS, rxbyte);
295 
296  rxTmpBuffer[rxCount++] = rxbyte;
297  if (rxCount < 4)
298  {
299  UAVTALK_QXTLOG_DEBUG("UAVTalk: ObjID->ObjID");
300  break;
301  }
302 
303  // Search for object, if not found reset state machine
304  rxObjId = (qint32)qFromLittleEndian<quint32>(rxTmpBuffer);
305  {
306  UAVObject *rxObj = objMngr->getObject(rxObjId);
307  if (rxObj == NULL && rxType != TYPE_OBJ_REQ)
308  {
309  stats.rxErrors++;
310  rxState = STATE_SYNC;
311  UAVTALK_QXTLOG_DEBUG("UAVTalk: ObjID->Sync (badtype)");
312  break;
313  }
314  else if (rxObj == NULL)
315  {
316  // This is a non-existing object, just skip to checksum
317  // and we'll send a NACK next.
318  rxState = STATE_CS;
319  UAVTALK_QXTLOG_DEBUG("UAVTalk: ObjID->CSum (no obj)");
320  rxInstId = 0;
321  rxCount = 0;
322  break;
323  }
324 
325  // Determine data length
326  if (rxType == TYPE_OBJ_REQ || rxType == TYPE_ACK || rxType == TYPE_NACK)
327  {
328  rxLength = 0;
329  }
330  else
331  {
332  rxLength = rxObj->getNumBytes();
333  }
334 
335  // Check length and determine next state
336  if (rxLength >= MAX_PAYLOAD_LENGTH)
337  {
338  stats.rxErrors++;
339  rxState = STATE_SYNC;
340  UAVTALK_QXTLOG_DEBUG("UAVTalk: ObjID->Sync (oversize)");
341  break;
342  }
343 
344  quint8 rxInstanceLength = (rxObj->isSingleInstance() ? 0 : 2);
345  if ((rxPacketLength + rxInstanceLength + rxLength) != packetSize)
346  { // packet error - mismatched packet size
347  stats.rxErrors++;
348  rxState = STATE_SYNC;
349  UAVTALK_QXTLOG_DEBUG("UAVTalk: ObjID->Sync (length mismatch)");
350  break;
351  }
352 
353  if (rxObj->isSingleInstance())
354  { // Check if this is a single instance object (i.e. if the instance ID field is coming next)
355  // If there is a payload get it, otherwise receive checksum
356  if (rxLength > 0)
357  {
358  rxState = STATE_DATA;
359  UAVTALK_QXTLOG_DEBUG("UAVTalk: ObjID->Data (needs data)");
360  }
361  else
362  {
363  rxState = STATE_CS;
364  UAVTALK_QXTLOG_DEBUG("UAVTalk: ObjID->Checksum");
365  }
366  rxInstId = 0;
367  rxCount = 0;
368  }
369  else
370  {
371  rxState = STATE_INSTID;
372  UAVTALK_QXTLOG_DEBUG("UAVTalk: ObjID->InstID");
373  rxCount = 0;
374  }
375  }
376 
377  break;
378 
379  case STATE_INSTID:
380 
381  // Update CRC
382  rxCS = updateCRC(rxCS, rxbyte);
383 
384  rxTmpBuffer[rxCount++] = rxbyte;
385  if (rxCount < 2)
386  {
387  UAVTALK_QXTLOG_DEBUG("UAVTalk: InstID->InstID");
388  break;
389  }
390 
391  rxInstId = (qint16)qFromLittleEndian<quint16>(rxTmpBuffer);
392 
393  rxCount = 0;
394 
395  // If there is a payload get it, otherwise receive checksum
396  if (rxLength > 0)
397  {
398  rxState = STATE_DATA;
399  UAVTALK_QXTLOG_DEBUG("UAVTalk: InstID->Data");
400  }
401  else
402  {
403  rxState = STATE_CS;
404  UAVTALK_QXTLOG_DEBUG("UAVTalk: InstID->CSum");
405  }
406  break;
407 
408  case STATE_DATA:
409 
410  // Update CRC
411  rxCS = updateCRC(rxCS, rxbyte);
412 
413  rxBuffer[rxCount++] = rxbyte;
414  if (rxCount < rxLength)
415  {
416  //UAVTALK_QXTLOG_DEBUG("UAVTalk: Data->Data");
417  break;
418  }
419 
420  rxState = STATE_CS;
421  UAVTALK_QXTLOG_DEBUG("UAVTalk: Data->CSum");
422  rxCount = 0;
423  break;
424 
425  case STATE_CS:
426 
427  // The CRC byte
428  rxCSPacket = rxbyte;
429 
430  if (rxCS != rxCSPacket)
431  { // packet error - faulty CRC
432  stats.rxErrors++;
433  rxState = STATE_SYNC;
434  UAVTALK_QXTLOG_DEBUG("UAVTalk: CSum->Sync (badcrc)");
435  break;
436  }
437 
438  if (rxPacketLength != packetSize + 1)
439  { // packet error - mismatched packet size
440  stats.rxErrors++;
441  rxState = STATE_SYNC;
442  UAVTALK_QXTLOG_DEBUG("UAVTalk: CSum->Sync (length mismatch)");
443  break;
444  }
445 
446  mutex->lock();
447  receiveObject(rxType, rxObjId, rxInstId, rxBuffer, rxLength);
448  if(useUDPMirror)
449  {
450  udpSocketTx->writeDatagram(rxDataArray,QHostAddress::LocalHost,udpSocketRx->localPort());
451  }
452  stats.rxObjectBytes += rxLength;
453  stats.rxObjects++;
454  mutex->unlock();
455 
456  rxState = STATE_SYNC;
457  UAVTALK_QXTLOG_DEBUG("UAVTalk: CSum->Sync (OK)");
458  break;
459 
460  default:
461  rxState = STATE_SYNC;
462  stats.rxErrors++;
463  UAVTALK_QXTLOG_DEBUG("UAVTalk: \?\?\?->Sync"); //Use the escape character for '?' so that the tripgraph isn't triggered.
464  }
465 
466  // Done
467  return true;
468 }

479 bool UAVTalk::receiveObject(quint8 type, quint32 objId, quint16 instId, quint8* data, qint32 length)
480 {
481  Q_UNUSED(length);
482  UAVObject* obj = NULL;
483  bool error = false;
484  bool allInstances = (instId == ALL_INSTANCES);
485 
486  // Process message type
487  switch (type) {
488  case TYPE_OBJ: // We have received an object.
489  // All instances, not allowed for OBJ messages
490  if (!allInstances)
491  {
492  // Get object and update its data
493  obj = updateObject(objId, instId, data);
494  if (obj == NULL)
495  error = true;
496  }
497  else
498  {
499  error = true;
500  }
501  break;
502  case TYPE_OBJ_ACK: // We have received an object and are asked for an ACK
503  // All instances, not allowed for OBJ_ACK messages
504  if (!allInstances)
505  {
506  // Get object and update its data
507  obj = updateObject(objId, instId, data);
508  // Transmit ACK
509  if ( obj != NULL )
510  {
511  transmitObject(obj, TYPE_ACK, false);
512  }
513  else
514  {
515  error = true;
516  }
517  }
518  else
519  {
520  error = true;
521  }
522  break;
523  case TYPE_OBJ_REQ: // We are being asked for an object
524  // Get object, if all instances are requested get instance 0 of the object
525  if (allInstances)
526  {
527  obj = objMngr->getObject(objId);
528  }
529  else
530  {
531  obj = objMngr->getObject(objId, instId);
532  }
533  // If object was found transmit it
534  if (obj != NULL)
535  {
536  transmitObject(obj, TYPE_OBJ, allInstances);
537  }
538  else
539  {
540  // Object was not found, transmit a NACK with the
541  // objId which was not found.
542  transmitNack(objId);
543  error = true;
544  }
545  break;
546  case TYPE_NACK: // We have received a NACK for an object that does not exist on the far end.
547  // (but should exist on our end)
548  // All instances, not allowed for NACK messages
549  if (!allInstances)
550  {
551  // Get object
552  obj = objMngr->getObject(objId, instId);
553  // Check if object exists:
554  if (obj != NULL)
555  {
556  emit nackReceived(obj);
557  }
558  else
559  {
560  error = true;
561  }
562  }
563  break;
564  case TYPE_ACK: // We have received a ACK, supposedly after sending an object with OBJ_ACK
565  // All instances, not allowed for ACK messages
566  if (!allInstances)
567  {
568  qDebug() << "Got ack for instance: " << instId;
569  // Get object
570  obj = objMngr->getObject(objId, instId);
571  // Check if we actually know this object (tiny chance the ObjID
572  // could be unknown and got through CRC check...)
573  if (obj != NULL)
574  {
575  emit ackReceived(obj);
576  }
577  else
578  {
579  error = true;
580  }
581  }
582  break;
583  default:
584  error = true;
585  }
586  // Done (exit value is "success", hence the "!" below)
587  return !error;
588 }

596 {
597  // Get object
598  UAVObject* obj = objMngr->getObject(objId, instId);
599  // If the instance does not exist create it
600  if (obj == NULL)
601  {
602  // Get the object type
603  UAVObject* tobj = objMngr->getObject(objId);

619  }
620  instobj->unpack(data);
621  return instobj;

639 bool UAVTalk::transmitObject(UAVObject* obj, quint8 type, bool allInstances)
640 {
641  // If all instances are requested on a single instance object it is an error
642  if (allInstances && obj->isSingleInstance())
643  {
644  allInstances = false;
645  }
646 
647  // Process message type
648  if ( type == TYPE_OBJ || type == TYPE_OBJ_ACK )
649  {
650  if (allInstances)
651  {
652  // Get number of instances
653  quint32 numInst = objMngr->getNumInstances(obj->getObjID());
654  // Send all instances
655  for (quint32 instId = 0; instId < numInst; ++instId)
656  {
657  UAVObject* inst = objMngr->getObject(obj->getObjID(), instId);
658  transmitSingleObject(inst, type, false);
659  }
660  return true;
661  }
662  else
663  {
664  return transmitSingleObject(obj, type, false);
665  }
666  }
667  else if (type == TYPE_OBJ_REQ)
668  {
669  return transmitSingleObject(obj, TYPE_OBJ_REQ, allInstances);
670  }
671  else if (type == TYPE_ACK)
672  {
673  if (!allInstances)
674  {
675  return transmitSingleObject(obj, TYPE_ACK, false);
676  }
677  else
678  {
679  return false;
680  }
681  }
682  else
683  {
684  return false;
685  }

694 bool UAVTalk::transmitNack(quint32 objId)
695 {
696  int dataOffset = 8;
697 
698  txBuffer[0] = SYNC_VAL;
699  txBuffer[1] = TYPE_NACK;
700  qToLittleEndian<quint32>(objId, &txBuffer[4]);
701 
702  // Calculate checksum
703  txBuffer[dataOffset] = updateCRC(0, txBuffer, dataOffset);
704 
705  qToLittleEndian<quint16>(dataOffset, &txBuffer[2]);
706 
707  // Send buffer, check that the transmit backlog does not grow above limit
708  if (io && io->isWritable() && io->bytesToWrite() < TX_BUFFER_SIZE )
709  {
710  io->write((const char*)txBuffer, dataOffset+CHECKSUM_LENGTH);
711  if(useUDPMirror)
712  {
713  udpSocketRx->writeDatagram((const char*)txBuffer,dataOffset+CHECKSUM_LENGTH,QHostAddress::LocalHost,udpSocketTx->localPort());
714  }
715  }
716  else
717  {
718  ++stats.txErrors;
719  return false;
720  }
721 
722  // Update stats
723  stats.txBytes += 8+CHECKSUM_LENGTH;
724 
725  // Done
726  return true;
727
