178 void TelemetryMonitorWidget::showTelemetry()
179 {
180  txIndex = (txValue-minValue)/(maxValue-minValue) * NODE_NUMELEM;
181  rxIndex = (rxValue-minValue)/(maxValue-minValue) * NODE_NUMELEM;
182 
183  if (m_connected)
184  this->setToolTip(QString("Tx: %0 bytes/sec\nRx: %1 bytes/sec").arg(txValue).arg(rxValue));
185  else
186  this->setToolTip(QString("Disconnected"));
187 
188  int i;
189  QGraphicsItem* node;
190 
191  for (i=0; i < txNodes.count(); i++) {
192  node = txNodes.at(i);
193  node->setVisible(m_connected && i < txIndex);
194  node->update();
195  }
196 
197  for (i=0; i < rxNodes.count(); i++) {
198  node = rxNodes.at(i);
199  node->setVisible(m_connected && i < rxIndex);
200  node->update();
201  }
202 
203  txSpeed->setPlainText(QString("%0").arg(txValue));
204  txSpeed->setVisible(m_connected);
205 
206  rxSpeed->setPlainText(QString("%0").arg(rxValue));
207  rxSpeed->setVisible(m_connected);
208 
209  update();
210 }
