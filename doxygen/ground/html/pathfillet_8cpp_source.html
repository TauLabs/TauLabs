00483 enum PathFillet::arc_center_results PathFillet::find_arc_center(float start_point[2], float end_point[2], float radius, float center[2], bool clockwise, bool minor)
00484 {
00485     // Sanity check
00486     if(fabsf(start_point[0] - end_point[0]) < 1e-6 && fabsf(start_point[1] - end_point[1]) < 1e-6){
00487         // This means that the start point and end point are directly on top of each other. In the
00488         // case of coincident points, there is not enough information to define the circle
00489         center[0]=NAN;
00490         center[1]=NAN;
00491         return COINCIDENT_POINTS;
00492     }
00493 
00494     float m_n, m_e, p_n, p_e, d, d2;
00495 
00496     // Center between start and end
00497     m_n = (start_point[0] + end_point[0]) / 2;
00498     m_e = (start_point[1] + end_point[1]) / 2;
00499 
00500     // Normal vector to the line between start and end points
00501     if ((clockwise == true && minor == true) ||
00502             (clockwise == false && minor == false)) { //clockwise minor OR counterclockwise major
00503         p_n = -(end_point[1] - start_point[1]);
00504         p_e =  (end_point[0] - start_point[0]);
00505     }
00506     else { //counterclockwise minor OR clockwise major
00507         p_n =  (end_point[1] - start_point[1]);
00508         p_e = -(end_point[0] - start_point[0]);
00509     }
00510 
00511     // Work out how far to go along the perpendicular bisector. First check there is a solution.
00512     d2 = radius*radius / (p_n*p_n + p_e*p_e) - 0.25f;
00513     if (d2 < 0) {
00514         if (d2 > -powf(radius*0.01f, 2)) // Make a 1% allowance for roundoff error
00515             d2 = 0;
00516         else {
00517             center[0]=NAN;
00518             center[1]=NAN;
00519             return INSUFFICIENT_RADIUS; // In this case, the radius wasn't big enough to connect the two points
00520         }
00521     }
00522 
00523     d = sqrtf(d2);
00524 
00525     if (fabsf(p_n) < 1e-3 && fabsf(p_e) < 1e-3) {
00526         center[0] = m_n;
00527         center[1] = m_e;
00528     }
00529     else {
00530         center[0] = m_n + p_n * d;
00531         center[1] = m_e + p_e * d;
00532     }
00533 
00534     return CENTER_FOUND;

00561 float PathFillet::angle_between_2d_vectors(float a[2], float b[2])
00562 {
00563     // We cannot directly use the vector calculus formula for cos(theta) and sin(theta) because each
00564     // is only unique on half the circle. Instead, we combine the two because tangent is unique across
00565     // [-pi,pi]. Use the definition of the cross-product for 2-D vectors, a x b = |a||b| sin(theta), and
00566     // the definition of the dot product, a.b = |a||b| cos(theta), and divide the first by the second,
00567     // yielding a x b / (a.b) = sin(theta)/cos(theta) == tan(theta)
00568     float theta = atan2f(a[0]*b[1] - a[1]*b[0],(a[0]*b[0] + a[1]*b[1]));
00569     return theta;
00570 }
