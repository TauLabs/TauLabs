483 enum PathFillet::arc_center_results PathFillet::find_arc_center(float start_point[2], float end_point[2], float radius, float center[2], bool clockwise, bool minor)
484 {
485  // Sanity check
486  if(fabsf(start_point[0] - end_point[0]) < 1e-6 && fabsf(start_point[1] - end_point[1]) < 1e-6){
487  // This means that the start point and end point are directly on top of each other. In the
488  // case of coincident points, there is not enough information to define the circle
489  center[0]=NAN;
490  center[1]=NAN;
491  return COINCIDENT_POINTS;
492  }
493 
494  float m_n, m_e, p_n, p_e, d, d2;
495 
496  // Center between start and end
497  m_n = (start_point[0] + end_point[0]) / 2;
498  m_e = (start_point[1] + end_point[1]) / 2;
499 
500  // Normal vector to the line between start and end points
501  if ((clockwise == true && minor == true) ||
502  (clockwise == false && minor == false)) { //clockwise minor OR counterclockwise major
503  p_n = -(end_point[1] - start_point[1]);
504  p_e = (end_point[0] - start_point[0]);
505  }
506  else { //counterclockwise minor OR clockwise major
507  p_n = (end_point[1] - start_point[1]);
508  p_e = -(end_point[0] - start_point[0]);
509  }
510 
511  // Work out how far to go along the perpendicular bisector. First check there is a solution.
512  d2 = radius*radius / (p_n*p_n + p_e*p_e) - 0.25f;
513  if (d2 < 0) {
514  if (d2 > -powf(radius*0.01f, 2)) // Make a 1% allowance for roundoff error
515  d2 = 0;
516  else {
517  center[0]=NAN;
518  center[1]=NAN;
519  return INSUFFICIENT_RADIUS; // In this case, the radius wasn't big enough to connect the two points
520  }
521  }
522 
523  d = sqrtf(d2);
524 
525  if (fabsf(p_n) < 1e-3 && fabsf(p_e) < 1e-3) {
526  center[0] = m_n;
527  center[1] = m_e;
528  }
529  else {
530  center[0] = m_n + p_n * d;
531  center[1] = m_e + p_e * d;
532  }
533 
534  return CENTER_FOUND;
535 }

561 float PathFillet::angle_between_2d_vectors(float a[2], float b[2])
562 {
563  // We cannot directly use the vector calculus formula for cos(theta) and sin(theta) because each
564  // is only unique on half the circle. Instead, we combine the two because tangent is unique across
565  // [-pi,pi]. Use the definition of the cross-product for 2-D vectors, a x b = |a||b| sin(theta), and
566  // the definition of the dot product, a.b = |a||b| cos(theta), and divide the first by the second,
567  // yielding a x b / (a.b) = sin(theta)/cos(theta) == tan(theta)
568  float theta = atan2f(a[0]*b[1] - a[1]*b[0],(a[0]*b[0] + a[1]*b[1]));
569  return theta;
