00109             QString element2 = element + "-" + value;
00110             if (m_renderer->elementExists(element2)) {
00111                 QGraphicsSvgItem *ind = new QGraphicsSvgItem();

00113                 ind->setElementId(element2);
00114                 ind->setParentItem(background);

00117                 ind->setTransform(matrix,false);
