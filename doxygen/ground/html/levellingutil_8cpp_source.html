68 void LevellingUtil::gyroMeasurementsUpdated(UAVObject *obj)
69 {
70  QMutexLocker locker(&m_measurementMutex);
71 
72  if(m_receivedGyroUpdates < m_gyroMeasurementCount) {
73  ExtensionSystem::PluginManager *pm = ExtensionSystem::PluginManager::instance();
74  UAVObjectManager * uavObjectManager = pm->getObject<UAVObjectManager>();
75  Q_ASSERT(uavObjectManager);
76 
77  Gyros * gyros = Gyros::GetInstance(uavObjectManager);
78  Gyros::DataFields gyrosData = gyros->getData();
79 
80  m_gyroX += gyrosData.x;
81  m_gyroY += gyrosData.y;
82  m_gyroZ += gyrosData.z;
83 
84  m_receivedGyroUpdates++;
85  emit progress(m_receivedGyroUpdates + m_receivedAccelUpdates, m_gyroMeasurementCount + m_accelMeasurementCount);
86  }
87  else if (m_receivedAccelUpdates >= m_accelMeasurementCount &&
88  m_receivedGyroUpdates >= m_gyroMeasurementCount && m_isMeasuring) {
89  stopMeasurement();
90  emit done(calculateLevellingData());
91  }
92 }

128 void LevellingUtil::startMeasurement()
129 {
130  QMutexLocker locker(&m_measurementMutex);
131 
132  m_isMeasuring = true;
133 
134  // Reset variables
135  m_receivedAccelUpdates = 0;
136  m_accelerometerX = 0;
137  m_accelerometerY = 0;
138  m_accelerometerZ = 0;
139 
140  m_receivedGyroUpdates = 0;
141  m_gyroX = 0;
142  m_gyroY = 0;
143  m_gyroZ = 0;
144 
145  ExtensionSystem::PluginManager *pm = ExtensionSystem::PluginManager::instance();
146  UAVObjectManager * uavObjectManager = pm->getObject<UAVObjectManager>();
147  Q_ASSERT(uavObjectManager);
148 
149  // Disable gyro bias correction to see raw data
150  AttitudeSettings::DataFields attitudeSettingsData = AttitudeSettings::GetInstance(uavObjectManager)->getData();
151  attitudeSettingsData.BiasCorrectGyro = AttitudeSettings::BIASCORRECTGYRO_FALSE;
152  AttitudeSettings::GetInstance(uavObjectManager)->setData(attitudeSettingsData);
153 
154  // Set up to receive updates for accels
155  UAVDataObject *uavObject = Accels::GetInstance(uavObjectManager);
156  connect(uavObject, SIGNAL(objectUpdated(UAVObject*)), this, SLOT(accelMeasurementsUpdated(UAVObject*)));
157 
158  // Set update period for accels
159  m_previousAccelMetaData = uavObject->getMetadata();
160  UAVObject::Metadata newMetaData = m_previousAccelMetaData;
161  UAVObject::SetFlightTelemetryUpdateMode(newMetaData, UAVObject::UPDATEMODE_PERIODIC);
162  newMetaData.flightTelemetryUpdatePeriod = m_accelMeasurementRate;
163  uavObject->setMetadata(newMetaData);
164 
165  // Set up to receive updates from gyros
166  uavObject = Gyros::GetInstance(uavObjectManager);
167  connect(uavObject, SIGNAL(objectUpdated(UAVObject*)), this, SLOT(gyroMeasurementsUpdated(UAVObject*)));
168 
169  // Set update period for gyros
170  m_previousGyroMetaData = uavObject->getMetadata();
171  newMetaData = m_previousGyroMetaData;
172  UAVObject::SetFlightTelemetryUpdateMode(newMetaData, UAVObject::UPDATEMODE_PERIODIC);
173  newMetaData.flightTelemetryUpdatePeriod = m_gyroMeasurementRate;
174  uavObject->setMetadata(newMetaData);
175 }

177 void LevellingUtil::stopMeasurement()
178 {
179  m_isMeasuring = false;
180 
181  //Stop timeout timer
182  m_timeoutTimer.stop();
183  disconnect(&m_timeoutTimer, SIGNAL(timeout()), this, SLOT(timeout()));
184 
185  ExtensionSystem::PluginManager *pm = ExtensionSystem::PluginManager::instance();
186  UAVObjectManager * uavObjectManager = pm->getObject<UAVObjectManager>();
187  Q_ASSERT(uavObjectManager);
188 
189  // Stop listening for updates from accels
190  UAVDataObject *uavObject = Accels::GetInstance(uavObjectManager);
191  disconnect(uavObject, SIGNAL(objectUpdated(UAVObject*)), this, SLOT(accelMeasurementsUpdated(UAVObject*)));
192  uavObject->setMetadata(m_previousAccelMetaData);
193 
194  // Stop listening for updates from gyros
195  uavObject = Gyros::GetInstance(uavObjectManager);
196  disconnect(uavObject, SIGNAL(objectUpdated(UAVObject*)), this, SLOT(gyroMeasurementsUpdated(UAVObject*)));
197  uavObject->setMetadata(m_previousGyroMetaData);
198 
199  // Enable gyro bias correction again
200  AttitudeSettings::DataFields attitudeSettingsData = AttitudeSettings::GetInstance(uavObjectManager)->getData();
201  attitudeSettingsData.BiasCorrectGyro = AttitudeSettings::BIASCORRECTGYRO_TRUE;
202  AttitudeSettings::GetInstance(uavObjectManager)->setData(attitudeSettingsData);
203 }

205 accelGyroBias LevellingUtil::calculateLevellingData()
206 {
207  accelGyroBias bias;
208  bias.m_accelerometerXBias = m_accelerometerX / (double)m_receivedAccelUpdates / ACCELERATION_SCALE;
209  bias.m_accelerometerYBias = m_accelerometerY / (double)m_receivedAccelUpdates / ACCELERATION_SCALE;
210  bias.m_accelerometerZBias = (m_accelerometerZ / (double)m_receivedAccelUpdates + G) / ACCELERATION_SCALE;
211 
212  bias.m_gyroXBias = m_gyroX / (double)m_receivedGyroUpdates * 100.0;
213  bias.m_gyroYBias = m_gyroY / (double)m_receivedGyroUpdates * 100.0;
214  bias.m_gyroZBias = m_gyroZ / (double)m_receivedGyroUpdates * 100.0;
215  return bias;
216 }
