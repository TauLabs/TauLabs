121  if (cmdNode) {
122  painter->drawText(0,4,text);

148 QVariant MixerNode::itemChange(GraphicsItemChange change, const QVariant &val)
149 {
150  QPointF newPos = val.toPointF();
151  double h = graph->sceneRect().height();
152 
153  switch (change) {
154  case ItemPositionChange: {
155 
156  if (!vertical)
157  break;
158 
159  // Force node to move vertically
160  newPos.setX(pos().x());
161 
162  // Stay inside graph
163  if (newPos.y() < 0)
164  newPos.setY(0);
165  //qDebug() << h << " - " << newPos.y();
166  if (newPos.y() > h)
167  newPos.setY(h);
168 
169  return newPos;
170  }
171  case ItemPositionHasChanged: {
172  foreach (Edge *edge, edgeList)
173  edge->adjust();
174 
175  update();
176 
177  graph->itemMoved(value());
178  break;
179  }
180  default:
181  break;
182  };
183 
184  return QGraphicsItem::itemChange(change, val);
185 }
