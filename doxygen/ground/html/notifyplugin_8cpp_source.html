173 void SoundNotifyPlugin::connectNotifications()
174 {
175  foreach(UAVDataObject* obj,lstNotifiedUAVObjects) {
176  if (obj != NULL)
177  disconnect(obj,SIGNAL(objectUpdated(UAVObject*)),this,SLOT(on_arrived_Notification(UAVObject*)));
178  }
179  if (phonon.mo != NULL) {
180  delete phonon.mo;
181  phonon.mo = NULL;
182  }
183 
184  if (!enableSound) return;
185 
186  ExtensionSystem::PluginManager *pm = ExtensionSystem::PluginManager::instance();
187  UAVObjectManager *objManager = pm->getObject<UAVObjectManager>();
188 
189  lstNotifiedUAVObjects.clear();
190  _pendingNotifications.clear();
191  _notificationList.append(_toRemoveNotifications);
192  _toRemoveNotifications.clear();
193 
194  //first, reject empty args and unknown fields.
195  foreach(NotificationItem* notify, _notificationList) {
196  notify->_isPlayed = false;
197  notify->isNowPlaying=false;
198 
199  if(notify->mute()) continue;
200  // check is all sounds presented for notification,
201  // if not - we must not subscribe to it at all
202  if(notify->toSoundList().isEmpty()) continue;
203 
204  UAVDataObject* obj = dynamic_cast<UAVDataObject*>( objManager->getObject(notify->getDataObject()) );
205  if (obj != NULL ) {
206  if (!lstNotifiedUAVObjects.contains(obj)) {
207  lstNotifiedUAVObjects.append(obj);
208 
209  connect(obj, SIGNAL(objectUpdated(UAVObject*)),
210  this, SLOT(on_arrived_Notification(UAVObject*)),
211  Qt::QueuedConnection);
212  }
213  } else {
214  qNotifyDebug() << "Error: Object is unknown (" << notify->getDataObject() << ").";
215  }
216  }
217 
218  if (_notificationList.isEmpty()) return;
219  // set notification message to current event
220  phonon.mo = Phonon::createPlayer(Phonon::NotificationCategory);
221  phonon.mo->clearQueue();
222  phonon.firstPlay = true;
223  QList<Phonon::AudioOutputDevice> audioOutputDevices =
224  Phonon::BackendCapabilities::availableAudioOutputDevices();
225  foreach(Phonon::AudioOutputDevice dev, audioOutputDevices) {
226  qNotifyDebug() << "Notify: Audio Output device: " << dev.name() << " - " << dev.description();
227  }
228  connect(phonon.mo, SIGNAL(stateChanged(Phonon::State,Phonon::State)),
229  this, SLOT(stateChanged(Phonon::State,Phonon::State)));
230 }

232 void SoundNotifyPlugin::on_arrived_Notification(UAVObject *object)
233 {
234  foreach(NotificationItem* ntf, _notificationList) {
235  if (object->getName() != ntf->getDataObject())
236  continue;
237 
238 
239  // skip duplicate notifications
240  if (_nowPlayingNotification == ntf)
241  continue;
242 
243  // skip periodical notifications
244  // this condition accepts:
245  // 1. Periodical notifications played firstly;
246  // NOTE: At first time it will be played, then it played only by timer,
247  // when conditions became false firstStart flag has been cleared and
248  // notification can be accepted again;
249  // 2. Once time notifications, they removed immediately after first playing;
250  // 3. Instant notifications(played one by one without interval);
251  if (ntf->retryValue() != NotificationItem::repeatInstantly && ntf->retryValue() != NotificationItem::repeatOncePerUpdate &&
252  ntf->retryValue() != NotificationItem::repeatOnce && ntf->_isPlayed)
253  continue;
254 
255  qNotifyDebug() << QString("new notification: | %1 | %2 | val1: %3 | val2: %4")
256  .arg(ntf->getDataObject())
257  .arg(ntf->getObjectField())
258  .arg(ntf->singleValue().toString())
259  .arg(ntf->valueRange2());
260 
261  checkNotificationRule(ntf, object);
262  }
263  connect(object, SIGNAL(objectUpdated(UAVObject*)),
264  this, SLOT(on_arrived_Notification(UAVObject*)), Qt::UniqueConnection);
265 }

268 void SoundNotifyPlugin::on_timerRepeated_Notification()
269 {
270  NotificationItem* notification = static_cast<NotificationItem*>(sender()->parent());
271  if (!notification)
272  return;
273  // skip duplicate notifications
274  // WARNING: generally we shoudn't ever trap here
275  // this means, that timer fires to early and notification overlap itself
276  if (_nowPlayingNotification == notification) {
277  qNotifyDebug() << "WARN: on_timerRepeated - notification was skipped!";
278  notification->restartTimer();
279  return;
280  }
281 
282  qNotifyDebug() << QString("repeatTimer: %1% | %2 | %3").arg(notification->getDataObject())
283  .arg(notification->getObjectField())
284  .arg(notification->toString());
285 
286  ExtensionSystem::PluginManager *pm = ExtensionSystem::PluginManager::instance();
287  UAVObjectManager *objManager = pm->getObject<UAVObjectManager>();
288  UAVObject* object = objManager->getObject(notification->getDataObject());
289  if (object)
290  checkNotificationRule(notification,object);
291 }

395 void SoundNotifyPlugin::checkNotificationRule(NotificationItem* notification, UAVObject* object)
396 {
397  if(notification->getDataObject()!=object->getName() || object->getField(notification->getObjectField())==NULL)
398  return;
399  bool condition=false;
400 
401  if (notification->mute())
402  return;
403 
404  int direction = notification->getCondition();
405  QString fieldName = notification->getObjectField();
406  UAVObjectField* field = object->getField(fieldName);
407 
408  if (field->getName().isEmpty())
409  return;
410 
411  QVariant value = field->getValue();
412  if(UAVObjectField::ENUM == field->getType()) {
413  qNotifyDebug()<<"Check range ENUM"<<value.toString()<<"|"<<notification->singleValue().toString()<<"|"<<field->getOptions()<<"|"<<
414  direction<<checkRange(value.toString(),
415  notification->singleValue().toString(),
416  field->getOptions(),
417  direction);;
418  condition = checkRange(value.toString(),
419  notification->singleValue().toString(),
420  field->getOptions(),
421  direction);
422  } else {
423  qNotifyDebug()<<"Check range VAL"<<value.toString()<<"|"<<notification->singleValue().toString()<<"|"<<field->getOptions()<<"|"<<
424  direction<<checkRange(value.toDouble(),
425  notification->singleValue().toDouble(),
426  notification->valueRange2(),
427  direction);
428  condition = checkRange(value.toDouble(),
429  notification->singleValue().toDouble(),
430  notification->valueRange2(),
431  direction);
432  }
433 
434  notification->_isPlayed = condition;
435  // if condition has been changed, and already in false state
436  // we should reset _isPlayed flag and stop repeat timer
437  if (!notification->_isPlayed) {
438  notification->stopTimer();
439  notification->setCurrentUpdatePlayed(false);
440  return;
441  }
442  if(notification->retryValue() == NotificationItem::repeatOncePerUpdate && notification->getCurrentUpdatePlayed())
443  return;
444 
445  if (!playNotification(notification)) {
446  if (!_pendingNotifications.contains(notification)
447  && (_nowPlayingNotification != notification)) {
448  notification->stopTimer();
449 
450  qNotifyDebug() << "add to pending list - " << notification->toString();
451  // if audio is busy, start expiration timer
452  //ms = (notification->getExpiredTimeout()[in sec])*1000
453  //QxtTimer::singleShot(notification->getExpireTimeout()*1000, this, SLOT(expirationTimerHandler(NotificationItem*)), qVariantFromValue(notification));
454  _pendingNotifications.append(notification);
455  notification->startExpireTimer();
456  connect(notification->getExpireTimer(), SIGNAL(timeout()),
457  this, SLOT(on_expiredTimer_Notification()), Qt::UniqueConnection);
458  }
459  }

462 bool SoundNotifyPlugin::playNotification(NotificationItem* notification)
463 {
464  if(!notification)
465  return false;
466 
467  // Check: race condition, if phonon.mo got deleted don't go further
468  if (phonon.mo == NULL)
469  return false;
470 
471  //qNotifyDebug() << "Phonon State: " << phonon.mo->state();
472 
473  if ((phonon.mo->state()==Phonon::PausedState)
474  || (phonon.mo->state()==Phonon::StoppedState)
475  || phonon.firstPlay)
476  {
477  _nowPlayingNotification = notification;
478  notification->stopExpireTimer();
479 
480  if (notification->retryValue() == NotificationItem::repeatOnce) {
481  _toRemoveNotifications.append(_notificationList.takeAt(_notificationList.indexOf(notification)));
482  }
483  else if(notification->retryValue() == NotificationItem::repeatOncePerUpdate)
484  notification->setCurrentUpdatePlayed(true);
485  else {
486  if (notification->retryValue() != NotificationItem::repeatInstantly) {
487  QRegExp rxlen("(\\d+)");
488  QString value;
489  int timer_value=0;
490  int pos = rxlen.indexIn(NotificationItem::retryValues.at(notification->retryValue()));
491  if (pos > -1) {
492  value = rxlen.cap(1); // "189"
493 
494  // needs to correct repeat timer value,
495  // acording to message play duration,
496  // we don't measure duration of each message,
497  // simply take average duration
498  enum { eAverageDurationSec = 8 };
499 
500  enum { eSecToMsec = 1000 };
501 
502  timer_value = (value.toInt() + eAverageDurationSec) * eSecToMsec;
503  }
504 
505  notification->startTimer(timer_value);
506  connect(notification->getTimer(), SIGNAL(timeout()),
507  this, SLOT(on_timerRepeated_Notification()), Qt::UniqueConnection);
508  }
509  }
510  phonon.mo->clear();
511  qNotifyDebug() << "play: " << notification->toString();
512  foreach (QString item, notification->toSoundList()) {
513  Phonon::MediaSource *ms = new Phonon::MediaSource(item);
514  ms->setAutoDelete(true);
515  phonon.mo->enqueue(*ms);
516  }
517  qNotifyDebug()<<"begin play";
518  phonon.mo->play();
519  qNotifyDebug()<<"end play";
520  phonon.firstPlay = false; // On Linux, you sometimes have to nudge Phonon to play 1 time before
521  // the state is not "Loading" anymore.
522  return true;
523 
524  }
525 
526  return false;
527 }
