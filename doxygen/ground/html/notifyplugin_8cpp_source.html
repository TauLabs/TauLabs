00235 void SoundNotifyPlugin::on_arrived_Notification(UAVObject *object)
00236 {
00237     foreach(NotificationItem* ntf, _notificationList) {
00238         if (object->getName() != ntf->getDataObject())
00239             continue;
00240 
00241 
00242         // skip duplicate notifications
00243         if (_nowPlayingNotification == ntf)
00244             continue;
00245 
00246         // skip periodical notifications
00247         // this condition accepts:
00248         // 1. Periodical notifications played firstly;
00249         //    NOTE: At first time it will be played, then it played only by timer,
00250         //          when conditions became false firstStart flag has been cleared and
00251         //          notification can be accepted again;
00252         // 2. Once time notifications, they removed immediately after first playing;
00253         // 3. Instant notifications(played one by one without interval);
00254         if (ntf->retryValue() != NotificationItem::repeatInstantly && ntf->retryValue() != NotificationItem::repeatOncePerUpdate &&
00255                 ntf->retryValue() != NotificationItem::repeatOnce && ntf->_isPlayed)
00256            continue;
00257 
00258         qNotifyDebug() << QString("new notification: | %1 | %2 | val1: %3 | val2: %4")
00259                                       .arg(ntf->getDataObject())
00260                                         .arg(ntf->getObjectField())
00261                                         .arg(ntf->singleValue().toString())
00262                                         .arg(ntf->valueRange2());
00263 
00264         checkNotificationRule(ntf, object);
00265     }

00466 {

00468         return false;

00470     // Check: race condition, if phonon.mo got deleted don't go further

00477         || (phonon.mo->state()==Phonon::StoppedState)

00480         _nowPlayingNotification = notification;
00481         notification->stopExpireTimer();
00482 
00483         if (notification->retryValue() == NotificationItem::repeatOnce) {
00484             _toRemoveNotifications.append(_notificationList.takeAt(_notificationList.indexOf(notification)));
00485         }
00486         else if(notification->retryValue() == NotificationItem::repeatOncePerUpdate)
00487             notification->setCurrentUpdatePlayed(true);

00489             if (notification->retryValue() != NotificationItem::repeatInstantly) {
00490                 QRegExp rxlen("(\\d+)");

00495                     value = rxlen.cap(1); // "189"
00496 
00497                     // needs to correct repeat timer value,
00498                     // acording to message play duration,
00499                     // we don't measure duration of each message,
00500                     // simply take average duration
00501                     enum { eAverageDurationSec = 8 };
00502 
00503                     enum { eSecToMsec = 1000 };
00504 
00505                     timer_value = (value.toInt() + eAverageDurationSec) * eSecToMsec;
00506                 }
00507 
00508                 notification->startTimer(timer_value);
00509                 connect(notification->getTimer(), SIGNAL(timeout()),
00510                         this, SLOT(on_timerRepeated_Notification()), Qt::UniqueConnection);
