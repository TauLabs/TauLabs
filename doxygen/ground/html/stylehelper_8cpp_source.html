00056 QColor StyleHelper::mergedColors(const QColor &colorA, const QColor &colorB, int factor)
00057 {
00058     const int maxFactor = 100;
00059     QColor tmp = colorA;
00060     tmp.setRed((tmp.red() * factor) / maxFactor + (colorB.red() * (maxFactor - factor)) / maxFactor);
00061     tmp.setGreen((tmp.green() * factor) / maxFactor + (colorB.green() * (maxFactor - factor)) / maxFactor);
00062     tmp.setBlue((tmp.blue() * factor) / maxFactor + (colorB.blue() * (maxFactor - factor)) / maxFactor);
00063     return tmp;
00064 }
