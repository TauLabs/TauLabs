459 double GLC_Mesh::volume()

467  const int materialsCount= materials.count();

474  }

481  triangleIndex.append(equivalentTrianglesIndexOfFansIndex(0, materialId));
482  }
483  }
484 
485  GLfloatVector vertices= m_MeshData.positionVector();
486  Q_ASSERT((triangleIndex.count() % 3) == 0);
487  const int triangleCount= triangleIndex.count() / 3;
488  for (int i= 0; i < triangleCount; ++i)
489  {
490  const int index= i * 3;
491  const double v1x= vertices.at(triangleIndex.at(index) * 3);
492  const double v1y= vertices.at(triangleIndex.at(index) * 3 + 1);
493  const double v1z= vertices.at(triangleIndex.at(index) * 3 + 2);
494 
495  const double v2x= vertices.at(triangleIndex.at(index + 1) * 3);
496  const double v2y= vertices.at(triangleIndex.at(index + 1) * 3 + 1);
497  const double v2z= vertices.at(triangleIndex.at(index + 1) * 3 + 2);
498 
499  const double v3x= vertices.at(triangleIndex.at(index + 2) * 3);
500  const double v3y= vertices.at(triangleIndex.at(index + 2) * 3 + 1);
501  const double v3z= vertices.at(triangleIndex.at(index + 2) * 3 + 2);
502 
503  resultVolume+= ((v2y - v1y) * (v3z - v1z) - (v2z - v1z) * (v3y - v1y)) * (v1x + v2x + v3x);
504  }
505 
506  resultVolume= resultVolume / 6.0;
507  }
