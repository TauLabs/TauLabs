00175 void ConsoleProcess::readStubOutput()
00176 {

00178         QByteArray out = m_stubSocket->readLine();
00179         out.chop(2); // \r\n
00180         if (out.startsWith("err:chdir ")) {
00181             emit processError(msgCannotChangeToWorkDir(workingDirectory(), winErrorMessage(out.mid(10).toInt())));
00182         } else if (out.startsWith("err:exec ")) {
00183             emit processError(msgCannotExecute(m_executable, winErrorMessage(out.mid(9).toInt())));
00184         } else if (out.startsWith("pid ")) {
00185             // Wil not need it any more
00186             delete m_tempFile;
00187             m_tempFile = 0;

00189             m_appPid = out.mid(4).toInt();

00191                     SYNCHRONIZE | PROCESS_QUERY_INFORMATION | PROCESS_TERMINATE,
