173 void UAVGadgetView::removeGadget()
174 {
175  if (!m_uavGadget)
176  return;
177  tl->removeWidget(m_uavGadget->widget());
178 
179  m_uavGadget->setParent(0);
180  m_uavGadget->widget()->setParent(0);
181  QWidget *toolBar = m_uavGadget->toolBar();
182  if (toolBar != 0) {
183  if (m_activeToolBar == toolBar) {
184  m_activeToolBar = m_defaultToolBar;
185  m_activeToolBar->setVisible(true);
186  }
187  m_toolBar->layout()->removeWidget(toolBar);
188  toolBar->setParent(0);
189  }
190  m_uavGadget = 0;
191 }

231 void UAVGadgetView::listSelectionActivated(int index)
232 {
233  if (index < 0) // this could happen when called from SplitterOrView::restoreState()
234  index = m_defaultIndex;
235  QString classId = m_uavGadgetList->itemData(index).toString();
236  if (m_uavGadget && (m_uavGadget->classId() == classId))
237  return;
238  UAVGadgetInstanceManager *im = ICore::instance()->uavGadgetInstanceManager();
239  IUAVGadget *gadgetToRemove = m_uavGadget;
240  IUAVGadget *gadget = im->createGadget(classId, this);
