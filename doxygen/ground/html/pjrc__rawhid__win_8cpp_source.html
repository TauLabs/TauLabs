209  DWORD n;

213  return -1;

215  return -1;
216

218 
219  ResetEvent(&rx_event);
220 
221  memset(&ov, 0, sizeof(ov));
222  ov.hEvent = rx_event;
223 
224  if (!ReadFile(hid->handle, buf, len, NULL, &ov))
225  {
226  DWORD err = GetLastError();
227 
228  if (err == ERROR_DEVICE_NOT_CONNECTED)

236

239  print_win32_err(err);
