222 void GCSControlGadgetOptionsPage::apply()
223 {
224  m_config->setControlsMode(options_page->controlsMode->currentIndex()+1);
225 
226  int roll=-1 , pitch=-1, yaw=-1, throttle=-1;
227  for (int i=0; i<chList.length(); i++) {
228  switch (chList.at(i)->currentIndex()) {
229  case 1:
230  roll = i;
231  break;
232  case 2:
233  pitch =i;
234  break;
235  case 3:
236  yaw = i;
237  break;
238  case 4:
239  throttle = i;
240  break;
241  }
242  }
243  m_config->setRPYTchannels(roll,pitch,yaw,throttle);
244 
245  m_config->setUDPControlSettings(options_page->udp_port->text().toInt(),options_page->udp_host->text());
246 
247  m_config->setGcsReceiverMode(options_page->gcsReceiverCB->checkState());
248 
249  for (unsigned int j = 0; j < 8; j++)
250  {
251  m_config->setbuttonSettingsAction(j,buttonActionList.at(j)->currentIndex());
252  m_config->setbuttonSettingsFunction(j,buttonFunctionList.at(j)->currentIndex());
253  m_config->setbuttonSettingsAmount(j,buttonValueList.at(j)->value());
254  m_config->setChannelReverse(j,chRevList.at(j)->isChecked());
255  }
