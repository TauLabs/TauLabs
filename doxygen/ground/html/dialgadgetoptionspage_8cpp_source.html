00203 void DialGadgetOptionsPage::on_uavObject1_currentIndexChanged(QString val) {
00204     options_page->objectField1->clear();
00205     ExtensionSystem::PluginManager *pm = ExtensionSystem::PluginManager::instance();
00206     UAVObjectManager *objManager = pm->getObject<UAVObjectManager>();
00207     UAVDataObject* obj = dynamic_cast<UAVDataObject*>( objManager->getObject(val) );
00208     QList<UAVObjectField*> fieldList = obj->getFields();
00209     foreach (UAVObjectField* field, fieldList) {
00210         if(field->getType() == UAVObjectField::STRING || field->getType() == UAVObjectField::ENUM )
00211             continue;
00212         if(field->getElementNames().count() > 1)
00213         {
00214             foreach(QString elemName , field->getElementNames())
00215             {
00216                 options_page->objectField1->addItem(field->getName() + "-" + elemName);
00217             }
00218         }
00219         else
00220             options_page->objectField1->addItem(field->getName());
00221     }
00222 }
