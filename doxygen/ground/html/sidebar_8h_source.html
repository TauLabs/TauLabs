00057  * optionally a list of tool buttons that are added to the toolbar above it.
00058  * The window title of the widget is displayed in the combo box.

00060  * The SideBarItem takes ownership over the widget.

00067     {}

00073     /* Should always return a new set of tool buttons.
00074      *
00075      * Workaround since there doesn't seem to be a nice way to remove widgets
00076      * that have been added to a QToolBar without either not deleting the
00077      * associated QAction or causing the QToolButton to be deleted.
00078      */
00079     virtual QList<QToolButton *> createToolBarWidgets()
00080     {
00081         return QList<QToolButton *>();
