293 // Display the instance
294 void GLC_3DViewInstance::render(glc::RenderFlag renderFlag, bool useLod, GLC_Viewport* pView)
295 {
296  //qDebug() << "GLC_3DViewInstance::render render properties= " << m_RenderProperties.renderingMode();
297  if (m_3DRep.isEmpty()) return;
298  const int bodyCount= m_3DRep.numberOfBody();
299 
300  if (bodyCount != m_ViewableGeomFlag.size())
301  {
302  m_ViewableGeomFlag.fill(true, bodyCount);
303  }
304 
305  m_RenderProperties.setRenderingFlag(renderFlag);
306 
307  // Save current OpenGL Matrix
308  GLC_Context::current()->glcPushMatrix();
309  OpenglVisProperties();
310 
311  // Change front face orientation if this instance absolute matrix is indirect
312  if (m_AbsoluteMatrix.type() == GLC_Matrix4x4::Indirect)
313  {
314  glFrontFace(GL_CW);
315  }
316  if(GLC_State::isInSelectionMode())
317  {
318  glColor3ubv(m_colorId); // D'ont use Alpha component
319  }
320 
321  if (useLod && (NULL != pView))
322  {
323  for (int i= 0; i < bodyCount; ++i)
324  {
325  if (m_ViewableGeomFlag.at(i))
326  {
327  const int lodValue= choseLod(m_3DRep.geomAt(i)->boundingBox(), pView, useLod);
328  if (lodValue <= 100)
329  {
330  m_3DRep.geomAt(i)->setCurrentLod(lodValue);
331  m_RenderProperties.setCurrentBodyIndex(i);
332  m_3DRep.geomAt(i)->render(m_RenderProperties);
333  }
334  }
335  }
336  }
337  else
338  {
339  for (int i= 0; i < bodyCount; ++i)
340  {
341  if (m_ViewableGeomFlag.at(i))
342  {
343  int lodValue= 0;
344  if (GLC_State::isPixelCullingActivated() && (NULL != pView))
345  {
346  lodValue= choseLod(m_3DRep.geomAt(i)->boundingBox(), pView, useLod);
347  }
348 
349  if (lodValue <= 100)
350  {
351  m_3DRep.geomAt(i)->setCurrentLod(m_DefaultLOD);
352  m_RenderProperties.setCurrentBodyIndex(i);
353  m_3DRep.geomAt(i)->render(m_RenderProperties);
354  }
355  }
356  }
357  }
358  // Restore OpenGL Matrix
359  GLC_Context::current()->glcPopMatrix();
360 
361  // Restore front face orientation if this instance absolute matrix is indirect
362  if (m_AbsoluteMatrix.type() == GLC_Matrix4x4::Indirect)
363  {
364  glFrontFace(GL_CCW);
365  }
366 
367 }

369 // Display the instance in Body selection mode
370 void GLC_3DViewInstance::renderForBodySelection()
371 {
372  Q_ASSERT(GLC_State::isInSelectionMode());
373  if (m_3DRep.isEmpty()) return;
374 
375  // Save previous rendering mode and set the rendering mode to BodySelection
376  glc::RenderMode previousRenderMode= m_RenderProperties.renderingMode();
377  m_RenderProperties.setRenderingMode(glc::BodySelection);
378 
379  // Save current OpenGL Matrix
380  GLC_Context::current()->glcPushMatrix();
381  OpenglVisProperties();
382 
383  GLubyte colorId[4];
384  const int size= m_3DRep.numberOfBody();
385  for (int i= 0; i < size; ++i)
386  {
387  GLC_Geometry* pGeom= m_3DRep.geomAt(i);
388  glc::encodeRgbId(pGeom->id(), colorId);
389  glColor3ubv(colorId);
390  pGeom->setCurrentLod(m_DefaultLOD);
391  m_RenderProperties.setCurrentBodyIndex(i);
392  pGeom->render(m_RenderProperties);
393  }
394 
395  // Restore rendering mode
396  m_RenderProperties.setRenderingMode(previousRenderMode);
397  // Restore OpenGL Matrix
398  GLC_Context::current()->glcPopMatrix();
399 }

401 // Display the instance in Primitive selection mode and return the body index
402 int GLC_3DViewInstance::renderForPrimitiveSelection(GLC_uint bodyId)
403 {
404  Q_ASSERT(GLC_State::isInSelectionMode());
405  if (m_3DRep.isEmpty()) return -1;
406  // Save previous rendering mode and set the rendering mode to BodySelection
407  glc::RenderMode previousRenderMode= m_RenderProperties.renderingMode();
408  m_RenderProperties.setRenderingMode(glc::PrimitiveSelection);
409 
410  // Save current OpenGL Matrix
411  GLC_Context::current()->glcPushMatrix();
412  OpenglVisProperties();
413 
414  const int size= m_3DRep.numberOfBody();
415  int i= 0;
416  bool continu= true;
417  while ((i < size) && continu)
418  {
419  GLC_Geometry* pGeom= m_3DRep.geomAt(i);
420  if (pGeom->id() == bodyId)
421  {
422  pGeom->setCurrentLod(0);
423  pGeom->render(m_RenderProperties);
424  continu= false;
425  }
426  else ++i;
427  }
428 
429  m_RenderProperties.setRenderingMode(previousRenderMode);
430 
431  // Restore OpenGL Matrix
432  GLC_Context::current()->glcPopMatrix();
433 
434  return i;
435 }

446 void GLC_3DViewInstance::computeBoundingBox(void)
447 {

462  m_pBoundingBox->transform(m_AbsoluteMatrix);
463 }

477 // Compute LOD
478 int GLC_3DViewInstance::choseLod(const GLC_BoundingBox& boundingBox, GLC_Viewport* pView, bool useLod)
479 {
480  if (NULL == pView) return 0;
481  double pixelCullingRatio= 0.0;
482  if (useLod)
483  {
484  pixelCullingRatio= pView->minimumDynamicPixelCullingRatio();
485  }
486  else
487  {
488  pixelCullingRatio= pView->minimumStaticPixelCullingRatio();
489  }
490 
491  const double diameter= boundingBox.boundingSphereRadius() * 2.0 * m_AbsoluteMatrix.scalingX();
492  GLC_Vector3d center(m_AbsoluteMatrix * boundingBox.center());
493 
494  const double dist= (center - pView->cameraHandle()->eye()).length();
495  const double cameraCover= dist * pView->viewTangent();
496 
497  double ratio= diameter / cameraCover * 100.0;
498  if (ratio > 100.0) ratio= 100.0;
499  ratio= 100.0 - ratio;
500 
501  if ((ratio > (100.0 - pixelCullingRatio)) && GLC_State::isPixelCullingActivated()) ratio= 110.0;
502  else if(useLod && (ratio > 50.0))
503  {
504  ratio= (ratio - 50.0) / 50.0 * 100.0;
505  if (ratio < static_cast<double>(m_DefaultLOD)) ratio= static_cast<double>(m_DefaultLOD);
506  }
507  else
508  {
509  ratio= static_cast<double>(m_DefaultLOD);
510  }
511 
512  return static_cast<int>(ratio);
513 }
