83 void Calibration::connectSensor(sensor_type sensor, bool con)

86  switch (sensor) {
87  case ACCEL:
88  {
89  Accels * accels = Accels::GetInstance(getObjectManager());
90  Q_ASSERT(accels);
91 
92  assignUpdateRate(accels, SENSOR_UPDATE_PERIOD);
93  connect(accels, SIGNAL(objectUpdated(UAVObject*)), this, SLOT(dataUpdated(UAVObject *)));
94  }
95  break;
96 
97  case MAG:
98  {
99  Magnetometer * mag = Magnetometer::GetInstance(getObjectManager());
100  Q_ASSERT(mag);
101 
102  assignUpdateRate(mag, SENSOR_UPDATE_PERIOD);
103  connect(mag, SIGNAL(objectUpdated(UAVObject*)), this, SLOT(dataUpdated(UAVObject *)));
104  }
105  break;
106 
107  case GYRO:
108  {
109  Gyros * gyros = Gyros::GetInstance(getObjectManager());
110  Q_ASSERT(gyros);
111 
112  assignUpdateRate(gyros, SENSOR_UPDATE_PERIOD);
113  connect(gyros, SIGNAL(objectUpdated(UAVObject*)), this, SLOT(dataUpdated(UAVObject *)));
114  }
115  break;
116 
117  }
118  } else {
119  switch (sensor) {
120  case ACCEL:
121  {
122  Accels * accels = Accels::GetInstance(getObjectManager());

173 void Calibration::dataUpdated(UAVObject * obj) {
174 
175  if (!timer.isActive()) {
176  // ignore updates that come in after the timer has expired
177  return;
178  }
179 
180  switch(calibration_state) {
181  case IDLE:
182  case SIX_POINT_WAIT1:
183  case SIX_POINT_WAIT2:
184  case SIX_POINT_WAIT3:
185  case SIX_POINT_WAIT4:
186  case SIX_POINT_WAIT5:
187  case SIX_POINT_WAIT6:
188  // Do nothing
189  return;
190  break;
191  case LEVELING:
192  // Store data while computing the level attitude
193  // and if completed go back to the idle state
194  if(storeLevelingMeasurement(obj)) {
195  //Disconnect sensors and reset metadata
196  connectSensor(GYRO, false);
197  connectSensor(ACCEL, false);
198  getObjectUtilManager()->setAllNonSettingsMetadata(originalMetaData);
199 
200  calibration_state = IDLE;
201 
202  emit showLevelingMessage(tr("Level computed"));
203  emit toggleControls(true);
204  emit levelingProgressChanged(0);
205  disconnect(&timer,SIGNAL(timeout()),this,SLOT(timeout()));
206  }
207  break;
208  case SIX_POINT_COLLECT1:
209  // These state collect each position for six point calibration and
210  // when enough data is acquired advance to the next step
211  if(storeSixPointMeasurement(obj,1)) {
212  // If all the data is collected advance to the next position
213  calibration_state = SIX_POINT_WAIT2;
214  emit showSixPointMessage(tr("Rotate left side down and press Save Position..."));
215  emit toggleSavePosition(true);
216  emit updatePlane(2);
217  disconnect(&timer,SIGNAL(timeout()),this,SLOT(timeout()));
218  }
219  break;
220  case SIX_POINT_COLLECT2:
221  if(storeSixPointMeasurement(obj,2)) {
222  // If all the data is collected advance to the next position
223  calibration_state = SIX_POINT_WAIT3;
224  emit showSixPointMessage(tr("Rotate upside down and press Save Position..."));
225  emit toggleSavePosition(true);
226  emit updatePlane(3);
227  disconnect(&timer,SIGNAL(timeout()),this,SLOT(timeout()));
228  }
229  break;
230  case SIX_POINT_COLLECT3:
231  if(storeSixPointMeasurement(obj,3)) {
232  // If all the data is collected advance to the next position
233  calibration_state = SIX_POINT_WAIT4;
234  emit showSixPointMessage(tr("Point right side down and press Save Position..."));
235  emit toggleSavePosition(true);
236  emit updatePlane(4);
237  disconnect(&timer,SIGNAL(timeout()),this,SLOT(timeout()));
238  }
239  break;
240  case SIX_POINT_COLLECT4:
241  if(storeSixPointMeasurement(obj,4)) {
242  // If all the data is collected advance to the next position
243  calibration_state = SIX_POINT_WAIT5;
244  emit showSixPointMessage(tr("Point nose up and press Save Position..."));
245  emit toggleSavePosition(true);
246  emit updatePlane(5);
247  disconnect(&timer,SIGNAL(timeout()),this,SLOT(timeout()));
248  }
249  break;
250  case SIX_POINT_COLLECT5:
251  if(storeSixPointMeasurement(obj,5)) {
252  // If all the data is collected advance to the next position
253  calibration_state = SIX_POINT_WAIT6;
254  emit showSixPointMessage(tr("Point nose down and press Save Position..."));
255  emit toggleSavePosition(true);
256  emit updatePlane(6);
257  disconnect(&timer,SIGNAL(timeout()),this,SLOT(timeout()));
258  }
259  break;
260  case SIX_POINT_COLLECT6:
261  // Store data points in the final position and if enough data
262  // has been computed attempt to calculate the scale and bias
263  // for the accel and optionally the mag.
264  if(storeSixPointMeasurement(obj,6)) {
265  // All data collected. Disconnect and reset all UAVOs, and compute value
266  connectSensor(ACCEL, false);
267  connectSensor(GYRO, false);
268  if (calibrateMag)
269  connectSensor(MAG, false);
270  getObjectUtilManager()->setAllNonSettingsMetadata(originalMetaData);
271 
272  calibration_state = IDLE;
273  emit toggleControls(true);
274  emit updatePlane(0);
275  emit sixPointProgressChanged(0);
276  disconnect(&timer,SIGNAL(timeout()),this,SLOT(timeout()));
277 
278  // Do calculation
279  int ret=computeScaleBias();
280  if (ret==CALIBRATION_SUCCESS)
281  emit showSixPointMessage(tr("Calibration succeeded"));
282  else{
283  //Return sensor calibration values to their original settings
284  resetSensorCalibrationToOriginalValues();
285 
286  if(ret==ACCELEROMETER_FAILED){
287  emit showSixPointMessage(tr("Acceleromter calibration failed. Original values have been written back to device. Perhaps you moved too much during the calculation? Please repeat calibration."));
288  }
289  else if(ret==MAGNETOMETER_FAILED){
290  emit showSixPointMessage(tr("Magnetometer calibration failed. Original values have been written back to device. Perhaps you performed the calibration near iron? Please repeat calibration."));
291  }
292  }
293  }
294  break;
295  case GYRO_TEMP_CAL:
296  if (storeTempCalMeasurement(obj)) {
297  // Disconnect and reset metadata
298  connectSensor(GYRO, false);
299  getObjectUtilManager()->setAllNonSettingsMetadata(originalMetaData);
300 
301  calibration_state = IDLE;
302  emit toggleControls(true);
303 
304  int ret = computeTempCal();
305  if (ret == CALIBRATION_SUCCESS) {
306  emit showTempCalMessage(tr("Temperature compensation calibration succeeded"));
307  } else {
308  emit showTempCalMessage(tr("Temperature compensation calibration succeeded"));
309  }
310  }
311  }

319 void Calibration::timeout()
320 {
321  // Reset metadata update rates
322  getObjectUtilManager()->setAllNonSettingsMetadata(originalMetaData);
323 
324  switch(calibration_state) {
325  case IDLE:
326  // Do nothing
327  return;
328  break;
329  case LEVELING:
330  // Disconnect appropriate sensors
331  connectSensor(GYRO, false);
332  connectSensor(ACCEL, false);
333  calibration_state = IDLE;
334  emit showLevelingMessage(tr("Leveling timed out ..."));
335  emit levelingProgressChanged(0);
336  case SIX_POINT_WAIT1:
337  case SIX_POINT_WAIT2:
338  case SIX_POINT_WAIT3:
339  case SIX_POINT_WAIT4:
340  case SIX_POINT_WAIT5:
341  case SIX_POINT_WAIT6:
342  // Do nothing, shouldn't happen
343  return;
344  break;
345  case SIX_POINT_COLLECT1:
346  case SIX_POINT_COLLECT2:
347  case SIX_POINT_COLLECT3:
348  case SIX_POINT_COLLECT4:
349  case SIX_POINT_COLLECT5:
350  case SIX_POINT_COLLECT6:
351  connectSensor(ACCEL, false);
352  connectSensor(GYRO, false);
353  if (calibrateMag)
354  connectSensor(MAG, false);
355  calibration_state = IDLE;
356  emit showSixPointMessage(tr("Six point data collection timed out"));
357  emit sixPointProgressChanged(0);
358  break;
359  case GYRO_TEMP_CAL:
360  connectSensor(GYRO, false);
361  calibration_state = IDLE;
362  emit showTempCalMessage(tr("Temperature calibration timed out"));
363  emit tempCalProgressChanged(0);
364  break;
365  }
366 
367  emit updatePlane(0);
368  emit toggleControls(true);
369 
370  disconnect(&timer,SIGNAL(timeout()),this,SLOT(timeout()));
371 
372  QMessageBox msgBox;
373  msgBox.setText(tr("Calibration timed out before receiving required updates."));
374  msgBox.setStandardButtons(QMessageBox::Ok);
375  msgBox.setDefaultButton(QMessageBox::Ok);
376  msgBox.exec();
377 }

398

403

441 void Calibration::doStartSixPoint()
442 {

459  // Save initial accelerometer settings
460  SensorSettings * sensorSettings = SensorSettings::GetInstance(getObjectManager());
461  Q_ASSERT(sensorSettings);
462  SensorSettings::DataFields sensorSettingsData = sensorSettings->getData();
463 
464  initialAccelsScale[0]=sensorSettingsData.AccelScale[SensorSettings::ACCELSCALE_X];
465  initialAccelsScale[1]=sensorSettingsData.AccelScale[SensorSettings::ACCELSCALE_Y];
466  initialAccelsScale[2]=sensorSettingsData.AccelScale[SensorSettings::ACCELSCALE_Z];

478 
479  // If calibrating the mag, remove any scaling

481  initialMagsScale[0]=sensorSettingsData.MagScale[SensorSettings::MAGSCALE_X];
482  initialMagsScale[1]=sensorSettingsData.MagScale[SensorSettings::MAGSCALE_Y];
483  initialMagsScale[2]=sensorSettingsData.MagScale[SensorSettings::MAGSCALE_Z];
484  initialMagsBias[0]= sensorSettingsData.MagBias[SensorSettings::MAGBIAS_X];

490  sensorSettingsData.MagScale[SensorSettings::MAGSCALE_Y] = 1;
491  sensorSettingsData.MagScale[SensorSettings::MAGSCALE_Z] = 1;
492  sensorSettingsData.MagBias[SensorSettings::MAGBIAS_X] = 0;
493  sensorSettingsData.MagBias[SensorSettings::MAGBIAS_Y] = 0;
494  sensorSettingsData.MagBias[SensorSettings::MAGBIAS_Z] = 0;
495  }
496 
497  sensorSettings->setData(sensorSettingsData);
498 
499  // Clear the accumulators
500  accel_accum_x.clear();
501  accel_accum_y.clear();
502  accel_accum_z.clear();
503  mag_accum_x.clear();
504  mag_accum_y.clear();
505  mag_accum_z.clear();
506 
507  // TODO: Document why the thread needs to wait 100ms.
508  Thread::usleep(100000);
509 
510  // Save previous sensor states
511  originalMetaData = getObjectUtilManager()->readAllNonSettingsMetadata();
512 
513  // Make all UAVObject rates update slowly
514  UAVObjectManager *objManager = getObjectManager();
515  QVector< QVector<UAVDataObject*> > objList = objManager->getDataObjects();

517  foreach (UAVDataObject* obj, list) {
518  if(!obj->isSettings()) {

524  }
525  }
526  }
527

529  connectSensor(ACCEL, true);
530  if(calibrateMag) {
531  connectSensor(MAG, true);

542  calibration_state = SIX_POINT_WAIT1;

727 bool Calibration::storeLevelingMeasurement(UAVObject *obj) {
728  Accels * accels = Accels::GetInstance(getObjectManager());
729  Gyros * gyros = Gyros::GetInstance(getObjectManager());
730 
731  // Accumulate samples until we have _at least_ NUM_SENSOR_UPDATES_LEVELING samples
732  if(obj->getObjID() == Accels::OBJID) {
733  Accels::DataFields accelsData = accels->getData();
734  accel_accum_x.append(accelsData.x);
735  accel_accum_y.append(accelsData.y);
736  accel_accum_z.append(accelsData.z);
737  } else if (obj->getObjID() == Gyros::OBJID) {
738  Gyros::DataFields gyrosData = gyros->getData();
739  gyro_accum_x.append(gyrosData.x);
740  gyro_accum_y.append(gyrosData.y);
741  gyro_accum_z.append(gyrosData.z);
742  gyro_accum_temp.append(gyrosData.temperature);
743  }
744 
745  // update the progress indicator
746  emit levelingProgressChanged((float) qMin(accel_accum_x.size(), gyro_accum_x.size()) / NUM_SENSOR_UPDATES_LEVELING * 100);
747 
748  // If we have enough samples, then stop sampling and compute the biases
749  if (accel_accum_x.size() >= NUM_SENSOR_UPDATES_LEVELING && gyro_accum_x.size() >= NUM_SENSOR_UPDATES_LEVELING) {
750  timer.stop();
751  disconnect(&timer,SIGNAL(timeout()),this,SLOT(timeout()));
752 
753  float x_gyro_bias = listMean(gyro_accum_x);
754  float y_gyro_bias = listMean(gyro_accum_y);
755  float z_gyro_bias = listMean(gyro_accum_z);
756  float temp = listMean(gyro_accum_temp);
757 
758  // Get the existing attitude settings
759  AttitudeSettings::DataFields attitudeSettingsData = AttitudeSettings::GetInstance(getObjectManager())->getData();
760  SensorSettings::DataFields sensorSettingsData = SensorSettings::GetInstance(getObjectManager())->getData();
761 
762  // Inverse rotation of sensor data, from body frame into sensor frame
763  double a_body[3] = { listMean(accel_accum_x), listMean(accel_accum_y), listMean(accel_accum_z) };
764  double a_sensor[3];
765  double Rsb[3][3]; // The initial board rotation
766  double rpy[3] = { attitudeSettingsData.BoardRotation[AttitudeSettings::BOARDROTATION_ROLL] * DEG2RAD / 100.0,
767  attitudeSettingsData.BoardRotation[AttitudeSettings::BOARDROTATION_PITCH] * DEG2RAD / 100.0,
768  attitudeSettingsData.BoardRotation[AttitudeSettings::BOARDROTATION_YAW] * DEG2RAD / 100.0};
769  Euler2R(rpy, Rsb);
770  rotate_vector(Rsb, a_body, a_sensor, true);
771 
772  // Temporary variables
773  double psi, theta, phi;
774 
775  // Keep existing yaw rotation
776  psi = attitudeSettingsData.BoardRotation[AttitudeSettings::BOARDROTATION_YAW] * DEG2RAD / 100.0;
777 
778  double cP = cos(psi);
779  double sP = sin(psi);
780 
781 
782  // the inversion of the rotation matrix multiplied by the basis vector
783  theta = atan2(-(cP * a_sensor[0] + sP * a_sensor[1]), -a_sensor[2]);
784  phi = atan2(-(sP * a_sensor[0] - cP * a_sensor[1]), -a_sensor[2] / cos(theta));
785 
786  attitudeSettingsData.BoardRotation[AttitudeSettings::BOARDROTATION_ROLL] = phi * RAD2DEG * 100.0;
787  attitudeSettingsData.BoardRotation[AttitudeSettings::BOARDROTATION_PITCH] = theta * RAD2DEG * 100.0;
788 
789  // Rotate the gyro bias from the old body frame into the sensor frame
790  // and then into the new body frame
791  double gyro_sensor[3];
792  double gyro_newbody[3];
793  double gyro_oldbody[3] = {x_gyro_bias, y_gyro_bias, z_gyro_bias};
794  double new_rpy[3] = { attitudeSettingsData.BoardRotation[AttitudeSettings::BOARDROTATION_ROLL] * DEG2RAD / 100.0,
795  attitudeSettingsData.BoardRotation[AttitudeSettings::BOARDROTATION_PITCH] * DEG2RAD / 100.0,
796  attitudeSettingsData.BoardRotation[AttitudeSettings::BOARDROTATION_YAW] * DEG2RAD / 100.0};
797  rotate_vector(Rsb, gyro_oldbody, gyro_sensor, true);
798  Euler2R(new_rpy, Rsb);
799  rotate_vector(Rsb, gyro_sensor, gyro_newbody, false);
800 
801  // Store these new biases, accounting for any temperature coefficients
802  sensorSettingsData.XGyroTempCoeff[0] = gyro_newbody[0] -
803  temp * sensorSettingsData.XGyroTempCoeff[1] -
804  pow(temp,2) * sensorSettingsData.XGyroTempCoeff[2] -
805  pow(temp,3) * sensorSettingsData.XGyroTempCoeff[3];
806  sensorSettingsData.YGyroTempCoeff[0] = gyro_newbody[1] -
807  temp * sensorSettingsData.YGyroTempCoeff[1] -
808  pow(temp,2) * sensorSettingsData.YGyroTempCoeff[2] -
809  pow(temp,3) * sensorSettingsData.YGyroTempCoeff[3];
810  sensorSettingsData.ZGyroTempCoeff[0] = gyro_newbody[2] -
811  temp * sensorSettingsData.ZGyroTempCoeff[1] -
812  pow(temp,2) * sensorSettingsData.ZGyroTempCoeff[2] -
813  pow(temp,3) * sensorSettingsData.ZGyroTempCoeff[3];
814  SensorSettings::GetInstance(getObjectManager())->setData(sensorSettingsData);
815 
816  // We offset the gyro bias by current bias to help precision
817  // Disable gyro bias correction to see raw data
818  AttitudeSettings *attitudeSettings = AttitudeSettings::GetInstance(getObjectManager());
819  Q_ASSERT(attitudeSettings);
820  attitudeSettingsData.BiasCorrectGyro = AttitudeSettings::BIASCORRECTGYRO_TRUE;
821  attitudeSettings->setData(attitudeSettingsData);
822  attitudeSettings->updated();
823 
824  // Inform the system that the calibration process has completed
825  emit calibrationCompleted();
826 
827  return true;
828  }
829  return false;
830 }

941 void Calibration::updateTempCompCalibrationDisplay()
942 {
943  unsigned int n_samples = gyro_accum_temp.size();
944 
945  // Construct the matrix of temperature.
946  Eigen::Matrix<double, Eigen::Dynamic, 4> X(n_samples, 4);
947 
948  // And the matrix of gyro samples.
949  Eigen::Matrix<double, Eigen::Dynamic, 3> Y(n_samples, 3);
950 
951  for (unsigned i = 0; i < n_samples; ++i) {
952  X(i,0) = 1;
953  X(i,1) = gyro_accum_temp[i];
954  X(i,2) = pow(gyro_accum_temp[i],2);
955  X(i,3) = pow(gyro_accum_temp[i],3);
956  Y(i,0) = gyro_accum_x[i];
957  Y(i,1) = gyro_accum_y[i];
958  Y(i,2) = gyro_accum_z[i];
959  }
960 
961  // Solve Y = X * B
962 
963  Eigen::Matrix<double, 4, 3> result;
964  // Use the cholesky-based Penrose pseudoinverse method.
965  (X.transpose() * X).ldlt().solve(X.transpose()*Y, &result);
966 
967  QList<double> xCoeffs, yCoeffs, zCoeffs;
968  xCoeffs.clear();
969  xCoeffs.append(result(0,0));
970  xCoeffs.append(result(1,0));
971  xCoeffs.append(result(2,0));
972  xCoeffs.append(result(3,0));
973  yCoeffs.clear();
974  yCoeffs.append(result(0,1));
975  yCoeffs.append(result(1,1));
976  yCoeffs.append(result(2,1));
977  yCoeffs.append(result(3,1));
978  zCoeffs.clear();
979  zCoeffs.append(result(0,2));
980  zCoeffs.append(result(1,2));
981  zCoeffs.append(result(2,2));

983 
984  if (xCurve != NULL)

1157 int Calibration::computeScaleBias()
1158 {
1159  // Regardless of calibration result, set board rotations back to user settings
1160  AttitudeSettings * attitudeSettings = AttitudeSettings::GetInstance(getObjectManager());
1161  Q_ASSERT(attitudeSettings);
1162  AttitudeSettings::DataFields attitudeSettingsData = attitudeSettings->getData();
1163  attitudeSettingsData.BoardRotation[AttitudeSettings::BOARDROTATION_ROLL] = initialBoardRotation[0];
1164  attitudeSettingsData.BoardRotation[AttitudeSettings::BOARDROTATION_PITCH] = initialBoardRotation[1];
1165  attitudeSettingsData.BoardRotation[AttitudeSettings::BOARDROTATION_YAW] = initialBoardRotation[2];
1166  attitudeSettings->setData(attitudeSettingsData);
1167 
1168  bool good_calibration = true;
1169 
1170  qDebug() << "Accel measurements";
1171  for(int i = 0; i < 6; i++)
1172  qDebug() << accel_data_x[i] << ", " << accel_data_y[i] << ", " << accel_data_z[i] << ";";
1173  qDebug() << "Mag measurements";
1174  for(int i = 0; i < 6; i++)
1175  qDebug() << mag_data_x[i] << ", " << mag_data_y[i] << ", " << mag_data_z[i] << ";";
1176 
1177  // Calibrate accelerometer
1178  double S[3], b[3];
1179  SixPointInConstFieldCal(accelLength, accel_data_x, accel_data_y, accel_data_z, S, b);
1180 
1181  SensorSettings * sensorSettings = SensorSettings::GetInstance(getObjectManager());
1182  Q_ASSERT(sensorSettings);
1183  SensorSettings::DataFields sensorSettingsData = sensorSettings->getData();
1184 
1185  //Assign calibration data
1186  sensorSettingsData.AccelBias[SensorSettings::ACCELBIAS_X] += (-sign(S[0]) * b[0]);
1187  sensorSettingsData.AccelBias[SensorSettings::ACCELBIAS_Y] += (-sign(S[1]) * b[1]);
1188  sensorSettingsData.AccelBias[SensorSettings::ACCELBIAS_Z] += (-sign(S[2]) * b[2]);
1189 
1190  sensorSettingsData.AccelScale[SensorSettings::ACCELSCALE_X] *= fabs(S[0]);
1191  sensorSettingsData.AccelScale[SensorSettings::ACCELSCALE_Y] *= fabs(S[1]);
1192  sensorSettingsData.AccelScale[SensorSettings::ACCELSCALE_Z] *= fabs(S[2]);
1193 
1194  // Check the accel calibration is good
1195  good_calibration &= sensorSettingsData.AccelScale[SensorSettings::ACCELSCALE_X] ==
1196  sensorSettingsData.AccelScale[SensorSettings::ACCELSCALE_X];
1197  good_calibration &= sensorSettingsData.AccelScale[SensorSettings::ACCELSCALE_Y] ==
1198  sensorSettingsData.AccelScale[SensorSettings::ACCELSCALE_Y];
1199  good_calibration &= sensorSettingsData.AccelScale[SensorSettings::ACCELSCALE_Z] ==
1200  sensorSettingsData.AccelScale[SensorSettings::ACCELSCALE_Z];
1201  good_calibration &= (sensorSettingsData.AccelBias[SensorSettings::ACCELBIAS_X] ==
1202  sensorSettingsData.AccelBias[SensorSettings::ACCELBIAS_X]);
1203  good_calibration &= (sensorSettingsData.AccelBias[SensorSettings::ACCELBIAS_Y] ==
1204  sensorSettingsData.AccelBias[SensorSettings::ACCELBIAS_Y]);
1205  good_calibration &= (sensorSettingsData.AccelBias[SensorSettings::ACCELBIAS_Z] ==
1206  sensorSettingsData.AccelBias[SensorSettings::ACCELBIAS_Z]);
1207 
1208  //This can happen if, for instance, HomeLocation.g_e == 0
1209  if((S[0]+S[1]+S[2])<0.0001){
1210  good_calibration=false;
1211  }
1212 
1213  if (calibrateMag) {
1214  good_calibration = true;
1215 
1216  // Work out the average vector length as nominal since mag scale normally close to 1
1217  // and we don't use the magnitude anyway. Avoids requiring home location.
1218  double m_x = 0, m_y = 0, m_z = 0, len = 0;
1219  for (int i = 0; i < 6; i++) {
1220  m_x += mag_data_x[i];
1221  m_y += mag_data_x[i];
1222  m_z += mag_data_x[i];
1223  }
1224  m_x /= 6;
1225  m_y /= 6;
1226  m_z /= 6;
1227  for (int i = 0; i < 6; i++) {
1228  len += sqrt(pow(mag_data_x[i] - m_x,2) + pow(mag_data_y[i] - m_y,2) + pow(mag_data_z[i] - m_z,2));
1229  }
1230  len /= 6;
1231 
1232  // Calibrate magnetomter
1233  double S[3], b[3];
1234  SixPointInConstFieldCal(len, mag_data_x, mag_data_y, mag_data_z, S, b);
1235 
1236  //Assign calibration data
1237  sensorSettingsData.MagBias[SensorSettings::MAGBIAS_X] += (-sign(S[0]) * b[0]);
1238  sensorSettingsData.MagBias[SensorSettings::MAGBIAS_Y] += (-sign(S[1]) * b[1]);
1239  sensorSettingsData.MagBias[SensorSettings::MAGBIAS_Z] += (-sign(S[2]) * b[2]);
1240  sensorSettingsData.MagScale[SensorSettings::MAGSCALE_X] *= fabs(S[0]);
1241  sensorSettingsData.MagScale[SensorSettings::MAGSCALE_Y] *= fabs(S[1]);
1242  sensorSettingsData.MagScale[SensorSettings::MAGSCALE_Z] *= fabs(S[2]);
1243 
1244  // Check the mag calibration is good
1245  good_calibration &= sensorSettingsData.MagBias[SensorSettings::MAGBIAS_X] ==
1246  sensorSettingsData.MagBias[SensorSettings::MAGBIAS_X];
1247  good_calibration &= sensorSettingsData.MagBias[SensorSettings::MAGBIAS_Y] ==
1248  sensorSettingsData.MagBias[SensorSettings::MAGBIAS_Y];
1249  good_calibration &= sensorSettingsData.MagBias[SensorSettings::MAGBIAS_Z] ==
1250  sensorSettingsData.MagBias[SensorSettings::MAGBIAS_Z] ;
1251  good_calibration &= sensorSettingsData.MagScale[SensorSettings::MAGSCALE_X] ==
1252  sensorSettingsData.MagScale[SensorSettings::MAGSCALE_X];

1390  double fAcc;
1391 
1392  int i , j, k, m;
1393 
1394  for(k=0; k<(nDim-1); k++) // base row of matrix
1395  {
1396  // search of line with max element
1397  fMaxElem = fabs( pfMatr[k*nDim + k] );
1398  m = k;
1399  for(i=k+1; i<nDim; i++)
1400  {
1401  if(fMaxElem < fabs(pfMatr[i*nDim + k]) )
1402  {
1403  fMaxElem = pfMatr[i*nDim + k];
1404  m = i;
1405  }
1406  }
1407 
1408  // permutation of base line (index k) and max element line(index m)
1409  if(m != k)
1410  {
1411  for(i=k; i<nDim; i++)
1412  {
1413  fAcc = pfMatr[k*nDim + i];
1414  pfMatr[k*nDim + i] = pfMatr[m*nDim + i];
1415  pfMatr[m*nDim + i] = fAcc;
1416  }
1417  fAcc = pfVect[k];
1418  pfVect[k] = pfVect[m];
1419  pfVect[m] = fAcc;
1420  }
1421 
1422  if( pfMatr[k*nDim + k] == 0.) return 0; // needs improvement !!!
1423 
1424  // triangulation of matrix with coefficients
1425  for(j=(k+1); j<nDim; j++) // current row of matrix
1426  {
1427  fAcc = - pfMatr[j*nDim + k] / pfMatr[k*nDim + k];
1428  for(i=k; i<nDim; i++)
1429  {
1430  pfMatr[j*nDim + i] = pfMatr[j*nDim + i] + fAcc*pfMatr[k*nDim + i];
1431  }
1432  pfVect[j] = pfVect[j] + fAcc*pfVect[k]; // free member recalculation
1433  }
1434  }
1435 
1436  for(k=(nDim-1); k>=0; k--)
1437  {
1438  pfSolution[k] = pfVect[k];
1439  for(i=(k+1); i<nDim; i++)
1440  {
1441  pfSolution[k] -= (pfMatr[k*nDim + i]*pfSolution[i]);
1442  }
1443  pfSolution[k] = pfSolution[k] / pfMatr[k*nDim + k];
1444  }

1469 int Calibration::SixPointInConstFieldCal( double ConstMag, double x[6], double y[6], double z[6], double S[3], double b[3] )
1470 {
1471  int i;
1472  double A[5][5];
1473  double f[5], c[5];
1474  double xp, yp, zp, Sx;
1475 
1476  // Fill in matrix A -
1477  // write six difference-in-magnitude equations of the form
1478  // Sx^2(x2^2-x1^2) + 2*Sx*bx*(x2-x1) + Sy^2(y2^2-y1^2) + 2*Sy*by*(y2-y1) + Sz^2(z2^2-z1^2) + 2*Sz*bz*(z2-z1) = 0
1479  // or in other words
1480  // 2*Sx*bx*(x2-x1)/Sx^2 + Sy^2(y2^2-y1^2)/Sx^2 + 2*Sy*by*(y2-y1)/Sx^2 + Sz^2(z2^2-z1^2)/Sx^2 + 2*Sz*bz*(z2-z1)/Sx^2 = (x1^2-x2^2)
1481  for (i=0;i<5;i++){
1482  A[i][0] = 2.0 * (x[i+1] - x[i]);
1483  A[i][1] = y[i+1]*y[i+1] - y[i]*y[i];
1484  A[i][2] = 2.0 * (y[i+1] - y[i]);
1485  A[i][3] = z[i+1]*z[i+1] - z[i]*z[i];
1486  A[i][4] = 2.0 * (z[i+1] - z[i]);
1487  f[i] = x[i]*x[i] - x[i+1]*x[i+1];
1488  }
1489 
1490  // solve for c0=bx/Sx, c1=Sy^2/Sx^2; c2=Sy*by/Sx^2, c3=Sz^2/Sx^2, c4=Sz*bz/Sx^2
1491  if ( !LinearEquationsSolving( 5, (double *)A, f, c) ) return 0;
1492 
1493  // use one magnitude equation and c's to find Sx - doesn't matter which - all give the same answer
1494  xp = x[0]; yp = y[0]; zp = z[0];
1495  Sx = sqrt(ConstMag*ConstMag / (xp*xp + 2*c[0]*xp + c[0]*c[0] + c[1]*yp*yp + 2*c[2]*yp + c[2]*c[2]/c[1] + c[3]*zp*zp + 2*c[4]*zp + c[4]*c[4]/c[3]));
1496 
1497  S[0] = Sx;
1498  b[0] = Sx*c[0];
1499  S[1] = sqrt(c[1]*Sx*Sx);
1500  b[1] = c[2]*Sx*Sx/S[1];
1501  S[2] = sqrt(c[3]*Sx*Sx);
1502  b[2] = c[4]*Sx*Sx/S[2];
1503 
1504  return 1;
1505 }
