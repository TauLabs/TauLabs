00755 bool Calibration::storeLevelingMeasurement(UAVObject *obj) {
00756     Accels * accels = Accels::GetInstance(getObjectManager());
00757     Gyros * gyros = Gyros::GetInstance(getObjectManager());
00758 
00759     // Accumulate samples until we have _at least_ NUM_SENSOR_UPDATES_LEVELING samples
00760     if(obj->getObjID() == Accels::OBJID) {
00761         Accels::DataFields accelsData = accels->getData();
00762         accel_accum_x.append(accelsData.x);
00763         accel_accum_y.append(accelsData.y);
00764         accel_accum_z.append(accelsData.z);
00765     } else if (obj->getObjID() == Gyros::OBJID) {
00766         Gyros::DataFields gyrosData = gyros->getData();
00767         gyro_accum_x.append(gyrosData.x);
00768         gyro_accum_y.append(gyrosData.y);
00769         gyro_accum_z.append(gyrosData.z);
00770         gyro_accum_temp.append(gyrosData.temperature);
00771     }
00772 
00773     // update the progress indicator
00774     emit levelingProgressChanged((float) qMin(accel_accum_x.size(),  gyro_accum_x.size()) / NUM_SENSOR_UPDATES_LEVELING * 100);
00775 
00776     // If we have enough samples, then stop sampling and compute the biases
00777     if (accel_accum_x.size() >= NUM_SENSOR_UPDATES_LEVELING && gyro_accum_x.size() >= NUM_SENSOR_UPDATES_LEVELING) {
00778         timer.stop();
00779         disconnect(&timer,SIGNAL(timeout()),this,SLOT(timeout()));
00780 
00781         float x_gyro_bias = listMean(gyro_accum_x);
00782         float y_gyro_bias = listMean(gyro_accum_y);
00783         float z_gyro_bias = listMean(gyro_accum_z);
00784         float temp = listMean(gyro_accum_temp);
00785 
00786         // Get the existing attitude settings
00787         AttitudeSettings::DataFields attitudeSettingsData = AttitudeSettings::GetInstance(getObjectManager())->getData();
00788         SensorSettings::DataFields sensorSettingsData = SensorSettings::GetInstance(getObjectManager())->getData();
00789 
00790         // Inverse rotation of sensor data, from body frame into sensor frame
00791         double a_body[3] = { listMean(accel_accum_x), listMean(accel_accum_y), listMean(accel_accum_z) };
00792         double a_sensor[3];
00793         double Rsb[3][3];  // The initial board rotation
00794         double rpy[3] = { attitudeSettingsData.BoardRotation[AttitudeSettings::BOARDROTATION_ROLL] * DEG2RAD / 100.0,
00795                           attitudeSettingsData.BoardRotation[AttitudeSettings::BOARDROTATION_PITCH] * DEG2RAD / 100.0,
00796                           attitudeSettingsData.BoardRotation[AttitudeSettings::BOARDROTATION_YAW] * DEG2RAD / 100.0};
00797         Euler2R(rpy, Rsb);
00798         rotate_vector(Rsb, a_body, a_sensor, true);
00799 
00800         // Temporary variables
00801         double psi, theta, phi;
00802 
00803         // Keep existing yaw rotation
00804         psi = attitudeSettingsData.BoardRotation[AttitudeSettings::BOARDROTATION_YAW] * DEG2RAD / 100.0;
00805 
00806         double cP = cos(psi);
00807         double sP = sin(psi);
00808 
00809 
00810         // the inversion of the rotation matrix multiplied by the basis vector
00811         theta = atan2(-(cP * a_sensor[0] + sP * a_sensor[1]), -a_sensor[2]);
00812         phi = atan2(-(sP * a_sensor[0] - cP * a_sensor[1]), -a_sensor[2] / cos(theta));
00813 
00814         attitudeSettingsData.BoardRotation[AttitudeSettings::BOARDROTATION_ROLL] = phi * RAD2DEG * 100.0;
00815         attitudeSettingsData.BoardRotation[AttitudeSettings::BOARDROTATION_PITCH] = theta * RAD2DEG * 100.0;
00816 
00817         // Rotate the gyro bias from the old body frame into the sensor frame
00818         // and then into the new body frame
00819         double gyro_sensor[3];
00820         double gyro_newbody[3];
00821         double gyro_oldbody[3] = {x_gyro_bias, y_gyro_bias, z_gyro_bias};
00822         double new_rpy[3] = { attitudeSettingsData.BoardRotation[AttitudeSettings::BOARDROTATION_ROLL] * DEG2RAD / 100.0,
00823                               attitudeSettingsData.BoardRotation[AttitudeSettings::BOARDROTATION_PITCH] * DEG2RAD / 100.0,
00824                               attitudeSettingsData.BoardRotation[AttitudeSettings::BOARDROTATION_YAW] * DEG2RAD / 100.0};
00825         rotate_vector(Rsb, gyro_oldbody, gyro_sensor, true);
00826         Euler2R(new_rpy, Rsb);
00827         rotate_vector(Rsb, gyro_sensor, gyro_newbody, false);
00828 
00829         // Store these new biases, accounting for any temperature coefficients
00830         sensorSettingsData.XGyroTempCoeff[0] = gyro_newbody[0] -
00831                 temp * sensorSettingsData.XGyroTempCoeff[1] -
00832                 pow(temp,2) * sensorSettingsData.XGyroTempCoeff[2] -
00833                 pow(temp,3) * sensorSettingsData.XGyroTempCoeff[3];
00834         sensorSettingsData.YGyroTempCoeff[0] = gyro_newbody[1] -
00835                 temp * sensorSettingsData.YGyroTempCoeff[1] -
00836                 pow(temp,2) * sensorSettingsData.YGyroTempCoeff[2] -
00837                 pow(temp,3) * sensorSettingsData.YGyroTempCoeff[3];
00838         sensorSettingsData.ZGyroTempCoeff[0] = gyro_newbody[2] -
00839                 temp * sensorSettingsData.ZGyroTempCoeff[1] -
00840                 pow(temp,2) * sensorSettingsData.ZGyroTempCoeff[2] -
00841                 pow(temp,3) * sensorSettingsData.ZGyroTempCoeff[3];
00842         SensorSettings::GetInstance(getObjectManager())->setData(sensorSettingsData);
00843 
00844         // We offset the gyro bias by current bias to help precision
00845         // Disable gyro bias correction to see raw data
00846         AttitudeSettings *attitudeSettings = AttitudeSettings::GetInstance(getObjectManager());
00847         Q_ASSERT(attitudeSettings);
00848         attitudeSettingsData.BiasCorrectGyro = AttitudeSettings::BIASCORRECTGYRO_TRUE;
00849         attitudeSettings->setData(attitudeSettingsData);
00850         attitudeSettings->updated();
00851 
00852         // Inform the system that the calibration process has completed
00853         emit calibrationCompleted();
00854 
00855         return true;
00856     }
00857     return false;
00858 }

01515 {
01516     int i;
01517     double A[5][5];
01518     double f[5], c[5];
01519     double xp, yp, zp, Sx;
01520 
01521     // Fill in matrix A -
01522     // write six difference-in-magnitude equations of the form
01523     // Sx^2(x2^2-x1^2) + 2*Sx*bx*(x2-x1) + Sy^2(y2^2-y1^2) + 2*Sy*by*(y2-y1) + Sz^2(z2^2-z1^2) + 2*Sz*bz*(z2-z1) = 0

01525     // 2*Sx*bx*(x2-x1)/Sx^2  + Sy^2(y2^2-y1^2)/Sx^2  + 2*Sy*by*(y2-y1)/Sx^2  + Sz^2(z2^2-z1^2)/Sx^2  + 2*Sz*bz*(z2-z1)/Sx^2  = (x1^2-x2^2)
01526     for (i=0;i<5;i++){
01527         A[i][0] = 2.0 * (x[i+1] - x[i]);
01528         A[i][1] = y[i+1]*y[i+1] - y[i]*y[i];
01529         A[i][2] = 2.0 * (y[i+1] - y[i]);
01530         A[i][3] = z[i+1]*z[i+1] - z[i]*z[i];
01531         A[i][4] = 2.0 * (z[i+1] - z[i]);
01532         f[i]    = x[i]*x[i] - x[i+1]*x[i+1];
01533     }
01534 
01535     // solve for c0=bx/Sx, c1=Sy^2/Sx^2; c2=Sy*by/Sx^2, c3=Sz^2/Sx^2, c4=Sz*bz/Sx^2
01536     if (  !LinearEquationsSolving( 5, (double *)A, f, c) ) return 0;
01537 
01538     // use one magnitude equation and c's to find Sx - doesn't matter which - all give the same answer
01539     xp = x[0]; yp = y[0]; zp = z[0];
01540     Sx = sqrt(ConstMag*ConstMag / (xp*xp + 2*c[0]*xp + c[0]*c[0] + c[1]*yp*yp + 2*c[2]*yp + c[2]*c[2]/c[1] + c[3]*zp*zp + 2*c[4]*zp + c[4]*c[4]/c[3]));
01541 
01542     S[0] = Sx;
01543     b[0] = Sx*c[0];
01544     S[1] = sqrt(c[1]*Sx*Sx);
01545     b[1] = c[2]*Sx*Sx/S[1];
01546     S[2] = sqrt(c[3]*Sx*Sx);
01547     b[2] = c[4]*Sx*Sx/S[2];
01548
