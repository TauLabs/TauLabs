00611 bool PluginManagerPrivate::loadQueue(PluginSpec *spec, QList<PluginSpec *> &queue,
00612         QList<PluginSpec *> &circularityCheckQueue)
00613 {
00614     if (queue.contains(spec))
00615         return true;
00616     // check for circular dependencies
00617     if (circularityCheckQueue.contains(spec)) {
00618         spec->d->hasError = true;
00619         spec->d->errorString = PluginManager::tr("Circular dependency detected:\n");
00620         int index = circularityCheckQueue.indexOf(spec);
00621         for (int i = index; i < circularityCheckQueue.size(); ++i) {
00622             spec->d->errorString.append(PluginManager::tr("%1(%2) depends on\n")
00623                 .arg(circularityCheckQueue.at(i)->name()).arg(circularityCheckQueue.at(i)->version()));
00624         }
00625         spec->d->errorString.append(PluginManager::tr("%1(%2)").arg(spec->name()).arg(spec->version()));
00626         return false;
00627     }
00628     circularityCheckQueue.append(spec);
00629     // check if we have the dependencies
00630     if (spec->state() == PluginSpec::Invalid || spec->state() == PluginSpec::Read) {
00631         spec->d->hasError = true;
00632         spec->d->errorString += "\n";
00633         spec->d->errorString += PluginManager::tr("Cannot load plugin because dependencies are not resolved");
00634         return false;
00635     }
00636     // add dependencies
00637     foreach (PluginSpec *depSpec, spec->dependencySpecs()) {
00638         if (!loadQueue(depSpec, queue, circularityCheckQueue)) {
00639             spec->d->hasError = true;
00640             spec->d->errorString =
00641                 PluginManager::tr("Cannot load plugin because dependency failed to load: %1(%2)\nReason: %3")
00642                     .arg(depSpec->name()).arg(depSpec->version()).arg(depSpec->errorString());
00643             return false;
00644         }
00645     }
00646     // add self
00647     queue.append(spec);
00648     return true;
00649 }

00655 void PluginManagerPrivate::loadPlugin(PluginSpec *spec, PluginSpec::State destState)
00656 {
00657     if (spec->hasError())
00658         return;
00659     if (destState == PluginSpec::Running) {
00660         spec->d->initializeExtensions();
00661         return;
00662     } else if (destState == PluginSpec::Deleted) {
00663         spec->d->kill();
00664         return;
00665     }
00666     foreach (PluginSpec *depSpec, spec->dependencySpecs()) {
00667         if (depSpec->state() != destState) {
00668             spec->d->hasError = true;
00669             spec->d->errorString =
00670                 PluginManager::tr("Cannot load plugin because dependency failed to load: %1(%2)\nReason: %3")
00671                     .arg(depSpec->name()).arg(depSpec->version()).arg(depSpec->errorString());
00672             return;
00673         }
00674     }
00675     if (destState == PluginSpec::Loaded)
00676         spec->d->loadLibrary();
00677     else if (destState == PluginSpec::Initialized)
00678         spec->d->initializePlugin();
00679     else if (destState == PluginSpec::Stopped)
