00067         // mag = (S**2 * x**2 + 2*S*b*x + b*b)
00068         // 0 = S**2 * (x1**2 - x2**2) + 2*S*B*(x1 - x2))
00069         // Fill in matrix A -
00070         // write six difference-in-magnitude equations of the form
00071         // Sx^2(x2^2-x1^2) + 2*Sx*bx*(x2-x1) + Sy^2(y2^2-y1^2) + 
00072         // 2*Sy*by*(y2-y1) + Sz^2(z2^2-z1^2) + 2*Sz*bz*(z2-z1) = 0

00074         // or in other words
00075         // 2*Sx*bx*(x2-x1)/Sx^2  + Sy^2(y2^2-y1^2)/Sx^2  +
00076         // 2*Sy*by*(y2-y1)/Sx^2  + Sz^2(z2^2-z1^2)/Sx^2  + 2*Sz*bz*(z2-z1)/Sx^2  =
00077         //       (x1^2-x2^2)
00078         Matrix<float, 5, 5> A;
00079         Matrix<float, 5, 1> f;
00080         for (unsigned i=0; i<5; i++){
00081                 A.row(i)[0] = 2.0 * (samples[i+1].x() - samples[i].x());
00082                 A.row(i)[1] = samples[i+1].y()* samples[i+1].y() - samples[i].y()*samples[i].y();
00083                 A.row(i)[2] = 2.0 * (samples[i+1].y() - samples[i].y());
00084                 A.row(i)[3] = samples[i+1].z()*samples[i+1].z() - samples[i].z()*samples[i].z();
00085                 A.row(i)[4] = 2.0 * (samples[i+1].z() - samples[i].z());
00086                 f[i]    = samples[i].x()*samples[i].x() - samples[i+1].x()*samples[i+1].x();
00087         }
00088         Matrix<float, 5, 1> c;
00089         A.lu().solve(f, &c);
