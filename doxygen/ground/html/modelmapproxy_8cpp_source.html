190 void ModelMapProxy::refreshOverlays()
191 {
192  myMap->deleteAllOverlays();
193  if(model->rowCount()<1)
194  return;
195  WayPointItem *wp_current = NULL;
196  WayPointItem *wp_next = NULL;
197  overlayType wp_next_overlay;
198 
199  // Get first waypoint type before stepping through path
200  wp_current = findWayPointNumber(0);
201  overlayType wp_current_overlay = overlayTranslate(model->data(model->index(0,FlightDataModel::MODE),Qt::UserRole).toInt());
202  createOverlay(wp_current,myMap->Home,wp_current_overlay,Qt::green);
203 
204  for(int x=0;x<model->rowCount();++x)
205  {
206  wp_current = findWayPointNumber(x);
207 
208  wp_next_overlay = overlayTranslate(model->data(model->index(x+1,FlightDataModel::MODE),Qt::UserRole).toInt());
209 
210  wp_next = findWayPointNumber(x+1);
211  createOverlay(wp_current, wp_next, wp_next_overlay, Qt::green,
212  model->data(model->index(x+1,FlightDataModel::MODE_PARAMS)).toFloat());
213  }
214 }

250 void ModelMapProxy::dataChanged(const QModelIndex &topLeft, const QModelIndex &bottomRight)
251 {
252  Q_UNUSED(bottomRight);
253 
254  // Abort if no corresponding graphical item
255  WayPointItem *item = findWayPointNumber(topLeft.row());
256  if(!item)
257  return;
258 
259  internals::PointLatLng latlng;
260  double altitude;
261  QModelIndex index;
262  QString desc;
263 
264  for (int x = topLeft.row(); x <= bottomRight.row(); x++) {
265  for (int column = topLeft.column(); column <= bottomRight.column(); column++) {
266  // Action depends on which columns were modified
267  switch(column)
268  {
269  case FlightDataModel::MODE:
270  refreshOverlays();
271  break;
272  case FlightDataModel::WPDESCRITPTION:
273  index=model->index(x,FlightDataModel::WPDESCRITPTION);
274  desc=index.data(Qt::DisplayRole).toString();
275  item->SetDescription(desc);
276  break;
277  case FlightDataModel::LATPOSITION:
278  latlng=item->Coord();
279  index=model->index(x,FlightDataModel::LATPOSITION);
280  latlng.SetLat(index.data(Qt::DisplayRole).toDouble());
281  item->SetCoord(latlng);
282  break;
283  case FlightDataModel::LNGPOSITION:
284  latlng=item->Coord();
285  index=model->index(x,FlightDataModel::LNGPOSITION);
286  latlng.SetLng(index.data(Qt::DisplayRole).toDouble());
287  item->SetCoord(latlng);
288  break;
289  case FlightDataModel::ALTITUDE:
290  index=model->index(x,FlightDataModel::ALTITUDE);
291  altitude=index.data(Qt::DisplayRole).toDouble();
292  item->SetAltitude(altitude);
293  break;
294  case FlightDataModel::MODE_PARAMS:
295  // Make sure to update radius of arcs
296  refreshOverlays();
297  break;
298  case FlightDataModel::LOCKED:
299  index=model->index(x,FlightDataModel::LOCKED);
300  item->setFlag(QGraphicsItem::ItemIsMovable,!index.data(Qt::DisplayRole).toBool());
301  break;
302  }
303  }
304  }
305 }
