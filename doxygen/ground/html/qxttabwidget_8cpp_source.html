272 void QxtTabWidget::insertTabAction(int index, QAction* before, QAction* action)
273 {
274  Q_ASSERT(index >= 0 && index < qxt_d().actions.count());
275 
276  if (!action)
277  {
278  qWarning("QxtTabWidget::insertTabAction: Attempt to insert a null action");
279  return;
280  }
281 
282  const Actions& actions = qxt_d().actions.at(index);
283  if (actions.contains(action))
284  removeTabAction(index, action);
285 
286  int pos = actions.indexOf(before);
287  if (pos < 0)
288  {
289  before = 0;
290  pos = actions.count();
291  }
292  qxt_d().actions[index].insert(pos, action);
293 
294  QActionEvent e(QEvent::ActionAdded, action, before);
295  QApplication::sendEvent(this, &e);
296 }

436 void QxtTabWidget::contextMenuEvent(QContextMenuEvent* event)
437 {
438  const QPoint& pos = event->pos();
439  if (!tabBar()->geometry().contains(pos))
440  return QTabWidget::contextMenuEvent(event);
441 
442  const int index = qxt_d().tabIndexAt(event->pos());
443  switch (qxt_d().policy)
444  {
445  case Qt::NoContextMenu:
446  event->ignore();
447  break;
448 
449  case Qt::PreventContextMenu:
450  event->accept();
451  break;
452 
453  case Qt::ActionsContextMenu:
454  if (index != -1 && qxt_d().actions.at(index).count())
455  {
456  QMenu::exec(qxt_d().actions.at(index), event->globalPos());
457  }
458  break;
459 
460  case Qt::CustomContextMenu:
461  if (index != -1)
462  {
463  emit tabContextMenuRequested(index, event->globalPos());
464  }
465  break;
466 
467  case Qt::DefaultContextMenu:
468  default:
469  if (index != -1)
470  {
471  tabContextMenuEvent(index, event);
472  }
473  break;
474  }
475 }
