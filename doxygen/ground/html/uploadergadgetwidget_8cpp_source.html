00799 void UploaderGadgetWidget::versionMatchCheck()
00800 {
00801     ExtensionSystem::PluginManager *pm = ExtensionSystem::PluginManager::instance();
00802     UAVObjectUtilManager *utilMngr = pm->getObject<UAVObjectUtilManager>();
00803     deviceDescriptorStruct boardDescription = utilMngr->getBoardDescriptionStruct();
00804     QByteArray uavoHashArray;
00805     QString uavoHash = QString::fromLatin1(Core::Constants::UAVOSHA1_STR);
00806     uavoHash.chop(2);
00807     uavoHash.remove(0,2);
00808     uavoHash=uavoHash.trimmed();
00809     bool ok;
00810     foreach(QString str,uavoHash.split(","))
00811     {
00812         uavoHashArray.append(str.toInt(&ok,16));
00813     }
00814 
00815     QByteArray fwVersion=boardDescription.uavoHash;
00816     if (fwVersion != uavoHashArray) {
00817 
00818         QString gcsDescription = QString::fromLatin1(Core::Constants::GCS_REVISION_STR);
00819         QString gcsGitHash = gcsDescription.mid(gcsDescription.indexOf(":")+1, 8);
00820         gcsGitHash.remove( QRegExp("^[0]*") );
00821         QString gcsGitDate = gcsDescription.mid(gcsDescription.indexOf(" ")+1, 14);
00822 
00823         QString gcsUavoHashStr;
00824         QString fwUavoHashStr;
00825         foreach(char i, fwVersion)
00826         {
00827             fwUavoHashStr.append(QString::number(i,16).right(2));
00828         }
00829         foreach(char i, uavoHashArray)
00830         {
00831             gcsUavoHashStr.append(QString::number(i,16).right(2));
00832         }
00833         QString gcsVersion = gcsGitDate + " (" + gcsGitHash + "-"+ gcsUavoHashStr.left(8) + ")";
00834         QString fwVersion = boardDescription.gitDate + " (" + boardDescription.gitHash + "-" + fwUavoHashStr.left(8) + ")";
00835 
00836         QString warning = QString(tr(
00837             "GCS and firmware versions of the UAV objects set do not match which can cause configuration problems. "
00838             "GCS version: %1 Firmware version: %2.")).arg(gcsVersion).arg(fwVersion);
00839         msg->showMessage(warning);
00840     }
