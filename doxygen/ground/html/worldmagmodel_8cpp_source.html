00092

00094         GeodeticToSpherical(&CoordGeodetic, &CoordSpherical);
00095 
00096         if (DateToYear(Month, Day, Year) < 0)
00097             return -5;  // error
00098 
00099         // Compute the geoMagnetic field elements and their time change
00100         if (Geomag(&CoordSpherical, &CoordGeodetic, &GeoMagneticElements) < 0)
00101             return -6;  // error

00137     int WorldMagModel::Geomag(WMMtype_CoordSpherical *CoordSpherical, WMMtype_CoordGeodetic *CoordGeodetic, WMMtype_GeoMagneticElements *GeoMagneticElements)
00138     /*
00139       The main subroutine that calls a sequence of WMM sub-functions to calculate the magnetic field elements for a single point.
00140       The function expects the model coefficients and point coordinates as input and returns the magnetic field elements and
00141       their rate of change. Though, this subroutine can be called successively to calculate a time series, profile or grid
00142       of magnetic field, these are better achieved by the subroutine WMM_Grid.
00143 
00144       INPUT: Ellip
00145       CoordSpherical
00146       CoordGeodetic
00147       TimedMagneticModel
00148 
00149       OUTPUT : GeoMagneticElements
00150     */
00151     {
00152         WMMtype_MagneticResults             MagneticResultsSph;
00153         WMMtype_MagneticResults             MagneticResultsGeo;
00154         WMMtype_MagneticResults             MagneticResultsSphVar;
00155         WMMtype_MagneticResults             MagneticResultsGeoVar;
00156         WMMtype_LegendreFunction            LegendreFunction;
00157         WMMtype_SphericalHarmonicVariables  SphVariables;
00158 
00159         // Compute Spherical Harmonic variables
00160         ComputeSphericalHarmonicVariables(CoordSpherical, MagneticModel.nMax, &SphVariables);
00161 
00162         // Compute ALF
00163         if (AssociatedLegendreFunction(CoordSpherical, MagneticModel.nMax, &LegendreFunction) < 0)
00164             return -1; // error
00165 
00166         // Accumulate the spherical harmonic coefficients
00167         Summation(&LegendreFunction, &SphVariables, CoordSpherical, &MagneticResultsSph);
00168 
00169         // Sum the Secular Variation Coefficients
00170         SecVarSummation(&LegendreFunction, &SphVariables, CoordSpherical, &MagneticResultsSphVar);
00171 
00172         // Map the computed Magnetic fields to Geodeitic coordinates
00173         RotateMagneticVector(CoordSpherical, CoordGeodetic, &MagneticResultsSph, &MagneticResultsGeo);
00174 
00175         // Map the secular variation field components to Geodetic coordinates
00176         RotateMagneticVector(CoordSpherical, CoordGeodetic, &MagneticResultsSphVar, &MagneticResultsGeoVar);
00177 
00178         // Calculate the Geomagnetic elements, Equation 18 , WMM Technical report
00179         CalculateGeoMagneticElements(&MagneticResultsGeo, GeoMagneticElements);
00180 
00181         // Calculate the secular variation of each of the Geomagnetic elements
00182         CalculateSecularVariation(&MagneticResultsGeoVar, GeoMagneticElements);
00183 
00184         return 0;   // OK
00185     }

00187     void WorldMagModel::ComputeSphericalHarmonicVariables(WMMtype_CoordSpherical *CoordSpherical, int nMax, WMMtype_SphericalHarmonicVariables *SphVariables)
00188     {
00189        /* Computes Spherical variables
00190       Variables computed are (a/r)^(n+2), cos_m(lamda) and sin_m(lambda) for spherical harmonic
00191       summations. (Equations 10-12 in the WMM Technical Report)
00192       INPUT   Ellip  data  structure with the following elements
00193       float a; semi-major axis of the ellipsoid
00194       float b; semi-minor axis of the ellipsoid
00195       float fla;  flattening
00196       float epssq; first eccentricity squared
00197       float eps;  first eccentricity
00198       float re; mean radius of  ellipsoid
00199       CoordSpherical    A data structure with the following elements
00200       float lambda; ( longitude)
00201       float phig; ( geocentric latitude )
00202       float r;            ( distance from the center of the ellipsoid)
00203       nMax   integer     ( Maxumum degree of spherical harmonic secular model)\
00204 
00205       OUTPUT  SphVariables  Pointer to the   data structure with the following elements
00206       float RelativeRadiusPower[WMM_MAX_MODEL_DEGREES+1];   [earth_reference_radius_km  sph. radius ]^n
00207       float cos_mlambda[WMM_MAX_MODEL_DEGREES+1]; cp(m)  - cosine of (mspherical coord. longitude)
00208       float sin_mlambda[WMM_MAX_MODEL_DEGREES+1];  sp(m)  - sine of (mspherical coord. longitude)
00209     */
00210         double cos_lambda = cos(CoordSpherical->lambda * DEG2RAD);
00211         double sin_lambda = sin(CoordSpherical->lambda * DEG2RAD);
00212 
00213         /* for n = 0 ... model_order, compute (Radius of Earth / Spherica radius r)^(n+2)
00214            for n  1..nMax-1 (this is much faster than calling pow MAX_N+1 times).      */
00215 
00216         SphVariables->RelativeRadiusPower[0] = (Ellip.re / CoordSpherical->r) * (Ellip.re / CoordSpherical->r);
00217         for (int n = 1; n <= nMax; n++)
00218             SphVariables->RelativeRadiusPower[n] = SphVariables->RelativeRadiusPower[n - 1] * (Ellip.re / CoordSpherical->r);
00219 
00220         /*
00221            Compute cos(m*lambda), sin(m*lambda) for m = 0 ... nMax
00222            cos(a + b) = cos(a)*cos(b) - sin(a)*sin(b)
00223            sin(a + b) = cos(a)*sin(b) + sin(a)*cos(b)
00224          */
00225         SphVariables->cos_mlambda[0] = 1.0;
00226         SphVariables->sin_mlambda[0] = 0.0;
00227 
00228         SphVariables->cos_mlambda[1] = cos_lambda;
00229         SphVariables->sin_mlambda[1] = sin_lambda;
00230         for (int m = 2; m <= nMax; m++)
00231         {
00232             SphVariables->cos_mlambda[m] = SphVariables->cos_mlambda[m - 1] * cos_lambda - SphVariables->sin_mlambda[m - 1] * sin_lambda;
00233             SphVariables->sin_mlambda[m] = SphVariables->cos_mlambda[m - 1] * sin_lambda + SphVariables->sin_mlambda[m - 1] * cos_lambda;
00234         }
00235     }

00237     int WorldMagModel::AssociatedLegendreFunction(WMMtype_CoordSpherical *CoordSpherical, int nMax, WMMtype_LegendreFunction *LegendreFunction)
00238     {
00239         /* Computes  all of the Schmidt-semi normalized associated Legendre
00240            functions up to degree nMax. If nMax <= 16, function WMM_PcupLow is used.
00241            Otherwise WMM_PcupHigh is called.
00242            INPUT  CoordSpherical        A data structure with the following elements
00243            float lambda; ( longitude)
00244            float phig; ( geocentric latitude )
00245            float r;       ( distance from the center of the ellipsoid)
00246            nMax         integer          ( Maxumum degree of spherical harmonic secular model)
00247            LegendreFunction Pointer to data structure with the following elements
00248            float *Pcup;  (  pointer to store Legendre Function  )
00249            float *dPcup; ( pointer to store  Derivative of Lagendre function )
00250 
00251            OUTPUT  LegendreFunction  Calculated Legendre variables in the data structure
00252          */
00253 
00254         double sin_phi = sin(CoordSpherical->phig * DEG2RAD);   // sin  (geocentric latitude)
00255 
00256         if (nMax <= 16 || (1 - fabs(sin_phi)) < 1.0e-10)        /* If nMax is less tha 16 or at the poles */
00257             PcupLow(LegendreFunction->Pcup, LegendreFunction->dPcup, sin_phi, nMax);
00258         else
00259         {
00260             if (PcupHigh(LegendreFunction->Pcup, LegendreFunction->dPcup, sin_phi, nMax) < 0)
00261                 return -1;  // error
00262         }
00263 
00264         return 0;   // OK
00265     }

00267     void WorldMagModel::Summation(  WMMtype_LegendreFunction *LegendreFunction,
00268                                     WMMtype_SphericalHarmonicVariables *SphVariables,
00269                                     WMMtype_CoordSpherical *CoordSpherical,
00270                                     WMMtype_MagneticResults *MagneticResults)
00271     {
00272         /* Computes Geomagnetic Field Elements X, Y and Z in Spherical coordinate system using spherical harmonic summation.
00273 
00274            The vector Magnetic field is given by -grad V, where V is Geomagnetic scalar potential
00275            The gradient in spherical coordinates is given by:
00276 
00277            dV ^     1 dV ^        1     dV ^
00278            grad V = -- r  +  - -- t  +  -------- -- p
00279            dr       r dt       r sin(t) dp
00280 
00281            INPUT :  LegendreFunction
00282            MagneticModel
00283            SphVariables
00284            CoordSpherical
00285            OUTPUT : MagneticResults
00286 
00287            Manoj Nair, June, 2009 Manoj.C.Nair@Noaa.Gov
00288          */
00289 
00290         MagneticResults->Bz = 0.0;
00291         MagneticResults->By = 0.0;
00292         MagneticResults->Bx = 0.0;
00293 
00294         for (int n = 1; n <= MagneticModel.nMax; n++)
00295         {
00296             for (int m = 0; m <= n; m++)
00297             {
00298                 int index = (n * (n + 1) / 2 + m);
00299 
00300 /*                  nMax        (n+2)     n     m            m           m
00301     Bz =   -SUM (a/r)   (n+1) SUM  [g cos(m p) + h sin(m p)] P (sin(phi))
00302             n=1               m=0   n            n           n  */
00303 /* Equation 12 in the WMM Technical report.  Derivative with respect to radius.*/
00304                 MagneticResults->Bz -=
00305                     SphVariables->RelativeRadiusPower[n] *
00306                      (get_main_field_coeff_g(index) *
00307                      SphVariables->cos_mlambda[m] + get_main_field_coeff_h(index) * SphVariables->sin_mlambda[m])
00308                     * (double)(n + 1) * LegendreFunction->Pcup[index];
00309 
00310 /*                1 nMax  (n+2)    n     m            m           m
00311     By =    SUM (a/r) (m)  SUM  [g cos(m p) + h sin(m p)] dP (sin(phi))
00312            n=1             m=0   n            n           n  */
00313 /* Equation 11 in the WMM Technical report. Derivative with respect to longitude, divided by radius. */
00314                 MagneticResults->By +=
00315                     SphVariables->RelativeRadiusPower[n] *
00316                     (get_main_field_coeff_g(index) *
00317                      SphVariables->sin_mlambda[m] - get_main_field_coeff_h(index) * SphVariables->cos_mlambda[m])
00318                     * (double)(m) * LegendreFunction->Pcup[index];
00319 /*                 nMax  (n+2) n     m            m           m
00320     Bx = - SUM (a/r)   SUM  [g cos(m p) + h sin(m p)] dP (sin(phi))
00321            n=1         m=0   n            n           n  */
00322 /* Equation 10  in the WMM Technical report. Derivative with respect to latitude, divided by radius. */
00323 
00324                 MagneticResults->Bx -=
00325                     SphVariables->RelativeRadiusPower[n] *
00326                     (get_main_field_coeff_g(index) *
00327                      SphVariables->cos_mlambda[m] + get_main_field_coeff_h(index) * SphVariables->sin_mlambda[m])
00328                     * LegendreFunction->dPcup[index];
00329 
00330             }
00331         }
00332 
00333         double cos_phi = cos(CoordSpherical->phig * DEG2RAD);
00334         if (fabs(cos_phi) > 1.0e-10)
00335         {
00336             MagneticResults->By = MagneticResults->By / cos_phi;
00337         }
00338         else
00339         {
00340             /* Special calculation for component - By - at Geographic poles.
00341              * If the user wants to avoid using this function,  please make sure that
00342              * the latitude is not exactly +/-90. An option is to make use the function
00343              * WMM_CheckGeographicPoles.
00344              */
00345             SummationSpecial(SphVariables, CoordSpherical, MagneticResults);
00346         }
00347     }

00417     void WorldMagModel::RotateMagneticVector(   WMMtype_CoordSpherical *CoordSpherical,
00418                                                 WMMtype_CoordGeodetic *CoordGeodetic,
00419                                                 WMMtype_MagneticResults *MagneticResultsSph,
00420                                                 WMMtype_MagneticResults *MagneticResultsGeo)
00421     {
00422         /* Rotate the Magnetic Vectors to Geodetic Coordinates
00423            Manoj Nair, June, 2009 Manoj.C.Nair@Noaa.Gov
00424            Equation 16, WMM Technical report
00425 
00426            INPUT : CoordSpherical : Data structure WMMtype_CoordSpherical with the following elements
00427            float lambda; ( longitude)
00428            float phig; ( geocentric latitude )
00429            float r;       ( distance from the center of the ellipsoid)
00430 
00431            CoordGeodetic : Data structure WMMtype_CoordGeodetic with the following elements
00432            float lambda; (longitude)
00433            float phi; ( geodetic latitude)
00434            float HeightAboveEllipsoid; (height above the ellipsoid (HaE) )
00435            float HeightAboveGeoid;(height above the Geoid )
00436 
00437            MagneticResultsSph : Data structure WMMtype_MagneticResults with the following elements
00438            float Bx;     North
00439            float By;       East
00440            float Bz;    Down
00441 
00442            OUTPUT: MagneticResultsGeo Pointer to the data structure WMMtype_MagneticResults, with the following elements
00443            float Bx;     North
00444            float By;       East
00445            float Bz;    Down
00446          */
00447 
00448         /* Difference between the spherical and Geodetic latitudes */
00449         double Psi = (CoordSpherical->phig - CoordGeodetic->phi) * DEG2RAD;
00450 
00451         /* Rotate spherical field components to the Geodeitic system */
00452         MagneticResultsGeo->Bz = MagneticResultsSph->Bx * sin(Psi) + MagneticResultsSph->Bz * cos(Psi);
00453         MagneticResultsGeo->Bx = MagneticResultsSph->Bx * cos(Psi) - MagneticResultsSph->Bz * sin(Psi);
00454         MagneticResultsGeo->By = MagneticResultsSph->By;
00455     }

00457     void WorldMagModel::CalculateGeoMagneticElements(WMMtype_MagneticResults *MagneticResultsGeo, WMMtype_GeoMagneticElements *GeoMagneticElements)
00458     {
00459         /* Calculate all the Geomagnetic elements from X,Y and Z components
00460            INPUT     MagneticResultsGeo   Pointer to data structure with the following elements
00461            float Bx;    ( North )
00462            float By;      ( East )
00463            float Bz;    ( Down )
00464            OUTPUT    GeoMagneticElements    Pointer to data structure with the following elements
00465            float Decl; (Angle between the magnetic field vector and true north, positive east)
00466            float Incl; Angle between the magnetic field vector and the horizontal plane, positive down

00468            float H; Horizontal Magnetic Field Strength
00469            float X; Northern component of the magnetic field vector
00470            float Y; Eastern component of the magnetic field vector
00471            float Z; Downward component of the magnetic field vector
00472          */
00473 
00474         GeoMagneticElements->X = MagneticResultsGeo->Bx;
00475         GeoMagneticElements->Y = MagneticResultsGeo->By;
00476         GeoMagneticElements->Z = MagneticResultsGeo->Bz;
00477 
00478         GeoMagneticElements->H = sqrt(MagneticResultsGeo->Bx * MagneticResultsGeo->Bx + MagneticResultsGeo->By * MagneticResultsGeo->By);
00479         GeoMagneticElements->F = sqrt(GeoMagneticElements->H * GeoMagneticElements->H + MagneticResultsGeo->Bz * MagneticResultsGeo->Bz);
00480         GeoMagneticElements->Decl = atan2(GeoMagneticElements->Y, GeoMagneticElements->X) * RAD2DEG;
00481         GeoMagneticElements->Incl = atan2(GeoMagneticElements->Z, GeoMagneticElements->H) * RAD2DEG;
00482     }

00484     void WorldMagModel::CalculateSecularVariation(WMMtype_MagneticResults *MagneticVariation, WMMtype_GeoMagneticElements *MagneticElements)
00485     {
00486         /* This takes the Magnetic Variation in x, y, and z and uses it to calculate the secular variation of each of the Geomagnetic elements.
00487         INPUT     MagneticVariation   Data structure with the following elements
00488                     float Bx;    ( North )
00489                     float By;     ( East )
00490                     float Bz;    ( Down )
00491         OUTPUT   MagneticElements   Pointer to the data  structure with the following elements updated
00492                 float Decldot; Yearly Rate of change in declination
00493                 float Incldot; Yearly Rate of change in inclination
00494                 float Fdot; Yearly rate of change in Magnetic field strength
00495                 float Hdot; Yearly rate of change in horizontal field strength
00496                 float Xdot; Yearly rate of change in the northern component
00497                 float Ydot; Yearly rate of change in the eastern component
00498                 float Zdot; Yearly rate of change in the downward component
00499                 float GVdot;Yearly rate of chnage in grid variation
00500         */
00501 
00502         MagneticElements->Xdot = MagneticVariation->Bx;
00503         MagneticElements->Ydot = MagneticVariation->By;
00504         MagneticElements->Zdot = MagneticVariation->Bz;
00505         MagneticElements->Hdot = (MagneticElements->X * MagneticElements->Xdot + MagneticElements->Y * MagneticElements->Ydot) / MagneticElements->H;   //See equation 19 in the WMM technical report
00506         MagneticElements->Fdot =
00507             (MagneticElements->X * MagneticElements->Xdot +
00508              MagneticElements->Y * MagneticElements->Ydot + MagneticElements->Z * MagneticElements->Zdot) / MagneticElements->F;
00509         MagneticElements->Decldot =
00510             180.0 / M_PI * (MagneticElements->X * MagneticElements->Ydot -
00511                     MagneticElements->Y * MagneticElements->Xdot) / (MagneticElements->H * MagneticElements->H);
00512         MagneticElements->Incldot =
00513             180.0 / M_PI * (MagneticElements->H * MagneticElements->Zdot -
00514                     MagneticElements->Z * MagneticElements->Hdot) / (MagneticElements->F * MagneticElements->F);
00515         MagneticElements->GVdot = MagneticElements->Decldot;
00516     }

00519     {
00520         /*      This function evaluates all of the Schmidt-semi normalized associated Legendre
00521             functions up to degree nMax. The functions are initially scaled by
00522             10^280 sin^m in order to minimize the effects of underflow at large m
00523             near the poles (see Holmes and Featherstone 2002, J. Geodesy, 76, 279-299).
00524             Note that this function performs the same operation as WMM_PcupLow.
00525             However this function also can be used for high degree (large nMax) models.
00526 
00527             Calling Parameters:
00528                 INPUT
00529                     nMax:        Maximum spherical harmonic degree to compute.
00530                     x:          cos(colatitude) or sin(latitude).
00531 
00532                 OUTPUT
00533                     Pcup:       A vector of all associated Legendgre polynomials evaluated at
00534                             x up to nMax. The lenght must by greater or equal to (nMax+1)*(nMax+2)/2.
00535                   dPcup:   Derivative of Pcup(x) with respect to latitude
00536             Notes:
00537 
00538           Adopted from the FORTRAN code written by Mark Wieczorek September 25, 2005.
00539 
00540           Manoj Nair, Nov, 2009 Manoj.C.Nair@Noaa.Gov
00541 
00542           Change from the previous version
00543           The prevous version computes the derivatives as
00544           dP(n,m)(x)/dx, where x = sin(latitude) (or cos(colatitude) ).
00545           However, the WMM Geomagnetic routines requires dP(n,m)(x)/dlatitude.
00546           Hence the derivatives are multiplied by sin(latitude).
00547           Removed the options for CS phase and normalizations.
00548 
00549           Note: In geomagnetism, the derivatives of ALF are usually found with
00550           respect to the colatitudes. Here the derivatives are found with respect
00551           to the latitude. The difference is a sign reversal for the derivative of
00552           the Associated Legendre Functions.
00553 
00554           The derivates can't be computed for latitude = |90| degrees.
00555             */
00556         double  f1[WMM_NUMPCUP];
00557         double  f2[WMM_NUMPCUP];
00558         double  PreSqr[WMM_NUMPCUP];
00559         int m;
00560 
00561         if (fabs(x) == 1.0)
00562         {
00563             // printf("Error in PcupHigh: derivative cannot be calculated at poles\n");
00564             return -2;
00565         }
00566 
00567         double scalef = 1.0e-280;
00568 
00569         for (int n = 0; n <= 2 * nMax + 1; ++n)
00570             PreSqr[n] = sqrt((double)(n));
00571 
00572         int k = 2;
00573 
00574         for (int n = 2; n <= nMax; n++)
00575         {
00576             k = k + 1;
00577             f1[k] = (double)(2 * n - 1) / n;
00578             f2[k] = (double)(n - 1) / n;
00579             for (int m = 1; m <= n - 2; m++)
00580             {
00581                 k = k + 1;
00582                 f1[k] = (double)(2 * n - 1) / PreSqr[n + m] / PreSqr[n - m];
00583                 f2[k] = PreSqr[n - m - 1] * PreSqr[n + m - 1] / PreSqr[n + m] / PreSqr[n - m];
00584             }
00585             k = k + 2;
00586         }
00587 
00588         /*z = sin (geocentric latitude) */
00589         double z = sqrt((1.0 - x) * (1.0 + x));
00590         double pm2 = 1.0;
00591         Pcup[0] = 1.0;
00592         dPcup[0] = 0.0;
00593         if (nMax == 0)
00594             return -3;
00595         double pm1 = x;
00596         Pcup[1] = pm1;
00597         dPcup[1] = z;
00598         k = 1;
00599 
00600         for (int n = 2; n <= nMax; n++)
00601         {
00602             k = k + n;
00603             double plm = f1[k] * x * pm1 - f2[k] * pm2;
00604             Pcup[k] = plm;
00605             dPcup[k] = (double)(n) * (pm1 - x * plm) / z;
00606             pm2 = pm1;
00607             pm1 = plm;
00608         }
00609 
00610         double pmm = PreSqr[2] * scalef;
00611         double rescalem = 1.0 / scalef;
00612         int kstart = 0;
00613 
00614         for (m = 1; m <= nMax - 1; ++m)
00615         {
00616             rescalem = rescalem * z;
00617 
00618             /* Calculate Pcup(m,m) */
00619             kstart = kstart + m + 1;
00620             pmm = pmm * PreSqr[2 * m + 1] / PreSqr[2 * m];
00621             Pcup[kstart] = pmm * rescalem / PreSqr[2 * m + 1];
00622             dPcup[kstart] = -((double)(m) * x * Pcup[kstart] / z);
00623             pm2 = pmm / PreSqr[2 * m + 1];
00624             /* Calculate Pcup(m+1,m) */
00625             k = kstart + m + 1;
00626             pm1 = x * PreSqr[2 * m + 1] * pm2;
00627             Pcup[k] = pm1 * rescalem;
00628             dPcup[k] = ((pm2 * rescalem) * PreSqr[2 * m + 1] - x * (double)(m + 1) * Pcup[k]) / z;
00629             /* Calculate Pcup(n,m) */
00630             for (int n = m + 2; n <= nMax; ++n)
00631             {
00632                 k = k + n;
00633                 double plm = x * f1[k] * pm1 - f2[k] * pm2;
00634                 Pcup[k] = plm * rescalem;
00635                 dPcup[k] = (PreSqr[n + m] * PreSqr[n - m] * (pm1 * rescalem) - (double)(n) * x * Pcup[k]) / z;
00636                 pm2 = pm1;
00637                 pm1 = plm;
00638             }
00639         }
00640 
00641         /* Calculate Pcup(nMax,nMax) */
00642         rescalem = rescalem * z;
00643         kstart = kstart + m + 1;
00644         pmm = pmm / PreSqr[2 * nMax];
00645         Pcup[kstart] = pmm * rescalem;
00646         dPcup[kstart] = -(double)(nMax) * x * Pcup[kstart] / z;
00647 
00648         // *********
00649 
00650         return 0;   // OK
00651     }

00653     void WorldMagModel::PcupLow(double *Pcup, double *dPcup, double x, int nMax)
00654     {
00655         /*   This function evaluates all of the Schmidt-semi normalized associated Legendre functions up to degree nMax.
00656 
00657             Calling Parameters:
00658                 INPUT
00659                     nMax:        Maximum spherical harmonic degree to compute.
00660                     x:          cos(colatitude) or sin(latitude).
00661 
00662                OUTPUT
00663                    Pcup:        A vector of all associated Legendgre polynomials evaluated at
00664                            x up to nMax.
00665                   dPcup: Derivative of Pcup(x) with respect to latitude
00666 
00667            Notes: Overflow may occur if nMax > 20 , especially for high-latitudes.
00668            Use WMM_PcupHigh for large nMax.
00669 
00670           Writted by Manoj Nair, June, 2009 . Manoj.C.Nair@Noaa.Gov.
00671 
00672           Note: In geomagnetism, the derivatives of ALF are usually found with
00673           respect to the colatitudes. Here the derivatives are found with respect
00674           to the latitude. The difference is a sign reversal for the derivative of
00675           the Associated Legendre Functions.
00676         */
00677 
00678         double schmidtQuasiNorm[WMM_NUMPCUP];
00679 
00680         Pcup[0] = 1.0;
00681         dPcup[0] = 0.0;
00682 
00683         /*sin (geocentric latitude) - sin_phi */
00684         double z = sqrt((1.0 - x) * (1.0 + x));
00685 
00686         /*       First, Compute the Gauss-normalized associated Legendre  functions */
00687         for (int n = 1; n <= nMax; n++)
00688         {
00689             for (int m = 0; m <= n; m++)
00690             {
00691                 int index = (n * (n + 1) / 2 + m);
00692                 if (n == m)
00693                 {
00694                     int index1 = (n - 1) * n / 2 + m - 1;
00695                     Pcup[index] = z * Pcup[index1];
00696                     dPcup[index] = z * dPcup[index1] + x * Pcup[index1];
00697                 }
00698                 else
00699                 if (n == 1 && m == 0)
00700                 {
00701                     int index1 = (n - 1) * n / 2 + m;
00702                     Pcup[index] = x * Pcup[index1];
00703                     dPcup[index] = x * dPcup[index1] - z * Pcup[index1];
00704                 }
00705                 else
00706                 if (n > 1 && n != m)
00707                 {
00708                     int index1 = (n - 2) * (n - 1) / 2 + m;
00709                     int index2 = (n - 1) * n / 2 + m;
00710                     if (m > n - 2)
00711                     {
00712                         Pcup[index] = x * Pcup[index2];
00713                         dPcup[index] = x * dPcup[index2] - z * Pcup[index2];
00714                     }
00715                     else
00716                     {
00717                         double k = (double)(((n - 1) * (n - 1)) - (m * m)) / (double)((2 * n - 1) * (2 * n - 3));
00718                         Pcup[index] = x * Pcup[index2] - k * Pcup[index1];
00719                         dPcup[index] = x * dPcup[index2] - z * Pcup[index2] - k * dPcup[index1];
00720                     }
00721                 }
00722             }
00723         }
00724 
00725         /*Compute the ration between the Gauss-normalized associated Legendre
00726           functions and the Schmidt quasi-normalized version. This is equivalent to
00727         sqrt((m==0?1:2)*(n-m)!/(n+m!))*(2n-1)!!/(n-m)!  */
00728 
00729         schmidtQuasiNorm[0] = 1.0;
00730         for (int n = 1; n <= nMax; n++)
00731         {
00732             int index = (n * (n + 1) / 2);
00733             int index1 = (n - 1) * n / 2;
00734             /* for m = 0 */
00735             schmidtQuasiNorm[index] = schmidtQuasiNorm[index1] * (double)(2 * n - 1) / (double)n;
00736 
00737             for (int m = 1; m <= n; m++)
00738             {
00739                 index = (n * (n + 1) / 2 + m);
00740                 index1 = (n * (n + 1) / 2 + m - 1);
00741                 schmidtQuasiNorm[index] = schmidtQuasiNorm[index1] * sqrt((double)((n - m + 1) * (m == 1 ? 2 : 1)) / (double)(n + m));
00742             }
00743         }
00744 
00745         /* Converts the  Gauss-normalized associated Legendre
00746               functions to the Schmidt quasi-normalized version using pre-computed
00747               relation stored in the variable schmidtQuasiNorm */
00748 
00749         for (int n = 1; n <= nMax; n++)
00750         {
00751             for (int m = 0; m <= n; m++)
00752             {
00753                 int index = (n * (n + 1) / 2 + m);
00754                 Pcup[index] = Pcup[index] * schmidtQuasiNorm[index];
00755                 dPcup[index] = -dPcup[index] * schmidtQuasiNorm[index];
00756                 /* The sign is changed since the new WMM routines use derivative with respect to latitude insted of co-latitude */
00757             }
00758         }
00759     }

00761     void WorldMagModel::SummationSpecial(WMMtype_SphericalHarmonicVariables *SphVariables, WMMtype_CoordSpherical *CoordSpherical, WMMtype_MagneticResults *MagneticResults)
00762     {
00763         /* Special calculation for the component By at Geographic poles.
00764            Manoj Nair, June, 2009 manoj.c.nair@noaa.gov
00765            INPUT: MagneticModel
00766            SphVariables
00767            CoordSpherical
00768            OUTPUT: MagneticResults
00769            CALLS : none
00770            See Section 1.4, "SINGULARITIES AT THE GEOGRAPHIC POLES", WMM Technical report
00771          */
00772 
00773         double PcupS[WMM_NUMPCUPS];
00774 
00775         PcupS[0] = 1;
00776         double schmidtQuasiNorm1 = 1.0;
00777 
00778         MagneticResults->By = 0.0;
00779         double sin_phi = sin(CoordSpherical->phig * DEG2RAD);
00780 
00781         for (int n = 1; n <= MagneticModel.nMax; n++)
00782         {
00783             /*Compute the ration between the Gauss-normalized associated Legendre
00784                functions and the Schmidt quasi-normalized version. This is equivalent to
00785                sqrt((m==0?1:2)*(n-m)!/(n+m!))*(2n-1)!!/(n-m)!  */
00786 
00787             int index = (n * (n + 1) / 2 + 1);
00788             double schmidtQuasiNorm2 = schmidtQuasiNorm1 * (double)(2 * n - 1) / (double)n;
00789             double schmidtQuasiNorm3 = schmidtQuasiNorm2 * sqrt((double)(n * 2) / (double)(n + 1));
00790             schmidtQuasiNorm1 = schmidtQuasiNorm2;
00791             if (n == 1)
00792             {
00793                 PcupS[n] = PcupS[n - 1];
00794             }
00795             else
00796             {
00797                 double k = (double)(((n - 1) * (n - 1)) - 1) / (double)((2 * n - 1) * (2 * n - 3));
00798                 PcupS[n] = sin_phi * PcupS[n - 1] - k * PcupS[n - 2];
00799             }
00800 
00801 /*                1 nMax  (n+2)    n     m            m           m
00802     By =    SUM (a/r) (m)  SUM  [g cos(m p) + h sin(m p)] dP (sin(phi))
00803            n=1             m=0   n            n           n  */
00804 /* Equation 11 in the WMM Technical report. Derivative with respect to longitude, divided by radius. */
00805 
00806             MagneticResults->By +=
00807                 SphVariables->RelativeRadiusPower[n] *
00808                 (get_main_field_coeff_g(index) *
00809                  SphVariables->sin_mlambda[1] - get_main_field_coeff_h(index) * SphVariables->cos_mlambda[1])
00810                 * PcupS[n] * schmidtQuasiNorm3;
00811         }
00812     }

00962     void WorldMagModel::GeodeticToSpherical(WMMtype_CoordGeodetic *CoordGeodetic, WMMtype_CoordSpherical *CoordSpherical)
00963     {
00964         // Converts Geodetic coordinates to Spherical coordinates
00965         // Convert geodetic coordinates, (defined by the WGS-84
00966         // reference ellipsoid), to Earth Centered Earth Fixed Cartesian
00967         // coordinates, and then to spherical coordinates.
00968 
00969         double CosLat = cos(CoordGeodetic->phi * DEG2RAD);
00970         double SinLat = sin(CoordGeodetic->phi * DEG2RAD);
00971 
00972         // compute the local radius of curvature on the WGS-84 reference ellipsoid
00973         double rc = Ellip.a / sqrt(1.0 - Ellip.epssq * SinLat * SinLat);
00974 
00975         // compute ECEF Cartesian coordinates of specified point (for longitude=0)
00976         double xp = (rc + CoordGeodetic->HeightAboveEllipsoid) * CosLat;
00977         double zp = (rc * (1.0 - Ellip.epssq) + CoordGeodetic->HeightAboveEllipsoid) * SinLat;
00978 
00979         // compute spherical radius and angle lambda and phi of specified point
00980         CoordSpherical->r = sqrt(xp * xp + zp * zp);
00981         CoordSpherical->phig = asin(zp / CoordSpherical->r) * RAD2DEG;  // geocentric latitude
00982         CoordSpherical->lambda = CoordGeodetic->lambda; // longitude
00983     }
