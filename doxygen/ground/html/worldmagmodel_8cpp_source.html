137  int WorldMagModel::Geomag(WMMtype_CoordSpherical *CoordSpherical, WMMtype_CoordGeodetic *CoordGeodetic, WMMtype_GeoMagneticElements *GeoMagneticElements)
138  /*
139  The main subroutine that calls a sequence of WMM sub-functions to calculate the magnetic field elements for a single point.
140  The function expects the model coefficients and point coordinates as input and returns the magnetic field elements and
141  their rate of change. Though, this subroutine can be called successively to calculate a time series, profile or grid
142  of magnetic field, these are better achieved by the subroutine WMM_Grid.
143 
144  INPUT: Ellip
145  CoordSpherical
146  CoordGeodetic
147  TimedMagneticModel
148 
149  OUTPUT : GeoMagneticElements
150  */
151  {
152  WMMtype_MagneticResults MagneticResultsSph;
153  WMMtype_MagneticResults MagneticResultsGeo;
154  WMMtype_MagneticResults MagneticResultsSphVar;
155  WMMtype_MagneticResults MagneticResultsGeoVar;
156  WMMtype_LegendreFunction LegendreFunction;
157  WMMtype_SphericalHarmonicVariables SphVariables;
158 
159  // Compute Spherical Harmonic variables
160  ComputeSphericalHarmonicVariables(CoordSpherical, MagneticModel.nMax, &SphVariables);
161 
162  // Compute ALF
163  if (AssociatedLegendreFunction(CoordSpherical, MagneticModel.nMax, &LegendreFunction) < 0)
164  return -1; // error
165 
166  // Accumulate the spherical harmonic coefficients
167  Summation(&LegendreFunction, &SphVariables, CoordSpherical, &MagneticResultsSph);
168 
169  // Sum the Secular Variation Coefficients
170  SecVarSummation(&LegendreFunction, &SphVariables, CoordSpherical, &MagneticResultsSphVar);
171 
172  // Map the computed Magnetic fields to Geodeitic coordinates
173  RotateMagneticVector(CoordSpherical, CoordGeodetic, &MagneticResultsSph, &MagneticResultsGeo);
174 
175  // Map the secular variation field components to Geodetic coordinates
176  RotateMagneticVector(CoordSpherical, CoordGeodetic, &MagneticResultsSphVar, &MagneticResultsGeoVar);
177 
178  // Calculate the Geomagnetic elements, Equation 18 , WMM Technical report
179  CalculateGeoMagneticElements(&MagneticResultsGeo, GeoMagneticElements);
180 
181  // Calculate the secular variation of each of the Geomagnetic elements
182  CalculateSecularVariation(&MagneticResultsGeoVar, GeoMagneticElements);
183 
184  return 0; // OK
185  }

187  void WorldMagModel::ComputeSphericalHarmonicVariables(WMMtype_CoordSpherical *CoordSpherical, int nMax, WMMtype_SphericalHarmonicVariables *SphVariables)
188  {
189  /* Computes Spherical variables
190  Variables computed are (a/r)^(n+2), cos_m(lamda) and sin_m(lambda) for spherical harmonic
191  summations. (Equations 10-12 in the WMM Technical Report)
192  INPUT Ellip data structure with the following elements
193  float a; semi-major axis of the ellipsoid
194  float b; semi-minor axis of the ellipsoid
195  float fla; flattening
196  float epssq; first eccentricity squared
197  float eps; first eccentricity
198  float re; mean radius of ellipsoid
199  CoordSpherical A data structure with the following elements
200  float lambda; ( longitude)
201  float phig; ( geocentric latitude )
202  float r; ( distance from the center of the ellipsoid)
203  nMax integer ( Maxumum degree of spherical harmonic secular model)\
204 
205  OUTPUT SphVariables Pointer to the data structure with the following elements
206  float RelativeRadiusPower[WMM_MAX_MODEL_DEGREES+1]; [earth_reference_radius_km sph. radius ]^n
207  float cos_mlambda[WMM_MAX_MODEL_DEGREES+1]; cp(m) - cosine of (mspherical coord. longitude)
208  float sin_mlambda[WMM_MAX_MODEL_DEGREES+1]; sp(m) - sine of (mspherical coord. longitude)
209  */
210  double cos_lambda = cos(CoordSpherical->lambda * DEG2RAD);
211  double sin_lambda = sin(CoordSpherical->lambda * DEG2RAD);
212 
213  /* for n = 0 ... model_order, compute (Radius of Earth / Spherica radius r)^(n+2)
214  for n 1..nMax-1 (this is much faster than calling pow MAX_N+1 times). */
215 
216  SphVariables->RelativeRadiusPower[0] = (Ellip.re / CoordSpherical->r) * (Ellip.re / CoordSpherical->r);
217  for (int n = 1; n <= nMax; n++)
218  SphVariables->RelativeRadiusPower[n] = SphVariables->RelativeRadiusPower[n - 1] * (Ellip.re / CoordSpherical->r);
219 
220  /*
221  Compute cos(m*lambda), sin(m*lambda) for m = 0 ... nMax
222  cos(a + b) = cos(a)*cos(b) - sin(a)*sin(b)
223  sin(a + b) = cos(a)*sin(b) + sin(a)*cos(b)
224  */
225  SphVariables->cos_mlambda[0] = 1.0;
226  SphVariables->sin_mlambda[0] = 0.0;
227 
228  SphVariables->cos_mlambda[1] = cos_lambda;
229  SphVariables->sin_mlambda[1] = sin_lambda;
230  for (int m = 2; m <= nMax; m++)
231  {
232  SphVariables->cos_mlambda[m] = SphVariables->cos_mlambda[m - 1] * cos_lambda - SphVariables->sin_mlambda[m - 1] * sin_lambda;
233  SphVariables->sin_mlambda[m] = SphVariables->cos_mlambda[m - 1] * sin_lambda + SphVariables->sin_mlambda[m - 1] * cos_lambda;
234  }
235  }

237  int WorldMagModel::AssociatedLegendreFunction(WMMtype_CoordSpherical *CoordSpherical, int nMax, WMMtype_LegendreFunction *LegendreFunction)
238  {
239  /* Computes all of the Schmidt-semi normalized associated Legendre
240  functions up to degree nMax. If nMax <= 16, function WMM_PcupLow is used.
241  Otherwise WMM_PcupHigh is called.
242  INPUT CoordSpherical A data structure with the following elements
243  float lambda; ( longitude)
244  float phig; ( geocentric latitude )
245  float r; ( distance from the center of the ellipsoid)
246  nMax integer ( Maxumum degree of spherical harmonic secular model)
247  LegendreFunction Pointer to data structure with the following elements
248  float *Pcup; ( pointer to store Legendre Function )
249  float *dPcup; ( pointer to store Derivative of Lagendre function )
250 
251  OUTPUT LegendreFunction Calculated Legendre variables in the data structure
252  */
253 
254  double sin_phi = sin(CoordSpherical->phig * DEG2RAD); // sin (geocentric latitude)
255 
256  if (nMax <= 16 || (1 - fabs(sin_phi)) < 1.0e-10) /* If nMax is less tha 16 or at the poles */
257  PcupLow(LegendreFunction->Pcup, LegendreFunction->dPcup, sin_phi, nMax);
258  else
259  {
260  if (PcupHigh(LegendreFunction->Pcup, LegendreFunction->dPcup, sin_phi, nMax) < 0)
261  return -1; // error
262  }
263 
264  return 0; // OK
265  }

267  void WorldMagModel::Summation( WMMtype_LegendreFunction *LegendreFunction,
268  WMMtype_SphericalHarmonicVariables *SphVariables,
269  WMMtype_CoordSpherical *CoordSpherical,
270  WMMtype_MagneticResults *MagneticResults)
271  {
272  /* Computes Geomagnetic Field Elements X, Y and Z in Spherical coordinate system using spherical harmonic summation.
273 
274  The vector Magnetic field is given by -grad V, where V is Geomagnetic scalar potential
275  The gradient in spherical coordinates is given by:
276 
277  dV ^ 1 dV ^ 1 dV ^
278  grad V = -- r + - -- t + -------- -- p
279  dr r dt r sin(t) dp
280 
281  INPUT : LegendreFunction
282  MagneticModel
283  SphVariables
284  CoordSpherical
285  OUTPUT : MagneticResults
286 
287  Manoj Nair, June, 2009 Manoj.C.Nair@Noaa.Gov
288  */
289 
290  MagneticResults->Bz = 0.0;
291  MagneticResults->By = 0.0;
292  MagneticResults->Bx = 0.0;
293 
294  for (int n = 1; n <= MagneticModel.nMax; n++)
295  {
296  for (int m = 0; m <= n; m++)
297  {
298  int index = (n * (n + 1) / 2 + m);
299 
300 /* nMax (n+2) n m m m
301  Bz = -SUM (a/r) (n+1) SUM [g cos(m p) + h sin(m p)] P (sin(phi))
302  n=1 m=0 n n n */
303 /* Equation 12 in the WMM Technical report. Derivative with respect to radius.*/
304  MagneticResults->Bz -=
305  SphVariables->RelativeRadiusPower[n] *
306  (get_main_field_coeff_g(index) *
307  SphVariables->cos_mlambda[m] + get_main_field_coeff_h(index) * SphVariables->sin_mlambda[m])
308  * (double)(n + 1) * LegendreFunction->Pcup[index];
309 
310 /* 1 nMax (n+2) n m m m
311  By = SUM (a/r) (m) SUM [g cos(m p) + h sin(m p)] dP (sin(phi))
312  n=1 m=0 n n n */
313 /* Equation 11 in the WMM Technical report. Derivative with respect to longitude, divided by radius. */
314  MagneticResults->By +=
315  SphVariables->RelativeRadiusPower[n] *
316  (get_main_field_coeff_g(index) *
317  SphVariables->sin_mlambda[m] - get_main_field_coeff_h(index) * SphVariables->cos_mlambda[m])
318  * (double)(m) * LegendreFunction->Pcup[index];
319 /* nMax (n+2) n m m m
320  Bx = - SUM (a/r) SUM [g cos(m p) + h sin(m p)] dP (sin(phi))
321  n=1 m=0 n n n */
322 /* Equation 10 in the WMM Technical report. Derivative with respect to latitude, divided by radius. */
323 
324  MagneticResults->Bx -=
325  SphVariables->RelativeRadiusPower[n] *
326  (get_main_field_coeff_g(index) *
327  SphVariables->cos_mlambda[m] + get_main_field_coeff_h(index) * SphVariables->sin_mlambda[m])
328  * LegendreFunction->dPcup[index];
329 
330  }
331  }
332 
333  double cos_phi = cos(CoordSpherical->phig * DEG2RAD);
334  if (fabs(cos_phi) > 1.0e-10)
335  {
336  MagneticResults->By = MagneticResults->By / cos_phi;
337  }
338  else
339  {
340  /* Special calculation for component - By - at Geographic poles.
341  * If the user wants to avoid using this function, please make sure that
342  * the latitude is not exactly +/-90. An option is to make use the function
343  * WMM_CheckGeographicPoles.
344  */
345  SummationSpecial(SphVariables, CoordSpherical, MagneticResults);
346  }
347  }

349  void WorldMagModel::SecVarSummation( WMMtype_LegendreFunction *LegendreFunction,
350  WMMtype_SphericalHarmonicVariables *SphVariables,
351  WMMtype_CoordSpherical *CoordSpherical,
352  WMMtype_MagneticResults *MagneticResults)
353  {
354  /*This Function sums the secular variation coefficients to get the secular variation of the Magnetic vector.
355  INPUT : LegendreFunction
356  MagneticModel
357  SphVariables
358  CoordSpherical
359  OUTPUT : MagneticResults
360  */
361 
362  MagneticModel.SecularVariationUsed = true;
363 
364  MagneticResults->Bz = 0.0;
365  MagneticResults->By = 0.0;
366  MagneticResults->Bx = 0.0;
367 
368  for (int n = 1; n <= MagneticModel.nMaxSecVar; n++)
369  {
370  for (int m = 0; m <= n; m++)
371  {
372  int index = (n * (n + 1) / 2 + m);
373 
374 /* nMax (n+2) n m m m
375  Bz = -SUM (a/r) (n+1) SUM [g cos(m p) + h sin(m p)] P (sin(phi))
376  n=1 m=0 n n n */
377 /* Derivative with respect to radius.*/
378  MagneticResults->Bz -=
379  SphVariables->RelativeRadiusPower[n] *
380  (get_secular_var_coeff_g(index) *
381  SphVariables->cos_mlambda[m] + get_secular_var_coeff_h(index) * SphVariables->sin_mlambda[m])
382  * (double)(n + 1) * LegendreFunction->Pcup[index];
383 
384 /* 1 nMax (n+2) n m m m
385  By = SUM (a/r) (m) SUM [g cos(m p) + h sin(m p)] dP (sin(phi))
386  n=1 m=0 n n n */
387 /* Derivative with respect to longitude, divided by radius. */
388  MagneticResults->By +=
389  SphVariables->RelativeRadiusPower[n] *
390  (get_secular_var_coeff_g(index) *
391  SphVariables->sin_mlambda[m] - get_secular_var_coeff_h(index) * SphVariables->cos_mlambda[m])
392  * (double)(m) * LegendreFunction->Pcup[index];
393 /* nMax (n+2) n m m m
394  Bx = - SUM (a/r) SUM [g cos(m p) + h sin(m p)] dP (sin(phi))
395  n=1 m=0 n n n */
396 /* Derivative with respect to latitude, divided by radius. */
397 
398  MagneticResults->Bx -=
399  SphVariables->RelativeRadiusPower[n] *
400  (get_secular_var_coeff_g(index) *
401  SphVariables->cos_mlambda[m] + get_secular_var_coeff_h(index) * SphVariables->sin_mlambda[m])
402  * LegendreFunction->dPcup[index];
403  }
404  }
405 
406  double cos_phi = cos(CoordSpherical->phig * DEG2RAD);
407  if (fabs(cos_phi) > 1.0e-10)
408  {
409  MagneticResults->By = MagneticResults->By / cos_phi;
410  }
411  else
412  { /* Special calculation for component By at Geographic poles */
413  SecVarSummationSpecial(SphVariables, CoordSpherical, MagneticResults);
414  }
415  }

417  void WorldMagModel::RotateMagneticVector( WMMtype_CoordSpherical *CoordSpherical,
418  WMMtype_CoordGeodetic *CoordGeodetic,
419  WMMtype_MagneticResults *MagneticResultsSph,
420  WMMtype_MagneticResults *MagneticResultsGeo)
421  {
422  /* Rotate the Magnetic Vectors to Geodetic Coordinates
423  Manoj Nair, June, 2009 Manoj.C.Nair@Noaa.Gov
424  Equation 16, WMM Technical report
425 
426  INPUT : CoordSpherical : Data structure WMMtype_CoordSpherical with the following elements
427  float lambda; ( longitude)
428  float phig; ( geocentric latitude )
429  float r; ( distance from the center of the ellipsoid)
430 
431  CoordGeodetic : Data structure WMMtype_CoordGeodetic with the following elements
432  float lambda; (longitude)
433  float phi; ( geodetic latitude)
434  float HeightAboveEllipsoid; (height above the ellipsoid (HaE) )
435  float HeightAboveGeoid;(height above the Geoid )
436 
437  MagneticResultsSph : Data structure WMMtype_MagneticResults with the following elements
438  float Bx; North
439  float By; East
440  float Bz; Down
441 
442  OUTPUT: MagneticResultsGeo Pointer to the data structure WMMtype_MagneticResults, with the following elements
443  float Bx; North
444  float By; East
445  float Bz; Down
446  */
447 
448  /* Difference between the spherical and Geodetic latitudes */
449  double Psi = (CoordSpherical->phig - CoordGeodetic->phi) * DEG2RAD;
450 
451  /* Rotate spherical field components to the Geodeitic system */
452  MagneticResultsGeo->Bz = MagneticResultsSph->Bx * sin(Psi) + MagneticResultsSph->Bz * cos(Psi);
453  MagneticResultsGeo->Bx = MagneticResultsSph->Bx * cos(Psi) - MagneticResultsSph->Bz * sin(Psi);
454  MagneticResultsGeo->By = MagneticResultsSph->By;

457  void WorldMagModel::CalculateGeoMagneticElements(WMMtype_MagneticResults *MagneticResultsGeo, WMMtype_GeoMagneticElements *GeoMagneticElements)
458  {
459  /* Calculate all the Geomagnetic elements from X,Y and Z components
460  INPUT MagneticResultsGeo Pointer to data structure with the following elements
461  float Bx; ( North )
462  float By; ( East )
463  float Bz; ( Down )
464  OUTPUT GeoMagneticElements Pointer to data structure with the following elements
465  float Decl; (Angle between the magnetic field vector and true north, positive east)
466  float Incl; Angle between the magnetic field vector and the horizontal plane, positive down
467  float F; Magnetic Field Strength
468  float H; Horizontal Magnetic Field Strength
469  float X; Northern component of the magnetic field vector
470  float Y; Eastern component of the magnetic field vector
471  float Z; Downward component of the magnetic field vector
472  */
473 
474  GeoMagneticElements->X = MagneticResultsGeo->Bx;
475  GeoMagneticElements->Y = MagneticResultsGeo->By;
476  GeoMagneticElements->Z = MagneticResultsGeo->Bz;
477 
478  GeoMagneticElements->H = sqrt(MagneticResultsGeo->Bx * MagneticResultsGeo->Bx + MagneticResultsGeo->By * MagneticResultsGeo->By);
479  GeoMagneticElements->F = sqrt(GeoMagneticElements->H * GeoMagneticElements->H + MagneticResultsGeo->Bz * MagneticResultsGeo->Bz);
480  GeoMagneticElements->Decl = atan2(GeoMagneticElements->Y, GeoMagneticElements->X) * RAD2DEG;
481  GeoMagneticElements->Incl = atan2(GeoMagneticElements->Z, GeoMagneticElements->H) * RAD2DEG;
482  }

484  void WorldMagModel::CalculateSecularVariation(WMMtype_MagneticResults *MagneticVariation, WMMtype_GeoMagneticElements *MagneticElements)
485  {
486  /* This takes the Magnetic Variation in x, y, and z and uses it to calculate the secular variation of each of the Geomagnetic elements.
487  INPUT MagneticVariation Data structure with the following elements
488  float Bx; ( North )
489  float By; ( East )
490  float Bz; ( Down )
491  OUTPUT MagneticElements Pointer to the data structure with the following elements updated
492  float Decldot; Yearly Rate of change in declination
493  float Incldot; Yearly Rate of change in inclination
494  float Fdot; Yearly rate of change in Magnetic field strength
495  float Hdot; Yearly rate of change in horizontal field strength
496  float Xdot; Yearly rate of change in the northern component
497  float Ydot; Yearly rate of change in the eastern component
498  float Zdot; Yearly rate of change in the downward component
499  float GVdot;Yearly rate of chnage in grid variation
500  */
501 
502  MagneticElements->Xdot = MagneticVariation->Bx;
503  MagneticElements->Ydot = MagneticVariation->By;
504  MagneticElements->Zdot = MagneticVariation->Bz;
505  MagneticElements->Hdot = (MagneticElements->X * MagneticElements->Xdot + MagneticElements->Y * MagneticElements->Ydot) / MagneticElements->H; //See equation 19 in the WMM technical report
506  MagneticElements->Fdot =
507  (MagneticElements->X * MagneticElements->Xdot +
508  MagneticElements->Y * MagneticElements->Ydot + MagneticElements->Z * MagneticElements->Zdot) / MagneticElements->F;
509  MagneticElements->Decldot =
510  180.0 / M_PI * (MagneticElements->X * MagneticElements->Ydot -
511  MagneticElements->Y * MagneticElements->Xdot) / (MagneticElements->H * MagneticElements->H);
512  MagneticElements->Incldot =
513  180.0 / M_PI * (MagneticElements->H * MagneticElements->Zdot -
514  MagneticElements->Z * MagneticElements->Hdot) / (MagneticElements->F * MagneticElements->F);
515  MagneticElements->GVdot = MagneticElements->Decldot;
516  }

518  int WorldMagModel::PcupHigh(double *Pcup, double *dPcup, double x, int nMax)
519  {
520  /* This function evaluates all of the Schmidt-semi normalized associated Legendre
521  functions up to degree nMax. The functions are initially scaled by
522  10^280 sin^m in order to minimize the effects of underflow at large m
523  near the poles (see Holmes and Featherstone 2002, J. Geodesy, 76, 279-299).
524  Note that this function performs the same operation as WMM_PcupLow.
525  However this function also can be used for high degree (large nMax) models.
526 
527  Calling Parameters:
528  INPUT
529  nMax: Maximum spherical harmonic degree to compute.
530  x: cos(colatitude) or sin(latitude).
531 
532  OUTPUT
533  Pcup: A vector of all associated Legendgre polynomials evaluated at
534  x up to nMax. The lenght must by greater or equal to (nMax+1)*(nMax+2)/2.
535  dPcup: Derivative of Pcup(x) with respect to latitude
536  Notes:
537 
538  Adopted from the FORTRAN code written by Mark Wieczorek September 25, 2005.
539 
540  Manoj Nair, Nov, 2009 Manoj.C.Nair@Noaa.Gov
541 
542  Change from the previous version
543  The prevous version computes the derivatives as
544  dP(n,m)(x)/dx, where x = sin(latitude) (or cos(colatitude) ).
545  However, the WMM Geomagnetic routines requires dP(n,m)(x)/dlatitude.
546  Hence the derivatives are multiplied by sin(latitude).
547  Removed the options for CS phase and normalizations.
548 
549  Note: In geomagnetism, the derivatives of ALF are usually found with
550  respect to the colatitudes. Here the derivatives are found with respect
551  to the latitude. The difference is a sign reversal for the derivative of
552  the Associated Legendre Functions.
553 
554  The derivates can't be computed for latitude = |90| degrees.
555  */
556  double f1[WMM_NUMPCUP];
557  double f2[WMM_NUMPCUP];
558  double PreSqr[WMM_NUMPCUP];
559  int m;
560 
561  if (fabs(x) == 1.0)
562  {
563  // printf("Error in PcupHigh: derivative cannot be calculated at poles\n");
564  return -2;
565  }
566 
567  double scalef = 1.0e-280;
568 
569  for (int n = 0; n <= 2 * nMax + 1; ++n)
570  PreSqr[n] = sqrt((double)(n));
571 
572  int k = 2;
573 
574  for (int n = 2; n <= nMax; n++)
575  {
576  k = k + 1;
577  f1[k] = (double)(2 * n - 1) / n;
578  f2[k] = (double)(n - 1) / n;
579  for (int m = 1; m <= n - 2; m++)
580  {
581  k = k + 1;
582  f1[k] = (double)(2 * n - 1) / PreSqr[n + m] / PreSqr[n - m];
583  f2[k] = PreSqr[n - m - 1] * PreSqr[n + m - 1] / PreSqr[n + m] / PreSqr[n - m];
584  }
585  k = k + 2;
586  }
587 
588  /*z = sin (geocentric latitude) */
589  double z = sqrt((1.0 - x) * (1.0 + x));
590  double pm2 = 1.0;
591  Pcup[0] = 1.0;
592  dPcup[0] = 0.0;
593  if (nMax == 0)
594  return -3;
595  double pm1 = x;
596  Pcup[1] = pm1;
597  dPcup[1] = z;
598  k = 1;
599 
600  for (int n = 2; n <= nMax; n++)
601  {
602  k = k + n;
603  double plm = f1[k] * x * pm1 - f2[k] * pm2;
604  Pcup[k] = plm;
605  dPcup[k] = (double)(n) * (pm1 - x * plm) / z;
606  pm2 = pm1;
607  pm1 = plm;
608  }
609 
610  double pmm = PreSqr[2] * scalef;
611  double rescalem = 1.0 / scalef;
612  int kstart = 0;
613 
614  for (m = 1; m <= nMax - 1; ++m)
615  {
616  rescalem = rescalem * z;
617 
618  /* Calculate Pcup(m,m) */
619  kstart = kstart + m + 1;
620  pmm = pmm * PreSqr[2 * m + 1] / PreSqr[2 * m];
621  Pcup[kstart] = pmm * rescalem / PreSqr[2 * m + 1];
622  dPcup[kstart] = -((double)(m) * x * Pcup[kstart] / z);
623  pm2 = pmm / PreSqr[2 * m + 1];
624  /* Calculate Pcup(m+1,m) */
625  k = kstart + m + 1;
626  pm1 = x * PreSqr[2 * m + 1] * pm2;
627  Pcup[k] = pm1 * rescalem;
628  dPcup[k] = ((pm2 * rescalem) * PreSqr[2 * m + 1] - x * (double)(m + 1) * Pcup[k]) / z;
629  /* Calculate Pcup(n,m) */
630  for (int n = m + 2; n <= nMax; ++n)
631  {
632  k = k + n;
633  double plm = x * f1[k] * pm1 - f2[k] * pm2;
634  Pcup[k] = plm * rescalem;
635  dPcup[k] = (PreSqr[n + m] * PreSqr[n - m] * (pm1 * rescalem) - (double)(n) * x * Pcup[k]) / z;
636  pm2 = pm1;
637  pm1 = plm;
638  }
639  }
640 
641  /* Calculate Pcup(nMax,nMax) */
642  rescalem = rescalem * z;
643  kstart = kstart + m + 1;
644  pmm = pmm / PreSqr[2 * nMax];
645  Pcup[kstart] = pmm * rescalem;
646  dPcup[kstart] = -(double)(nMax) * x * Pcup[kstart] / z;
647 
648  // *********
649 
650  return 0; // OK
651  }

653  void WorldMagModel::PcupLow(double *Pcup, double *dPcup, double x, int nMax)
654  {
655  /* This function evaluates all of the Schmidt-semi normalized associated Legendre functions up to degree nMax.
656 
657  Calling Parameters:
658  INPUT
659  nMax: Maximum spherical harmonic degree to compute.
660  x: cos(colatitude) or sin(latitude).
661 
662  OUTPUT
663  Pcup: A vector of all associated Legendgre polynomials evaluated at
664  x up to nMax.
665  dPcup: Derivative of Pcup(x) with respect to latitude
666 
667  Notes: Overflow may occur if nMax > 20 , especially for high-latitudes.
668  Use WMM_PcupHigh for large nMax.
669 
670  Writted by Manoj Nair, June, 2009 . Manoj.C.Nair@Noaa.Gov.
671 
672  Note: In geomagnetism, the derivatives of ALF are usually found with
673  respect to the colatitudes. Here the derivatives are found with respect
674  to the latitude. The difference is a sign reversal for the derivative of
675  the Associated Legendre Functions.
676  */
677 
678  double schmidtQuasiNorm[WMM_NUMPCUP];
679 
680  Pcup[0] = 1.0;
681  dPcup[0] = 0.0;
682 
683  /*sin (geocentric latitude) - sin_phi */
684  double z = sqrt((1.0 - x) * (1.0 + x));
685 
686  /* First, Compute the Gauss-normalized associated Legendre functions */
687  for (int n = 1; n <= nMax; n++)
688  {
689  for (int m = 0; m <= n; m++)
690  {
691  int index = (n * (n + 1) / 2 + m);
692  if (n == m)
693  {
694  int index1 = (n - 1) * n / 2 + m - 1;
695  Pcup[index] = z * Pcup[index1];
696  dPcup[index] = z * dPcup[index1] + x * Pcup[index1];
697  }
698  else
699  if (n == 1 && m == 0)
700  {
701  int index1 = (n - 1) * n / 2 + m;
702  Pcup[index] = x * Pcup[index1];
703  dPcup[index] = x * dPcup[index1] - z * Pcup[index1];
704  }
705  else
706  if (n > 1 && n != m)
707  {
708  int index1 = (n - 2) * (n - 1) / 2 + m;
709  int index2 = (n - 1) * n / 2 + m;
710  if (m > n - 2)
711  {
712  Pcup[index] = x * Pcup[index2];
713  dPcup[index] = x * dPcup[index2] - z * Pcup[index2];
714  }
715  else
716  {
717  double k = (double)(((n - 1) * (n - 1)) - (m * m)) / (double)((2 * n - 1) * (2 * n - 3));
718  Pcup[index] = x * Pcup[index2] - k * Pcup[index1];
719  dPcup[index] = x * dPcup[index2] - z * Pcup[index2] - k * dPcup[index1];
720  }
721  }
722  }
723  }
724 
725  /*Compute the ration between the Gauss-normalized associated Legendre
726  functions and the Schmidt quasi-normalized version. This is equivalent to
727  sqrt((m==0?1:2)*(n-m)!/(n+m!))*(2n-1)!!/(n-m)! */
728 
729  schmidtQuasiNorm[0] = 1.0;
730  for (int n = 1; n <= nMax; n++)
731  {
732  int index = (n * (n + 1) / 2);
733  int index1 = (n - 1) * n / 2;
734  /* for m = 0 */
735  schmidtQuasiNorm[index] = schmidtQuasiNorm[index1] * (double)(2 * n - 1) / (double)n;
736 
737  for (int m = 1; m <= n; m++)
738  {
739  index = (n * (n + 1) / 2 + m);
740  index1 = (n * (n + 1) / 2 + m - 1);
741  schmidtQuasiNorm[index] = schmidtQuasiNorm[index1] * sqrt((double)((n - m + 1) * (m == 1 ? 2 : 1)) / (double)(n + m));
742  }
743  }
744 
745  /* Converts the Gauss-normalized associated Legendre
746  functions to the Schmidt quasi-normalized version using pre-computed
747  relation stored in the variable schmidtQuasiNorm */
748 
749  for (int n = 1; n <= nMax; n++)
750  {
751  for (int m = 0; m <= n; m++)
752  {
753  int index = (n * (n + 1) / 2 + m);
754  Pcup[index] = Pcup[index] * schmidtQuasiNorm[index];
755  dPcup[index] = -dPcup[index] * schmidtQuasiNorm[index];
756  /* The sign is changed since the new WMM routines use derivative with respect to latitude insted of co-latitude */
757  }
758  }
759  }

761  void WorldMagModel::SummationSpecial(WMMtype_SphericalHarmonicVariables *SphVariables, WMMtype_CoordSpherical *CoordSpherical, WMMtype_MagneticResults *MagneticResults)
762  {
763  /* Special calculation for the component By at Geographic poles.
764  Manoj Nair, June, 2009 manoj.c.nair@noaa.gov
765  INPUT: MagneticModel
766  SphVariables
767  CoordSpherical
768  OUTPUT: MagneticResults
769  CALLS : none
770  See Section 1.4, "SINGULARITIES AT THE GEOGRAPHIC POLES", WMM Technical report
771  */
772 
773  double PcupS[WMM_NUMPCUPS];
774 
775  PcupS[0] = 1;
776  double schmidtQuasiNorm1 = 1.0;
777 
778  MagneticResults->By = 0.0;
779  double sin_phi = sin(CoordSpherical->phig * DEG2RAD);
780 
781  for (int n = 1; n <= MagneticModel.nMax; n++)
782  {
783  /*Compute the ration between the Gauss-normalized associated Legendre
784  functions and the Schmidt quasi-normalized version. This is equivalent to
785  sqrt((m==0?1:2)*(n-m)!/(n+m!))*(2n-1)!!/(n-m)! */
786 
787  int index = (n * (n + 1) / 2 + 1);
788  double schmidtQuasiNorm2 = schmidtQuasiNorm1 * (double)(2 * n - 1) / (double)n;
789  double schmidtQuasiNorm3 = schmidtQuasiNorm2 * sqrt((double)(n * 2) / (double)(n + 1));
790  schmidtQuasiNorm1 = schmidtQuasiNorm2;
791  if (n == 1)
792  {
793  PcupS[n] = PcupS[n - 1];
794  }
795  else
796  {
797  double k = (double)(((n - 1) * (n - 1)) - 1) / (double)((2 * n - 1) * (2 * n - 3));
798  PcupS[n] = sin_phi * PcupS[n - 1] - k * PcupS[n - 2];
799  }
800 
801 /* 1 nMax (n+2) n m m m
802  By = SUM (a/r) (m) SUM [g cos(m p) + h sin(m p)] dP (sin(phi))
803  n=1 m=0 n n n */
804 /* Equation 11 in the WMM Technical report. Derivative with respect to longitude, divided by radius. */
805 
806  MagneticResults->By +=
807  SphVariables->RelativeRadiusPower[n] *
808  (get_main_field_coeff_g(index) *
809  SphVariables->sin_mlambda[1] - get_main_field_coeff_h(index) * SphVariables->cos_mlambda[1])
810  * PcupS[n] * schmidtQuasiNorm3;
811  }
812  }

962  void WorldMagModel::GeodeticToSpherical(WMMtype_CoordGeodetic *CoordGeodetic, WMMtype_CoordSpherical *CoordSpherical)
963  {
964  // Converts Geodetic coordinates to Spherical coordinates
965  // Convert geodetic coordinates, (defined by the WGS-84
966  // reference ellipsoid), to Earth Centered Earth Fixed Cartesian
967  // coordinates, and then to spherical coordinates.
968 
969  double CosLat = cos(CoordGeodetic->phi * DEG2RAD);
970  double SinLat = sin(CoordGeodetic->phi * DEG2RAD);
971 
972  // compute the local radius of curvature on the WGS-84 reference ellipsoid
973  double rc = Ellip.a / sqrt(1.0 - Ellip.epssq * SinLat * SinLat);
974 
975  // compute ECEF Cartesian coordinates of specified point (for longitude=0)
976  double xp = (rc + CoordGeodetic->HeightAboveEllipsoid) * CosLat;
977  double zp = (rc * (1.0 - Ellip.epssq) + CoordGeodetic->HeightAboveEllipsoid) * SinLat;
978 
979  // compute spherical radius and angle lambda and phi of specified point
980  CoordSpherical->r = sqrt(xp * xp + zp * zp);
981  CoordSpherical->phig = asin(zp / CoordSpherical->r) * RAD2DEG; // geocentric latitude
982  CoordSpherical->lambda = CoordGeodetic->lambda; // longitude
983  }
