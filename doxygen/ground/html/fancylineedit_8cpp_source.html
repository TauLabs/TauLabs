00098 bool FancyLineEditPrivate::eventFilter(QObject *obj, QEvent *event)
00099 {
00100     if (!m_menu || obj != m_menuLabel)
00101         return QObject::eventFilter(obj, event);
00102 
00103     switch (event->type()) {
00104     case QEvent::MouseButtonPress: {
00105         const QMouseEvent *me = static_cast<QMouseEvent *>(event);
00106         m_menu->exec(me->globalPos());
00107         return true;
00108     }
00109     case QEvent::FocusIn:
00110         if (m_menuTabFocusTrigger) {
00111             m_lineEdit->setFocus();
00112             m_menu->exec(m_menuLabel->mapToGlobal(m_menuLabel->rect().center()));
00113             return true;
00114         }
00115     default:
00116         break;
00117     }
00118     return QObject::eventFilter(obj, event);
00119 }
