00121

00123     if (state)

00125         mccInitialData = mdata;

00127         UAVObject::SetFlightTelemetryUpdateMode(mdata, UAVObject::UPDATEMODE_ONCHANGE);
00128         UAVObject::SetGcsTelemetryAcked(mdata, false);
00129         UAVObject::SetGcsTelemetryUpdateMode(mdata, UAVObject::UPDATEMODE_ONCHANGE);
00130         mdata.gcsTelemetryUpdatePeriod = 100;
00131         m_gcscontrol->checkBoxUDPControl->setEnabled(true);
00132 
00133     }
00134     else
00135     {
00136         mdata = mccInitialData;
00137         toggleUDPControl(false);
00138         m_gcscontrol->checkBoxUDPControl->setEnabled(false);
00139     }
00140     obj->setMetadata(mdata);
00141 }
