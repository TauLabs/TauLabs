37 UAVGadgetDecorator::UAVGadgetDecorator(IUAVGadget *gadget, QList<IUAVGadgetConfiguration*> *configurations) :
38  IUAVGadget(gadget->classId(), gadget->parent()),
39  m_gadget(gadget),
40  m_toolbar(new QComboBox),
41  m_activeConfiguration(0),
42  m_configurations(configurations)
43 {
44  m_gadget->setParent(this);
45  m_toolbar->setMinimumContentsLength(15);
46  foreach (IUAVGadgetConfiguration *config, *m_configurations)
47  m_toolbar->addItem(config->name());
48  connect(m_toolbar, SIGNAL(activated(int)), this, SLOT(loadConfiguration(int)));
49 
50  // If a gadget configuration exists, use the first configuration when

53  // Must call loadConfiguration(), or else GCS crashes when
54  // changing widgets using Edit Gadgets mode.
55  loadConfiguration(0);

58  updateToolbar();
59 }
