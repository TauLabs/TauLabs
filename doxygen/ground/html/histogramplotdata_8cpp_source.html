92 bool HistogramData::append(UAVObject* obj)
93 {
94 
95  //Empty histogram data set
96  xData->clear();
97  yData->clear();
98 
99  if (uavObjectName == obj->getName()) {
100 
101  //Get the field of interest
102  UAVObjectField* field = obj->getField(uavFieldName);
103 
104  //Bad place to do this
105  double step = binWidth;
106  if (step < 1e-6) //Don't allow step size to be 0.
107  step =1e-6;
108 
109  if (numberOfBins > MAX_NUMBER_OF_INTERVALS)
110  numberOfBins = MAX_NUMBER_OF_INTERVALS;
111 
112  if (field) {
113  double currentValue = valueAsDouble(obj, field, haveSubField, uavSubFieldName) * pow(10, scalePower);
114 
115  // Extend interval, if necessary
116  if(!histogramInterval->empty()){
117  while (currentValue < histogramInterval->front().minValue()
118  && histogramInterval->size() <= (int) numberOfBins){
119  histogramInterval->prepend(QwtInterval(histogramInterval->front().minValue() - step, histogramInterval->front().minValue()));
120  histogramBins->prepend(QwtIntervalSample(0,histogramInterval->front()));
121  }
122 
123  while (currentValue > histogramInterval->back().maxValue()
124  && histogramInterval->size() <= (int) numberOfBins){
125  histogramInterval->append(QwtInterval(histogramInterval->back().maxValue(), histogramInterval->back().maxValue() + step));
126  histogramBins->append(QwtIntervalSample(0,histogramInterval->back()));
127  }
128 
129  // If the histogram reaches its max size, pop one off the end and return
130  // This is a graceful way not to lock up the GCS if the bin width
131  // is inappropriate, or if there is an extremely distant outlier.
132  if (histogramInterval->size() > (int) numberOfBins )
133  {
134  histogramBins->pop_back();
135  histogramInterval->pop_back();
136  return false;
137  }
138 
139  // Test all intervals. This isn't particularly effecient, especially if we have just
140  // extended the interval and thus know for sure that the point lies on the extremity.
141  // On top of that, some kind of search by bisection would be better.
142  for (int i=0; i < histogramInterval->size(); i++ ){
143  if(histogramInterval->at(i).contains(currentValue)){
144  histogramBins->replace(i, QwtIntervalSample(histogramBins->at(i).value + 1, histogramInterval->at(i)));
145  break;
146  }
147 
148  }
149  }
150  else{
151  // Create first interval
152  double tmp=0;
153  if (tmp < currentValue){
154  while (tmp < currentValue){
155  tmp+=step;
156  }
157  histogramInterval->append(QwtInterval(tmp-step, tmp));
158  }
159  else{
160  while (tmp > step){
161  tmp-=step;
162  }
163  histogramInterval->append(QwtInterval(tmp, tmp+step));
164  }
165 
166  histogramBins->append(QwtIntervalSample(0,histogramInterval->front()));
167  }
168 
169 
170  return true;
171  }
172  }
173 
174  return false;
175 }
