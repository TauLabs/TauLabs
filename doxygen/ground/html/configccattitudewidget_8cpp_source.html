159 void ConfigCCAttitudeWidget::displayPlane(int position)
160 {
161  QString displayElement;
162  switch(position) {
163  case 1:
164  displayElement = "plane-horizontal";
165  break;
166  case 2:
167  displayElement = "plane-left";
168  break;
169  case 3:
170  displayElement = "plane-flip";
171  break;
172  case 4:
173  displayElement = "plane-right";
174  break;
175  case 5:
176  displayElement = "plane-up";
177  break;
178  case 6:
179  displayElement = "plane-down";
180  break;
181  default:
182  return;
183  }
184 
185  paperplane->setElementId(displayElement);
186  ui->sixPointHelp->setSceneRect(paperplane->boundingRect());
187  ui->sixPointHelp->fitInView(paperplane,Qt::KeepAspectRatio);
188 }
