00231 // Extract the wrapped text from a text edit, which performs
00232 // the wrapping only optically.
00233 static QString wrappedText(const QTextEdit *e)
00234 {
00235     const QChar newLine = QLatin1Char('\n');
00236     QString rc;
00237     QTextCursor cursor(e->document());
00238     cursor.movePosition(QTextCursor::Start);    
00239     while (!cursor.atEnd()) {
00240         cursor.select(QTextCursor::LineUnderCursor);
00241         rc += cursor.selectedText();
00242         rc += newLine;
00243         cursor.movePosition(QTextCursor::EndOfLine); // Mac needs it
00244         cursor.movePosition(QTextCursor::Right);
00245     }
00246     return rc;
00247 }
