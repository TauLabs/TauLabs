105 void TaskMonitorUpdateAll(void)
106 {
107 #if defined(DIAG_TASKS)
108  TaskInfoData data;
109  int n;
110 
111  // Lock
112  xSemaphoreTakeRecursive(lock, portMAX_DELAY);
113 
114 #if ( configGENERATE_RUN_TIME_STATS == 1 )
115  uint32_t currentTime;
116  uint32_t deltaTime;
117 
118  /*
119  * Calculate the amount of elapsed run time between the last time we
120  * measured and now. Scale so that we can convert task run times
121  * directly to percentages.
122  */
123  currentTime = portGET_RUN_TIME_COUNTER_VALUE();
124  deltaTime = ((currentTime - lastMonitorTime) / 100) ? : 1; /* avoid divide-by-zero if the interval is too small */
125  lastMonitorTime = currentTime;
126 #endif
127 
128  // Update all task information
129  for (n = 0; n < TASKINFO_RUNNING_NUMELEM; ++n)
130  {
131  if (handles[n] != 0)
132  {
133  data.Running[n] = TASKINFO_RUNNING_TRUE;
134 #if defined(ARCH_POSIX) || defined(ARCH_WIN32)
135  data.StackRemaining[n] = 10000;
136 #else
137  data.StackRemaining[n] = uxTaskGetStackHighWaterMark(handles[n]) * 4;
138 #endif
139 #if ( configGENERATE_RUN_TIME_STATS == 1 )
140  /* Generate run time stats */
141  data.RunningTime[n] = uxTaskGetRunTime(handles[n]) / deltaTime;
142 #endif
143 
144  }
145  else
146  {
147  data.Running[n] = TASKINFO_RUNNING_FALSE;
148  data.StackRemaining[n] = 0;
149  data.RunningTime[n] = 0;
150  }
151  }
152 
153  // Update object
154  TaskInfoSet(&data);
155 
156  // Done
157  xSemaphoreGiveRecursive(lock);
158 #endif
159 }
