309 static void prvProcessExpiredTimer( portTickType xNextExpireTime, portTickType xTimeNow )
310 {
311 xTIMER *pxTimer;
312 portBASE_TYPE xResult;
313 
314  /* Remove the timer from the list of active timers. A check has already
315  been performed to ensure the list is not empty. */
316  pxTimer = ( xTIMER * ) listGET_OWNER_OF_HEAD_ENTRY( pxCurrentTimerList );
317  uxListRemove( &( pxTimer->xTimerListItem ) );
318  traceTIMER_EXPIRED( pxTimer );
319 
320  /* If the timer is an auto reload timer then calculate the next
321  expiry time and re-insert the timer in the list of active timers. */
322  if( pxTimer->uxAutoReload == ( unsigned portBASE_TYPE ) pdTRUE )
323  {
324  /* This is the only time a timer is inserted into a list using
325  a time relative to anything other than the current time. It
326  will therefore be inserted into the correct list relative to
327  the time this task thinks it is now, even if a command to
328  switch lists due to a tick count overflow is already waiting in
329  the timer queue. */
330  if( prvInsertTimerInActiveList( pxTimer, ( xNextExpireTime + pxTimer->xTimerPeriodInTicks ), xTimeNow, xNextExpireTime ) == pdTRUE )
331  {
332  /* The timer expired before it was added to the active timer
333  list. Reload it now. */
334  xResult = xTimerGenericCommand( pxTimer, tmrCOMMAND_START, xNextExpireTime, NULL, tmrNO_DELAY );
335  configASSERT( xResult );
336  ( void ) xResult;
337  }
338  }
339 
340  /* Call the timer callback. */
341  pxTimer->pxCallbackFunction( ( xTimerHandle ) pxTimer );
342 }

345 static void prvTimerTask( void *pvParameters )
346 {
347 portTickType xNextExpireTime;
348 portBASE_TYPE xListWasEmpty;
349 
350  /* Just to avoid compiler warnings. */
351  ( void ) pvParameters;
352 
353  for( ;; )
354  {
355  /* Query the timers list to see if it contains any timers, and if so,
356  obtain the time at which the next timer will expire. */
357  xNextExpireTime = prvGetNextExpireTime( &xListWasEmpty );
358 
359  /* If a timer has expired, process it. Otherwise, block this task
360  until either a timer does expire, or a command is received. */
361  prvProcessTimerOrBlockTask( xNextExpireTime, xListWasEmpty );
362 
363  /* Empty the command queue. */
364  prvProcessReceivedCommands();
365  }
366 }

369 static void prvProcessTimerOrBlockTask( portTickType xNextExpireTime, portBASE_TYPE xListWasEmpty )
370 {
371 portTickType xTimeNow;
372 portBASE_TYPE xTimerListsWereSwitched;
373 
374  vTaskSuspendAll();
375  {
376  /* Obtain the time now to make an assessment as to whether the timer
377  has expired or not. If obtaining the time causes the lists to switch
378  then don't process this timer as any timers that remained in the list
379  when the lists were switched will have been processed within the
380  prvSampelTimeNow() function. */
381  xTimeNow = prvSampleTimeNow( &xTimerListsWereSwitched );
382  if( xTimerListsWereSwitched == pdFALSE )
383  {
384  /* The tick count has not overflowed, has the timer expired? */
385  if( ( xListWasEmpty == pdFALSE ) && ( xNextExpireTime <= xTimeNow ) )
386  {
387  xTaskResumeAll();
388  prvProcessExpiredTimer( xNextExpireTime, xTimeNow );
389  }
390  else
391  {
392  /* The tick count has not overflowed, and the next expire
393  time has not been reached yet. This task should therefore
394  block to wait for the next expire time or a command to be
395  received - whichever comes first. The following line cannot
396  be reached unless xNextExpireTime > xTimeNow, except in the
397  case when the current timer list is empty. */
398  vQueueWaitForMessageRestricted( xTimerQueue, ( xNextExpireTime - xTimeNow ) );
399 
400  if( xTaskResumeAll() == pdFALSE )
401  {
402  /* Yield to wait for either a command to arrive, or the block time
403  to expire. If a command arrived between the critical section being
404  exited and this yield then the yield will not cause the task
405  to block. */
406  portYIELD_WITHIN_API();
407  }
408  }
409  }
410  else
411  {
412  xTaskResumeAll();
413  }
414  }
415 }

418 static portTickType prvGetNextExpireTime( portBASE_TYPE *pxListWasEmpty )
419 {
420 portTickType xNextExpireTime;
421 
422  /* Timers are listed in expiry time order, with the head of the list
423  referencing the task that will expire first. Obtain the time at which
424  the timer with the nearest expiry time will expire. If there are no
425  active timers then just set the next expire time to 0. That will cause
426  this task to unblock when the tick count overflows, at which point the
427  timer lists will be switched and the next expiry time can be
428  re-assessed. */
429  *pxListWasEmpty = listLIST_IS_EMPTY( pxCurrentTimerList );
430  if( *pxListWasEmpty == pdFALSE )
431  {
432  xNextExpireTime = listGET_ITEM_VALUE_OF_HEAD_ENTRY( pxCurrentTimerList );
433  }
434  else
435  {
436  /* Ensure the task unblocks when the tick count rolls over. */
437  xNextExpireTime = ( portTickType ) 0U;
438  }
439 
440  return xNextExpireTime;
441 }

445 {
446 portTickType xTimeNow;
447 PRIVILEGED_DATA static portTickType xLastTime = ( portTickType ) 0U;
448 
449  xTimeNow = xTaskGetTickCount();

467 static portBASE_TYPE prvInsertTimerInActiveList( xTIMER *pxTimer, portTickType xNextExpiryTime, portTickType xTimeNow, portTickType xCommandTime )
468 {
469 portBASE_TYPE xProcessTimerNow = pdFALSE;
470 
471  listSET_LIST_ITEM_VALUE( &( pxTimer->xTimerListItem ), xNextExpiryTime );
472  listSET_LIST_ITEM_OWNER( &( pxTimer->xTimerListItem ), pxTimer );
473 
474  if( xNextExpiryTime <= xTimeNow )
475  {
476  /* Has the expiry time elapsed between the command to start/reset a
477  timer was issued, and the time the command was processed? */
478  if( ( ( portTickType ) ( xTimeNow - xCommandTime ) ) >= pxTimer->xTimerPeriodInTicks )
479  {
480  /* The time between a command being issued and the command being
481  processed actually exceeds the timers period. */
482  xProcessTimerNow = pdTRUE;
483  }
484  else
485  {
486  vListInsert( pxOverflowTimerList, &( pxTimer->xTimerListItem ) );
487  }
488  }
489  else
490  {
491  if( ( xTimeNow < xCommandTime ) && ( xNextExpiryTime >= xCommandTime ) )
492  {
493  /* If, since the command was issued, the tick count has overflowed
494  but the expiry time has not, then the timer must have already passed
495  its expiry time and should be processed immediately. */
496  xProcessTimerNow = pdTRUE;
497  }
498  else
499  {
500  vListInsert( pxCurrentTimerList, &( pxTimer->xTimerListItem ) );
501  }
502  }
503 
504  return xProcessTimerNow;
505 }

508 static void prvProcessReceivedCommands( void )
509 {
510 xTIMER_MESSAGE xMessage;
511 xTIMER *pxTimer;
512 portBASE_TYPE xTimerListsWereSwitched, xResult;
513 portTickType xTimeNow;
514 
515  /* In this case the xTimerListsWereSwitched parameter is not used, but it
516  must be present in the function call. */
517  xTimeNow = prvSampleTimeNow( &xTimerListsWereSwitched );
518 
519  while( xQueueReceive( xTimerQueue, &xMessage, tmrNO_DELAY ) != pdFAIL )
520  {
521  pxTimer = xMessage.pxTimer;
522 
523  /* Is the timer already in a list of active timers? When the command
524  is trmCOMMAND_PROCESS_TIMER_OVERFLOW, the timer will be NULL as the
525  command is to the task rather than to an individual timer. */
526  if( pxTimer != NULL )
527  {
528  if( listIS_CONTAINED_WITHIN( NULL, &( pxTimer->xTimerListItem ) ) == pdFALSE )
529  {
530  /* The timer is in a list, remove it. */
531  uxListRemove( &( pxTimer->xTimerListItem ) );
532  }
533  }
534 
535  traceTIMER_COMMAND_RECEIVED( pxTimer, xMessage.xMessageID, xMessage.xMessageValue );
536 
537  switch( xMessage.xMessageID )
538  {
539  case tmrCOMMAND_START :
540  /* Start or restart a timer. */
541  if( prvInsertTimerInActiveList( pxTimer, xMessage.xMessageValue + pxTimer->xTimerPeriodInTicks, xTimeNow, xMessage.xMessageValue ) == pdTRUE )
542  {
543  /* The timer expired before it was added to the active timer
544  list. Process it now. */
545  pxTimer->pxCallbackFunction( ( xTimerHandle ) pxTimer );
546 
547  if( pxTimer->uxAutoReload == ( unsigned portBASE_TYPE ) pdTRUE )
548  {
549  xResult = xTimerGenericCommand( pxTimer, tmrCOMMAND_START, xMessage.xMessageValue + pxTimer->xTimerPeriodInTicks, NULL, tmrNO_DELAY );
550  configASSERT( xResult );
551  ( void ) xResult;
552  }
553  }
554  break;
555 
556  case tmrCOMMAND_STOP :
557  /* The timer has already been removed from the active list.
558  There is nothing to do here. */
559  break;
560 
561  case tmrCOMMAND_CHANGE_PERIOD :
562  pxTimer->xTimerPeriodInTicks = xMessage.xMessageValue;
563  configASSERT( ( pxTimer->xTimerPeriodInTicks > 0 ) );
564  prvInsertTimerInActiveList( pxTimer, ( xTimeNow + pxTimer->xTimerPeriodInTicks ), xTimeNow, xTimeNow );
565  break;
566 
567  case tmrCOMMAND_DELETE :
568  /* The timer has already been removed from the active list,
569  just free up the memory. */
570  vPortFree( pxTimer );
571  break;
572 
573  default :
574  /* Don't expect to get here. */
575  break;
576  }
577  }
578 }

581 static void prvSwitchTimerLists( portTickType xLastTime )
582 {
583 portTickType xNextExpireTime, xReloadTime;
584 xList *pxTemp;
585 xTIMER *pxTimer;
586 portBASE_TYPE xResult;
587 
588  /* Remove compiler warnings if configASSERT() is not defined. */
589  ( void ) xLastTime;
590 
591  /* The tick count has overflowed. The timer lists must be switched.
592  If there are any timers still referenced from the current timer list
593  then they must have expired and should be processed before the lists
594  are switched. */
595  while( listLIST_IS_EMPTY( pxCurrentTimerList ) == pdFALSE )
596  {
597  xNextExpireTime = listGET_ITEM_VALUE_OF_HEAD_ENTRY( pxCurrentTimerList );
598 
599  /* Remove the timer from the list. */
600  pxTimer = ( xTIMER * ) listGET_OWNER_OF_HEAD_ENTRY( pxCurrentTimerList );
601  uxListRemove( &( pxTimer->xTimerListItem ) );
602 
603  /* Execute its callback, then send a command to restart the timer if
604  it is an auto-reload timer. It cannot be restarted here as the lists
605  have not yet been switched. */
606  pxTimer->pxCallbackFunction( ( xTimerHandle ) pxTimer );
607 
608  if( pxTimer->uxAutoReload == ( unsigned portBASE_TYPE ) pdTRUE )
609  {
610  /* Calculate the reload value, and if the reload value results in
611  the timer going into the same timer list then it has already expired
612  and the timer should be re-inserted into the current list so it is
613  processed again within this loop. Otherwise a command should be sent
614  to restart the timer to ensure it is only inserted into a list after
615  the lists have been swapped. */
616  xReloadTime = ( xNextExpireTime + pxTimer->xTimerPeriodInTicks );
617  if( xReloadTime > xNextExpireTime )
618  {
619  listSET_LIST_ITEM_VALUE( &( pxTimer->xTimerListItem ), xReloadTime );
620  listSET_LIST_ITEM_OWNER( &( pxTimer->xTimerListItem ), pxTimer );
621  vListInsert( pxCurrentTimerList, &( pxTimer->xTimerListItem ) );
622  }
623  else
624  {
625  xResult = xTimerGenericCommand( pxTimer, tmrCOMMAND_START, xNextExpireTime, NULL, tmrNO_DELAY );
626  configASSERT( xResult );
627  ( void ) xResult;
628  }
629  }
630  }
631 
632  pxTemp = pxCurrentTimerList;
633  pxCurrentTimerList = pxOverflowTimerList;
634  pxOverflowTimerList = pxTemp;
635 }

659 {
660 portBASE_TYPE xTimerIsInActiveList;
661 xTIMER *pxTimer = ( xTIMER * ) xTimer;
662 
663  /* Is the timer in the list of active timers? */
664  taskENTER_CRITICAL();
665  {
666  /* Checking to see if it is in the NULL list in effect checks to see if
667  it is referenced from either the current or the overflow timer lists in
668  one go, but the logic has to be reversed, hence the '!'. */
669  xTimerIsInActiveList = !( listIS_CONTAINED_WITHIN( NULL, &( pxTimer->xTimerListItem ) ) );
670  }
671  taskEXIT_CRITICAL();
672
