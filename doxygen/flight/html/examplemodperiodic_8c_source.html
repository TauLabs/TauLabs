00076 static void exampleTask(void *parameters)
00077 {
00078         ExampleSettingsData settings;

00081         portTickType lastSysTime;

00083         // Main task loop
00084         lastSysTime = xTaskGetTickCount();
00085         while (1) {
00086                 // Update settings with latest value
00087                 ExampleSettingsGet(&settings);
00088 
00089                 // Get the object data
00090                 ExampleObject2Get(&data);
00091 
00092                 // Determine how to update the data
00093                 if (settings.StepDirection == EXAMPLESETTINGS_STEPDIRECTION_UP) {
00094                         step = settings.StepSize;
00095                 } else {
00096                         step = -settings.StepSize;
00097                 }
00098 
00099                 // Update the data
00100                 data.Field1 += step;
00101                 data.Field2 += step;
00102                 data.Field3 += step;
00103                 data.Field4[0] += step;
00104                 data.Field4[1] += step;
00105 
00106                 // Update the ExampleObject, after this function is called
00107                 // notifications to any other modules listening to that object
00108                 // will be sent and the GCS object will be updated through the
00109                 // telemetry link. All operations will take place asynchronously
00110                 // and the following call will return immediately.
00111                 ExampleObject2Set(&data);
00112 
00113                 // Since this module executes at fixed time intervals, we need to
00114                 // block the task until it is time for the next update.
00115                 // The settings field is in ms, to convert to RTOS ticks we need
