00460                 if (!lowThrottle) {

00463                                 case ARM_STATE_DISARMING_TIMEOUT:

00469                                 default:
00470                                         // Nothing needs to be done in the other states
00471                                         break;

00477                 if (settings->Arming == MANUALCONTROLSETTINGS_ARMING_ALWAYSARMED) {

00483                 // the state will not be changed when the throttle is not low

00486 
00487                 // Calc channel see assumptions7
00488                 int8_t sign = ((settings->Arming-MANUALCONTROLSETTINGS_ARMING_ROLLLEFT)%2) ? -1 : 1;
00489                 switch ( (settings->Arming-MANUALCONTROLSETTINGS_ARMING_ROLLLEFT)/2 ) {
00490                         case ARMING_CHANNEL_ROLL:    armingInputLevel = sign * cmd->Roll;    break;
00491                         case ARMING_CHANNEL_PITCH:   armingInputLevel = sign * cmd->Pitch;   break;
00492                         case ARMING_CHANNEL_YAW:     armingInputLevel = sign * cmd->Yaw;     break;
00493                 }

00496                 bool manualDisarm = false;
00497

00504                         case ARM_STATE_DISARMED:
00505                                 set_armed_if_changed(FLIGHTSTATUS_ARMED_DISARMED);
00506 
00507                                 if (manualArm) {
00508                                         armedDisarmStart = lastSysTime;
00509                                         arm_state = ARM_STATE_ARMING_MANUAL;
00510                                 }
00511 
00512                                 break;
00513 
00514                         case ARM_STATE_ARMING_MANUAL:
00515                                 set_armed_if_changed(FLIGHTSTATUS_ARMED_ARMING);
00516

00518                                         arm_state = ARM_STATE_ARMED;
00519                                 else if (!manualArm)
00520                                         arm_state = ARM_STATE_DISARMED;
00521                                 break;
00522 
00523                         case ARM_STATE_ARMED:
00524                                 // When we get here, the throttle is low,
00525                                 // we go immediately to disarming due to timeout, also when the disarming mechanism is not enabled

00529                                 break;

00533                                 if ((settings->ArmedTimeout != 0) && (timeDifferenceMs(armedDisarmStart, lastSysTime) > settings->ArmedTimeout))

00536                                 // Switch to disarming due to manual control when needed
00537                                 if (manualDisarm) {
00538                                         armedDisarmStart = lastSysTime;

00540                                 }
00541                                 break;

00662 static bool updateRcvrActivity(struct rcvr_activity_fsm * fsm)
00663 {
00664         bool activity_updated = false;
00665 
00666         if (fsm->group >= MANUALCONTROLSETTINGS_CHANNELGROUPS_NONE) {
00667                 /* We're out of range, reset things */
00668                 resetRcvrActivity(fsm);
00669         }
00670 
00671         extern uint32_t pios_rcvr_group_map[];
00672         if (!pios_rcvr_group_map[fsm->group]) {
00673                 /* Unbound group, skip it */
00674                 goto group_completed;
00675         }
00676 
00677         if (fsm->sample_count == 0) {
00678                 /* Take a sample of each channel in this group */
00679                 updateRcvrActivitySample(pios_rcvr_group_map[fsm->group],
00680                                         fsm->prev,
00681                                         NELEMENTS(fsm->prev));
00682                 fsm->sample_count++;
00683                 return (false);
00684         }
00685 
00686         /* Compare with previous sample */
00687         activity_updated = updateRcvrActivityCompare(pios_rcvr_group_map[fsm->group], fsm);
00688 
00689 group_completed:
00690         /* Reset the sample counter */
00691         fsm->sample_count = 0;
00692 
00693         /* Find the next active group, but limit search so we can't loop forever here */
00694         for (uint8_t i = 0; i < MANUALCONTROLSETTINGS_CHANNELGROUPS_NONE; i++) {
00695                 /* Move to the next group */
00696                 fsm->group++;
00697                 if (fsm->group >= MANUALCONTROLSETTINGS_CHANNELGROUPS_NONE) {
00698                         /* Wrap back to the first group */
00699                         fsm->group = 0;
00700                 }
00701                 if (pios_rcvr_group_map[fsm->group]) {
00702                         /* 
00703                          * Found an active group, take a sample here to avoid an
00704                          * extra 20ms delay in the main thread so we can speed up
00705                          * this algorithm.
00706                          */
00707                         updateRcvrActivitySample(pios_rcvr_group_map[fsm->group],
00708                                                 fsm->prev,
00709                                                 NELEMENTS(fsm->prev));
00710                         fsm->sample_count++;
00711                         break;
00712                 }
00713         }
00714 
00715         return (activity_updated);
00716 }
