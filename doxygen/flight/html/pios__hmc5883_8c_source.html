00396 int32_t PIOS_HMC5883_Test(void)
00397 {
00398         int32_t failed = 0;
00399         uint8_t registers[3] = {0,0,0};
00400         uint8_t status;
00401         uint8_t ctrl_a_read;
00402         uint8_t ctrl_b_read;    
00403         uint8_t mode_read;
00404         struct pios_sensor_mag_data values;
00405 
00406         /* Verify that ID matches (HMC5883 ID is null-terminated ASCII string "H43") */
00407         char id[4];
00408         PIOS_HMC5883_ReadID((uint8_t *)id);
00409         if((id[0] != 'H') || (id[1] != '4') || (id[2] != '3')) // Expect H43
00410                 return -1;
00411 
00412         /* Backup existing configuration */
00413         if (PIOS_HMC5883_Read(PIOS_HMC5883_CONFIG_REG_A,registers,3) != 0)
00414                 return -1;
00415 
00416         /* Stop the device and read out last value */
00417         PIOS_DELAY_WaitmS(10);
00418         if (PIOS_HMC5883_Write(PIOS_HMC5883_MODE_REG, PIOS_HMC5883_MODE_IDLE) != 0) 
00419                 return -1;
00420         if( PIOS_HMC5883_Read(PIOS_HMC5883_DATAOUT_STATUS_REG, &status,1) != 0)
00421                 return -1;
00422         if (PIOS_HMC5883_ReadMag(&values) != 0)
00423                 return -1;
00424 
00425         /*
00426          * Put HMC5883 into self test mode
00427          * This is done by placing measurement config into positive (0x01) or negative (0x10) bias
00428          * and then placing the mode register into single-measurement mode.  This causes the HMC5883
00429          * to create an artificial magnetic field of ~1.1 Gauss.
00430          *
00431          * If gain were PIOS_HMC5883_GAIN_2_5, for example, X and Y will read around +766 LSB
00432          * (1.16 Ga * 660 LSB/Ga) and Z would read around +713 LSB (1.08 Ga * 660 LSB/Ga)
00433          *
00434          * Changing measurement config back to PIOS_HMC5883_MEASCONF_NORMAL will leave self-test mode.
00435          */
00436         PIOS_DELAY_WaitmS(10);
00437         if (PIOS_HMC5883_Write(PIOS_HMC5883_CONFIG_REG_A, PIOS_HMC5883_MEASCONF_BIAS_POS | PIOS_HMC5883_ODR_15) != 0)
00438                 return -1;
00439         PIOS_DELAY_WaitmS(10);
00440         if (PIOS_HMC5883_Write(PIOS_HMC5883_CONFIG_REG_B, PIOS_HMC5883_GAIN_8_1) != 0) 
00441                 return -1;
00442         PIOS_DELAY_WaitmS(10);
00443         if (PIOS_HMC5883_Write(PIOS_HMC5883_MODE_REG, PIOS_HMC5883_MODE_SINGLE) != 0) 
00444                 return -1;
00445 
00446         /* Must wait for value to be updated */
00447         PIOS_DELAY_WaitmS(200);
00448 
00449         if (PIOS_HMC5883_ReadMag(&values) != 0)
00450                 return -1;
00451 
00452         PIOS_HMC5883_Read(PIOS_HMC5883_CONFIG_REG_A, &ctrl_a_read,1);
00453         PIOS_HMC5883_Read(PIOS_HMC5883_CONFIG_REG_B, &ctrl_b_read,1);
00454         PIOS_HMC5883_Read(PIOS_HMC5883_MODE_REG, &mode_read,1);
00455         PIOS_HMC5883_Read(PIOS_HMC5883_DATAOUT_STATUS_REG, &status,1);
00456 
00457         /* Restore backup configuration */
00458         PIOS_DELAY_WaitmS(10);
00459         if (PIOS_HMC5883_Write(PIOS_HMC5883_CONFIG_REG_A, registers[0]) != 0)
00460                 return -1;
00461         PIOS_DELAY_WaitmS(10);
00462         if (PIOS_HMC5883_Write(PIOS_HMC5883_CONFIG_REG_B, registers[1]) != 0) 
00463                 return -1;
00464         PIOS_DELAY_WaitmS(10);
00465         if (PIOS_HMC5883_Write(PIOS_HMC5883_MODE_REG, registers[2]) != 0) 
00466                 return -1;
00467 
00468         return failed;
00469 }
