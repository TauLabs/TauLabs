00101 /*
00102  * Correct sensor drift, using the DCM approach from W. Premerlani et. al
00103  */
00104 void Premerlani_GPS(float *accels, float *gyros, float Rbe[3][3], const float delT, bool GPS_Drift_Compensation, GlobalAttitudeVariables *glblAtt, float *omegaCorrP)
00105 {
00106         float errYaw_b[3] = { 0, 0, 0 };
00107         float errRollPitch_b[3] = { 0, 0, 0 };
00108         
00109         float normOmegaScalar = VectorMagnitude(gyros);
00110         
00111         //Correct roll-pitch drift via GPS and accelerometer
00112         //The math is derived from Roll-Pitch Gyro Drift Compensation, Rev.3, by W. Premerlani
00113 #if defined (PIOS_INCLUDE_GPS)
00114         if (drft->gpsPresent_flag && GPS_Drift_Compensation) {
00115                 float accels_e[3];
00116                 
00117                 //Rotate accelerometer readings into Earth frame. Note that we need to take the transpose of Rbe.
00118                 rot_mult(Rbe, accels, accels_e, TRUE);
00119                 
00120                 //Integrate accelerometer measurements in Earth frame
00121                 drft->accels_e_integrator[0] += accels_e[0] * delT;
00122                 drft->accels_e_integrator[1] += accels_e[1] * delT;
00123                 drft->accels_e_integrator[2] += accels_e[2] * delT;
00124                 
00125                 drft->delT_between_GPS += delT;
00126                 
00127                 //Check if the GPS has new information.
00128                 if (!
00129                     (drft->
00130                      gpsVelocityDataConsumption_flag & GPS_CONSUMED_BY_RPY)) {
00131                                  
00132                                  //Compute drift correction, errRollPitch_b, from GPS
00133                                  rollPitch_drift_GPS(Rbe, drft->accels_e_integrator,
00134                                                                                         drft->delT_between_GPS,
00135                                                                                         errRollPitch_b);
00136                                  
00137                                  //Reset integrator
00138                                  memset(drft->accels_e_integrator, 0,
00139                                                   sizeof(drft->accels_e_integrator));
00140                                  
00141                                  //Mark GPS data as consumed by this function
00142                                  drft->gpsVelocityDataConsumption_flag |=
00143                             GPS_CONSUMED_BY_RPY;
00144                                  
00145                                  drft->delT_between_GPS = 0;
00146                                  
00147                          }
00148         }
00149 #endif
00150         
00151         if (!GPS_Drift_Compensation) {
00152 #if defined (PIOS_INCLUDE_GPS) && 0 || defined (PIOS_INCLUDE_MAGNETOMETER)
00153                 if (!(drft->gpsVelocityDataConsumption_flag & GPS_CONSUMED_BY_Y)) {
00154                         // We're actually using new GPS data here, but it's already been stored in old by the previous function
00155                         yaw_drift_MagGPS(Rbe, true, drft->magNewData_flag, errYaw_b);   
00156                         
00157                         // Mark GPS data as consumed by this function
00158                         drft->gpsVelocityDataConsumption_flag |= GPS_CONSUMED_BY_Y;
00159                 } else {
00160                         // In addition to calculating the roll-pitch-yaw error, we can calculate yaw drift, errYaw_b, based on GPS and attitude data
00161                         // We're actually using new GPS data here, but it's already been stored in old by the previous function
00162                         yaw_drift_MagGPS(Rbe, false, drft->magNewData_flag, errYaw_b);  
00163                 }
00164                 
00165                 // Reset flag. Not the best place to do it, but it's messy anywhere else
00166                 if (drft->magNewData_flag) {
00167                         drft->magNewData_flag = false;
00168                 }
00169 #endif
00170                 //In addition, we can calculate roll-pitch error with only the aid of an accelerometer
00171 #if defined(PIOS_GPS_PROVIDES_AIRSPEED)
00172                 AirspeedActualData airspeedActualData;
00173                 AirspeedActualGet(&airspeedActualData);
00174                 float airspeed_tas = airspeedActualData.TrueAirspeed;
00175 #else
00176                 float airspeed_tas = 0;
00177 #endif
00178                 rollPitch_drift_accel(accels, gyros, Rbe, airspeed_tas,
00179                                                                          errRollPitch_b);
00180         }
00181         
00182         // Calculate gyro drift, based on all errors
00183         gyro_drift(gyros, errYaw_b, errRollPitch_b, normOmegaScalar, delT, omegaCorrP, drft->omegaCorrI);
00184         
00185         //Calculate final drift response
00186         gyros[0] += omegaCorrP[0] + drft->omegaCorrI[0];
00187         gyros[1] += omegaCorrP[1] + drft->omegaCorrI[1];
00188         gyros[2] += omegaCorrP[2] + drft->omegaCorrI[2];
00189         
00190         //Add 0.0001% of proportional error back into gyroscope bias offset. This keeps DC elements out of the raw gyroscope data.
00191         glblAtt->gyro_correct_int[0] += omegaCorrP[0] / 1000000.0f;
00192         glblAtt->gyro_correct_int[1] += omegaCorrP[1] / 1000000.0f;
00193         
00194         // Because most crafts wont get enough information from gravity to zero yaw gyro, we try
00195         // and make it average zero (weakly)
00196         glblAtt->gyro_correct_int[2] += -gyros[2] * glblAtt->yawBiasRate;
00197 }

00199 /*
00200  * Calculate the error, as indicated by the accelerometer. It's important to use the true airspeed here, and not CAS or EAS
00201  */
00202 void rollPitch_drift_accel(float accels[3], float gyros[3], float Rbe[3][3],
00203                                                                         float airspeed_tas, float *errRollPitch_b)
00204 {
00205         float g_ref[3];
00206         float errAccelsRollPitch_b[3];
00207         
00208         //Fuselage Z vector is simply the Z column of the rotation matrix
00209         float fuselageZ[3] = { Rbe[0][2], Rbe[1][2], Rbe[2][2] };
00210         
00211         //Calculate centripetal acceleration
00212         float acc_centripetal[3] =
00213         { 0, gyros[2] * airspeed_tas, -gyros[1] * airspeed_tas };
00214         
00215         //Combine measurements and centripetal accelerations. This should give the gravity reference vector
00216         g_ref[0] = accels[0];   //Remember that for fixed-wing, acc_centripetal[0] = 0
00217         g_ref[1] = accels[1] - acc_centripetal[1];
00218         g_ref[2] = accels[2] - acc_centripetal[2];
00219         
00220         //Normalize acceleration vector
00221         float normG = sqrtf(g_ref[0] * g_ref[0] + g_ref[1] * g_ref[1] + g_ref[2] * g_ref[2]);
00222         g_ref[0] /= normG;
00223         g_ref[1] /= normG;
00224         g_ref[2] /= normG;
00225         
00226         //Error is cross product of reference vector and estimated orientation. Reverse operation in order to have correct sign for application to error sum
00227         CrossProduct((const float *)fuselageZ, (const float *)g_ref, errAccelsRollPitch_b);
00228         
00229         //Add errors into global error vector
00230         errRollPitch_b[0] += errAccelsRollPitch_b[0] * drft->accelsKp;
00231         errRollPitch_b[1] += errAccelsRollPitch_b[1] * drft->accelsKp;
00232         errRollPitch_b[2] += errAccelsRollPitch_b[2] * drft->accelsKp;
00233 }

00235 /*
00236  * This function takes in a number of different error rotations, weights them, and applies
00237  *  them to the gyroscope readings. In addition, there is a function to increase the correction 
00238  *  at high speeds in order to keep the gyro from becoming "dizzy".
00239  */
00240 void gyro_drift(float gyro[3], float errYaw_b[3], float errRollPitch_b[3],
00241                                          float normOmegaScalar, float delT, float *omegaCorrP,
00242                                          float *omegaCorrI)
00243 {
00244         float kpyaw;
00245         float kprollpitch;
00246         
00247         // boost the KPs at high spin rate, to compensate for increased error due to calibration error
00248         // above 50 degrees/second, scale by rotation rate divided by 50. As per Fast Rotations, by William Premerlani. 
00249         if (normOmegaScalar < (50.0f)) {
00250                 kpyaw = drft->yawKp;
00251                 kprollpitch = drft->rollPitchKp;
00252         } else if (normOmegaScalar < (500.0f)) {
00253                 kpyaw = (normOmegaScalar / 50.0f) * drft->yawKp;
00254                 kprollpitch = (normOmegaScalar / 50.0f) * drft->rollPitchKp;
00255         } else {
00256                 kpyaw = 10.0f * drft->yawKp;
00257                 kprollpitch = 10.0f * drft->rollPitchKp;
00258         }
00259         
00260         //Compute proportional correction.
00261         omegaCorrP[0] = errRollPitch_b[0] * kprollpitch + errYaw_b[0] * kpyaw;
00262         omegaCorrP[1] = errRollPitch_b[1] * kprollpitch + errYaw_b[1] * kpyaw;
00263         omegaCorrP[2] = errRollPitch_b[2] * kprollpitch + errYaw_b[2] * kpyaw;
00264         
00265         // Compute integral correction. Turn off the offset integrator while spinning at high speeds,
00266         // it doesn't work in that case, and it only causes trouble.
00267         if (normOmegaScalar < MAXIMUM_SPIN_DCM_INTEGRAL) {
00268                 omegaCorrI[0] += (errRollPitch_b[0] * drft->rollPitchKi + errYaw_b[0] * drft->yawKi) * delT;
00269                 omegaCorrI[1] += (errRollPitch_b[1] * drft->rollPitchKi + errYaw_b[1] * drft->yawKi) * delT;
00270                 omegaCorrI[2] += (errRollPitch_b[2] * drft->rollPitchKi + errYaw_b[2] * drft->yawKi) * delT;
00271         }
00272 }

00274 /*
00275  * From Roll-Pitch Gyro Drift Compensation, Rev 3. William Premerlani, 2012.
00276  */
00277 static void rollPitch_drift_GPS(float Rbe[3][3], float accels_e_int[3],
00278                                                                  float delT_between_updates, float *errRollPitch_b)
00279 {
00280         float errRollPitch_e[3];
00281         float dGPSdt_e[3];
00282         
00283         GPSVelocityData gpsVelocity;
00284         GPSVelocityGet(&gpsVelocity);
00285         
00286         dGPSdt_e[0] = (gpsVelocity.North - drft->GPSV_old[0]) / delT_between_updates;
00287         dGPSdt_e[1] = (gpsVelocity.East - drft->GPSV_old[1]) / delT_between_updates;
00288         dGPSdt_e[2] = -GRAVITY + (gpsVelocity.Down - drft->GPSV_old[2]) / delT_between_updates;
00289         
00290         drft->GPSV_old[0] = gpsVelocity.North;
00291         drft->GPSV_old[1] = gpsVelocity.East;
00292         drft->GPSV_old[2] = gpsVelocity.Down;
00293         
00294         float normdGPSdt_e = VectorMagnitude(dGPSdt_e);
00295         
00296         //Take cross product of integrated accelerometer measurements with integrated earth frame accelerations. We should be using normalized dGPSdt, but we perform that calculation in the following line(s).
00297         CrossProduct((const float *)accels_e_int, (const float *)dGPSdt_e, errRollPitch_e);
00298         
00299         //Scale cross product
00300         errRollPitch_e[0] /= (normdGPSdt_e * delT_between_updates);
00301         errRollPitch_e[1] /= (normdGPSdt_e * delT_between_updates);
00302         errRollPitch_e[2] /= (normdGPSdt_e * delT_between_updates);
00303         
00304         //Rotate earth drift error back into body frame;
00305         rot_mult(Rbe, errRollPitch_e, errRollPitch_b, FALSE);
00306 }

00312 void yaw_drift_MagGPS(float Rbe[3][3], bool gpsNewData_flag, bool magNewData_flag, float *errYaw_b)

00314 #if defined (PIOS_INCLUDE_GPS) || defined (PIOS_INCLUDE_MAGNETOMETER)

00316         //      Form the horizontal direction over ground based on rmat
00317         float horizDirOverGndRmat[3];
00318         
00319         //Define forward vector in earth frame, Rbe' * [1;0;0], and eliminate vertical component from vector
00320         horizDirOverGndRmat[0] = Rbe[0][0];
00321         horizDirOverGndRmat[1] = Rbe[0][1];
00322         horizDirOverGndRmat[2] = 0;
00323         
00324 #if defined (PIOS_INCLUDE_GPS)
00325         if (gpsNewData_flag) {
00326                 
00327                 GPSVelocityData gpsVelocity;
00328                 GPSVelocityGet(&gpsVelocity);
00329                 
00330                 if (fabs(gpsVelocity.North) > GPS_SPEED_MIN
00331                     || fabs(gpsVelocity.East) > GPS_SPEED_MIN) {
00332                         float errorGPSYaw_e;
00333                         float errorGPSYaw_b[3];
00334                         float normGPS =
00335                         sqrtf(gpsVelocity.North * gpsVelocity.North +
00336                                         gpsVelocity.East * gpsVelocity.East);
00337                         float horizDirOverGndGPS[3] = { gpsVelocity.North / normGPS, gpsVelocity.East / normGPS, 0 };   //Normalized vector
00338                         
00339                         // vector cross product to get the rotation error in ground frame. However, save several processor cycles by 
00340                         // condensing the math to take advantage of the cross products null entries on the x and y elements.
00341                         // errorGPSYaw_e = horizDirOverGndRmat X horizDirOverGndGPS
00342                         errorGPSYaw_e =
00343                         horizDirOverGndRmat[0] * horizDirOverGndGPS[1] -
00344                         horizDirOverGndGPS[0] * horizDirOverGndRmat[1];
00345                         
00346                         // Rotate error to body frame. Again, take advantage of the yaw error vector [0;0;errorGPSYaw_e]'s null entries
00347                         // errorGPSYaw_b = Rbe * errorGPSYaw_e;
00348                         errorGPSYaw_b[0] = Rbe[0][2] * errorGPSYaw_e;
00349                         errorGPSYaw_b[1] = Rbe[1][2] * errorGPSYaw_e;
00350                         errorGPSYaw_b[2] = Rbe[2][2] * errorGPSYaw_e;
00351                         
00352                         //Sum error with existing error
00353                         errYaw_b[0] += errorGPSYaw_b[0] * GPS_YAW_KP;
00354                         errYaw_b[1] += errorGPSYaw_b[1] * GPS_YAW_KP;
00355                         errYaw_b[2] += errorGPSYaw_b[2] * GPS_YAW_KP;
00356                 }
00357         }
00358 #endif
00359         
00360         // TODO: Create a flag to indicate whether mag data is present (or check for updates of MagnetometerData)
00361 #if defined (PIOS_INCLUDE_MAGNETOMETER)
00362         if (magNewData_flag) {
00363                 MagnetometerData magnetometerData;
00364                 MagnetometerGet(&magnetometerData);
00365                 
00366                 float errorMagYaw_e;
00367                 float errorMagYaw_b[3];

00371                 // Rotate magnetometer to earth frame and eliminate vertical component

00392                 errorMagYaw_b[2] = Rbe[2][2] * errorMagYaw_e;

00394                 errYaw_b[0] += errorMagYaw_b[0] * MAG_YAW_KP;
00395                 errYaw_b[1] += errorMagYaw_b[1] * MAG_YAW_KP;
