46 void PIOS_DEBUG_Init(const struct pios_tim_channel * channels, uint8_t num_channels)
47 {
48 #ifdef PIOS_ENABLE_DEBUG_PINS
49  PIOS_Assert(channels);
50  PIOS_Assert(num_channels);
51 
52  /* Store away the GPIOs we've been given */
53  debug_channels = channels;
54  debug_num_channels = num_channels;
55 
56  /* Configure the GPIOs we've been given */
57  for (uint8_t i = 0; i < num_channels; i++) {
58  const struct pios_tim_channel * chan = &channels[i];
59 
60  // Initialise pins as standard output pins
61  GPIO_InitTypeDef GPIO_InitStructure;
62  GPIO_StructInit(&GPIO_InitStructure);
63  GPIO_InitStructure.GPIO_Mode = GPIO_Mode_Out_PP;
64  GPIO_InitStructure.GPIO_Speed = GPIO_Speed_50MHz;
65  GPIO_InitStructure.GPIO_Pin = chan->init->GPIO_Pin;
66 
67  /* Initialize the GPIO */
68  GPIO_Init(chan->init->port, &GPIO_InitStructure);
69 
70  /* Set the pin low */
71  GPIO_WriteBit(chan->init->port, chan->init->GPIO_Pin, Bit_RESET);
72  }
73 #endif // PIOS_ENABLE_DEBUG_PINS
74 }

113 void PIOS_DEBUG_PinValue8Bit(uint8_t value)
114 {
115 #ifdef PIOS_ENABLE_DEBUG_PINS
116  if (!debug_channels) {
117  return;
118  }
119 
120  uint32_t bsrr_l = ( ((~value)&0x0F)<<(16+6) ) | ((value & 0x0F)<<6);
121  uint32_t bsrr_h = ( ((~value)&0xF0)<<(16+6-4) ) | ((value & 0xF0)<<(6-4));
122 
123  PIOS_IRQ_Disable();

126  * This is sketchy since it assumes a particular ordering
