00090 /*
00091  * Calculate airspeed as a function of GPS groundspeed and vehicle attitude.
00092  *  From "IMU Wind Estimation (Theory)", by William Premerlani.
00093  *  The idea is that V_gps=V_air+V_wind. If we assume wind constant, => 
00094  *  V_gps_2-V_gps_1 = (V_air_2+V_wind_2) -(V_air_1+V_wind_1) = V_air_2 - V_air_1.
00095  *  If we assume airspeed constant, => V_gps_2-V_gps_1 = |V|*(f_2 - f1),
00096  *  where "f" is the fuselage vector in earth coordinates.
00097  *  We then solve for |V| = |V_gps_2-V_gps_1|/ |f_2 - f1|.
00098  */

00107                 float q[4] ={attData.q1, attData.q2, attData.q3, attData.q4};
00108                 
00109                 //Calculate rotation matrix
00110                 Quaternion2R(q, Rbe);
00111         }
00112         
00113         //Calculate the cos(angle) between the two fuselage basis vectors
00114         float cosDiff=(Rbe[0][0]*gps->RbeCol1_old[0]) + (Rbe[0][1]*gps->RbeCol1_old[1]) + (Rbe[0][2]*gps->RbeCol1_old[2]);
00115         
00116         //If there's more than a 5 degree difference between two fuselage measurements, then we have sufficient delta to continue.
00117         if (fabs(cosDiff) < cos(5.0f*DEG2RAD)) {
