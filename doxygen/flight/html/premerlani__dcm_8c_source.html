00108 /*
00109  * Correct sensor drift, using the DCM approach from W. Premerlani et. al
00110  */
00111 void Premerlani_DCM(float *accels, float *gyros, float Rbe[3][3], const float delT, bool GPS_Drift_Compensation, GlobalAttitudeVariables *glblAtt, float *omegaCorrP)
00112 {
00113         float errYaw_b[3] = { 0, 0, 0 };
00114         float errRollPitch_b[3] = { 0, 0, 0 };
00115         
00116         float normOmegaScalar = VectorMagnitude(gyros);
00117         
00118         //Correct roll-pitch drift via GPS and accelerometer
00119         //The math is derived from Roll-Pitch Gyro Drift Compensation, Rev.3, by W. Premerlani
00120 #if defined (PIOS_INCLUDE_GPS)
00121         if (drft->gpsPresent_flag && GPS_Drift_Compensation) {
00122                 float accels_e[3];
00123                 
00124                 //Rotate accelerometer readings into Earth frame. Note that we need to take the transpose of Rbe.
00125                 rot_mult(Rbe, accels, accels_e, TRUE);
00126                 
00127                 //Integrate accelerometer measurements in Earth frame
00128                 drft->accels_e_integrator[0] += accels_e[0] * delT;
00129                 drft->accels_e_integrator[1] += accels_e[1] * delT;
00130                 drft->accels_e_integrator[2] += accels_e[2] * delT;
00131                 
00132                 drft->delT_between_GPS += delT;
00133                 
00134                 //Check if the GPS has new information.
00135                 if (!
00136                     (drft->
00137                      gpsVelocityDataConsumption_flag & GPS_CONSUMED_BY_RPY)) {
00138                                  
00139                                  //Compute drift correction, errRollPitch_b, from GPS
00140                                  rollPitch_drift_GPS(Rbe, drft->accels_e_integrator,
00141                                                                                         drft->delT_between_GPS,
00142                                                                                         errRollPitch_b);
00143                                  
00144                                  //Reset integrator
00145                                  memset(drft->accels_e_integrator, 0,
00146                                                   sizeof(drft->accels_e_integrator));
00147                                  
00148                                  //Mark GPS data as consumed by this function
00149                                  drft->gpsVelocityDataConsumption_flag |=
00150                             GPS_CONSUMED_BY_RPY;
00151                                  
00152                                  drft->delT_between_GPS = 0;
00153                                  
00154                          }
00155         }
00156 #endif
00157         
00158         if (!GPS_Drift_Compensation) {
00159 #if defined (PIOS_INCLUDE_GPS) && 0 || defined (PIOS_INCLUDE_MAGNETOMETER)
00160                 if (!(drft->gpsVelocityDataConsumption_flag & GPS_CONSUMED_BY_Y)) {
00161                         // We're actually using new GPS data here, but it's already been stored in old by the previous function
00162                         yaw_drift_MagGPS(Rbe, true, drft->magNewData_flag, errYaw_b);   
00163                         
00164                         // Mark GPS data as consumed by this function
00165                         drft->gpsVelocityDataConsumption_flag |= GPS_CONSUMED_BY_Y;
00166                 } else {
00167                         // In addition to calculating the roll-pitch-yaw error, we can calculate yaw drift, errYaw_b, based on GPS and attitude data
00168                         // We're actually using new GPS data here, but it's already been stored in old by the previous function
00169                         yaw_drift_MagGPS(Rbe, false, drft->magNewData_flag, errYaw_b);  
00170                 }
00171                 
00172                 // Reset flag. Not the best place to do it, but it's messy anywhere else
00173                 if (drft->magNewData_flag) {
00174                         drft->magNewData_flag = false;
00175                 }
00176 #endif
00177                 //In addition, we can calculate roll-pitch error with only the aid of an accelerometer
00178 #if defined(PIOS_GPS_PROVIDES_AIRSPEED)
00179                 AirspeedActualData airspeedActualData;
00180                 AirspeedActualGet(&airspeedActualData);
00181                 float airspeed_tas = airspeedActualData.TrueAirspeed;
00182 #else
00183                 float airspeed_tas = 0;
00184 #endif
00185                 rollPitch_drift_accel(accels, gyros, Rbe, airspeed_tas,
00186                                                                          errRollPitch_b);
00187         }
00188         
00189         // Calculate gyro drift, based on all errors
00190         gyro_drift(gyros, errYaw_b, errRollPitch_b, normOmegaScalar, delT, omegaCorrP, drft->omegaCorrI);
00191         
00192         //Calculate final drift response
00193         gyros[0] += omegaCorrP[0] + drft->omegaCorrI[0];
00194         gyros[1] += omegaCorrP[1] + drft->omegaCorrI[1];
00195         gyros[2] += omegaCorrP[2] + drft->omegaCorrI[2];
00196         
00197         //Add 0.0001% of proportional error back into gyroscope bias offset. This keeps DC elements out of the raw gyroscope data.
00198         glblAtt->gyro_correct_int[0] += omegaCorrP[0] / 1000000.0f;
00199         glblAtt->gyro_correct_int[1] += omegaCorrP[1] / 1000000.0f;
00200         
00201         // Because most crafts wont get enough information from gravity to zero yaw gyro, we try
00202         // and make it average zero (weakly)
00203         glblAtt->gyro_correct_int[2] += -gyros[2] * glblAtt->yawBiasRate;
00204 }

00206 /*
00207  * Calculate the error, as indicated by the accelerometer. It's important to use the true airspeed here, and not CAS or EAS
00208  */
00209 static void rollPitch_drift_accel(float accels[3], float gyros[3], float Rbe[3][3],
00210                                                                         float airspeed_tas, float *errRollPitch_b)
00211 {
00212         float g_ref[3];
00213         float errAccelsRollPitch_b[3];
00214         
00215         //Fuselage Z vector is simply the Z column of the rotation matrix
00216         float fuselageZ[3] = { Rbe[0][2], Rbe[1][2], Rbe[2][2] };
00217         
00218         //Calculate centripetal acceleration
00219         float acc_centripetal[3] =
00220         { 0, gyros[2] * airspeed_tas, -gyros[1] * airspeed_tas };
00221         
00222         //Combine measurements and centripetal accelerations. This should give the gravity reference vector
00223         g_ref[0] = accels[0];   //Remember that for fixed-wing, acc_centripetal[0] = 0
00224         g_ref[1] = accels[1] - acc_centripetal[1];
00225         g_ref[2] = accels[2] - acc_centripetal[2];
00226         
00227         //Normalize acceleration vector
00228         float normG = sqrtf(g_ref[0] * g_ref[0] + g_ref[1] * g_ref[1] + g_ref[2] * g_ref[2]);
00229         g_ref[0] /= normG;
00230         g_ref[1] /= normG;
00231         g_ref[2] /= normG;
00232         
00233         //Error is cross product of reference vector and estimated orientation. Reverse operation in order to have correct sign for application to error sum
00234         CrossProduct((const float *)fuselageZ, (const float *)g_ref, errAccelsRollPitch_b);
00235         
00236         //Add errors into global error vector
00237         errRollPitch_b[0] += errAccelsRollPitch_b[0] * drft->accelsKp;
00238         errRollPitch_b[1] += errAccelsRollPitch_b[1] * drft->accelsKp;
00239         errRollPitch_b[2] += errAccelsRollPitch_b[2] * drft->accelsKp;
00240 }

00242 /*
00243  * This function takes in a number of different error rotations, weights them, and applies
00244  *  them to the gyroscope readings. In addition, there is a function to increase the correction 
00245  *  at high speeds in order to keep the gyro from becoming "dizzy".
00246  */
00247 static void gyro_drift(float gyro[3], float errYaw_b[3], float errRollPitch_b[3],
00248                                          float normOmegaScalar, float delT, float *omegaCorrP,
00249                                          float *omegaCorrI)
00250 {
00251         float kpyaw;
00252         float kprollpitch;
00253         
00254         // boost the KPs at high spin rate, to compensate for increased error due to calibration error
00255         // above 50 degrees/second, scale by rotation rate divided by 50. As per Fast Rotations, by William Premerlani. 
00256         if (normOmegaScalar < (50.0f)) {
00257                 kpyaw = drft->yawKp;
00258                 kprollpitch = drft->rollPitchKp;
00259         } else if (normOmegaScalar < (500.0f)) {
00260                 kpyaw = (normOmegaScalar / 50.0f) * drft->yawKp;
00261                 kprollpitch = (normOmegaScalar / 50.0f) * drft->rollPitchKp;
00262         } else {
00263                 kpyaw = 10.0f * drft->yawKp;
00264                 kprollpitch = 10.0f * drft->rollPitchKp;
00265         }
00266         
00267         //Compute proportional correction.
00268         omegaCorrP[0] = errRollPitch_b[0] * kprollpitch + errYaw_b[0] * kpyaw;
00269         omegaCorrP[1] = errRollPitch_b[1] * kprollpitch + errYaw_b[1] * kpyaw;
00270         omegaCorrP[2] = errRollPitch_b[2] * kprollpitch + errYaw_b[2] * kpyaw;
00271         
00272         // Compute integral correction. Turn off the offset integrator while spinning at high speeds,
00273         // it doesn't work in that case, and it only causes trouble.
00274         if (normOmegaScalar < MAXIMUM_SPIN_DCM_INTEGRAL) {
00275                 omegaCorrI[0] += (errRollPitch_b[0] * drft->rollPitchKi + errYaw_b[0] * drft->yawKi) * delT;
00276                 omegaCorrI[1] += (errRollPitch_b[1] * drft->rollPitchKi + errYaw_b[1] * drft->yawKi) * delT;
00277                 omegaCorrI[2] += (errRollPitch_b[2] * drft->rollPitchKi + errYaw_b[2] * drft->yawKi) * delT;
00278         }
00279 }

00282  * From Roll-Pitch Gyro Drift Compensation, Rev 3. William Premerlani, 2012.
00283  */
00284 static void rollPitch_drift_GPS(float Rbe[3][3], float accels_e_int[3],
00285                                                                  float delT_between_updates, float *errRollPitch_b)
00286 {
00287         float errRollPitch_e[3];
00288         float dGPSdt_e[3];
00289         
00290         GPSVelocityData gpsVelocity;
00291         GPSVelocityGet(&gpsVelocity);
00292         
00293         dGPSdt_e[0] = (gpsVelocity.North - drft->GPSV_old[0]) / delT_between_updates;
00294         dGPSdt_e[1] = (gpsVelocity.East - drft->GPSV_old[1]) / delT_between_updates;
00295         dGPSdt_e[2] = -GRAVITY + (gpsVelocity.Down - drft->GPSV_old[2]) / delT_between_updates;
00296         
00297         drft->GPSV_old[0] = gpsVelocity.North;
00298         drft->GPSV_old[1] = gpsVelocity.East;
00299         drft->GPSV_old[2] = gpsVelocity.Down;
00300         
00301         float normdGPSdt_e = VectorMagnitude(dGPSdt_e);
00302         
00303         //Take cross product of integrated accelerometer measurements with integrated earth frame accelerations. We should be using normalized dGPSdt, but we perform that calculation in the following line(s).
00304         CrossProduct((const float *)accels_e_int, (const float *)dGPSdt_e, errRollPitch_e);
00305         
00306         //Scale cross product
00307         errRollPitch_e[0] /= (normdGPSdt_e * delT_between_updates);
00308         errRollPitch_e[1] /= (normdGPSdt_e * delT_between_updates);
00309         errRollPitch_e[2] /= (normdGPSdt_e * delT_between_updates);
00310         
00311         //Rotate earth drift error back into body frame;
00312         rot_mult(Rbe, errRollPitch_e, errRollPitch_b, FALSE);
00313 }
