120 /*
121  * Correct sensor drift, using the DCM approach from W. Premerlani et. al
122  */
123 void Premerlani_DCM(float *accels, float *gyros, float Rbe[3][3], const float delT, bool GPS_Drift_Compensation, GlobalAttitudeVariables *glblAtt, float *omegaCorrP)
124 {
125  float errYaw_b[3] = { 0, 0, 0 };
126  float errRollPitch_b[3] = { 0, 0, 0 };
127 
128  float normOmegaScalar = VectorMagnitude(gyros);
129 
130  //Correct roll-pitch drift via GPS and accelerometer
131  //The math is derived from Roll-Pitch Gyro Drift Compensation, Rev.3, by W. Premerlani
132 #if defined (PIOS_INCLUDE_GPS)
133  if (drft->gpsPresent_flag && GPS_Drift_Compensation) {
134  float accels_e[3];
135 
136  //Rotate accelerometer readings into Earth frame. Note that we need to take the transpose of Rbe.
137  rot_mult(Rbe, accels, accels_e, TRUE);
138 
139  //Integrate accelerometer measurements in Earth frame
140  drft->accels_e_integrator[0] += accels_e[0] * delT;
141  drft->accels_e_integrator[1] += accels_e[1] * delT;
142  drft->accels_e_integrator[2] += accels_e[2] * delT;
143 
144  drft->delT_between_GPS += delT;
145 
146  //Check if the GPS has new information.
147  if (!
148  (drft->
149  gpsVelocityDataConsumption_flag & GPS_CONSUMED_BY_RPY)) {
150 
151  //Compute drift correction, errRollPitch_b, from GPS
152  rollPitch_drift_GPS(Rbe, drft->accels_e_integrator,
153  drft->delT_between_GPS,
154  errRollPitch_b);
155 
156  //Reset integrator
157  memset(drft->accels_e_integrator, 0,
158  sizeof(drft->accels_e_integrator));
159 
160  //Mark GPS data as consumed by this function
161  drft->gpsVelocityDataConsumption_flag |=
162  GPS_CONSUMED_BY_RPY;
163 
164  drft->delT_between_GPS = 0;
165 
166  }
167  }
168 #endif
169 
170  if (!GPS_Drift_Compensation) {
171 #if defined (PIOS_INCLUDE_GPS) && 0 || defined (PIOS_INCLUDE_MAGNETOMETER)
172  if (!(drft->gpsVelocityDataConsumption_flag & GPS_CONSUMED_BY_Y)) {
173  // We're actually using new GPS data here, but it's already been stored in old by the previous function
174  yaw_drift_MagGPS(Rbe, true, drft->magNewData_flag, errYaw_b);
175 
176  // Mark GPS data as consumed by this function
177  drft->gpsVelocityDataConsumption_flag |= GPS_CONSUMED_BY_Y;
178  } else {
179  // In addition to calculating the roll-pitch-yaw error, we can calculate yaw drift, errYaw_b, based on GPS and attitude data
180  // We're actually using new GPS data here, but it's already been stored in old by the previous function
181  yaw_drift_MagGPS(Rbe, false, drft->magNewData_flag, errYaw_b);
182  }
183 
184  // Reset flag. Not the best place to do it, but it's messy anywhere else
185  if (drft->magNewData_flag) {
186  drft->magNewData_flag = false;
187  }
188 #endif
189  //In addition, we can calculate roll-pitch error with only the aid of an accelerometer
190 #if defined(PIOS_GPS_PROVIDES_AIRSPEED)
191  AirspeedActualData airspeedActualData;
192  AirspeedActualGet(&airspeedActualData);
193  float airspeed_tas = airspeedActualData.TrueAirspeed;
194 #else
195  float airspeed_tas = 0;
196 #endif
197  rollPitch_drift_accel(accels, gyros, Rbe, airspeed_tas,
198  errRollPitch_b);
199  }
200 
201  // Calculate gyro drift, based on all errors
202  gyro_drift(gyros, errYaw_b, errRollPitch_b, normOmegaScalar, delT, omegaCorrP, drft->omegaCorrI);
203 
204  //Calculate final drift response
205  gyros[0] += omegaCorrP[0] + drft->omegaCorrI[0];
206  gyros[1] += omegaCorrP[1] + drft->omegaCorrI[1];
207  gyros[2] += omegaCorrP[2] + drft->omegaCorrI[2];
208 
209  //Add 0.0001% of proportional error back into gyroscope bias offset. This keeps DC elements out of the raw gyroscope data.
210  glblAtt->gyro_correct_int[0] += omegaCorrP[0] / 1000000.0f;
211  glblAtt->gyro_correct_int[1] += omegaCorrP[1] / 1000000.0f;
212 
213  // Because most crafts wont get enough information from gravity to zero yaw gyro, we try
214  // and make it average zero (weakly)
215  glblAtt->gyro_correct_int[2] += -gyros[2] * glblAtt->yawBiasRate;
216 }

218 /*
219  * Calculate the error, as indicated by the accelerometer. It's important to use the true airspeed here, and not CAS or EAS
220  */
221 static void rollPitch_drift_accel(float accels[3], float gyros[3], float Rbe[3][3],
222  float airspeed_tas, float *errRollPitch_b)
223 {
224  float g_ref[3];
225  float errAccelsRollPitch_b[3];
226 
227  //Fuselage Z vector is simply the Z column of the rotation matrix
228  float fuselageZ[3] = { Rbe[0][2], Rbe[1][2], Rbe[2][2] };
229 
230  //Calculate centripetal acceleration
231  float acc_centripetal[3] =
232  { 0, gyros[2] * airspeed_tas, -gyros[1] * airspeed_tas };
233 
234  //Combine measurements and centripetal accelerations. This should give the gravity reference vector
235  g_ref[0] = accels[0]; //Remember that for fixed-wing, acc_centripetal[0] = 0
236  g_ref[1] = accels[1] - acc_centripetal[1];
237  g_ref[2] = accels[2] - acc_centripetal[2];
238 
239  //Normalize acceleration vector
240  float normG = sqrtf(g_ref[0] * g_ref[0] + g_ref[1] * g_ref[1] + g_ref[2] * g_ref[2]);
241  g_ref[0] /= normG;
242  g_ref[1] /= normG;
243  g_ref[2] /= normG;
244 
245  //Error is cross product of reference vector and estimated orientation. Reverse operation in order to have correct sign for application to error sum
246  CrossProduct((const float *)fuselageZ, (const float *)g_ref, errAccelsRollPitch_b);
247 
248  //Add errors into global error vector
249  errRollPitch_b[0] += errAccelsRollPitch_b[0] * drft->accelsKp;
250  errRollPitch_b[1] += errAccelsRollPitch_b[1] * drft->accelsKp;
251  errRollPitch_b[2] += errAccelsRollPitch_b[2] * drft->accelsKp;
252 }

254 /*
255  * This function takes in a number of different error rotations, weights them, and applies
256  * them to the gyroscope readings. In addition, there is a function to increase the correction
257  * at high speeds in order to keep the gyro from becoming "dizzy".
258  */
259 static void gyro_drift(float gyro[3], float errYaw_b[3], float errRollPitch_b[3],
260  float normOmegaScalar, float delT, float *omegaCorrP,
261  float *omegaCorrI)
262 {
263  float kpyaw;
264  float kprollpitch;
265 
266  // boost the KPs at high spin rate, to compensate for increased error due to calibration error
267  // above 50 degrees/second, scale by rotation rate divided by 50. As per Fast Rotations, by William Premerlani.
268  if (normOmegaScalar < (50.0f)) {
269  kpyaw = drft->yawKp;
270  kprollpitch = drft->rollPitchKp;
271  } else if (normOmegaScalar < (500.0f)) {
272  kpyaw = (normOmegaScalar / 50.0f) * drft->yawKp;
273  kprollpitch = (normOmegaScalar / 50.0f) * drft->rollPitchKp;
274  } else {
275  kpyaw = 10.0f * drft->yawKp;
276  kprollpitch = 10.0f * drft->rollPitchKp;
277  }
278 
279  //Compute proportional correction.
280  omegaCorrP[0] = errRollPitch_b[0] * kprollpitch + errYaw_b[0] * kpyaw;
281  omegaCorrP[1] = errRollPitch_b[1] * kprollpitch + errYaw_b[1] * kpyaw;
282  omegaCorrP[2] = errRollPitch_b[2] * kprollpitch + errYaw_b[2] * kpyaw;
283 
284  // Compute integral correction. Turn off the offset integrator while spinning at high speeds,
285  // it doesn't work in that case, and it only causes trouble.
286  if (normOmegaScalar < MAXIMUM_SPIN_DCM_INTEGRAL) {
287  omegaCorrI[0] += (errRollPitch_b[0] * drft->rollPitchKi + errYaw_b[0] * drft->yawKi) * delT;
288  omegaCorrI[1] += (errRollPitch_b[1] * drft->rollPitchKi + errYaw_b[1] * drft->yawKi) * delT;
289  omegaCorrI[2] += (errRollPitch_b[2] * drft->rollPitchKi + errYaw_b[2] * drft->yawKi) * delT;
290  }
291 }

293 /*
294  * From Roll-Pitch Gyro Drift Compensation, Rev 3. William Premerlani, 2012.
295  */
296 static void rollPitch_drift_GPS(float Rbe[3][3], float accels_e_int[3],
297  float delT_between_updates, float *errRollPitch_b)
298 {
299  float errRollPitch_e[3];
300  float dGPSdt_e[3];
301 
302  GPSVelocityData gpsVelocity;
303  GPSVelocityGet(&gpsVelocity);
304 
305  dGPSdt_e[0] = (gpsVelocity.North - drft->GPSV_old[0]) / delT_between_updates;
306  dGPSdt_e[1] = (gpsVelocity.East - drft->GPSV_old[1]) / delT_between_updates;
307  dGPSdt_e[2] = -GRAVITY + (gpsVelocity.Down - drft->GPSV_old[2]) / delT_between_updates;
308 
309  drft->GPSV_old[0] = gpsVelocity.North;
310  drft->GPSV_old[1] = gpsVelocity.East;
311  drft->GPSV_old[2] = gpsVelocity.Down;
312 
313  float normdGPSdt_e = VectorMagnitude(dGPSdt_e);
314 
315  //Take cross product of integrated accelerometer measurements with integrated earth frame accelerations. We should be using normalized dGPSdt, but we perform that calculation in the following line(s).
316  CrossProduct((const float *)accels_e_int, (const float *)dGPSdt_e, errRollPitch_e);
317 
318  //Scale cross product
319  errRollPitch_e[0] /= (normdGPSdt_e * delT_between_updates);
320  errRollPitch_e[1] /= (normdGPSdt_e * delT_between_updates);
321  errRollPitch_e[2] /= (normdGPSdt_e * delT_between_updates);
322 
323  //Rotate earth drift error back into body frame;
324  rot_mult(Rbe, errRollPitch_e, errRollPitch_b, FALSE);
325 }
