31 // *****************************************************************
32 // RFM22B hardware layer
33 //
34 // This module uses the RFM22B's internal packet handling hardware to
35 // encapsulate our own packet data.
36 //
37 // The RFM22B internal hardware packet handler configuration is as follows ..
38 //
39 // 4-byte (32-bit) preamble .. alternating 0's & 1's
40 // 4-byte (32-bit) sync
41 // 1-byte packet length (number of data bytes to follow)
42 // 0 to 255 user data bytes
43 //
44 // Our own packet data will also contain it's own header and 32-bit CRC
45 // as a single 16-bit CRC is not sufficient for wireless comms.
46 //
47 // *****************************************************************

567  rfm22b_dev->send_ppm = false;
568  rfm22b_dev->datarate = RFM22B_DEFAULT_RX_DATARATE;

571  rfm22b_dev->com_config_cb = NULL;
572  rfm22b_dev->rx_in_cb = NULL;
573  rfm22b_dev->tx_out_cb = NULL;

576  rfm22b_dev->stats.packets_per_sec = 0;
577  rfm22b_dev->stats.rx_good = 0;
578  rfm22b_dev->stats.rx_corrected = 0;

595  vSemaphoreCreateBinary( rfm22b_dev->isrPending );

598  uint8_t crcs[] = { 0, 0, 0, 0 };

651 void PIOS_RFM22B_InjectEvent(struct pios_rfm22b_dev *rfm22b_dev, enum pios_rfm22b_event event, bool inISR)
652 {
653 
654  // Store the event.
655  if (xQueueSend(rfm22b_dev->eventQueue, &event, portMAX_DELAY) != pdTRUE)
656  return;
657 
658  // Signal the semaphore to wake up the handler thread.
659  if (inISR) {
660  portBASE_TYPE pxHigherPriorityTaskWoken;

662  // Something went fairly seriously wrong
663  rfm22b_dev->errors++;
664  }
665  portEND_SWITCHING_ISR(pxHigherPriorityTaskWoken);
666  }
667  else
668  {
669  if (xSemaphoreGive(rfm22b_dev->isrPending) != pdTRUE) {
670  // Something went fairly seriously wrong
671  rfm22b_dev->errors++;
672  }
673  }
674 }

848 static void PIOS_RFM22B_Task(void *parameters)
849 {
850  struct pios_rfm22b_dev *rfm22b_dev = (struct pios_rfm22b_dev *)parameters;
851  if (!PIOS_RFM22B_validate(rfm22b_dev))
852  return;
853  portTickType lastEventTicks = xTaskGetTickCount();
854  portTickType lastStatusTicks = lastEventTicks;
855  portTickType lastPPMTicks = lastEventTicks;
856 
857  while(1)
858  {
859 #ifdef PIOS_WDG_RFM22B
860  // Update the watchdog timer
861  PIOS_WDG_UpdateFlag(PIOS_WDG_RFM22B);
862 #endif /* PIOS_WDG_RFM22B */
863 
864  // Wait for a signal indicating an external interrupt or a pending send/receive request.
865  if (xSemaphoreTake(rfm22b_dev->isrPending, ISR_TIMEOUT / portTICK_RATE_MS) == pdTRUE) {
866  lastEventTicks = xTaskGetTickCount();
867 
868  // Process events through the state machine.
869  enum pios_rfm22b_event event;
870  while (xQueueReceive(rfm22b_dev->eventQueue, &event, 0) == pdTRUE)
871  {
872  if ((event == RFM22B_EVENT_INT_RECEIVED) &&
873  ((rfm22b_dev->state == RFM22B_STATE_UNINITIALIZED) || (rfm22b_dev->state == RFM22B_STATE_INITIALIZING)))
874  continue;
875  rfm22_process_event(rfm22b_dev, event);
876  }
877  }
878  else
879  {
880  // Has it been too long since the last event?
881  portTickType curTicks = xTaskGetTickCount();
882  if (timeDifferenceMs(lastEventTicks, curTicks) > PIOS_RFM22B_SUPERVISOR_TIMEOUT)
883  {
884  // Transsition through an error event.
885  rfm22_process_event(rfm22b_dev, RFM22B_EVENT_ERROR);
886 
887  // Clear the event queue.
888  enum pios_rfm22b_event event;
889  while (xQueueReceive(rfm22b_dev->eventQueue, &event, 0) == pdTRUE)
890  ;
891  lastEventTicks = xTaskGetTickCount();
892  }
893  }
894 
895  portTickType curTicks = xTaskGetTickCount();
896  // Have we been sending this packet too long?
897  if ((rfm22b_dev->packet_start_ticks > 0) && (timeDifferenceMs(rfm22b_dev->packet_start_ticks, curTicks) > (rfm22b_dev->max_packet_time * 3)))
898  rfm22_process_event(rfm22b_dev, RFM22B_EVENT_TIMEOUT);
899 
900  // Have it been too long since we received a packet
901  else if ((rfm22b_dev->rx_complete_ticks > 0) && (timeDifferenceMs(rfm22b_dev->rx_complete_ticks, curTicks) > DISCONNECT_TIMEOUT_MS))
902  rfm22_process_event(rfm22b_dev, RFM22B_EVENT_ERROR);
903 
904  else
905  {
906 
907  // Are we waiting for an ACK?
908  if (rfm22b_dev->prev_tx_packet)
909  {
910 
911  // Should we resend the packet?
912  if (timeDifferenceMs(rfm22b_dev->tx_complete_ticks, curTicks) > rfm22b_dev->max_ack_delay)
913  {
914  rfm22b_dev->tx_complete_ticks = curTicks;
915  rfm22_process_event(rfm22b_dev, RFM22B_EVENT_ACK_TIMEOUT);
916  }
917  }
918  else
919  {
920 
921  // Queue up a PPM packet if it's time.
922  if (timeDifferenceMs(lastPPMTicks, curTicks) > PPM_UPDATE_PERIOD_MS)
923  {
924  rfm22_sendPPM(rfm22b_dev);
925  lastPPMTicks = curTicks;
926  }
927 
928  // Queue up a status packet if it's time.
929  if (timeDifferenceMs(lastStatusTicks, curTicks) > RADIOSTATS_UPDATE_PERIOD_MS)
930  {
931  rfm22_sendStatus(rfm22b_dev);
932  lastStatusTicks = curTicks;
933  }
934  }
935 
936  }
937 
938  // Send a packet if it's our time slice
939  rfm22b_dev->time_to_send = (((curTicks - rfm22b_dev->time_to_send_offset) & 0x6) == 0);
940 #ifdef PIOS_RFM22B_DEBUG_ON_TELEM
941  if (rfm22b_dev->time_to_send)
942  D4_LED_ON;
943  else
944  D4_LED_OFF;
945 #endif
946  if (rfm22b_dev->time_to_send)
947  rfm22_process_event(rfm22b_dev, RFM22B_EVENT_TX_START);
948  }
949 }

1137  return RFM22B_EVENT_NUM_EVENTS;

1227 static void rfm22_calculateLinkQuality(struct pios_rfm22b_dev *rfm22b_dev)
1228 {
1229  // Add the RX packet statistics
1230  rfm22b_dev->stats.rx_good = 0;
1231  rfm22b_dev->stats.rx_corrected = 0;
1232  rfm22b_dev->stats.rx_error = 0;
1233  rfm22b_dev->stats.tx_resent = 0;
1234  for (uint8_t i = 0; i < RFM22B_RX_PACKET_STATS_LEN; ++i)
1235  {
1236  uint32_t val = rfm22b_dev->rx_packet_stats[i];
1237  for (uint8_t j = 0; j < 16; ++j)
1238  {
1239  switch ((val >> (j * 2)) & 0x3)
1240  {
1241  case RFM22B_GOOD_RX_PACKET:
1242  rfm22b_dev->stats.rx_good++;
1243  break;
1244  case RFM22B_CORRECTED_RX_PACKET:
1245  rfm22b_dev->stats.rx_corrected++;
1246  break;
1247  case RFM22B_ERROR_RX_PACKET:
1248  rfm22b_dev->stats.rx_error++;
1249  break;
1250  case RFM22B_RESENT_TX_PACKET:
1251  rfm22b_dev->stats.tx_resent++;
1252  break;
1253  }
1254  }
1255  }
1256 
1257  // Calculate the link quality metric, which is related to the number of good packets in relation to the number of bad packets.
1258  // Note: This assumes that the number of packets sampled for the stats is 64.
1259  // Using this equation, error and resent packets are counted as -2, and corrected packets are counted as -1.
1260  // The range is 0 (all error or resent packets) to 128 (all good packets).
1261  rfm22b_dev->stats.link_quality = 64 + rfm22b_dev->stats.rx_good - rfm22b_dev->stats.rx_error - rfm22b_dev->stats.tx_resent;
1262 }

1348  else

1350

1370  {
1371  p = (PHPacketHandle)&(rfm22b_dev->status_packet);
1372  rfm22b_dev->send_status = false;
1373  }

1400

1403  D2_LED_OFF;
1404  D3_LED_TOGGLE;
1405 #endif

1694  if (!rfm22_readStatus(rfm22b_dev))

1696

1698  if (rfm22b_dev->int_status1 & RFM22_is1_ifferr)
1699  return RFM22B_EVENT_FAILURE;
1700

1702  if (rfm22b_dev->int_status1 & RFM22_is1_irxffafull)

1705  // read the total length of the packet data
1706  uint16_t len = rfm22_read(rfm22b_dev, RFM22_received_packet_length);
1707 
1708  // The received packet is going to be larger than the specified length
1709  if ((rfm22b_dev->rx_buffer_wr + RX_FIFO_HI_WATERMARK) > len)
1710  return RFM22B_EVENT_FAILURE;
1711

1713  if (((rfm22b_dev->rx_buffer_wr + RX_FIFO_HI_WATERMARK) >= len) && !(rfm22b_dev->int_status1 & RFM22_is1_ipkvalid))
1714  return RFM22B_EVENT_FAILURE;
1715

1717  rfm22_claimBus(rfm22b_dev);
1718  rfm22_assertCs(rfm22b_dev);
1719  PIOS_SPI_TransferByte(rfm22b_dev->spi_id,RFM22_fifo_access & 0x7F);

1722  rfm22_releaseBus(rfm22b_dev);
1723  }
1724

1726  if (rfm22b_dev->int_status1 & RFM22_is1_icrerror)
1727  return RFM22B_EVENT_FAILURE;
1728 
1729  // Valid packet received
1730  if (rfm22b_dev->int_status1 & RFM22_is1_ipkvalid)
1731  {
1732 
1733  // read the total length of the packet data
1734  uint32_t len = rfm22_read(rfm22b_dev, RFM22_received_packet_length);
1735

1737  if (rfm22b_dev->rx_buffer_wr < len)

1742  rfm22_assertCs(rfm22b_dev);

1746  rfm22_releaseBus(rfm22b_dev);
1747  }

1750  return RFM22B_EVENT_FAILURE;

1753  enum pios_rfm22b_event ret_event = RFM22B_EVENT_RX_COMPLETE;
1754  if (rfm22b_dev->rx_buffer_wr > 0)
1755  {
1756  rfm22b_dev->stats.rx_byte_count += rfm22b_dev->rx_buffer_wr;
1757  // Check the packet for errors.
1758  if (rfm22_receivePacket(rfm22b_dev, &(rfm22b_dev->rx_packet), rfm22b_dev->rx_buffer_wr))

1769  {
1770  // Send the data to the com port
1771  bool rx_need_yield;
1772  if (rfm22b_dev->rx_in_cb)
1773  (rfm22b_dev->rx_in_cb)(rfm22b_dev->rx_in_context, rfm22b_dev->rx_packet.data, rfm22b_dev->rx_packet.header.data_size, NULL, &rx_need_yield);

1777  break;
1778  case PACKET_TYPE_ACK:
1779  case PACKET_TYPE_ACK_RTS:

1782  case PACKET_TYPE_NACK:

1786  break;

1791  rfm22b_dev->rx_buffer_wr = 0;
1792  rfm22b_dev->rx_complete_ticks = xTaskGetTickCount();

2031  rfm22b_dev->pair_stats[min_idx].rssi = rssi;
2032  rfm22b_dev->pair_stats[min_idx].afc_correction = afc;

2034  }
