00079         PIOS_SYS_SerialNumberGet((char *)sn);
00080 
00081         /* Concatenate the device serial number and the appropriate suffix ("+BL" or "+FW") into the USB serial number */
00082         uint8_t * utf8 = &(usb_serial_number[2]);

00084         utf8 = PIOS_USB_UTIL_AsciiToUtf8(utf8, (uint8_t *)PIOS_USB_BOARD_SN_SUFFIX, sizeof(PIOS_USB_BOARD_SN_SUFFIX)-1);
00085 
00086         PIOS_USBHOOK_RegisterString(USB_STRING_DESC_PRODUCT, (uint8_t *)&usb_product_id, sizeof(usb_product_id));
