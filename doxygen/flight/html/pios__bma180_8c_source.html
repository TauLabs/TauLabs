258 static int32_t PIOS_BMA180_Config()
259 {
260  /*
261  0x35 = 0x81 //smp-skip = 1 for less interrupts
262  0x33 = 0x81 //shadow-dis = 1, update MSB and LSB synchronously
263  0x27 = 0x01 //dis-i2c
264  0x21 = 0x02 //new_data_int = 1
265  */
266 
267  PIOS_DELAY_WaituS(20);
268 
269  if(PIOS_BMA180_Validate(dev) != 0)
270  return -1;
271 
272  while(PIOS_BMA180_EnableEeprom() != 0);
273  while(PIOS_BMA180_SetReg(BMA_RESET, BMA_RESET_CODE) != 0);
274  while(PIOS_BMA180_SetReg(BMA_OFFSET_LSB1, 0x81) != 0);
275  while(PIOS_BMA180_SetReg(BMA_GAIN_Y, 0x81) != 0);
276  while(PIOS_BMA180_SetReg(BMA_CTRREG3, 0xFF) != 0);
277  while(PIOS_BMA180_SelectBW(dev->cfg->bandwidth) != 0);
278  while(PIOS_BMA180_SetRange(dev->cfg->range) != 0);
279  while(PIOS_BMA180_DisableEeprom() != 0);
280 
281  return 0;
282 }

291 static int32_t PIOS_BMA180_SelectBW(enum bma180_bandwidth bw)
292 {
293  if(PIOS_BMA180_Validate(dev) != 0)
294  return -1;
295 
296  dev->bandwidth = bw;
297 
298  uint8_t reg;

301  return PIOS_BMA180_SetReg(BMA_BW_ADDR, reg);

345 int32_t PIOS_BMA180_ReadAccels(struct pios_bma180_data * data)
346 {
347  // To save memory use same buffer for in and out but offset by
348  // a byte
349  uint8_t buf[7] = {BMA_X_LSB_ADDR | 0x80,0,0,0,0,0};
350  uint8_t rec[7] = {0,0,0,0,0,0};
351 
352  if(PIOS_BMA180_ClaimBus() != 0)
353  return -1;
354  if(PIOS_SPI_TransferBlock(dev->spi_id,&buf[0],&rec[0],7,NULL) != 0)
355  return -2;
356  PIOS_BMA180_ReleaseBus();
357 
358  // | MSB | LSB | 0 | new_data |
359  data->x = ((rec[2] << 8) | rec[1]);
360  data->y = ((rec[4] << 8) | rec[3]);
361  data->z = ((rec[6] << 8) | rec[5]);
362  data->x /= 4;
363  data->y /= 4;
364  data->z /= 4;
365 
366  return 0; // return number of remaining entries
367 }

421 int32_t PIOS_BMA180_Test()
422 {
423  // Read chip ID then version ID
424  uint8_t buf[3] = {0x80 | BMA_CHIPID_ADDR, 0, 0};
425  uint8_t rec[3] = {0,0, 0};
426  int32_t retval;
427 
428  if(PIOS_BMA180_ClaimBus() != 0)
429  return -1;
430  retval = PIOS_SPI_TransferBlock(dev->spi_id,&buf[0],&rec[0],sizeof(buf),NULL);
431  PIOS_BMA180_ReleaseBus();
432 
433  if(retval != 0)
434  return -2;
435 
436  struct pios_bma180_data data;

438  return -3;

440  if(rec[1] != 0x3)
441  return -4;
442 
443  if(rec[2] < 0x12)
444  return -5;

452 int32_t bma180_irqs = 0;
453 bool PIOS_BMA180_IRQHandler(void)
454 {
455  bma180_irqs++;
456 
457  const static uint8_t pios_bma180_req_buf[7] = {BMA_X_LSB_ADDR | 0x80,0,0,0,0,0};
458  uint8_t pios_bma180_dmabuf[8];
459 
460  // If we can't get the bus then just move on for efficiency
461  bool woken = false;
462  if(PIOS_BMA180_ClaimBusISR(&woken) != 0) {
463  return woken; // Something else is using bus, miss this data
464  }
465 
466  PIOS_SPI_TransferBlock(dev->spi_id,pios_bma180_req_buf,(uint8_t *) pios_bma180_dmabuf,
467  sizeof(pios_bma180_dmabuf), NULL);
468 
469  // TODO: Make this conversion depend on configuration scale
470  struct pios_bma180_data data;
471 
472  // Don't release bus till data has copied
473  PIOS_BMA180_ReleaseBus(&woken);
474 
475  // Must not return before releasing bus
476  if(fifoBuf_getFree(&dev->fifo) < sizeof(data))
477  return woken;
478 
479  // Bottom two bits indicate new data and are constant zeros. Don't right
480  // shift because it drops sign bit
481  data.x = ((pios_bma180_dmabuf[2] << 8) | pios_bma180_dmabuf[1]);
482  data.y = ((pios_bma180_dmabuf[4] << 8) | pios_bma180_dmabuf[3]);
483  data.z = ((pios_bma180_dmabuf[6] << 8) | pios_bma180_dmabuf[5]);
484  data.x /= 4;
485  data.y /= 4;
486  data.z /= 4;
487  data.temperature = pios_bma180_dmabuf[7];
488 
489  fifoBuf_putData(&dev->fifo, (uint8_t *) &data, sizeof(data));
490 
491  return woken;
492 }
