00074 /*
00075  * Reads the airspeed and returns CAS (calibrated airspeed) in the case of success. 
00076  * In the case of a failed read, returns -1.
00077  */
00078 float PIOS_MPXV7002_ReadAirspeed(uint8_t airspeedADCPin)
00079 {
00080         float sensorVal = PIOS_MPXV7002_Measure(airspeedADCPin);
00081         
00082         //Calculate dynamic pressure, as per docs
00083         float Qc = 5.0f*(((sensorVal - calibrationOffset)/4096.0f*3.3f)/VCC - 0.5f);
00084 
00085         //Saturate Qc on the lower bound, in order to make sure we don't have negative airspeeds. No need
00086         // to saturate on the upper bound, we'll handle that later with calibratedAirspeed.
00087         if (Qc < 0) {
00088                 Qc=0;
00089         }
00090         
00091         //Compute calibraterd airspeed, as per http://en.wikipedia.org/wiki/Calibrated_airspeed
00092         float calibratedAirspeed = A0*sqrtf(5.0f*(pow(Qc/P0+1.0f,POWER)-1.0f));
00093         
00094         //Upper bound airspeed. No need to lower bound it, that comes from Qc
00095         if (calibratedAirspeed > 59) { //in [m/s]
00096                 calibratedAirspeed=59;
00097         }
00098         
00099         
00100         return calibratedAirspeed;
00101 }
