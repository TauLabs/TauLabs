00286 static float scale(float val, float inMin, float inMax, float outMin, float outMax)
00287 {
00288         // bound input value
00289         if (val > inMax) val = inMax;
00290         if (val < inMin) val = inMin;
00291 
00292         // normalize input value to [0..1]
00293         if (inMax <= inMin)
00294                 val = 0.0;
00295         else
00296                 val = (val - inMin) / (inMax - inMin);
00297 
00298         // update output bounds
00299         if (outMin > outMax) {
00300                 float t = outMin;
00301                 outMin = outMax;
00302                 outMax = t;
00303                 val = 1.0 - val;
00304         }
00305 
00306         return (outMax - outMin) * val + outMin;
