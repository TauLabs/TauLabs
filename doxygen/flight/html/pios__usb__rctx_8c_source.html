00161 void PIOS_USB_RCTX_Update(uint32_t usbrctx_id, const uint16_t channel[], const int16_t channel_min[], const int16_t channel_max[], uint8_t num_channels)

00166         PIOS_Assert(valid);

00168         if (!PIOS_USB_CheckAvailable(usb_rctx_dev->lower_id)) {
00169                 return;

00171 
00172         for (uint8_t i = 0; 
00173              i < PIOS_USB_RCTX_NUM_CHANNELS && i < num_channels;
00174              i++) {
00175                 int16_t min = channel_min[i];
00176                 int16_t max = channel_max[i];
00177                 uint16_t val = channel[i];
00178 
00179                 if (channel_min[i] > channel_max[i]) {
00180                         /* This channel is reversed, flip min and max */
00181                         min = channel_max[i];
00182                         max = channel_min[i];
00183 
00184                         /* and flip val to be an offset from the lower end of the range */
00185                         val = channel_min[i] - channel[i] + channel_max[i];
00186                 }
00187 
00188                 /* Scale channel linearly between min and max */
00189                 if (min == max) {
00190                         val = 0;
00191                 } else {
00192                         if (val < min) val = min;
00193                         if (val > max) val = max;
00194 
00195                         val = (val - min) * (65535 / (max - min));
00196                 }
00197 
00198                 usb_rctx_dev->report.vals[i] = val;
00199         }
00200 
00201         if (GetEPTxStatus(usb_rctx_dev->cfg->data_tx_ep) == EP_TX_VALID) {
00202                 /* Endpoint is already transmitting */
00203                 return;
00204         }
