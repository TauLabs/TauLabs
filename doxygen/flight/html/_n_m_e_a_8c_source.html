110 int parse_nmea_stream (uint8_t c, char *gps_rx_buffer, GPSPositionData *GpsData, struct GPS_RX_STATS *gpsRxStats)
111 {
112  static uint8_t rx_count = 0;
113  static bool start_flag = false;
114  static bool found_cr = false;
115 
116  // detect start while acquiring stream
117  if (!start_flag && (c == '$')) // NMEA identifier found
118  {
119  start_flag = true;
120  found_cr = false;
121  rx_count = 0;
122  }
123  else
124  if (!start_flag)
125  return PARSER_ERROR;
126 
127  if (rx_count >= NMEA_MAX_PACKET_LENGTH)
128  {
129  // The buffer is already full and we haven't found a valid NMEA sentence.
130  // Flush the buffer and note the overflow event.
131  gpsRxStats->gpsRxOverflow++;
132  start_flag = false;
133  found_cr = false;
134  rx_count = 0;
135  return PARSER_OVERRUN;
136  }
137  else
138  {
139  gps_rx_buffer[rx_count] = c;
140  rx_count++;
141  }
142 
143  // look for ending '\r\n' sequence
144  if (!found_cr && (c == '\r') )
145  found_cr = true;
146  else
147  if (found_cr && (c != '\n') )
148  found_cr = false; // false end flag
149  else
150  if (found_cr && (c == '\n') )
151  {
152  // The NMEA functions require a zero-terminated string
153  // As we detected \r\n, the string as for sure 2 bytes long, we will also strip the \r\n
154  gps_rx_buffer[rx_count-2] = 0;
155 
156  // prepare to parse next sentence
157  start_flag = false;
158  found_cr = false;
159  rx_count = 0;
160  // Our rxBuffer must look like this now:
161  // [0] = '$'
162  // ... = zero or more bytes of sentence payload
163  // [end_pos - 1] = '\r'
164  // [end_pos] = '\n'
165  //
166  // Prepare to consume the sentence from the buffer
167 
168  // Validate the checksum over the sentence
169  if (!NMEA_checksum(&gps_rx_buffer[1]))
170  { // Invalid checksum. May indicate dropped characters on Rx.
171  //PIOS_DEBUG_PinHigh(2);
172  gpsRxStats->gpsRxChkSumError++;
173  //PIOS_DEBUG_PinLow(2);
174  return PARSER_ERROR;
175  }
176  else
177  { // Valid checksum, use this packet to update the GPS position
178  if (!NMEA_update_position(&gps_rx_buffer[1], GpsData)) {
179  //PIOS_DEBUG_PinHigh(2);
180  gpsRxStats->gpsRxParserError++;
181  //PIOS_DEBUG_PinLow(2);
182  }
183  else
184  gpsRxStats->gpsRxReceived++;;
185 
186  return PARSER_COMPLETE;
187  }
188  }
189  return PARSER_INCOMPLETE;
190 }

218 bool NMEA_checksum(char *nmea_sentence)
219 {
220  uint8_t checksum_computed = 0;
221  uint8_t checksum_received;
222 
223  while (*nmea_sentence != '\0' && *nmea_sentence != '*') {
224  checksum_computed ^= *nmea_sentence;
225  nmea_sentence++;
226  }
227 
228  /* Make sure we're now pointing at the checksum */
229  if (*nmea_sentence == '\0') {
230  /* Buffer ran out before we found a checksum marker */
231  return false;
232  }
233 
234  /* Load the checksum from the buffer */
235  checksum_received = strtol(nmea_sentence + 1, NULL, 16);
236 
237  //PIOS_COM_SendFormattedStringNonBlocking(COM_DEBUG_USART,"$%d=%d\r\n",checksum_received,checksum_computed);
238 
239  return (checksum_computed == checksum_received);
240 }

249 /* Parse a number encoded in a string of the format:
250  * [-]NN.nnnnn
251  * into a signed whole part and an unsigned fractional part.
252  * The fract_units field indicates the units of the fractional part as
253  * 1 whole = 10^fract_units fract

255 static bool NMEA_parse_real(int32_t * whole, uint32_t * fract, uint8_t * fract_units, char *field)
256 {
257  char *s = field;
258  char *field_w;
259  char *field_f;
260 
261  PIOS_DEBUG_Assert(whole);
262  PIOS_DEBUG_Assert(fract);
263  PIOS_DEBUG_Assert(fract_units);
264 
265  field_w = strsep(&s, ".");
266  field_f = s;
267 
268  *whole = strtol(field_w, NULL, 10);
269 
270  if (field_w) {
271  /* decimal was found so we may have a fractional part */
272  *fract = strtoul(field_f, NULL, 10);
273  *fract_units = strlen(field_f);
274  } else {
275  /* no decimal was found, fractional part is zero */
276  *fract = 0;
277  *fract_units = 0;
278  }
279 
280  return true;
281 }

300 /*
301  * Parse a field in the format:
302  * DD[D]MM.mmmm[mm]
303  * into a fixed-point representation in units of (degrees * 1e-7)
304  */
305 static bool NMEA_latlon_to_fixed_point(int32_t * latlon, char *nmea_latlon, bool negative)
306 {
307  int32_t num_DDDMM;
308  uint32_t num_m;
309  uint8_t units;
310 
311  /* Sanity checks */
312  PIOS_DEBUG_Assert(nmea_latlon);
313  PIOS_DEBUG_Assert(latlon);
314 
315  if (*nmea_latlon == '\0') { /* empty lat/lon field */
316  return false;
317  }
318 
319  if (!NMEA_parse_real(&num_DDDMM, &num_m, &units, nmea_latlon)) {
320  return false;
321  }
322 
323  /* scale up the mmmm[mm] field apropriately depending on # of digits */
324  /* not using 1eN notation because that forces fixed point and lost precision */
325  switch (units) {
326  case 0:
327  /* no digits, value is zero so no scaling */
328  break;
329  case 1: /* m */
330  num_m *= 1000000; /* m000000 */
331  break;
332  case 2: /* mm */
333  num_m *= 100000; /* mm00000 */
334  break;
335  case 3: /* mmm */
336  num_m *= 10000; /* mmm0000 */
337  break;
338  case 4: /* mmmm */
339  num_m *= 1000; /* mmmm000 */
340  break;
341  case 5: /* mmmmm */
342  num_m *= 100; /* mmmmm00 */
343  break;
344  case 6: /* mmmmmm */
345  num_m *= 10; /* mmmmmm0 */
346  break;
347  default:
348  /* unhandled format */
349  num_m = 0.0f;
350  break;
351  }
352 
353  *latlon = (num_DDDMM / 100) * 10000000; /* scale the whole degrees */
354  *latlon += (num_DDDMM % 100) * 10000000 / 60; /* add in the scaled decimal whole minutes */
355  *latlon += num_m / 60; /* add in the scaled decimal fractional minutes */
356 
357  if (negative)
358  *latlon *= -1;
359 
360  return true;
361 }

370 bool NMEA_update_position(char *nmea_sentence, GPSPositionData *GpsData)
371 {
372  char* p = nmea_sentence;
373  char* params[MAX_NB_PARAMS];
374  uint8_t nbParams;
375 
376 #ifdef DEBUG_MSG_IN
377  DEBUG_MSG("\"%s\"\n", nmea_sentence);
378 #endif
379 
380  // Split the nmea sentence it its parameters, separated by ","
381  // Sample NMEA message: "GPRMC,000131.736,V,,,,,0.00,0.00,060180,,,N*43"
382 
383  // The first parameter starts at the beginning of the message
384  params[0] = p;
385  nbParams = 1;
386  while (*p != 0) {
387  if (*p == '*') {
388  // After the * comes the "CRC", we are done,
389  *p = 0; // Zero-terminate this parameter
390  break;
391  } else if (*p == ',') {
392  // This is the end of this parameter
393  *p = 0; // Zero-terminate this parameter
394  // Start new parameter
395  if (nbParams==MAX_NB_PARAMS)
396  break;
397  params[nbParams] = p+1; // For sure there is something at p+1 because at p there is ","
398  nbParams++;
399  }
400  p++;
401  }
402 
403 
404 #ifdef DEBUG_PARAMS
405  int i;
406  for (i=0;i<nbParams; i++) {
407  DEBUG_MSG(" %d \"%s\"\n", i, params[i]);
408  }
409 #endif
410 
411  // The first parameter is the message name, lets see if we find a parser for it
412  const struct nmea_parser *parser;
413  parser = NMEA_find_parser_by_prefix(params[0]);
414  if (!parser) {
415  // No parser found
416  DEBUG_MSG(" NO PARSER (\"%s\")\n", params[0]);
417  return false;
418  }
419 
420  #ifdef DEBUG_MGSID_IN
421  DEBUG_MSG("%s %d ", params[0]);
422  #endif
423  // Send the message to the parser and get it update the GpsData
424  // Information from various different NMEA messages are temporarily
425  // cumulated in the GpsData structure. An actual GPSPosition update
426  // is triggered by GGA messages only. This message type sets the
427  // gpsDataUpdated flag to request this.
428  bool gpsDataUpdated = false;
429 
430  if (!parser->handler(GpsData, &gpsDataUpdated, params, nbParams)) {
431  // Parse failed
432  DEBUG_MSG("PARSE FAILED (\"%s\")\n", params[0]);
433  if (gpsDataUpdated && (GpsData->Status == GPSPOSITION_STATUS_NOFIX)) {
434  GPSPositionSet(GpsData);
435  }
436  return false;
437  }
438 
439 
440  // All is fine :) Update object if data has changed
441  if (gpsDataUpdated) {
442  #ifdef DEBUG_MGSID_IN
443  DEBUG_MSG("U");
444  #endif
445  GPSPositionSet(GpsData);
446  }
447 
448  #ifdef DEBUG_MGSID_IN
449  DEBUG_MSG("\n");
450  #endif
451  return true;
452 }

643 static bool nmeaProcessGPGSV(GPSPositionData * GpsData, bool* gpsDataUpdated, char* param[], uint8_t nbParam)
644 {
645  if (nbParam < 4)
646  return false;
647 
648 #ifdef NMEA_DEBUG_GSV
649  DEBUG_MSG("\n Sentence=%s/%s\n", param[2], param[1]);
650  DEBUG_MSG(" Sats=%s\n", param[3]);
651 #endif
652 
653  uint8_t nbSentences = atoi(param[1]);
654  uint8_t currSentence = atoi(param[2]);
655 
656  *gpsDataUpdated = false;
657 
658  if (nbSentences < 1 || nbSentences > 8 || currSentence < 1 || currSentence > nbSentences)
659  return false;
660 
661  gsv_partial.SatsInView = atoi(param[3]);
662 
663  // Find out if this is the first sentence in the GSV set
664  if (currSentence == 1) {
665  if (gsv_expected_mask != gsv_processed_mask) {
666  // We are starting over when we haven't yet finished our previous GSV group
667  gsv_incomplete_error++;
668  }
669 
670  // First GSV sentence in the sequence, reset our expected_mask
671  gsv_expected_mask = (1 << nbSentences) - 1;
672  }
673 
674  uint8_t current_sentence_id = (1 << (currSentence - 1));
675  if (gsv_processed_mask & current_sentence_id) {
676  /* Duplicate sentence in this GSV set */
677  gsv_duplicate_error++;
678  } else {
679  /* Note that we've seen this sentence */
680  gsv_processed_mask |= current_sentence_id;
681  }
682 
683  uint8_t parIdx = 4;
684 
685 #ifdef NMEA_DEBUG_GSV
686  DEBUG_MSG(" PRN:");
687 #endif
688 
689  /* Make sure this sentence can fit in our GPSSatellites object */
690  if ((currSentence * 4) <= NELEMENTS(gsv_partial.PRN)) {
691  /* Process 4 blocks of satellite info */
692  for (uint8_t i = 0; parIdx+4 <= nbParam && i < 4; i++) {
693  uint8_t sat_index = ((currSentence - 1) * 4) + i;
694

697  gsv_partial.Elevation[sat_index] = NMEA_real_to_float(param[parIdx++]);
698  gsv_partial.Azimuth[sat_index] = NMEA_real_to_float(param[parIdx++]);
699  gsv_partial.SNR[sat_index] = atoi(param[parIdx++]);

701  DEBUG_MSG(" %d", gsv_partial.PRN[sat_index]);
702 #endif
703  }
704  }
705 #ifdef NMEA_DEBUG_GSV
706  DEBUG_MSG("\n");
707 #endif
708 
709 
710  /* Find out if we're finished processing all GSV sentences in the set */
711  if ((gsv_expected_mask != 0) && (gsv_processed_mask == gsv_expected_mask)) {
712  /* GSV set has been fully processed. Update the GPSSatellites object. */
713  GPSSatellitesSet(&gsv_partial);
714  memset((void *)&gsv_partial, 0, sizeof(gsv_partial));
715  gsv_expected_mask = 0;
716  gsv_processed_mask = 0;
717  }
718 
719  return true;
