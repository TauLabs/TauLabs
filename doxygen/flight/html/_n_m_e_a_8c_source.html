00110 int parse_nmea_stream (uint8_t c, char *gps_rx_buffer, GPSPositionData *GpsData, struct GPS_RX_STATS *gpsRxStats)
00111 {
00112         static uint8_t rx_count = 0;
00113         static bool start_flag = false;
00114         static bool found_cr = false;
00115 
00116         // detect start while acquiring stream
00117         if (!start_flag && (c == '$')) // NMEA identifier found
00118         {
00119                 start_flag = true;
00120                 found_cr = false;
00121                 rx_count = 0;
00122         }
00123         else
00124         if (!start_flag)
00125                 return PARSER_ERROR;
00126 
00127         if (rx_count >= NMEA_MAX_PACKET_LENGTH)
00128         {
00129                 // The buffer is already full and we haven't found a valid NMEA sentence.
00130                 // Flush the buffer and note the overflow event.
00131                 gpsRxStats->gpsRxOverflow++;
00132                 start_flag = false;
00133                 found_cr = false;
00134                 rx_count = 0;
00135                 return PARSER_OVERRUN;
00136         }
00137         else
00138         {
00139                 gps_rx_buffer[rx_count] = c;
00140                 rx_count++;
00141         }
00142 
00143         // look for ending '\r\n' sequence
00144         if (!found_cr && (c == '\r') )
00145                 found_cr = true;
00146         else
00147         if (found_cr && (c != '\n') )
00148                 found_cr = false;  // false end flag
00149         else
00150         if (found_cr && (c == '\n') )
00151         {
00152                 // The NMEA functions require a zero-terminated string
00153                 // As we detected \r\n, the string as for sure 2 bytes long, we will also strip the \r\n
00154                 gps_rx_buffer[rx_count-2] = 0;
00155 
00156                 // prepare to parse next sentence
00157                 start_flag = false;
00158                 found_cr = false;
00159                 rx_count = 0;
00160                 // Our rxBuffer must look like this now:
00161                 //   [0]           = '$'
00162                 //   ...           = zero or more bytes of sentence payload
00163                 //   [end_pos - 1] = '\r'
00164                 //   [end_pos]     = '\n'
00165                 //
00166                 // Prepare to consume the sentence from the buffer
00167 
00168                 // Validate the checksum over the sentence
00169                 if (!NMEA_checksum(&gps_rx_buffer[1]))
00170                 {       // Invalid checksum.  May indicate dropped characters on Rx.
00171                         //PIOS_DEBUG_PinHigh(2);
00172                         gpsRxStats->gpsRxChkSumError++;
00173                         //PIOS_DEBUG_PinLow(2);
00174                         return PARSER_ERROR;
00175                 }
00176                 else
00177                 {       // Valid checksum, use this packet to update the GPS position
00178                         if (!NMEA_update_position(&gps_rx_buffer[1], GpsData)) {
00179                                 //PIOS_DEBUG_PinHigh(2);
00180                                 gpsRxStats->gpsRxParserError++;
00181                                 //PIOS_DEBUG_PinLow(2);
00182                         }
00183                         else
00184                                 gpsRxStats->gpsRxReceived++;;
00185 
00186                         return PARSER_COMPLETE;
00187                 }
00188         }
00189         return PARSER_INCOMPLETE;
00190 }

00218 bool NMEA_checksum(char *nmea_sentence)
00219 {
00220         uint8_t checksum_computed = 0;
00221         uint8_t checksum_received;
00222 
00223         while (*nmea_sentence != '\0' && *nmea_sentence != '*') {
00224                 checksum_computed ^= *nmea_sentence;
00225                 nmea_sentence++;
00226         }
00227 
00228         /* Make sure we're now pointing at the checksum */
00229         if (*nmea_sentence == '\0') {
00230                 /* Buffer ran out before we found a checksum marker */
00231                 return false;
00232         }
00233 
00234         /* Load the checksum from the buffer */
00235         checksum_received = strtol(nmea_sentence + 1, NULL, 16);
00236 
00237         //PIOS_COM_SendFormattedStringNonBlocking(COM_DEBUG_USART,"$%d=%d\r\n",checksum_received,checksum_computed);
00238 
00239         return (checksum_computed == checksum_received);
00240 }

00249 /* Parse a number encoded in a string of the format:
00250  *   [-]NN.nnnnn
00251  * into a signed whole part and an unsigned fractional part.
00252  * The fract_units field indicates the units of the fractional part as
00253  *   1 whole = 10^fract_units fract
00254  */
00255 static bool NMEA_parse_real(int32_t * whole, uint32_t * fract, uint8_t * fract_units, char *field)
00256 {
00257         char *s = field;
00258         char *field_w;
00259         char *field_f;
00260 
00261         PIOS_DEBUG_Assert(whole);
00262         PIOS_DEBUG_Assert(fract);
00263         PIOS_DEBUG_Assert(fract_units);
00264 
00265         field_w = strsep(&s, ".");
00266         field_f = s;
00267 
00268         *whole = strtol(field_w, NULL, 10);
00269 
00270         if (field_w) {
00271                 /* decimal was found so we may have a fractional part */
00272                 *fract = strtoul(field_f, NULL, 10);
00273                 *fract_units = strlen(field_f);
00274         } else {
00275                 /* no decimal was found, fractional part is zero */
00276                 *fract = 0;
00277                 *fract_units = 0;
00278         }
00279 
00280         return true;
00281 }

00300 /*
00301  * Parse a field in the format:
00302  *    DD[D]MM.mmmm[mm]
00303  * into a fixed-point representation in units of (degrees * 1e-7)
00304  */
00305 static bool NMEA_latlon_to_fixed_point(int32_t * latlon, char *nmea_latlon, bool negative)
00306 {
00307         int32_t num_DDDMM;
00308         uint32_t num_m;
00309         uint8_t units;
00310 
00311         /* Sanity checks */
00312         PIOS_DEBUG_Assert(nmea_latlon);
00313         PIOS_DEBUG_Assert(latlon);
00314 
00315         if (*nmea_latlon == '\0') { /* empty lat/lon field */
00316                 return false;
00317         }
00318 
00319         if (!NMEA_parse_real(&num_DDDMM, &num_m, &units, nmea_latlon)) {
00320                 return false;
00321         }
00322 
00323         /* scale up the mmmm[mm] field apropriately depending on # of digits */
00324         /* not using 1eN notation because that forces fixed point and lost precision */
00325         switch (units) {
00326         case 0:
00327                 /* no digits, value is zero so no scaling */
00328                 break;
00329         case 1:         /* m       */
00330                 num_m *= 1000000;       /* m000000 */
00331                 break;
00332         case 2:         /* mm      */
00333                 num_m *= 100000;        /* mm00000 */
00334                 break;
00335         case 3:         /* mmm     */
00336                 num_m *= 10000; /* mmm0000 */
00337                 break;
00338         case 4:         /* mmmm    */
00339                 num_m *= 1000;  /* mmmm000 */
00340                 break;
00341         case 5:         /* mmmmm   */
00342                 num_m *= 100;   /* mmmmm00 */
00343                 break;
00344         case 6:         /* mmmmmm  */
00345                 num_m *= 10;    /* mmmmmm0 */
00346                 break;
00347         default:
00348                 /* unhandled format */
00349                 num_m = 0.0f;
00350                 break;
00351         }
00352 
00353         *latlon = (num_DDDMM / 100) * 10000000;        /* scale the whole degrees */
00354         *latlon += (num_DDDMM % 100) * 10000000 / 60;  /* add in the scaled decimal whole minutes */
00355         *latlon += num_m / 60;                    /* add in the scaled decimal fractional minutes */
00356 
00357         if (negative)
00358                 *latlon *= -1;
00359 
00360         return true;
00361 }

00373         char* params[MAX_NB_PARAMS];

00375 
00376 #ifdef DEBUG_MSG_IN

00380         // Split the nmea sentence it its parameters, separated by ","
00381         // Sample NMEA message: "GPRMC,000131.736,V,,,,,0.00,0.00,060180,,,N*43"
00382 
00383         // The first parameter starts at the beginning of the message
00384         params[0] = p;
00385         nbParams = 1;
00386         while (*p != 0) {
00387                 if (*p == '*') {
00388                         // After the * comes the "CRC", we are done,
00389                         *p = 0;         // Zero-terminate this parameter
00390                         break;
00391                 } else if (*p == ',') {
00392                         // This is the end of this parameter
00393                         *p = 0;         // Zero-terminate this parameter
00394                         // Start new parameter
00395                         if (nbParams==MAX_NB_PARAMS)
00396                                 break;
00397                         params[nbParams] = p+1; // For sure there is something at p+1 because at p there is ","
00398                         nbParams++;
00399                 }
00400                 p++;
00401         }
00402 
00403 
00404 #ifdef DEBUG_PARAMS
00405         int i;
00406         for (i=0;i<nbParams; i++) {
00407                 DEBUG_MSG(" %d \"%s\"\n", i, params[i]);
00408         }
00409 #endif
00410 
00411         // The first parameter is the message name, lets see if we find a parser for it
00412         const struct nmea_parser *parser;
00413         parser = NMEA_find_parser_by_prefix(params[0]);
00414         if (!parser) {
00415                 // No parser found
00416                 DEBUG_MSG(" NO PARSER (\"%s\")\n", params[0]);
00417                 return false;
00418         }
00419 
00420         #ifdef DEBUG_MGSID_IN
00421                 DEBUG_MSG("%s %d ", params[0]);
00422         #endif
00423         // Send the message to the parser and get it update the GpsData

00428         bool gpsDataUpdated = false;
00429 
00430         if (!parser->handler(GpsData, &gpsDataUpdated, params, nbParams)) {
00431                 // Parse failed
00432                 DEBUG_MSG("PARSE FAILED (\"%s\")\n", params[0]);
00433                 if (gpsDataUpdated && (GpsData->Status == GPSPOSITION_STATUS_NOFIX)) {
00434                         GPSPositionSet(GpsData);
00435                 }
00436                 return false;
00437         }
00438 
00439 
00440         // All is fine :)  Update object if data has changed
00441         if (gpsDataUpdated) {
00442                 #ifdef DEBUG_MGSID_IN
00443                         DEBUG_MSG("U");
00444                 #endif
00445                 GPSPositionSet(GpsData);
00446         }
