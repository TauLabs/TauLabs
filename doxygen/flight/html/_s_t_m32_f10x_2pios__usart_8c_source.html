138 int32_t PIOS_USART_Init(uint32_t * usart_id, const struct pios_usart_cfg * cfg)
139 {
140  PIOS_DEBUG_Assert(usart_id);
141  PIOS_DEBUG_Assert(cfg);
142 
143  struct pios_usart_dev * usart_dev;
144 
145  usart_dev = (struct pios_usart_dev *) PIOS_USART_alloc();
146  if (!usart_dev) goto out_fail;
147 
148  /* Bind the configuration to the device instance */
149  usart_dev->cfg = cfg;
150 
151  /* Enable the USART Pins Software Remapping */
152  if (usart_dev->cfg->remap) {
153  GPIO_PinRemapConfig(usart_dev->cfg->remap, ENABLE);
154  }
155 
156  /* Initialize the USART Rx and Tx pins */
157  GPIO_Init(usart_dev->cfg->rx.gpio, &usart_dev->cfg->rx.init);
158  GPIO_Init(usart_dev->cfg->tx.gpio, &usart_dev->cfg->tx.init);
159 
160  /* Enable USART clock */
161  switch ((uint32_t)usart_dev->cfg->regs) {
162  case (uint32_t)USART1:
163  RCC_APB2PeriphClockCmd(RCC_APB2Periph_USART1, ENABLE);
164  break;
165  case (uint32_t)USART2:
166  RCC_APB1PeriphClockCmd(RCC_APB1Periph_USART2, ENABLE);
167  break;
168  case (uint32_t)USART3:
169  RCC_APB1PeriphClockCmd(RCC_APB1Periph_USART3, ENABLE);
170  break;
171  }
172 
173  /* Configure the USART */
174  USART_Init(usart_dev->cfg->regs, &usart_dev->cfg->init);
175 
176  *usart_id = (uint32_t)usart_dev;
177 
178  /* Configure USART Interrupts */
179  switch ((uint32_t)usart_dev->cfg->regs) {
180  case (uint32_t)USART1:
181  PIOS_USART_1_id = (uint32_t)usart_dev;
182  break;
183  case (uint32_t)USART2:
184  PIOS_USART_2_id = (uint32_t)usart_dev;
185  break;
186  case (uint32_t)USART3:
187  PIOS_USART_3_id = (uint32_t)usart_dev;
188  break;
189  }
190  NVIC_Init(&usart_dev->cfg->irq.init);
191  USART_ITConfig(usart_dev->cfg->regs, USART_IT_RXNE, ENABLE);
192  USART_ITConfig(usart_dev->cfg->regs, USART_IT_TXE, ENABLE);
193 
194  /* Enable USART */
195  USART_Cmd(usart_dev->cfg->regs, ENABLE);
196 
197  return(0);
198 
199 out_fail:
200  return(-1);
201 }

276 static void PIOS_USART_generic_irq_handler(uint32_t usart_id)
277 {
278  struct pios_usart_dev * usart_dev = (struct pios_usart_dev *)usart_id;
279 
280  bool valid = PIOS_USART_validate(usart_dev);
281  PIOS_Assert(valid);
282 
283  /* Force read of dr after sr to make sure to clear error flags */
284  volatile uint16_t sr = usart_dev->cfg->regs->SR;
285  volatile uint8_t dr = usart_dev->cfg->regs->DR;
286 
287  /* Check if RXNE flag is set */
288  bool rx_need_yield = false;
289  if (sr & USART_SR_RXNE) {
290  uint8_t byte = dr;
291  if (usart_dev->rx_in_cb) {
292  uint16_t rc;
293  rc = (usart_dev->rx_in_cb)(usart_dev->rx_in_context, &byte, 1, NULL, &rx_need_yield);
294  if (rc < 1) {
295  /* Lost bytes on rx */
296  usart_dev->rx_dropped += 1;
297  }
298  }
299  }
300 
301  /* Check if TXE flag is set */
302  bool tx_need_yield = false;
303  if (sr & USART_SR_TXE) {
304  if (usart_dev->tx_out_cb) {
305  uint8_t b;
306  uint16_t bytes_to_send;
307 
308  bytes_to_send = (usart_dev->tx_out_cb)(usart_dev->tx_out_context, &b, 1, NULL, &tx_need_yield);
309 
310  if (bytes_to_send > 0) {
311  /* Send the byte we've been given */
312  usart_dev->cfg->regs->DR = b;
313  } else {
314  /* No bytes to send, disable TXE interrupt */
315  USART_ITConfig(usart_dev->cfg->regs, USART_IT_TXE, DISABLE);
316  }
317  } else {
318  /* No bytes to send, disable TXE interrupt */
319  USART_ITConfig(usart_dev->cfg->regs, USART_IT_TXE, DISABLE);
320  }
321  }
322 
323 #if defined(PIOS_INCLUDE_FREERTOS)
324  if (rx_need_yield || tx_need_yield) {
325  vPortYieldFromISR();
326  }
327 #endif /* PIOS_INCLUDE_FREERTOS */
328 }
