74 /*
75  * Reads the airspeed and returns CAS (calibrated airspeed) in the case of success.
76  * In the case of a failed read, returns -1.
77  */
78 float PIOS_MPXV5004_ReadAirspeed(uint8_t airspeedADCPin)
79 {
80  float sensorVal = PIOS_MPXV5004_Measure(airspeedADCPin);
81 
82  //Calculate dynamic pressure, as per docs
83  float Qc = 5.0f*(((sensorVal - calibrationOffset)/4096.0f*3.3f)/VCC - 0.2f);
84 
85  //Saturate Qc on the lower bound, in order to make sure we don't have negative airspeeds. No need
86  // to saturate on the upper bound, we'll handle that later with calibratedAirspeed.
87  if (Qc < 0) {
88  Qc=0;
89  }
90 
91  //Compute calibraterd airspeed, as per http://en.wikipedia.org/wiki/Calibrated_airspeed
92  float calibratedAirspeed = A0*sqrtf(5.0f*(pow(Qc/P0+1.0f,POWER)-1.0f));
93 
94  //Upper bound airspeed. No need to lower bound it, that comes from Qc
95  if (calibratedAirspeed > 80) { //in [m/s]
96  calibratedAirspeed=80;
97  }
