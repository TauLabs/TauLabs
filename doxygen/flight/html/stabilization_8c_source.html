93 int32_t StabilizationStart()
94 {
95  // Initialize variables
96  // Create object queue
97  queue = xQueueCreate(MAX_QUEUE_SIZE, sizeof(UAVObjEvent));
98 
99  // Listen for updates.
100  // AttitudeActualConnectQueue(queue);
101  GyrosConnectQueue(queue);
102 
103  StabilizationSettingsConnectCallback(SettingsUpdatedCb);
104  SettingsUpdatedCb(StabilizationSettingsHandle());
105 
106  // Start main task
107  xTaskCreate(stabilizationTask, (signed char*)"Stabilization", STACK_SIZE_BYTES/4, NULL, TASK_PRIORITY, &taskHandle);
108  TaskMonitorAdd(TASKINFO_RUNNING_STABILIZATION, taskHandle);
109  PIOS_WDG_RegisterFlag(PIOS_WDG_STABILIZATION);
110  return 0;
111 }

136 static void stabilizationTask(void* parameters)
137 {
138  UAVObjEvent ev;
139 
140  uint32_t timeval = PIOS_DELAY_GetRaw();
141 
142  ActuatorDesiredData actuatorDesired;
143  StabilizationDesiredData stabDesired;
144  RateDesiredData rateDesired;
145  AttitudeActualData attitudeActual;
146  GyrosData gyrosData;
147  FlightStatusData flightStatus;
148 
149  float *stabDesiredAxis = &stabDesired.Roll;
150  float *actuatorDesiredAxis = &actuatorDesired.Roll;
151  float *rateDesiredAxis = &rateDesired.Roll;
152 
153  SettingsUpdatedCb((UAVObjEvent *) NULL);
154 
155  // Main task loop
156  ZeroPids();
157  while(1) {
158  float dT;
159 
160  PIOS_WDG_UpdateFlag(PIOS_WDG_STABILIZATION);
161 
162  // Wait until the AttitudeRaw object is updated, if a timeout then go to failsafe
163  if ( xQueueReceive(queue, &ev, FAILSAFE_TIMEOUT_MS / portTICK_RATE_MS) != pdTRUE )
164  {
165  AlarmsSet(SYSTEMALARMS_ALARM_STABILIZATION,SYSTEMALARMS_ALARM_WARNING);
166  continue;
167  }
168 
169  dT = PIOS_DELAY_DiffuS(timeval) * 1.0e-6f;
170  timeval = PIOS_DELAY_GetRaw();
171 
172  FlightStatusGet(&flightStatus);
173  StabilizationDesiredGet(&stabDesired);
174  AttitudeActualGet(&attitudeActual);
175  GyrosGet(&gyrosData);
176  ActuatorDesiredGet(&actuatorDesired);
177 #if defined(RATEDESIRED_DIAGNOSTICS)
178  RateDesiredGet(&rateDesired);
179 #endif
180 
181 #if defined(PIOS_QUATERNION_STABILIZATION)
182  // Quaternion calculation of error in each axis. Uses more memory.
183  float rpy_desired[3];
184  float q_desired[4];
185  float q_error[4];
186  float local_error[3];
187 
188  // Essentially zero errors for anything in rate or none
189  if(stabDesired.StabilizationMode[STABILIZATIONDESIRED_STABILIZATIONMODE_ROLL] == STABILIZATIONDESIRED_STABILIZATIONMODE_ATTITUDE)
190  rpy_desired[0] = stabDesired.Roll;
191  else
192  rpy_desired[0] = attitudeActual.Roll;
193 
194  if(stabDesired.StabilizationMode[STABILIZATIONDESIRED_STABILIZATIONMODE_PITCH] == STABILIZATIONDESIRED_STABILIZATIONMODE_ATTITUDE)
195  rpy_desired[1] = stabDesired.Pitch;
196  else
197  rpy_desired[1] = attitudeActual.Pitch;
198 
199  if(stabDesired.StabilizationMode[STABILIZATIONDESIRED_STABILIZATIONMODE_YAW] == STABILIZATIONDESIRED_STABILIZATIONMODE_ATTITUDE)
200  rpy_desired[2] = stabDesired.Yaw;
201  else
202  rpy_desired[2] = attitudeActual.Yaw;
203 
204  RPY2Quaternion(rpy_desired, q_desired);
205  quat_inverse(q_desired);
206  quat_mult(q_desired, &attitudeActual.q1, q_error);
207  quat_inverse(q_error);
208  Quaternion2RPY(q_error, local_error);
209 
210 #else
211  // Simpler algorithm for CC, less memory
212  float local_error[3] = {stabDesired.Roll - attitudeActual.Roll,
213  stabDesired.Pitch - attitudeActual.Pitch,
214  stabDesired.Yaw - attitudeActual.Yaw};
215  local_error[2] = circular_modulus_deg(local_error[2]);
216 #endif
217 
218  float gyro_filtered[3];
219  gyro_filtered[0] = gyro_filtered[0] * gyro_alpha + gyrosData.x * (1 - gyro_alpha);
220  gyro_filtered[1] = gyro_filtered[1] * gyro_alpha + gyrosData.y * (1 - gyro_alpha);
221  gyro_filtered[2] = gyro_filtered[2] * gyro_alpha + gyrosData.z * (1 - gyro_alpha);
222 
223  // A flag to track which stabilization mode each axis is in
224  static uint8_t previous_mode[MAX_AXES] = {255,255,255};
225  bool error = false;
226 
227  //Run the selected stabilization algorithm on each axis:
228  for(uint8_t i=0; i< MAX_AXES; i++)
229  {
230  // Check whether this axis mode needs to be reinitialized
231  bool reinit = (stabDesired.StabilizationMode[i] != previous_mode[i]);
232  previous_mode[i] = stabDesired.StabilizationMode[i];
233 
234  // Apply the selected control law
235  switch(stabDesired.StabilizationMode[i])
236  {
237  case STABILIZATIONDESIRED_STABILIZATIONMODE_RATE:
238  if(reinit)
239  pids[PID_RATE_ROLL + i].iAccumulator = 0;
240 
241  // Store to rate desired variable for storing to UAVO
242  rateDesiredAxis[i] = bound(stabDesiredAxis[i], settings.ManualRate[i]);
243 
244  // Compute the inner loop
245  actuatorDesiredAxis[i] = pid_apply_setpoint(&pids[PID_RATE_ROLL + i], rateDesiredAxis[i], gyro_filtered[i], dT);
246  actuatorDesiredAxis[i] = bound(actuatorDesiredAxis[i],1.0f);
247 
248  break;
249 
250  case STABILIZATIONDESIRED_STABILIZATIONMODE_ATTITUDE:
251  if(reinit) {
252  pids[PID_ATT_ROLL + i].iAccumulator = 0;
253  pids[PID_RATE_ROLL + i].iAccumulator = 0;
254  }
255 
256  // Compute the outer loop
257  rateDesiredAxis[i] = pid_apply(&pids[PID_ATT_ROLL + i], local_error[i], dT);
258  rateDesiredAxis[i] = bound(rateDesiredAxis[i], settings.MaximumRate[i]);
259 
260  // Compute the inner loop
261  actuatorDesiredAxis[i] = pid_apply_setpoint(&pids[PID_RATE_ROLL + i], rateDesiredAxis[i], gyro_filtered[i], dT);
262  actuatorDesiredAxis[i] = bound(actuatorDesiredAxis[i],1.0f);
263 
264  break;
265 
266  case STABILIZATIONDESIRED_STABILIZATIONMODE_VIRTUALBAR:
267  // Store for debugging output
268  rateDesiredAxis[i] = stabDesiredAxis[i];
269 
270  // Run a virtual flybar stabilization algorithm on this axis
271  stabilization_virtual_flybar(gyro_filtered[i], rateDesiredAxis[i], &actuatorDesiredAxis[i], dT, reinit, i, &settings);
272 
273  break;
274  case STABILIZATIONDESIRED_STABILIZATIONMODE_WEAKLEVELING:
275  {
276  if (reinit)
277  pids[PID_RATE_ROLL + i].iAccumulator = 0;
278 
279  float weak_leveling = local_error[i] * weak_leveling_kp;
280  weak_leveling = bound(weak_leveling, weak_leveling_max);
281 
282  // Compute desired rate as input biased towards leveling
283  rateDesiredAxis[i] = stabDesiredAxis[i] + weak_leveling;
284  actuatorDesiredAxis[i] = pid_apply_setpoint(&pids[PID_RATE_ROLL + i], rateDesiredAxis[i], gyro_filtered[i], dT);
285  actuatorDesiredAxis[i] = bound(actuatorDesiredAxis[i],1.0f);
286 
287  break;
288  }
289  case STABILIZATIONDESIRED_STABILIZATIONMODE_AXISLOCK:
290  if (reinit)
291  pids[PID_RATE_ROLL + i].iAccumulator = 0;
292 
293  if(fabs(stabDesiredAxis[i]) > max_axislock_rate) {
294  // While getting strong commands act like rate mode
295  rateDesiredAxis[i] = stabDesiredAxis[i];
296  axis_lock_accum[i] = 0;
297  } else {
298  // For weaker commands or no command simply attitude lock (almost) on no gyro change
299  axis_lock_accum[i] += (stabDesiredAxis[i] - gyro_filtered[i]) * dT;
300  axis_lock_accum[i] = bound(axis_lock_accum[i], max_axis_lock);
301  rateDesiredAxis[i] = pid_apply(&pids[PID_ATT_ROLL + i], axis_lock_accum[i], dT);
302  }
303 
304  rateDesiredAxis[i] = bound(rateDesiredAxis[i], settings.MaximumRate[i]);
305 
306  actuatorDesiredAxis[i] = pid_apply_setpoint(&pids[PID_RATE_ROLL + i], rateDesiredAxis[i], gyro_filtered[i], dT);
307  actuatorDesiredAxis[i] = bound(actuatorDesiredAxis[i],1.0f);
308 
309  break;
310 
311  case STABILIZATIONDESIRED_STABILIZATIONMODE_RELAYRATE:
312  // Store to rate desired variable for storing to UAVO
313  rateDesiredAxis[i] = bound(stabDesiredAxis[i], settings.ManualRate[i]);
314 
315  // Run the relay controller which also estimates the oscillation parameters
316  stabilization_relay_rate(rateDesiredAxis[i] - gyro_filtered[i], &actuatorDesiredAxis[i], i, reinit);
317  actuatorDesiredAxis[i] = bound(actuatorDesiredAxis[i],1.0);
318 
319  break;
320 
321  case STABILIZATIONDESIRED_STABILIZATIONMODE_RELAYATTITUDE:
322  if(reinit)
323  pids[PID_ATT_ROLL + i].iAccumulator = 0;
324 
325  // Compute the outer loop like attitude mode
326  rateDesiredAxis[i] = pid_apply(&pids[PID_ATT_ROLL + i], local_error[i], dT);
327  rateDesiredAxis[i] = bound(rateDesiredAxis[i], settings.MaximumRate[i]);
328 
329  // Run the relay controller which also estimates the oscillation parameters
330  stabilization_relay_rate(rateDesiredAxis[i] - gyro_filtered[i], &actuatorDesiredAxis[i], i, reinit);
331  actuatorDesiredAxis[i] = bound(actuatorDesiredAxis[i],1.0);
332 
333  break;
334 
335  case STABILIZATIONDESIRED_STABILIZATIONMODE_COORDINATEDFLIGHT:
336  switch (i) {
337  case YAW:
338  if ( stabDesired.StabilizationMode[ROLL]==STABILIZATIONDESIRED_STABILIZATIONMODE_ATTITUDE && //If we are in roll attitude mode..
339  fabs(stabDesired.Roll) > 3.0f && //...and we've requested more than 3 degrees of roll...
340  fabs(stabDesired.Yaw) < 0.02) { //...and we currently have no yaw input within a 2% deadband
341  float accelsDataY;
342  AccelsyGet(&accelsDataY);
343 
344  //Reset integral if we have changed roll to opposite direction from rudder. This implies that we have changed desired turning direction.
345  if ((stabDesired.Roll > 0 && actuatorDesiredAxis[i] < 0) || (stabDesired.Roll < 0 && actuatorDesiredAxis[i] > 0)){
346  pids[PID_RATE_YAW].iAccumulator = 0;
347  }
348 
349  //Coordinate flight can simply be seen as ensuring that there is no lateral acceleration in the
350  // body frame. As such, we use the (noisy) accelerometer data as our measurement. Ideally, at
351  // some point in the future we will estimate acceleration and then we can use the estimated value
352  // instead of the measured value.
353  float errorSlip = -accelsDataY;
354 
355  //Apply a 1 second rise time low-pass filter to the accelerometer driven output.
356  // This reduces jitter in the tail, and helps dampen aileron oscillations
357  // due to the decreased turning radius as rudder input is applied.
358  float alpha = 1.0f-dT/(dT+1.0f);
359 
360  float command=(1-alpha)*pid_apply(&pids[PID_RATE_YAW], errorSlip, dT) + alpha*actuatorDesiredAxis[i];
361  actuatorDesiredAxis[i] = bound(command,1.0);
362  }
363  else{ //Else, yaw input is either passed through manually or we're not requesting
364  actuatorDesiredAxis[i] = bound(stabDesiredAxis[i], 1.0);
365  pids[PID_RATE_YAW].iAccumulator = 0;
366  pids[PID_ATT_YAW].iAccumulator = 0;
367  }
368  break;
369  case ROLL:
370  case PITCH:
371  default:
372  //Coordinated Flight has no effect in these modes. Trigger a configuration error.
373  error = true;
374  break;
375  }
376 
377  break;
378 
379  case STABILIZATIONDESIRED_STABILIZATIONMODE_NONE:
380  actuatorDesiredAxis[i] = bound(stabDesiredAxis[i],1.0f);
381  break;
382  default:
383  error = true;
384  break;
385  }
386  }
387 
388  if (settings.VbarPiroComp == STABILIZATIONSETTINGS_VBARPIROCOMP_TRUE)
389  stabilization_virtual_flybar_pirocomp(gyro_filtered[2], dT);
390 
391 #if defined(RATEDESIRED_DIAGNOSTICS)
392  RateDesiredSet(&rateDesired);
393 #endif
394 
395  // Save dT
396  actuatorDesired.UpdateTime = dT * 1000;
397  actuatorDesired.Throttle = stabDesired.Throttle;
398 
399  if(flightStatus.FlightMode != FLIGHTSTATUS_FLIGHTMODE_MANUAL) {
400  ActuatorDesiredSet(&actuatorDesired);
401  } else {
402  // Force all axes to reinitialize when engaged
403  for(uint8_t i=0; i< MAX_AXES; i++)
404  previous_mode[i] = 255;
405  }
406 
407  if(flightStatus.Armed != FLIGHTSTATUS_ARMED_ARMED ||
408  (lowThrottleZeroIntegral && stabDesired.Throttle < 0))
409  {
410  // Force all axes to reinitialize when engaged
411  for(uint8_t i=0; i< MAX_AXES; i++)
412  previous_mode[i] = 255;
413  }
414 
415  // Clear or set alarms. Done like this to prevent toggling each cycle
416  // and hammering system alarms
417  if (error)
418  AlarmsSet(SYSTEMALARMS_ALARM_STABILIZATION,SYSTEMALARMS_ALARM_ERROR);
419  else
420  AlarmsClear(SYSTEMALARMS_ALARM_STABILIZATION);
421  }
422 }

442 static float bound(float val, float range)
443 {
444  if(val < -range) {
445  val = -range;
446  } else if(val > range) {
447  val = range;
448  }
449  return val;

452 static void SettingsUpdatedCb(UAVObjEvent * ev)
453 {
454  StabilizationSettingsGet(&settings);
455 
456  // Set the roll rate PID constants
457  pid_configure(&pids[PID_RATE_ROLL], settings.RollRatePID[STABILIZATIONSETTINGS_ROLLRATEPID_KP],
458  settings.RollRatePID[STABILIZATIONSETTINGS_ROLLRATEPID_KI],
459  settings.RollRatePID[STABILIZATIONSETTINGS_ROLLRATEPID_KD],
460  settings.RollRatePID[STABILIZATIONSETTINGS_ROLLRATEPID_ILIMIT]);
461 
462  // Set the pitch rate PID constants
463  pid_configure(&pids[PID_RATE_PITCH], settings.PitchRatePID[STABILIZATIONSETTINGS_PITCHRATEPID_KP],
464  settings.PitchRatePID[STABILIZATIONSETTINGS_PITCHRATEPID_KI],
465  settings.PitchRatePID[STABILIZATIONSETTINGS_PITCHRATEPID_KD],
466  settings.PitchRatePID[STABILIZATIONSETTINGS_PITCHRATEPID_ILIMIT]);
467 
468  // Set the yaw rate PID constants
469  pid_configure(&pids[PID_RATE_YAW], settings.YawRatePID[STABILIZATIONSETTINGS_YAWRATEPID_KP],
470  settings.YawRatePID[STABILIZATIONSETTINGS_YAWRATEPID_KI],
471  settings.YawRatePID[STABILIZATIONSETTINGS_YAWRATEPID_KD],
472  settings.YawRatePID[STABILIZATIONSETTINGS_YAWRATEPID_ILIMIT]);
473 
474  // Set the roll attitude PI constants
475  pid_configure(&pids[PID_ATT_ROLL], settings.RollPI[STABILIZATIONSETTINGS_ROLLPI_KP],
476  settings.RollPI[STABILIZATIONSETTINGS_ROLLPI_KI], 0,
477  settings.RollPI[STABILIZATIONSETTINGS_ROLLPI_ILIMIT]);
478 
479  // Set the pitch attitude PI constants
480  pid_configure(&pids[PID_ATT_PITCH], settings.PitchPI[STABILIZATIONSETTINGS_PITCHPI_KP],
481  settings.PitchPI[STABILIZATIONSETTINGS_PITCHPI_KI], 0,
482  settings.PitchPI[STABILIZATIONSETTINGS_PITCHPI_ILIMIT]);
483 
484  // Set the yaw attitude PI constants
485  pid_configure(&pids[PID_ATT_YAW], settings.YawPI[STABILIZATIONSETTINGS_YAWPI_KP],
486  settings.YawPI[STABILIZATIONSETTINGS_YAWPI_KI], 0,
487  settings.YawPI[STABILIZATIONSETTINGS_YAWPI_ILIMIT]);
488 
489  // Set up the derivative term
490  pid_configure_derivative(settings.DerivativeCutoff, settings.DerivativeGamma);
491 
492  // Maximum deviation to accumulate for axis lock
493  max_axis_lock = settings.MaxAxisLock;
494  max_axislock_rate = settings.MaxAxisLockRate;
495 
496  // Settings for weak leveling
497  weak_leveling_kp = settings.WeakLevelingKp;
498  weak_leveling_max = settings.MaxWeakLevelingRate;
499 
500  // Whether to zero the PID integrals while throttle is low
501  lowThrottleZeroIntegral = settings.LowThrottleZeroIntegral == STABILIZATIONSETTINGS_LOWTHROTTLEZEROINTEGRAL_TRUE;
502 
503  // The dT has some jitter iteration to iteration that we don't want to
504  // make thie result unpredictable. Still, it's nicer to specify the constant
505  // based on a time (in ms) rather than a fixed multiplier. The error between
506  // update rates on OP (~300 Hz) and CC (~475 Hz) is negligible for this
507  // calculation
508  const float fakeDt = 0.0025;
509  if(settings.GyroTau < 0.0001)
510  gyro_alpha = 0; // not trusting this to resolve to 0
511  else
512  gyro_alpha = expf(-fakeDt / settings.GyroTau);
513 
514  // Compute time constant for vbar decay term based on a tau
515  vbar_decay = expf(-fakeDt / settings.VbarTau);
516 }
