97 static void manualControlTask(void *parameters)
98 {
99  /* Make sure disarmed on power up */
100  FlightStatusData flightStatus;
101  FlightStatusGet(&flightStatus);
102  flightStatus.Armed = FLIGHTSTATUS_ARMED_DISARMED;
103 
104  // Main task loop
105  lastSysTime = xTaskGetTickCount();
106 
107  // Select failsafe before run
108  failsafe_control_select(true);
109 
110  while (1) {
111 
112  // Process periodic data for each of the controllers, including reading
113  // all available inputs
114  failsafe_control_update();
115  transmitter_control_update();
116  tablet_control_update();
117 
118  static enum control_selection last_control_selection = CONTROL_SELECTION_FAILSAFE;
119  enum control_events control_events = CONTROL_EVENTS_NONE;
120 
121  // Control logic to select the valid controller
122  enum control_selection control_selection = transmitter_control_selected_controller();
123  bool reset_controller = control_selection != last_control_selection;
124 
125  switch(control_selection) {
126  case CONTROL_SELECTION_FAILSAFE:
127  failsafe_control_select(reset_controller);
128  control_events = failsafe_control_get_events();
129  break;
130  case CONTROL_SELECTION_TRANSMITTER:
131  transmitter_control_select(reset_controller);
132  control_events = transmitter_control_get_events();
133  break;
134  case CONTROL_SELECTION_TABLET:
135  if (tablet_control_select(reset_controller) == 0) {
136  control_events = tablet_control_get_events();
137  } else {
138  // Failure in tablet control. This would be better if done
139  // at the selection stage before the tablet is even used.
140  failsafe_control_select(false);
141  control_events = failsafe_control_get_events();
142  }
143  break;
144  }
145  last_control_selection = control_selection;
146 
147  // TODO: This can evolve into a full FSM like I2C possibly
148  switch(control_events) {
149  case CONTROL_EVENTS_NONE:
150  break;
151  case CONTROL_EVENTS_ARM:
152  control_event_arm();
153  break;
154  case CONTROL_EVENTS_ARMING:
155  control_event_arming();
156  break;
157  case CONTROL_EVENTS_DISARM:
158  control_event_disarm();
159  break;
160  }
161 
162  // Wait until next update
163  vTaskDelayUntil(&lastSysTime, UPDATE_PERIOD_MS / portTICK_RATE_MS);
164  PIOS_WDG_UpdateFlag(PIOS_WDG_MANUAL);
165  }
166 }

169 static int32_t control_event_arm()
170 {
171  if(ok_to_arm()) {
172  FlightStatusData flightStatus;
173  FlightStatusGet(&flightStatus);
174  if (flightStatus.Armed != FLIGHTSTATUS_ARMED_ARMED) {
175  flightStatus.Armed = FLIGHTSTATUS_ARMED_ARMED;
176  FlightStatusSet(&flightStatus);
177  }
178  }
179  return 0;
180 }

211 static bool ok_to_arm(void)
212 {
213  // read alarms
214  SystemAlarmsData alarms;
215  SystemAlarmsGet(&alarms);
216 
217  // Check each alarm
218  for (int i = 0; i < SYSTEMALARMS_ALARM_NUMELEM; i++)
219  {
220  if (alarms.Alarm[i] >= SYSTEMALARMS_ALARM_ERROR &&
221  i != SYSTEMALARMS_ALARM_GPS &&
222  i != SYSTEMALARMS_ALARM_TELEMETRY)
223  {
224  return false;
225  }
226  }
227 
228  return true;
229 }
