00104 static void manualControlTask(void *parameters)
00105 {
00106         /* Make sure disarmed on power up */
00107         FlightStatusData flightStatus;
00108         FlightStatusGet(&flightStatus);
00109         flightStatus.Armed = FLIGHTSTATUS_ARMED_DISARMED;
00110 
00111         // Main task loop
00112         lastSysTime = xTaskGetTickCount();
00113 
00114         // Select failsafe before run
00115         failsafe_control_select(true);
00116 
00117         while (1) {
00118 
00119                 // Process periodic data for each of the controllers, including reading
00120                 // all available inputs
00121                 failsafe_control_update();
00122                 transmitter_control_update();
00123                 tablet_control_update();
00124 
00125                 static enum control_selection last_control_selection = CONTROL_SELECTION_FAILSAFE;
00126                 enum control_events control_events = CONTROL_EVENTS_NONE;
00127 
00128                 // Control logic to select the valid controller
00129                 enum control_selection control_selection = transmitter_control_selected_controller();
00130                 bool reset_controller = control_selection != last_control_selection;
00131 
00132                 switch(control_selection) {
00133                 case CONTROL_SELECTION_FAILSAFE:
00134                         failsafe_control_select(reset_controller);
00135                         control_events = failsafe_control_get_events();
00136                         break;
00137                 case CONTROL_SELECTION_TRANSMITTER:
00138                         transmitter_control_select(reset_controller);
00139                         control_events = transmitter_control_get_events();
00140                         break;
00141                 case CONTROL_SELECTION_TABLET:
00142                         if (tablet_control_select(reset_controller) == 0) {
00143                                 control_events = tablet_control_get_events();
00144                         } else {
00145                                 // Failure in tablet control.  This would be better if done
00146                                 // at the selection stage before the tablet is even used.
00147                                 failsafe_control_select(false);
00148                                 control_events = failsafe_control_get_events();
00149                         }
00150                         break;
00151                 }
00152                 last_control_selection = control_selection;
00153 
00154                 // TODO: This can evolve into a full FSM like I2C possibly
00155                 switch(control_events) {
00156                 case CONTROL_EVENTS_NONE:
00157                         break;
00158                 case CONTROL_EVENTS_ARM:
00159                         control_event_arm();
00160                         break;
00161                 case CONTROL_EVENTS_ARMING:
00162                         control_event_arming();
00163                         break;
00164                 case CONTROL_EVENTS_DISARM:
00165                         control_event_disarm();
00166                         break;
00167                 }
00168 
00169                 // Wait until next update
00170                 vTaskDelayUntil(&lastSysTime, UPDATE_PERIOD_MS / portTICK_RATE_MS);
00171                 PIOS_WDG_UpdateFlag(PIOS_WDG_MANUAL);
00172         }
