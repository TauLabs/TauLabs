124 /* Initialise S.Bus receiver interface */
125 int32_t PIOS_SBus_Init(uint32_t *sbus_id,
126  const struct pios_sbus_cfg *cfg,
127  const struct pios_com_driver *driver,
128  uint32_t lower_id)
129 {
130  PIOS_DEBUG_Assert(sbus_id);
131  PIOS_DEBUG_Assert(cfg);
132  PIOS_DEBUG_Assert(driver);
133 
134  struct pios_sbus_dev *sbus_dev;
135 
136  sbus_dev = (struct pios_sbus_dev *)PIOS_SBus_Alloc();
137  if (!sbus_dev) goto out_fail;
138 
139  /* Bind the configuration to the device instance */
140  sbus_dev->cfg = cfg;
141 
142  PIOS_SBus_ResetState(&(sbus_dev->state));
143 
144  *sbus_id = (uint32_t)sbus_dev;
145 
146  /* Enable inverter clock and enable the inverter */
147  if (cfg->gpio_clk_func != NULL)
148  (*cfg->gpio_clk_func)(cfg->gpio_clk_periph, ENABLE);
149  if (cfg->inv.gpio != NULL)
150  {
151  GPIO_Init(cfg->inv.gpio, (GPIO_InitTypeDef*)&cfg->inv.init);
152  GPIO_WriteBit(cfg->inv.gpio, cfg->inv.init.GPIO_Pin, cfg->gpio_inv_enable);
153  }
154 
155  /* Set comm driver callback */
156  (driver->bind_rx_cb)(lower_id, PIOS_SBus_RxInCallback, *sbus_id);
157 
158  if (!PIOS_RTC_RegisterTickCallback(PIOS_SBus_Supervisor, *sbus_id)) {
159  PIOS_DEBUG_Assert(0);
160  }
161 
162  return 0;
163 
164 out_fail:
165  return -1;
166 }

175 static int32_t PIOS_SBus_Get(uint32_t rcvr_id, uint8_t channel)
176 {

178 
179  if (!PIOS_SBus_Validate(sbus_dev))

195 static void PIOS_SBus_UnrollChannels(struct pios_sbus_state *state)
196 {
197  uint8_t *s = state->received_data;
198  uint16_t *d = state->channel_data;
199 
200 #define F(v,s) (((v) >> (s)) & 0x7ff)
201 
202  /* unroll channels 1-8 */
203  *d++ = F(s[0] | s[1] << 8, 0);
204  *d++ = F(s[1] | s[2] << 8, 3);
205  *d++ = F(s[2] | s[3] << 8 | s[4] << 16, 6);
206  *d++ = F(s[4] | s[5] << 8, 1);
207  *d++ = F(s[5] | s[6] << 8, 4);
208  *d++ = F(s[6] | s[7] << 8 | s[8] << 16, 7);
209  *d++ = F(s[8] | s[9] << 8, 2);
210  *d++ = F(s[9] | s[10] << 8, 5);
211 
212  /* unroll channels 9-16 */
213  *d++ = F(s[11] | s[12] << 8, 0);
214  *d++ = F(s[12] | s[13] << 8, 3);
215  *d++ = F(s[13] | s[14] << 8 | s[15] << 16, 6);
216  *d++ = F(s[15] | s[16] << 8, 1);
217  *d++ = F(s[16] | s[17] << 8, 4);
218  *d++ = F(s[17] | s[18] << 8 | s[19] << 16, 7);
219  *d++ = F(s[19] | s[20] << 8, 2);
220  *d++ = F(s[20] | s[21] << 8, 5);
221 
222  /* unroll discrete channels 17 and 18 */
223  *d++ = (s[22] & SBUS_FLAG_DC1) ? SBUS_VALUE_MAX : SBUS_VALUE_MIN;
224  *d++ = (s[22] & SBUS_FLAG_DC2) ? SBUS_VALUE_MAX : SBUS_VALUE_MIN;
225 }

227 /* Update decoder state processing input byte from the S.Bus stream */
228 static void PIOS_SBus_UpdateState(struct pios_sbus_state *state, uint8_t b)
229 {
230  /* should not process any data until new frame is found */
231  if (!state->frame_found)
232  return;
233 
234  if (state->byte_count == 0) {
235  if (b != SBUS_SOF_BYTE) {
236  /* discard the whole frame if the 1st byte is not correct */
237  state->frame_found = 0;
238  } else {
239  /* do not store the SOF byte */
240  state->byte_count++;
241  }
242  return;
243  }
244 
245  /* do not store last frame byte as well */
246  if (state->byte_count < SBUS_FRAME_LENGTH - 1) {
247  /* store next byte */
248  state->received_data[state->byte_count - 1] = b;
249  state->byte_count++;
250  } else {
251  if (b == SBUS_EOF_BYTE) {
252  /* full frame received */
253  uint8_t flags = state->received_data[SBUS_FRAME_LENGTH - 3];
254  if (flags & SBUS_FLAG_FL) {
255  /* frame lost, do not update */
256  } else if (flags & SBUS_FLAG_FS) {
257  /* failsafe flag active */
258  PIOS_SBus_ResetChannels(state);
259  } else {
260  /* data looking good */
261  PIOS_SBus_UnrollChannels(state);
262  state->failsafe_timer = 0;
263  }
264  } else {
265  /* discard whole frame */
266  }
267 
268  /* prepare for the next frame */
269  state->frame_found = 0;
270  }
271 }

273 /* Comm byte received callback */
274 static uint16_t PIOS_SBus_RxInCallback(uint32_t context,
275  uint8_t *buf,
276  uint16_t buf_len,
277  uint16_t *headroom,
278  bool *need_yield)
279 {
280  struct pios_sbus_dev *sbus_dev = (struct pios_sbus_dev *)context;
281 
282  bool valid = PIOS_SBus_Validate(sbus_dev);
283  PIOS_Assert(valid);
284 
285  struct pios_sbus_state *state = &(sbus_dev->state);
286 
287  /* process byte(s) and clear receive timer */
288  for (uint8_t i = 0; i < buf_len; i++) {
289  PIOS_SBus_UpdateState(state, buf[i]);
290  state->receive_timer = 0;
291  }
292 
293  /* Always signal that we can accept another byte */
294  if (headroom)
295  *headroom = SBUS_FRAME_LENGTH;
296 
297  /* We never need a yield */
298  *need_yield = false;
299 
300  /* Always indicate that all bytes were consumed */
301  return buf_len;
302 }

316 static void PIOS_SBus_Supervisor(uint32_t sbus_id)
317 {
318  struct pios_sbus_dev *sbus_dev = (struct pios_sbus_dev *)sbus_id;
319 
320  bool valid = PIOS_SBus_Validate(sbus_dev);
321  PIOS_Assert(valid);
322 
323  struct pios_sbus_state *state = &(sbus_dev->state);
324 
325  /* waiting for new frame if no bytes were received in 3.2ms */
326  if (++state->receive_timer > 2) {
327  state->frame_found = 1;
328  state->byte_count = 0;
329  state->receive_timer = 0;
330  }
331 
332  /* activate failsafe if no frames have arrived in 102.4ms */
333  if (++state->failsafe_timer > 64) {
334  PIOS_SBus_ResetChannels(state);
335  state->failsafe_timer = 0;
336  }
337 }
