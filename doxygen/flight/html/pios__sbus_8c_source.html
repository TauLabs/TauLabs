00228 static void PIOS_SBus_UpdateState(struct pios_sbus_state *state, uint8_t b)
00229 {
00230         /* should not process any data until new frame is found */

00232                 return;
00233 
00234         if (state->byte_count == 0) {
00235                 if (b != SBUS_SOF_BYTE) {
00236                         /* discard the whole frame if the 1st byte is not correct */
00237                         state->frame_found = 0;
00238                 } else {
00239                         /* do not store the SOF byte */
00240                         state->byte_count++;
00241                 }
00242                 return;
00243         }
00244 
00245         /* do not store last frame byte as well */
00246         if (state->byte_count < SBUS_FRAME_LENGTH - 1) {
00247                 /* store next byte */
00248                 state->received_data[state->byte_count - 1] = b;
00249                 state->byte_count++;
00250         } else {
00251                 if (b == SBUS_EOF_BYTE) {
00252                         /* full frame received */
00253                         uint8_t flags = state->received_data[SBUS_FRAME_LENGTH - 3];
00254                         if (flags & SBUS_FLAG_FL) {
00255                                 /* frame lost, do not update */
00256                         } else if (flags & SBUS_FLAG_FS) {
00257                                 /* failsafe flag active */
00258                                 PIOS_SBus_ResetChannels(state);
00259                         } else {
00260                                 /* data looking good */
00261                                 PIOS_SBus_UnrollChannels(state);
00262                                 state->failsafe_timer = 0;
00263                         }
00264                 } else {
00265                         /* discard whole frame */
00266                 }
00267 
00268                 /* prepare for the next frame */
