00395 float ProcessMixer(const int index, const float curve1, const float curve2,
00396                    const MixerSettingsData* mixerSettings, ActuatorDesiredData* desired, const float period)
00397 {
00398         static float lastFilteredResult[MAX_MIX_ACTUATORS];
00399         const Mixer_t * mixers = (Mixer_t *)&mixerSettings->Mixer1Type; //pointer to array of mixers in UAVObjects
00400         const Mixer_t * mixer = &mixers[index];
00401 
00402         float result = (((float)mixer->matrix[MIXERSETTINGS_MIXER1VECTOR_THROTTLECURVE1] / 128.0f) * curve1) +
00403                        (((float)mixer->matrix[MIXERSETTINGS_MIXER1VECTOR_THROTTLECURVE2] / 128.0f) * curve2) +
00404                        (((float)mixer->matrix[MIXERSETTINGS_MIXER1VECTOR_ROLL] / 128.0f) * desired->Roll) +
00405                        (((float)mixer->matrix[MIXERSETTINGS_MIXER1VECTOR_PITCH] / 128.0f) * desired->Pitch) +
00406                        (((float)mixer->matrix[MIXERSETTINGS_MIXER1VECTOR_YAW] / 128.0f) * desired->Yaw);
00407 
00408         if(mixer->type == MIXERSETTINGS_MIXER1TYPE_MOTOR)
00409         {
00410                 if(result < 0.0f) //idle throttle
00411                 {
00412                         result = 0.0f;
00413                 }
00414 
00415                 //feed forward
00416                 float accumulator = filterAccumulator[index];
00417                 accumulator += (result - lastResult[index]) * mixerSettings->FeedForward;
00418                 lastResult[index] = result;
00419                 result += accumulator;
00420                 if(period !=0)
00421                 {
00422                         if(accumulator > 0.0f)
00423                         {
00424                                 float filter = mixerSettings->AccelTime / period;
00425                                 if(filter <1)
00426                                 {
00427                                         filter = 1;
00428                                 }
00429                                 accumulator -= accumulator / filter;
00430                         }else
00431                         {
00432                                 float filter = mixerSettings->DecelTime / period;
00433                                 if(filter <1)
00434                                 {
00435                                         filter = 1;
00436                                 }
00437                                 accumulator -= accumulator / filter;
00438                         }
00439                 }
00440                 filterAccumulator[index] = accumulator;
00441                 result += accumulator;
00442 
00443                 //acceleration limit
00444                 float dt = result - lastFilteredResult[index];
00445                 float maxDt = mixerSettings->MaxAccel * period;
00446                 if(dt > maxDt) //we are accelerating too hard
00447                 {
00448                         result = lastFilteredResult[index] + maxDt;
00449                 }
00450                 lastFilteredResult[index] = result;
00451         }
00452 
00453         return(result);
00454 }

00461 static float MixerCurve(const float throttle, const float* curve, uint8_t elements)
00462 {
00463         float scale = throttle * (float) (elements - 1);
00464         int idx1 = scale;
00465         scale -= (float)idx1; //remainder
00466         if(curve[0] < -1)
00467         {
00468                 return(throttle);
00469         }
00470         if (idx1 < 0)
00471         {
00472                 idx1 = 0; //clamp to lowest entry in table
00473                 scale = 0;
00474         }
00475         int idx2 = idx1 + 1;
00476         if(idx2 >= elements)
00477         {
00478                 idx2 = elements -1; //clamp to highest entry in table
00479                 if(idx1 >= elements)
00480                 {
00481                         idx1 = elements -1;
00482                 }
00483         }
00484         return curve[idx1] * (1.0f - scale) + curve[idx2] * scale;
00485 }
