150 static void actuatorTask(void* parameters)
151 {
152  UAVObjEvent ev;
153  portTickType lastSysTime;
154  portTickType thisSysTime;
155  float dT = 0.0f;
156 
157  ActuatorCommandData command;
158  ActuatorDesiredData desired;
159  MixerStatusData mixerStatus;
160  FlightStatusData flightStatus;
161 
162  /* Read initial values of ActuatorSettings */
163  ActuatorSettingsData actuatorSettings;
164  actuator_settings_updated = false;
165  ActuatorSettingsGet(&actuatorSettings);
166 
167  /* Read initial values of MixerSettings */
168  MixerSettingsData mixerSettings;
169  mixer_settings_updated = false;
170  MixerSettingsGet(&mixerSettings);
171 
172  /* Force an initial configuration of the actuator update rates */
173  actuator_update_rate_if_changed(&actuatorSettings, true);
174 
175  // Go to the neutral (failsafe) values until an ActuatorDesired update is received
176  setFailsafe(&actuatorSettings, &mixerSettings);
177 
178  // Main task loop
179  lastSysTime = xTaskGetTickCount();
180  while (1)
181  {
182  PIOS_WDG_UpdateFlag(PIOS_WDG_ACTUATOR);
183 
184  // Wait until the ActuatorDesired object is updated
185  uint8_t rc = xQueueReceive(queue, &ev, FAILSAFE_TIMEOUT_MS / portTICK_RATE_MS);
186 
187  /* Process settings updated events even in timeout case so we always act on the latest settings */
188  if (actuator_settings_updated) {
189  actuator_settings_updated = false;
190  ActuatorSettingsGet (&actuatorSettings);
191  actuator_update_rate_if_changed (&actuatorSettings, false);
192  }
193  if (mixer_settings_updated) {
194  mixer_settings_updated = false;
195  MixerSettingsGet (&mixerSettings);
196  }
197 
198  if (rc != pdTRUE) {
199  /* Update of ActuatorDesired timed out. Go to failsafe */
200  setFailsafe(&actuatorSettings, &mixerSettings);
201  continue;
202  }
203 
204  // Check how long since last update
205  thisSysTime = xTaskGetTickCount();
206  if(thisSysTime > lastSysTime) // reuse dt in case of wraparound
207  dT = (thisSysTime - lastSysTime) / portTICK_RATE_MS / 1000.0f;
208  lastSysTime = thisSysTime;
209 
210  FlightStatusGet(&flightStatus);
211  ActuatorDesiredGet(&desired);
212  ActuatorCommandGet(&command);
213 
214 #if defined(MIXERSTATUS_DIAGNOSTICS)
215  MixerStatusGet(&mixerStatus);
216 #endif
217  int nMixers = 0;
218  Mixer_t * mixers = (Mixer_t *)&mixerSettings.Mixer1Type;
219  for(int ct=0; ct < MAX_MIX_ACTUATORS; ct++)
220  {
221  if(mixers[ct].type != MIXERSETTINGS_MIXER1TYPE_DISABLED)
222  {
223  nMixers ++;
224  }
225  }
226  if((nMixers < 2) && !ActuatorCommandReadOnly()) //Nothing can fly with less than two mixers.
227  {
228  setFailsafe(&actuatorSettings, &mixerSettings); // So that channels like PWM buzzer keep working
229  continue;
230  }
231 
232  AlarmsClear(SYSTEMALARMS_ALARM_ACTUATOR);
233 
234  bool armed = flightStatus.Armed == FLIGHTSTATUS_ARMED_ARMED;
235  bool positiveThrottle = desired.Throttle >= 0.00f;
236  bool spinWhileArmed = actuatorSettings.MotorsSpinWhileArmed == ACTUATORSETTINGS_MOTORSSPINWHILEARMED_TRUE;
237 
238  float curve1 = MixerCurve(desired.Throttle,mixerSettings.ThrottleCurve1,MIXERSETTINGS_THROTTLECURVE1_NUMELEM);
239 
240  //The source for the secondary curve is selectable
241  float curve2 = 0;
242  AccessoryDesiredData accessory;
243  switch(mixerSettings.Curve2Source) {
244  case MIXERSETTINGS_CURVE2SOURCE_THROTTLE:
245  curve2 = MixerCurve(desired.Throttle,mixerSettings.ThrottleCurve2,MIXERSETTINGS_THROTTLECURVE2_NUMELEM);
246  break;
247  case MIXERSETTINGS_CURVE2SOURCE_ROLL:
248  curve2 = MixerCurve(desired.Roll,mixerSettings.ThrottleCurve2,MIXERSETTINGS_THROTTLECURVE2_NUMELEM);
249  break;
250  case MIXERSETTINGS_CURVE2SOURCE_PITCH:
251  curve2 = MixerCurve(desired.Pitch,mixerSettings.ThrottleCurve2,
252  MIXERSETTINGS_THROTTLECURVE2_NUMELEM);
253  break;
254  case MIXERSETTINGS_CURVE2SOURCE_YAW:
255  curve2 = MixerCurve(desired.Yaw,mixerSettings.ThrottleCurve2,MIXERSETTINGS_THROTTLECURVE2_NUMELEM);
256  break;
257  case MIXERSETTINGS_CURVE2SOURCE_COLLECTIVE:
258  ManualControlCommandCollectiveGet(&curve2);
259  curve2 = MixerCurve(curve2,mixerSettings.ThrottleCurve2,
260  MIXERSETTINGS_THROTTLECURVE2_NUMELEM);
261  break;
262  case MIXERSETTINGS_CURVE2SOURCE_ACCESSORY0:
263  case MIXERSETTINGS_CURVE2SOURCE_ACCESSORY1:
264  case MIXERSETTINGS_CURVE2SOURCE_ACCESSORY2:
265  case MIXERSETTINGS_CURVE2SOURCE_ACCESSORY3:
266  case MIXERSETTINGS_CURVE2SOURCE_ACCESSORY4:
267  case MIXERSETTINGS_CURVE2SOURCE_ACCESSORY5:
268  if(AccessoryDesiredInstGet(mixerSettings.Curve2Source - MIXERSETTINGS_CURVE2SOURCE_ACCESSORY0,&accessory) == 0)
269  curve2 = MixerCurve(accessory.AccessoryVal,mixerSettings.ThrottleCurve2,MIXERSETTINGS_THROTTLECURVE2_NUMELEM);
270  else
271  curve2 = 0;
272  break;
273  }
274 
275  float * status = (float *)&mixerStatus; //access status objects as an array of floats
276 
277  for(int ct=0; ct < MAX_MIX_ACTUATORS; ct++)
278  {
279  if(mixers[ct].type == MIXERSETTINGS_MIXER1TYPE_DISABLED) {
280  // Set to minimum if disabled. This is not the same as saying PWM pulse = 0 us
281  status[ct] = -1;
282  command.Channel[ct] = 0;
283  continue;
284  }
285 
286  if((mixers[ct].type == MIXERSETTINGS_MIXER1TYPE_MOTOR) || (mixers[ct].type == MIXERSETTINGS_MIXER1TYPE_SERVO))
287  status[ct] = ProcessMixer(ct, curve1, curve2, &mixerSettings, &desired, dT);
288  else
289  status[ct] = -1;
290 
291 
292 
293  // Motors have additional protection for when to be on
294  if(mixers[ct].type == MIXERSETTINGS_MIXER1TYPE_MOTOR) {
295 
296  // If not armed or motors aren't meant to spin all the time
297  if( !armed ||
298  (!spinWhileArmed && !positiveThrottle))
299  {
300  filterAccumulator[ct] = 0;
301  lastResult[ct] = 0;
302  status[ct] = -1; //force min throttle
303  }
304  // If armed meant to keep spinning,
305  else if ((spinWhileArmed && !positiveThrottle) ||
306  (status[ct] < 0) )
307  status[ct] = 0;
308  }
309 
310  // If an accessory channel is selected for direct bypass mode
311  // In this configuration the accessory channel is scaled and mapped
312  // directly to output. Note: THERE IS NO SAFETY CHECK HERE FOR ARMING
313  // these also will not be updated in failsafe mode. I'm not sure what
314  // the correct behavior is since it seems domain specific. I don't love
315  // this code
316  if( (mixers[ct].type >= MIXERSETTINGS_MIXER1TYPE_ACCESSORY0) &&
317  (mixers[ct].type <= MIXERSETTINGS_MIXER1TYPE_ACCESSORY5))
318  {
319  if(AccessoryDesiredInstGet(mixers[ct].type - MIXERSETTINGS_MIXER1TYPE_ACCESSORY0,&accessory) == 0)
320  status[ct] = accessory.AccessoryVal;
321  else
322  status[ct] = -1;
323  }
324  if( (mixers[ct].type >= MIXERSETTINGS_MIXER1TYPE_CAMERAROLL) &&
325  (mixers[ct].type <= MIXERSETTINGS_MIXER1TYPE_CAMERAYAW))
326  {
327  CameraDesiredData cameraDesired;
328  if( CameraDesiredGet(&cameraDesired) == 0 ) {
329  switch(mixers[ct].type) {
330  case MIXERSETTINGS_MIXER1TYPE_CAMERAROLL:
331  status[ct] = cameraDesired.Roll;
332  break;
333  case MIXERSETTINGS_MIXER1TYPE_CAMERAPITCH:
334  status[ct] = cameraDesired.Pitch;
335  break;
336  case MIXERSETTINGS_MIXER1TYPE_CAMERAYAW:
337  status[ct] = cameraDesired.Yaw;
338  break;
339  default:
340  break;
341  }
342  }
343  else
344  status[ct] = -1;
345  }
346  }
347 
348  for(int i = 0; i < MAX_MIX_ACTUATORS; i++)
349  command.Channel[i] = scaleChannel(status[i],
350  actuatorSettings.ChannelMax[i],
351  actuatorSettings.ChannelMin[i],
352  actuatorSettings.ChannelNeutral[i]);
353 
354  // Store update time
355  command.UpdateTime = 1000.0f*dT;
356  if(1000.0f*dT > command.MaxUpdateTime)
357  command.MaxUpdateTime = 1000.0f*dT;
358 
359  // Update output object
360  ActuatorCommandSet(&command);
361  // Update in case read only (eg. during servo configuration)
362  ActuatorCommandGet(&command);
363 
364 #if defined(MIXERSTATUS_DIAGNOSTICS)
365  MixerStatusSet(&mixerStatus);
366 #endif
367 
368 
369  // Update servo outputs
370  bool success = true;
371 
372  for (int n = 0; n < ACTUATORCOMMAND_CHANNEL_NUMELEM; ++n)
373  {
374  success &= set_channel(n, command.Channel[n], &actuatorSettings);
375  }
376 
377  if(!success) {
378  command.NumFailedUpdates++;
379  ActuatorCommandSet(&command);
380  AlarmsSet(SYSTEMALARMS_ALARM_ACTUATOR, SYSTEMALARMS_ALARM_CRITICAL);
381  }
382 
383  }
384 }

391 float ProcessMixer(const int index, const float curve1, const float curve2,
392  const MixerSettingsData* mixerSettings, ActuatorDesiredData* desired, const float period)
393 {
394  static float lastFilteredResult[MAX_MIX_ACTUATORS];
395  const Mixer_t * mixers = (Mixer_t *)&mixerSettings->Mixer1Type; //pointer to array of mixers in UAVObjects
396  const Mixer_t * mixer = &mixers[index];
397 
398  float result = (((float)mixer->matrix[MIXERSETTINGS_MIXER1VECTOR_THROTTLECURVE1] / 128.0f) * curve1) +
399  (((float)mixer->matrix[MIXERSETTINGS_MIXER1VECTOR_THROTTLECURVE2] / 128.0f) * curve2) +
400  (((float)mixer->matrix[MIXERSETTINGS_MIXER1VECTOR_ROLL] / 128.0f) * desired->Roll) +
401  (((float)mixer->matrix[MIXERSETTINGS_MIXER1VECTOR_PITCH] / 128.0f) * desired->Pitch) +
402  (((float)mixer->matrix[MIXERSETTINGS_MIXER1VECTOR_YAW] / 128.0f) * desired->Yaw);
403 
404  if(mixer->type == MIXERSETTINGS_MIXER1TYPE_MOTOR)
405  {
406  if(result < 0.0f) //idle throttle
407  {
408  result = 0.0f;
409  }
410 
411  //feed forward
412  float accumulator = filterAccumulator[index];
413  accumulator += (result - lastResult[index]) * mixerSettings->FeedForward;
414  lastResult[index] = result;
415  result += accumulator;
416  if(period !=0)
417  {
418  if(accumulator > 0.0f)
419  {
420  float filter = mixerSettings->AccelTime / period;
421  if(filter <1)
422  {
423  filter = 1;
424  }
425  accumulator -= accumulator / filter;
426  }else
427  {
428  float filter = mixerSettings->DecelTime / period;
429  if(filter <1)
430  {
431  filter = 1;
432  }
433  accumulator -= accumulator / filter;
434  }
435  }
436  filterAccumulator[index] = accumulator;
437  result += accumulator;
438 
439  //acceleration limit
440  float dt = result - lastFilteredResult[index];
441  float maxDt = mixerSettings->MaxAccel * period;
442  if(dt > maxDt) //we are accelerating too hard
443  {
444  result = lastFilteredResult[index] + maxDt;
445  }
446  lastFilteredResult[index] = result;
447  }
448 
449  return(result);
450 }

457 static float MixerCurve(const float throttle, const float* curve, uint8_t elements)
458 {
459  float scale = throttle * (float) (elements - 1);
460  int idx1 = scale;
461  scale -= (float)idx1; //remainder
462  if(curve[0] < -1)
463  {
464  return(throttle);
465  }
466  if (idx1 < 0)
467  {
468  idx1 = 0; //clamp to lowest entry in table
469  scale = 0;
470  }
471  int idx2 = idx1 + 1;
472  if(idx2 >= elements)
473  {
474  idx2 = elements -1; //clamp to highest entry in table
475  if(idx1 >= elements)
476  {
477  idx1 = elements -1;
478  }
479  }
480  return curve[idx1] * (1.0f - scale) + curve[idx2] * scale;
481 }

487 static int16_t scaleChannel(float value, int16_t max, int16_t min, int16_t neutral)
488 {
489  int16_t valueScaled;
490  // Scale
491  if ( value >= 0.0f)
492  {
493  valueScaled = (int16_t)(value*((float)(max-neutral))) + neutral;
494  }
495  else
496  {
497  valueScaled = (int16_t)(value*((float)(neutral-min))) + neutral;
498  }
499 
500  if (max>min)
501  {
502  if( valueScaled > max ) valueScaled = max;
503  if( valueScaled < min ) valueScaled = min;
504  }
505  else
506  {
507  if( valueScaled < max ) valueScaled = max;
508  if( valueScaled > min ) valueScaled = min;
509  }
510 
511  return valueScaled;
512 }

564 static bool set_channel(uint8_t mixer_channel, uint16_t value, const ActuatorSettingsData * actuatorSettings)
565 {
566  switch(actuatorSettings->ChannelType[mixer_channel]) {
567  case ACTUATORSETTINGS_CHANNELTYPE_PWMALARMBUZZER:
568  case ACTUATORSETTINGS_CHANNELTYPE_ARMINGLED:
569  case ACTUATORSETTINGS_CHANNELTYPE_INFOLED:
570  {
571  // This is for buzzers that take a PWM input
572 
573  static uint32_t currBuzzTune = 0;
574  static uint32_t currBuzzTuneState;
575 
576  static uint32_t currArmingTune = 0;
577  static uint32_t currArmingTuneState;
578 
579  static uint32_t currInfoTune = 0;
580  static uint32_t currInfoTuneState;
581 
582  uint32_t newTune = 0;
583  if(actuatorSettings->ChannelType[mixer_channel] == ACTUATORSETTINGS_CHANNELTYPE_PWMALARMBUZZER)
584  {
585 
586  // Decide what tune to play
587  if (AlarmsGet(SYSTEMALARMS_ALARM_BATTERY) > SYSTEMALARMS_ALARM_WARNING) {
588  newTune = 0b11110110110000; // pause, short, short, short, long
589  } else if (AlarmsGet(SYSTEMALARMS_ALARM_GPS) >= SYSTEMALARMS_ALARM_WARNING) {
590  newTune = 0x80000000; // pause, short
591  } else {
592  newTune = 0;
593  }
594 
595  // Do we need to change tune?
596  if (newTune != currBuzzTune) {
597  currBuzzTune = newTune;
598  currBuzzTuneState = currBuzzTune;
599  }
600  }
601  else // ACTUATORSETTINGS_CHANNELTYPE_ARMINGLED || ACTUATORSETTINGS_CHANNELTYPE_INFOLED
602  {
603  uint8_t arming;
604  FlightStatusArmedGet(&arming);
605  //base idle tune
606  newTune = 0x80000000; // 0b1000...
607 
608  // Merge the error pattern for InfoLed
609  if(actuatorSettings->ChannelType[mixer_channel] == ACTUATORSETTINGS_CHANNELTYPE_INFOLED)
610  {
611  if (AlarmsGet(SYSTEMALARMS_ALARM_BATTERY) > SYSTEMALARMS_ALARM_WARNING)
612  {
613  newTune |= 0b00000000001111111011111110000000;
614  }
615  else if(AlarmsGet(SYSTEMALARMS_ALARM_GPS) >= SYSTEMALARMS_ALARM_WARNING)
616  {
617  newTune |= 0b00000000000000110110110000000000;
618  }
619  }
620  // fast double blink pattern if armed
621  if (arming == FLIGHTSTATUS_ARMED_ARMED)
622  newTune |= 0xA0000000; // 0b101000...
623 
624  // Do we need to change tune?
625  if(actuatorSettings->ChannelType[mixer_channel] == ACTUATORSETTINGS_CHANNELTYPE_ARMINGLED)
626  {
627  if (newTune != currArmingTune) {
628  currArmingTune = newTune;
629  // note: those are both updated so that Info and Arming are in sync if used simultaneously
630  currArmingTuneState = currArmingTune;
631  currInfoTuneState = currInfoTune;
632  }
633  }
634  else
635  {
636  if (newTune != currInfoTune) {
637  currInfoTune = newTune;
638  currArmingTuneState = currArmingTune;
639  currInfoTuneState = currInfoTune;
640  }
641  }
642  }
643 
644  // Play tune
645  bool buzzOn = false;
646  static portTickType lastSysTime = 0;
647  portTickType thisSysTime = xTaskGetTickCount();
648  portTickType dT = 0;
649 
650  // For now, only look at the battery alarm, because functions like AlarmsHasCritical() can block for some time; to be discussed
651  if (currBuzzTune||currArmingTune||currInfoTune) {
652  if(thisSysTime > lastSysTime)
653  dT = thisSysTime - lastSysTime;
654  if(actuatorSettings->ChannelType[mixer_channel] == ACTUATORSETTINGS_CHANNELTYPE_PWMALARMBUZZER)
655  buzzOn = (currBuzzTuneState&1); // Buzz when the LS bit is 1
656  else if(actuatorSettings->ChannelType[mixer_channel] == ACTUATORSETTINGS_CHANNELTYPE_ARMINGLED)
657  buzzOn = (currArmingTuneState&1);
658  else if(actuatorSettings->ChannelType[mixer_channel] == ACTUATORSETTINGS_CHANNELTYPE_INFOLED)
659  buzzOn = (currInfoTuneState&1);
660 
661  if (dT > 80) {
662  // Go to next bit in alarm_seq_state
663  currArmingTuneState >>=1;
664  currInfoTuneState >>= 1;
665  currBuzzTuneState >>= 1;
666 
667  if (currBuzzTuneState == 0)
668  currBuzzTuneState = currBuzzTune; // All done, re-start the tune
669  if (currArmingTuneState == 0)
670  currArmingTuneState = currArmingTune;
671  if (currInfoTuneState == 0)
672  currInfoTuneState = currInfoTune;
673  lastSysTime = thisSysTime;
674  }
675  }
676  PIOS_Servo_Set(actuatorSettings->ChannelAddr[mixer_channel],
677  buzzOn?actuatorSettings->ChannelMax[mixer_channel]:actuatorSettings->ChannelMin[mixer_channel]);
678  return true;
679  }
680  case ACTUATORSETTINGS_CHANNELTYPE_PWM:
681  PIOS_Servo_Set(actuatorSettings->ChannelAddr[mixer_channel], value);
682  return true;
683 #if defined(PIOS_INCLUDE_I2C_ESC)
684  case ACTUATORSETTINGS_CHANNELTYPE_MK:
685  return PIOS_SetMKSpeed(actuatorSettings->ChannelAddr[mixer_channel],value);
686  case ACTUATORSETTINGS_CHANNELTYPE_ASTEC4:
687  return PIOS_SetAstec4Speed(actuatorSettings->ChannelAddr[mixer_channel],value);
688  break;
689 #endif
690  default:
691  return false;
692  }
693 
694  return false;
695 
696 }
