00190 /*==============================================================================
00191  * Increase program data space. As malloc and related functions depend on this,
00192  * it is useful to have a working implementation. The following suffices for a
00193  * standalone system; it exploits the symbol _end automatically defined by the
00194  * GNU linker.
00195  */
00196 caddr_t _sbrk(int incr)
00197 {
00198     extern char _end;           /* Defined by the linker */
00199     static char *heap_end;
00200     char *prev_heap_end;
00201     char * stack_ptr;
00202 
00203     if (heap_end == 0)
00204     {
00205       heap_end = &_end;
00206     }
00207 
00208     prev_heap_end = heap_end;
00209     asm volatile ("MRS %0, msp" : "=r" (stack_ptr) );
00210     if (heap_end + incr > stack_ptr)
00211     {
00212       _write_r ((void *)0, 1, "Heap and stack collision\n", 25);
00213       _exit (1);
00214     }
00215 
00216     heap_end += incr;
00217     return (caddr_t) prev_heap_end;
00218 }

00300 static unsigned dbl2stri(char *outbfr, double dbl, unsigned dec_digits)
00301 {
00302    static char local_bfr[128] ;
00303    char *output = (outbfr == 0) ? local_bfr : outbfr ;
00304 
00305    //*******************************************
00306    //  extract negative info
00307    //*******************************************
00308    if (dbl < 0.0) {
00309       *output++ = '-' ;
00310       dbl *= -1.0 ;
00311    } else {
00312       if (use_leading_plus) {
00313          *output++ = '+' ;
00314       }
00315 
00316    }
00317 
00318    //  handling rounding by adding .5LSB to the floating-point data
00319    if (dec_digits < 8) {
00320       dbl += round_nums[dec_digits] ;
00321    }
00322 
00323    //**************************************************************************
00324    //  construct fractional multiplier for specified number of digits.
00325    //**************************************************************************
00326    uint mult = 1 ;
00327    uint idx ;
00328    for (idx=0; idx < dec_digits; idx++)
00329       mult *= 10 ;
00330 
00331    // printf("mult=%u\n", mult) ;
00332    uint wholeNum = (uint) dbl ;
00333    uint decimalNum = (uint) ((dbl - wholeNum) * mult);
00334 
00335    //*******************************************
00336    //  convert integer portion
00337    //*******************************************
00338    char tbfr[40] ;
00339    idx = 0 ;
00340    while (wholeNum != 0) {
00341       tbfr[idx++] = '0' + (wholeNum % 10) ;
00342       wholeNum /= 10 ;
00343    }
00344    // printf("%.3f: whole=%s, dec=%d\n", dbl, tbfr, decimalNum) ;
00345    if (idx == 0) {
00346       *output++ = '0' ;
00347    } else {
00348       while (idx > 0) {
00349          *output++ = tbfr[idx-1] ;  //lint !e771
00350          idx-- ;
00351       }
00352    }
00353    if (dec_digits > 0) {
00354       *output++ = '.' ;
00355 
00356       //*******************************************
00357       //  convert fractional portion
00358       //*******************************************
00359       idx = 0 ;
00360       while (decimalNum != 0) {
00361          tbfr[idx++] = '0' + (decimalNum % 10) ;
00362          decimalNum /= 10 ;
00363       }
00364       //  pad the decimal portion with 0s as necessary;
00365       //  We wouldn't want to report 3.093 as 3.93, would we??
00366       while (idx < dec_digits) {
00367          tbfr[idx++] = '0' ;
00368       }
00369       // printf("decimal=%s\n", tbfr) ;
00370       if (idx == 0) {
00371          *output++ = '0' ;
00372       } else {
00373          while (idx > 0) {
00374             *output++ = tbfr[idx-1] ;
00375             idx-- ;
00376          }
00377       }
00378    }
00379    *output = 0 ;
00380 
00381    //  prepare output
