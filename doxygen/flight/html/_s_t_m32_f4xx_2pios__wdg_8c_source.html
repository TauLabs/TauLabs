00122 bool PIOS_WDG_UpdateFlag(uint16_t flag) 
00123 {       
00124         // we can probably avoid using a semaphore here which will be good for
00125         // efficiency and not blocking critical tasks.  race condition could 
00126         // overwrite their flag update, but unlikely to block _all_ of them 
00127         // for the timeout window
00128         uint16_t cur_flags = RTC_ReadBackupRegister(PIOS_WDG_REGISTER);
00129         
00130         if((cur_flags | flag) == wdg_configuration.used_flags) {
00131                 PIOS_WDG_Clear();
00132                 RTC_WriteBackupRegister(PIOS_WDG_REGISTER, flag);
00133                 return true;
00134         } else {
00135                 RTC_WriteBackupRegister(PIOS_WDG_REGISTER, cur_flags | flag);
00136                 return false;
00137         }
00138                 
00139 }
