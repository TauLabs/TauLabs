38 uint32_t STOPWATCH_Init(u32 resolution, TIM_TypeDef* TIM) {
39  uint32_t STOPWATCH_TIMER_RCC;
40  switch ((uint32_t) TIM) {
41  case (uint32_t)TIM1:
42  STOPWATCH_TIMER_RCC = RCC_APB2Periph_TIM1;
43  break;
44  case (uint32_t)TIM2:
45  STOPWATCH_TIMER_RCC = RCC_APB1Periph_TIM2;
46  break;
47  case (uint32_t)TIM3:
48  STOPWATCH_TIMER_RCC = RCC_APB1Periph_TIM3;
49  break;
50  case (uint32_t)TIM4:
51  STOPWATCH_TIMER_RCC = RCC_APB1Periph_TIM4;
52  break;
53  case (uint32_t)TIM5:
54  STOPWATCH_TIMER_RCC = RCC_APB1Periph_TIM5;
55  break;
56  case (uint32_t)TIM6:
57  STOPWATCH_TIMER_RCC = RCC_APB1Periph_TIM6;
58  break;
59  case (uint32_t)TIM7:
60  STOPWATCH_TIMER_RCC = RCC_APB1Periph_TIM7;
61  break;
62  case (uint32_t)TIM8:
63  STOPWATCH_TIMER_RCC = RCC_APB2Periph_TIM8;
64  break;
65  default:
66  /* Unsupported timer */
67  while(1);
68  }
69 
70  // enable timer clock
71  if (STOPWATCH_TIMER_RCC == RCC_APB2Periph_TIM1 || STOPWATCH_TIMER_RCC
72  == RCC_APB2Periph_TIM8)
73  RCC_APB2PeriphClockCmd(STOPWATCH_TIMER_RCC, ENABLE);
74  else
75  RCC_APB1PeriphClockCmd(STOPWATCH_TIMER_RCC, ENABLE);
76 
77  // time base configuration
78  TIM_TimeBaseInitTypeDef TIM_TimeBaseStructure;
79  TIM_TimeBaseStructure.TIM_Period = 0xffff; // max period
80  TIM_TimeBaseStructure.TIM_Prescaler = (72 * resolution) - 1; // <resolution> uS accuracy @ 72 MHz
81  TIM_TimeBaseStructure.TIM_ClockDivision = 0;
82  TIM_TimeBaseStructure.TIM_CounterMode = TIM_CounterMode_Up;
83  TIM_TimeBaseInit(TIM, &TIM_TimeBaseStructure);
84 
85  // enable interrupt request
86  TIM_ITConfig(TIM, TIM_IT_Update, ENABLE);
87 
88  // start counter
89  TIM_Cmd(TIM, ENABLE);
90 
91  return 0; // no error
92 }
