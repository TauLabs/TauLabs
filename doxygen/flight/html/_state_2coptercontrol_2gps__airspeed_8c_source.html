00094 /*
00095  * Calculate airspeed as a function of GPS groundspeed and vehicle attitude.
00096  *  From "IMU Wind Estimation (Theory)", by William Premerlani.
00097  *  The idea is that V_gps=V_air+V_wind. If we assume wind constant, => 
00098  *  V_gps_2-V_gps_1 = (V_air_2+V_wind_2) -(V_air_1+V_wind_1) = V_air_2 - V_air_1.
00099  *  If we assume airspeed constant, => V_gps_2-V_gps_1 = |V|*(f_2 - f1),
00100  *  where "f" is the fuselage vector in earth coordinates.
00101  *  We then solve for |V| = |V_gps_2-V_gps_1|/ |f_2 - f1|.
00102  */
00103 void gps_airspeed_update(const GPSVelocityData * gpsVelData,
00104                          float staticAirDensity)
00105 {
00106         float Rbe[3][3];
00107         compute_rbe(Rbe);
00108 
00109         //Calculate the cos(angle) between the two fuselage basis vectors
00110         float cosDiff = (Rbe[0][0] * gps->RbeCol1_old[0]) +
00111             (Rbe[0][1] * gps->RbeCol1_old[1]) +
00112             (Rbe[0][2] * gps->RbeCol1_old[2]);
00113 
00114     //If there's more than a 5 degree difference between two fuselage
00115     //measurements, then we have sufficient delta to continue.
00116         if (fabs(cosDiff) < cos(5.0f * DEG2RAD)) {
00117                 float gps_airspeed;
00118 
00119                 GPSVelocityData gpsVelData;
00120                 GPSVelocityGet(&gpsVelData);
00121 
00122         //Calculate the norm^2 of the difference between the two GPS vectors
00123                 float normDiffGPS2 =
00124                     powf(gpsVelData.North - gps->gpsVelOld_N,
00125                          2.0f) + powf(gpsVelData.East - gps->gpsVelOld_E,
00126                                       2.0f) + powf(gpsVelData.Down -
00127                                                    gps->gpsVelOld_D, 2.0f);
00128 
00129         //Calculate the norm^2 of the difference between the two fuselage
00130         //vectors
00131                 float normDiffAttitude2 =
00132                     powf(Rbe[0][0] - gps->RbeCol1_old[0],
00133                          2.0f) + powf(Rbe[0][1] - gps->RbeCol1_old[1],
00134                                       2.0f) + powf(Rbe[0][2] -
00135                                                    gps->RbeCol1_old[2], 2.0f);
00136 
00137                 //Airspeed magnitude is the ratio between the two difference norms
00138                 gps_airspeed = sqrtf(normDiffGPS2 / normDiffAttitude2);
00139 
00140         //Check to see if gps airspeed estimate is reasonable (remember, it can
00141         //only be positive, so only need to check upper limit)
00142                 if (gps_airspeed < 300) {       //NEED TO SATURATE, BUT NOT VERY GOOD TO SATURATE LIKE THIS. PROBABLY SHOULD THROW OUT ANY READINGS THAT ARE TOO FAR OUTSIDE THE CURRENT SPEED
00143                         //Save old variables for next pass
00144                         gps->gpsVelOld_N = gpsVelData.North;
00145                         gps->gpsVelOld_E = gpsVelData.East;
00146                         gps->gpsVelOld_D = gpsVelData.Down;
00147 
00148                         gps->RbeCol1_old[0] = Rbe[0][0];
00149                         gps->RbeCol1_old[1] = Rbe[0][1];
00150                         gps->RbeCol1_old[2] = Rbe[0][2];
00151 
00152                         //Low pass filter
00153                         const float alpha = .2;
00154                         float gps_airspeed_old;
00155                         AirspeedActualTrueAirspeedGet(&gps_airspeed_old);
00156 
00157                         gps_airspeed =
00158                             gps_airspeed * alpha + (1 -
00159                                                     alpha) * gps_airspeed_old;
00160 
00161                         // Do not update airspeed data in simulation mode
00162                         if (!AirspeedActualReadOnly()) {
00163                                 AirspeedActualTrueAirspeedSet(&gps_airspeed);
00164 
00165                 //Calculate calibrated airspeed,
00166                 //http://en.wikipedia.org/wiki/True_airspeed
00167                                 gps_airspeed *=
00168                                     sqrtf(staticAirDensity /
00169                                           STANDARD_AIR_DENSITY);
00170                                 AirspeedActualCalibratedAirspeedSet
00171                                     (&gps_airspeed);
00172                         }
00173                 } else {
00174                         // Do not update airspeed data in simulation mode
00175                         if (!AirspeedActualReadOnly()) {
00176                                 AirspeedActualTrueAirspeedGet(&gps_airspeed);   //<--Why do we get the airspeed just to set it? So that the object is listed as updated?
00177                                 AirspeedActualTrueAirspeedSet(&gps_airspeed);
00178                         }
00179                 }
00180 
00181         }
00182 }
