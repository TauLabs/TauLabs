97 /*
98  * Calculate airspeed as a function of GPS groundspeed and vehicle attitude.
99  * From "IMU Wind Estimation (Theory)", by William Premerlani.
100  * The idea is that V_gps=V_air+V_wind. If we assume wind constant, =>
101  * V_gps_2-V_gps_1 = (V_air_2+V_wind_2) -(V_air_1+V_wind_1) = V_air_2 - V_air_1.
102  * If we assume airspeed constant, => V_gps_2-V_gps_1 = |V|*(f_2 - f1),
103  * where "f" is the fuselage vector in earth coordinates.
104  * We then solve for |V| = |V_gps_2-V_gps_1|/ |f_2 - f1|.
105  */
106 void gps_airspeed_update(const GPSVelocityData * gpsVelData,
107  float staticAirDensity)
108 {
109  float Rbe[3][3];
110  compute_rbe(Rbe);
111 
112  //Calculate the cos(angle) between the two fuselage basis vectors
113  float cosDiff = (Rbe[0][0] * gps->RbeCol1_old[0]) +
114  (Rbe[0][1] * gps->RbeCol1_old[1]) +
115  (Rbe[0][2] * gps->RbeCol1_old[2]);
116 
117  //If there's more than a 5 degree difference between two fuselage
118  //measurements, then we have sufficient delta to continue.
119  if (fabs(cosDiff) < cos(5.0f * DEG2RAD)) {
120  float gps_airspeed;
121 
122  GPSVelocityData gpsVelData;
123  GPSVelocityGet(&gpsVelData);
124 
125  //Calculate the norm^2 of the difference between the two GPS vectors
126  float normDiffGPS2 =
127  powf(gpsVelData.North - gps->gpsVelOld_N,
128  2.0f) + powf(gpsVelData.East - gps->gpsVelOld_E,
129  2.0f) + powf(gpsVelData.Down -
130  gps->gpsVelOld_D, 2.0f);
131 
132  //Calculate the norm^2 of the difference between the two fuselage
133  //vectors
134  float normDiffAttitude2 =
135  powf(Rbe[0][0] - gps->RbeCol1_old[0],
136  2.0f) + powf(Rbe[0][1] - gps->RbeCol1_old[1],
137  2.0f) + powf(Rbe[0][2] -
138  gps->RbeCol1_old[2], 2.0f);
139 
140  //Airspeed magnitude is the ratio between the two difference norms
141  gps_airspeed = sqrtf(normDiffGPS2 / normDiffAttitude2);
142 
143  //Check to see if gps airspeed estimate is reasonable (remember, it can
144  //only be positive, so only need to check upper limit)
145  if (gps_airspeed < 300) { //NEED TO SATURATE, BUT NOT VERY GOOD TO SATURATE LIKE THIS. PROBABLY SHOULD THROW OUT ANY READINGS THAT ARE TOO FAR OUTSIDE THE CURRENT SPEED
146  //Save old variables for next pass
147  gps->gpsVelOld_N = gpsVelData.North;
148  gps->gpsVelOld_E = gpsVelData.East;
149  gps->gpsVelOld_D = gpsVelData.Down;
150 
151  gps->RbeCol1_old[0] = Rbe[0][0];
152  gps->RbeCol1_old[1] = Rbe[0][1];
153  gps->RbeCol1_old[2] = Rbe[0][2];
154 
155  //Low pass filter
156  const float alpha = .2;
157  float gps_airspeed_old;
158  AirspeedActualTrueAirspeedGet(&gps_airspeed_old);
159 
160  gps_airspeed =
161  gps_airspeed * alpha + (1 -
162  alpha) * gps_airspeed_old;
163 
164  // Do not update airspeed data in simulation mode
165  if (!AirspeedActualReadOnly()) {
166  AirspeedActualTrueAirspeedSet(&gps_airspeed);
167 
168  //Calculate calibrated airspeed,
169  //http://en.wikipedia.org/wiki/True_airspeed
170  gps_airspeed *=
171  sqrtf(staticAirDensity /
172  STANDARD_AIR_DENSITY);
173  AirspeedActualCalibratedAirspeedSet
174  (&gps_airspeed);
175  }
176  } else {
177  // Do not update airspeed data in simulation mode
178  if (!AirspeedActualReadOnly()) {
179  AirspeedActualTrueAirspeedGet(&gps_airspeed); //<--Why do we get the airspeed just to set it? So that the object is listed as updated?
180  AirspeedActualTrueAirspeedSet(&gps_airspeed);
181  }
182  }
183 
184  }
