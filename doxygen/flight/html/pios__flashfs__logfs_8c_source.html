00120 {

00346 static int32_t logfs_raw_copy_bytes (const struct logfs_state * logfs, uintptr_t src_addr, uint16_t src_size, uintptr_t dst_addr)
00347 {
00348 #define RAW_COPY_BLOCK_SIZE 16
00349         uint8_t data_block[RAW_COPY_BLOCK_SIZE];
00350 
00351         while (src_size) {
00352                 uint16_t blk_size;
00353                 if (src_size >= RAW_COPY_BLOCK_SIZE) {
00354                         /* Copy a full block */
00355                         blk_size = RAW_COPY_BLOCK_SIZE;
00356                 } else {
00357                         /* Copy the remainder */
00358                         blk_size = src_size;
00359                 }
00360 
00361                 /* Read a block of data from source */
00362                 if (logfs->driver->read_data(logfs->flash_id,
00363                                                 src_addr,
00364                                                 data_block,
00365                                                 blk_size) != 0) {
00366                         /* Failed to read next chunk from source */
00367                         return -1;
00368                 }
00369 
00370                 /* Write a block of data to destination */
00371                 if (logfs->driver->write_data(logfs->flash_id,
00372                                                 dst_addr,
00373                                                 data_block,
00374                                                 blk_size) != 0) {
00375                         /* Failed to write chunk to destination */
00376                         return -2;
00377                 }
00378 
00379                 /* Update the src/dst pointers */
00380                 src_size -= blk_size;
00381                 src_addr += blk_size;
00382                 dst_addr += blk_size;
00383         }
00384 
00385         return 0;

00561                         /* Found the active arena */

00563                         break;

00569                                 break;

00693         /* First slot in the arena is reserved for arena header, skip it. */
00694         if (*curr_slot == 0) *curr_slot = 1;

00702                                                 slot_addr,
00703                                                 (uint8_t *)slot_hdr,
00704                                                 sizeof (*slot_hdr)) != 0) {
00705                         return -2;
00706                 }

00708                         /* We hit the end of the log */
00709                         break;

00712                         slot_hdr->obj_id      == obj_id &&

00767 static int8_t logfs_reserve_free_slot (struct logfs_state * logfs, uint16_t * slot_id, struct slot_header * slot_hdr, uint32_t obj_id, uint16_t obj_inst_id, uint16_t obj_size)
00768 {
00769         PIOS_Assert(slot_id);
00770         PIOS_Assert(slot_hdr);
00771 
00772         if (logfs->num_free_slots < 1) {
00773                 /* No free slots to allocate */
00774                 return -1;
00775         }
00776 
00777         if (obj_size > (logfs->cfg->slot_size - sizeof (slot_hdr))) {
00778                 /* This object is too big for the slot */
00779                 return -2;
00780         }
00781 
00782         uint16_t candidate_slot_id = (logfs->cfg->arena_size / logfs->cfg->slot_size) - logfs->num_free_slots;
00783         PIOS_Assert(candidate_slot_id > 0);
00784

00786

00788                                         slot_addr,
00789                                         (uint8_t *)slot_hdr,
00790                                         sizeof (*slot_hdr)) != 0) {
00791                 /* Failed to read slot header for candidate slot */
00792                 return -3;
00793         }
00794 
00795         if (slot_hdr->state != SLOT_STATE_EMPTY) {
00796                 /* Candidate slot isn't empty!  Something is broken. */
00797                 PIOS_DEBUG_Assert(0);
00798                 return -4;
00799         }
00800 
00801         /* Mark this slot as RESERVED */
00802         slot_hdr->state       = SLOT_STATE_RESERVED;
00803         slot_hdr->obj_id      = obj_id;
00804         slot_hdr->obj_inst_id = obj_inst_id;
00805         slot_hdr->obj_size    = obj_size;

00807         if (logfs->driver->write_data(logfs->flash_id,
00808                                         slot_addr,
00809                                         (uint8_t *)slot_hdr,
00810                                         sizeof(*slot_hdr)) != 0) {
00811                 /* Failed to write the slot header */
00812                 return -5;
00813         }
00814 
00815         /* FIXME: If the header write (above) failed, may have partially written data, thus corrupting that slot but we would have missed decrementing this counter */
00816         logfs->num_free_slots--;

00822 /* NOTE: Must be called while holding the flash transaction lock */
00823 static int8_t logfs_append_to_log (struct logfs_state * logfs, uint32_t obj_id, uint16_t obj_inst_id, uint8_t * obj_data, uint16_t obj_size)
00824 {
00825         /* Reserve a free slot for our new object */
00826         uint16_t free_slot_id;
00827         struct slot_header slot_hdr;
00828         if (logfs_reserve_free_slot (logfs, &free_slot_id, &slot_hdr, obj_id, obj_inst_id, obj_size) != 0) {
00829                 /* Failed to reserve a free slot */
00830                 return -1;
00831         }
00832 
00833         /* Compute slot address */
00834         uintptr_t slot_addr = logfs_get_addr (logfs, logfs->active_arena_id, free_slot_id);
00835 
00836         /* Write the data into the reserved slot, starting after the slot header */
00837         uintptr_t slot_offset = sizeof(slot_hdr);
00838         while (obj_size > 0) {
00839                 /* Individual writes must fit entirely within a single page buffer. */
00840                 uint16_t page_remaining = logfs->cfg->page_size - (slot_offset % logfs->cfg->page_size);
00841                 uint16_t write_size = MIN(obj_size, page_remaining);
00842                 if (logfs->driver->write_data (logfs->flash_id,
00843                                                 slot_addr + slot_offset,
00844                                                 obj_data,
00845                                                 write_size) != 0) {
00846                         /* Failed to write the object data to the slot */
00847                         return -2;
00848                 }
00849 
00850                 /* Update our accounting */
00851                 obj_data    += write_size;
00852                 slot_offset += write_size;
00853                 obj_size    -= write_size;
00854         }
00855 
00856         /* Mark this slot active in one atomic step */
00857         slot_hdr.state = SLOT_STATE_ACTIVE;
00858         if (logfs->driver->write_data (logfs->flash_id,
00859                                         slot_addr,
00860                                         (uint8_t *)&slot_hdr,
00861                                         sizeof(slot_hdr)) != 0) {
00862                 /* Failed to mark the slot active */
00863                 return -4;
00864         }
00865 
00866         /* Object has been successfully written to the slot */
00867         logfs->num_active_slots++;
00868         return 0;
00869 }
