00360 {
00361         static int temp_counter = -1;

00363         static const float TEMP_MIN = -10;
00364         static const float TEMP_MAX = 60;

00366         if (temperature < TEMP_MIN)
00367                 temperature = TEMP_MIN;
00368         if (temperature > TEMP_MAX)

00372                 temp_accum += temperature;

00394 static void mag_calibration_prelemari(MagnetometerData *mag)

00399         static float B2[3] = {0, 0, 0};

00401         MagBiasData magBias;
00402         MagBiasGet(&magBias);

00404         // Remove the current estimate of the bias
00405         mag->x -= magBias.x;

00407         mag->z -= magBias.z;

00409         // First call

00411                 B2[0] = mag->x;
00412                 B2[1] = mag->y;
00413                 B2[2] = mag->z;
00414                 return;
00415         }
00416 
00417         float B1[3] = {mag->x, mag->y, mag->z};
00418         float norm_diff = sqrtf(powf(B2[0] - B1[0],2) + powf(B2[1] - B1[1],2) + powf(B2[2] - B1[2],2));
00419         if (norm_diff > MIN_NORM_DIFFERENCE) {
00420                 float norm_b1 = sqrtf(B1[0]*B1[0] + B1[1]*B1[1] + B1[2]*B1[2]);
00421                 float norm_b2 = sqrtf(B2[0]*B2[0] + B2[1]*B2[1] + B2[2]*B2[2]);
00422                 float scale = insSettings.MagBiasNullingRate * (norm_b2 - norm_b1) / norm_diff;
00423                 float b_error[3] = {(B2[0] - B1[0]) * scale, (B2[1] - B1[1]) * scale, (B2[2] - B1[2]) * scale};
00424 
00425                 magBias.x += b_error[0];
00426                 magBias.y += b_error[1];
00427                 magBias.z += b_error[2];
00428 
00429                 MagBiasSet(&magBias);
00430 
00431                 // Store this value to compare against next update
00432                 B2[0] = B1[0]; B2[1] = B1[1]; B2[2] = B1[2];
00433         }

00473         B_e[2] = R[0][2] * mag->x + R[1][2] * mag->y + R[2][2] * mag->z;
00474

00476         float sy = sinf(attitude.Yaw * DEG2RAD);
00477         
00478         xy[0] =  cy * B_e[0] + sy * B_e[1];
00479         xy[1] = -sy * B_e[0] + cy * B_e[1];

00481         float xy_norm = sqrtf(xy[0]*xy[0] + xy[1]*xy[1]);
00482         
00483         delta[0] = -rate * (xy[0] / xy_norm * Rxy - xy[0]);
00484         delta[1] = -rate * (xy[1] / xy_norm * Rxy - xy[1]);
00485         delta[2] = -rate * (Rz - B_e[2]);
00486         
00487         if (delta[0] == delta[0] && delta[1] == delta[1] && delta[2] == delta[2]) {
