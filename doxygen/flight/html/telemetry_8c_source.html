88 int32_t TelemetryStart(void)
89 {
90  // Process all registered objects and connect queue for updates
91  UAVObjIterate(&registerObject);
92 
93  // Listen to objects of interest
94  GCSTelemetryStatsConnectQueue(priorityQueue);
95 
96  // Start telemetry tasks
97  xTaskCreate(telemetryTxTask, (signed char *)"TelTx", STACK_SIZE_BYTES/4, NULL, TASK_PRIORITY_TX, &telemetryTxTaskHandle);

99  TaskMonitorAdd(TASKINFO_RUNNING_TELEMETRYTX, telemetryTxTaskHandle);
100  TaskMonitorAdd(TASKINFO_RUNNING_TELEMETRYRX, telemetryRxTaskHandle);

102 #if defined(PIOS_TELEM_PRIORITY_QUEUE)
103  xTaskCreate(telemetryTxPriTask, (signed char *)"TelPriTx", STACK_SIZE_BYTES/4, NULL, TASK_PRIORITY_TXPRI, &telemetryTxPriTaskHandle);
104  TaskMonitorAdd(TASKINFO_RUNNING_TELEMETRYTXPRI, telemetryTxPriTaskHandle);
105 #endif
106 
107  return 0;
108 }

153 static void registerObject(UAVObjHandle obj)
154 {
155  if (UAVObjIsMetaobject(obj)) {
156  /* Only connect change notifications for meta objects. No periodic updates */
157  UAVObjConnectQueue(obj, priorityQueue, EV_MASK_ALL_UPDATES);
158  return;
159  } else {
160  UAVObjMetadata metadata;
161  UAVObjUpdateMode updateMode;
162  UAVObjGetMetadata(obj, &metadata);
163  updateMode = UAVObjGetTelemetryUpdateMode(&metadata);
164 
165  /* Only create a periodic event for objects that are periodic */
166  if ((updateMode == UPDATEMODE_PERIODIC) ||
167  (updateMode == UPDATEMODE_THROTTLED)) {
168  // Setup object for periodic updates
169  UAVObjEvent ev = {
170  .obj = obj,
171  .instId = UAVOBJ_ALL_INSTANCES,
172  .event = EV_UPDATED_PERIODIC,
173  };
174  EventPeriodicQueueCreate(&ev, queue, 0);
175  }
176 
177  // Setup object for telemetry updates
178  updateObject(obj, EV_NONE);
179  }
180 }

186 static void updateObject(UAVObjHandle obj, int32_t eventType)
187 {
188  UAVObjMetadata metadata;
189  UAVObjUpdateMode updateMode;
190  int32_t eventMask;
191 
192  if (UAVObjIsMetaobject(obj)) {
193  /* This function updates the periodic updates for the object.
194  * Meta Objects cannot have periodic updates.
195  */
196  PIOS_Assert(false);
197  return;
198  }
199 
200  // Get metadata
201  UAVObjGetMetadata(obj, &metadata);
202  updateMode = UAVObjGetTelemetryUpdateMode(&metadata);
203 
204  // Setup object depending on update mode
205  switch (updateMode) {
206  case UPDATEMODE_PERIODIC:
207  // Set update period
208  setUpdatePeriod(obj, metadata.telemetryUpdatePeriod);
209  // Connect queue
210  eventMask = EV_UPDATED_PERIODIC | EV_UPDATED_MANUAL | EV_UPDATE_REQ;
211  UAVObjConnectQueue(obj, priorityQueue, eventMask);
212  break;
213  case UPDATEMODE_ONCHANGE:
214  // Set update period
215  setUpdatePeriod(obj, 0);
216  // Connect queue
217  eventMask = EV_UPDATED | EV_UPDATED_MANUAL | EV_UPDATE_REQ;
218  UAVObjConnectQueue(obj, priorityQueue, eventMask);
219  break;
220  case UPDATEMODE_THROTTLED:
221  if ((eventType == EV_UPDATED_PERIODIC) || (eventType == EV_NONE)) {
222  // If we received a periodic update, we can change back to update on change
223  eventMask = EV_UPDATED | EV_UPDATED_MANUAL | EV_UPDATE_REQ;
224  // Set update period on initialization and metadata change
225  if (eventType == EV_NONE)
226  setUpdatePeriod(obj, metadata.telemetryUpdatePeriod);
227  } else {
228  // Otherwise, we just received an object update, so switch to periodic for the timeout period to prevent more updates
229  eventMask = EV_UPDATED_PERIODIC | EV_UPDATED_MANUAL | EV_UPDATE_REQ;
230  }
231  UAVObjConnectQueue(obj, priorityQueue, eventMask);
232  break;
233  case UPDATEMODE_MANUAL:
234  // Set update period
235  setUpdatePeriod(obj, 0);
236  // Connect queue
237  eventMask = EV_UPDATED_MANUAL | EV_UPDATE_REQ;
238  UAVObjConnectQueue(obj, priorityQueue, eventMask);
239  break;
240  }
241 }

246 static void processObjEvent(UAVObjEvent * ev)
247 {
248  UAVObjMetadata metadata;
249  UAVObjUpdateMode updateMode;
250  FlightTelemetryStatsData flightStats;
251  int32_t retries;
252  int32_t success;
253 
254  if (ev->obj == 0) {
255  updateTelemetryStats();
256  } else if (ev->obj == GCSTelemetryStatsHandle()) {
257  gcsTelemetryStatsUpdated();
258  } else {
259  FlightTelemetryStatsGet(&flightStats);
260  // Get object metadata
261  UAVObjGetMetadata(ev->obj, &metadata);
262  updateMode = UAVObjGetTelemetryUpdateMode(&metadata);
263 
264  // Act on event
265  retries = 0;
266  success = -1;
267  if (ev->event == EV_UPDATED || ev->event == EV_UPDATED_MANUAL || ((ev->event == EV_UPDATED_PERIODIC) && (updateMode != UPDATEMODE_THROTTLED))) {
268  // Send update to GCS (with retries)
269  while (retries < MAX_RETRIES && success == -1) {
270  success = UAVTalkSendObject(uavTalkCon, ev->obj, ev->instId, UAVObjGetTelemetryAcked(&metadata), REQ_TIMEOUT_MS); // call blocks until ack is received or timeout
271  ++retries;
272  }
273  // Update stats
274  txRetries += (retries - 1);
275  if (success == -1) {
276  ++txErrors;
277  }
278  } else if (ev->event == EV_UPDATE_REQ) {
279  // Request object update from GCS (with retries)
280  while (retries < MAX_RETRIES && success == -1) {
281  success = UAVTalkSendObjectRequest(uavTalkCon, ev->obj, ev->instId, REQ_TIMEOUT_MS); // call blocks until update is received or timeout
282  ++retries;
283  }
284  // Update stats
285  txRetries += (retries - 1);
286  if (success == -1) {
287  ++txErrors;
288  }
289  }
290  // If this is a metaobject then make necessary telemetry updates
291  if (UAVObjIsMetaobject(ev->obj)) {
292  updateObject(UAVObjGetLinkedObj(ev->obj), EV_NONE); // linked object will be the actual object the metadata are for
293  } else {
294  if (updateMode == UPDATEMODE_THROTTLED) {
295  // If this is UPDATEMODE_THROTTLED, the event mask changes on every event.
296  updateObject(ev->obj, ev->event);
297  }
298  }
299  }
300 }

341 static void telemetryRxTask(void *parameters)
342 {
343 
344  // Task loop
345  while (1) {
346  uint32_t inputPort = getComPort();
347 
348  if (inputPort) {
349  // Block until data are available
350  uint8_t serial_data[1];
351  uint16_t bytes_to_process;
352 
353  bytes_to_process = PIOS_COM_ReceiveBuffer(inputPort, serial_data, sizeof(serial_data), 500);
354  if (bytes_to_process > 0) {
355  for (uint8_t i = 0; i < bytes_to_process; i++) {
356  UAVTalkProcessInputStream(uavTalkCon,serial_data[i]);
357  }
358  }
359  } else {
360  vTaskDelay(5);
361  }
362  }
363 }

419 static void updateTelemetryStats()
420 {
421  UAVTalkStats utalkStats;
422  FlightTelemetryStatsData flightStats;
423  GCSTelemetryStatsData gcsStats;
424  uint8_t forceUpdate;
425  uint8_t connectionTimeout;
426  uint32_t timeNow;
427 
428  // Get stats
429  UAVTalkGetStats(uavTalkCon, &utalkStats);
430  UAVTalkResetStats(uavTalkCon);
431 
432  // Get object data
433  FlightTelemetryStatsGet(&flightStats);
434  GCSTelemetryStatsGet(&gcsStats);
435 
436  // Update stats object
437  if (flightStats.Status == FLIGHTTELEMETRYSTATS_STATUS_CONNECTED) {
438  flightStats.RxDataRate = (float)utalkStats.rxBytes / ((float)STATS_UPDATE_PERIOD_MS / 1000.0);
439  flightStats.TxDataRate = (float)utalkStats.txBytes / ((float)STATS_UPDATE_PERIOD_MS / 1000.0);
440  flightStats.RxFailures += utalkStats.rxErrors;
441  flightStats.TxFailures += txErrors;
442  flightStats.TxRetries += txRetries;
443  txErrors = 0;
444  txRetries = 0;
445  } else {
446  flightStats.RxDataRate = 0;
447  flightStats.TxDataRate = 0;
448  flightStats.RxFailures = 0;
449  flightStats.TxFailures = 0;
450  flightStats.TxRetries = 0;
451  txErrors = 0;
452  txRetries = 0;
453  }
454 
455  // Check for connection timeout
456  timeNow = xTaskGetTickCount() * portTICK_RATE_MS;
457  if (utalkStats.rxObjects > 0) {
458  timeOfLastObjectUpdate = timeNow;
459  }
460  if ((timeNow - timeOfLastObjectUpdate) > CONNECTION_TIMEOUT_MS) {
461  connectionTimeout = 1;
462  } else {
463  connectionTimeout = 0;
464  }
465 
466  // Update connection state
467  forceUpdate = 1;
468  if (flightStats.Status == FLIGHTTELEMETRYSTATS_STATUS_DISCONNECTED) {
469  // Wait for connection request
470  if (gcsStats.Status == GCSTELEMETRYSTATS_STATUS_HANDSHAKEREQ) {
471  flightStats.Status = FLIGHTTELEMETRYSTATS_STATUS_HANDSHAKEACK;
472  }
473  } else if (flightStats.Status == FLIGHTTELEMETRYSTATS_STATUS_HANDSHAKEACK) {
474  // Wait for connection
475  if (gcsStats.Status == GCSTELEMETRYSTATS_STATUS_CONNECTED) {
476  flightStats.Status = FLIGHTTELEMETRYSTATS_STATUS_CONNECTED;
477  } else if (gcsStats.Status == GCSTELEMETRYSTATS_STATUS_DISCONNECTED) {
478  flightStats.Status = FLIGHTTELEMETRYSTATS_STATUS_DISCONNECTED;
479  }
480  } else if (flightStats.Status == FLIGHTTELEMETRYSTATS_STATUS_CONNECTED) {
481  if (gcsStats.Status != GCSTELEMETRYSTATS_STATUS_CONNECTED || connectionTimeout) {
482  flightStats.Status = FLIGHTTELEMETRYSTATS_STATUS_DISCONNECTED;
483  } else {
484  forceUpdate = 0;
485  }
486  } else {
487  flightStats.Status = FLIGHTTELEMETRYSTATS_STATUS_DISCONNECTED;
488  }
489 
490  // Update the telemetry alarm
491  if (flightStats.Status == FLIGHTTELEMETRYSTATS_STATUS_CONNECTED) {
492  AlarmsClear(SYSTEMALARMS_ALARM_TELEMETRY);
493  } else {
494  AlarmsSet(SYSTEMALARMS_ALARM_TELEMETRY, SYSTEMALARMS_ALARM_ERROR);
495  }
496 
497  // Update object
498  FlightTelemetryStatsSet(&flightStats);
499 
500  // Force telemetry update if not connected
501  if (forceUpdate) {
502  FlightTelemetryStatsUpdated();
503  }
504 }
