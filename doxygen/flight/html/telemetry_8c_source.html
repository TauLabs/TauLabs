00184 static void updateObject(UAVObjHandle obj, int32_t eventType)
00185 {
00186         UAVObjMetadata metadata;
00187         UAVObjUpdateMode updateMode;
00188         int32_t eventMask;
00189 
00190         if (UAVObjIsMetaobject(obj)) {
00191                 /* This function updates the periodic updates for the object.
00192                  * Meta Objects cannot have periodic updates.
00193                  */
00194                 PIOS_Assert(false);
00195                 return;
00196         }
00197 
00198         // Get metadata
00199         UAVObjGetMetadata(obj, &metadata);
00200         updateMode = UAVObjGetTelemetryUpdateMode(&metadata);
00201 
00202         // Setup object depending on update mode
00203         switch (updateMode) {
00204         case UPDATEMODE_PERIODIC:
00205                 // Set update period
00206                 setUpdatePeriod(obj, metadata.telemetryUpdatePeriod);
00207                 // Connect queue
00208                 eventMask = EV_UPDATED_PERIODIC | EV_UPDATED_MANUAL | EV_UPDATE_REQ;
00209                 UAVObjConnectQueue(obj, priorityQueue, eventMask);
00210                 break;
00211         case UPDATEMODE_ONCHANGE:
00212                 // Set update period
00213                 setUpdatePeriod(obj, 0);
00214                 // Connect queue
00215                 eventMask = EV_UPDATED | EV_UPDATED_MANUAL | EV_UPDATE_REQ;
00216                 UAVObjConnectQueue(obj, priorityQueue, eventMask);
00217                 break;
00218         case UPDATEMODE_THROTTLED:
00219                 if ((eventType == EV_UPDATED_PERIODIC) || (eventType == EV_NONE)) {
00220                         // If we received a periodic update, we can change back to update on change
00221                         eventMask = EV_UPDATED | EV_UPDATED_MANUAL | EV_UPDATE_REQ;
00222                         // Set update period on initialization and metadata change
00223                         if (eventType == EV_NONE)
00224                                 setUpdatePeriod(obj, metadata.telemetryUpdatePeriod);
00225                 } else {
00226                         eventMask = getEventMask(obj, priorityQueue);
00227                         if (eventMask & EV_UPDATED_PERIODIC) {
00228                                 // If periodic flag is already set then we have previously sent an update during
00229                                 // the timeout period and this update would be missed, so set the dirty flag.
00230                                 // Once setting EV_UPDATED_THROTTLED_DIRTY, there is no need to listen for EV_UPDATED.
00231                                 eventMask = EV_UPDATED_PERIODIC | EV_UPDATE_REQ | EV_UPDATED_THROTTLED_DIRTY;
00232                         } else { //If periodic is not set then we just received an object
00233                                 // update so switch to periodic for the timeout period to prevent
00234                                 // sending more updates.  Listen to the EV_UPDATED flag still to
00235                                 // catch any updates that would overwise never get sent.
00236                                 eventMask = EV_UPDATED_PERIODIC | EV_UPDATED | EV_UPDATED_MANUAL | EV_UPDATE_REQ;
00237                         }
00238                 }
00239                 UAVObjConnectQueue(obj, priorityQueue, eventMask);
00240                 break;
00241         case UPDATEMODE_MANUAL:
00242                 // Set update period
00243                 setUpdatePeriod(obj, 0);
00244                 // Connect queue
00245                 eventMask = EV_UPDATED_MANUAL | EV_UPDATE_REQ;
00246                 UAVObjConnectQueue(obj, priorityQueue, eventMask);
00247                 break;
00248         }

00290                                 ++retries;
00291                         }
00292                         // Update stats
00293                         txRetries += (retries - 1);
00294                         if (success == -1) {
00295                                 ++txErrors;
00296                         }
00297                 } else if (ev->event == EV_UPDATED_PERIODIC && updateMode == UPDATEMODE_THROTTLED) {
00298                         // Get the event mask
00299                         int32_t eventMask = getEventMask(ev->obj, priorityQueue);
00300 
00301                         if (eventMask & EV_UPDATED_THROTTLED_DIRTY) { // If EV_UPDATED_THROTTLED_DIRTY flag is set then send the data like normal.
00302                                 // Send update to GCS (with retries)
00303                                 while (retries < MAX_RETRIES && success == -1) {
00304                                         success = UAVTalkSendObject(uavTalkCon, ev->obj, ev->instId, UAVObjGetTelemetryAcked(&metadata), REQ_TIMEOUT_MS);       // call blocks until ack is received or timeout
00305                                         ++retries;
00306                                 }
00307                                 // Update stats
00308                                 txRetries += (retries - 1);
00309                                 if (success == -1) {
00310                                         ++txErrors;
00311                                 }
00312                         }
00313                 }
00314                 // If this is a metaobject then make necessary telemetry updates
00315                 if (UAVObjIsMetaobject(ev->obj)) {
00316                         updateObject(UAVObjGetLinkedObj(ev->obj), EV_NONE);     // linked object will be the actual object the metadata are for
00317                 } else {

00319                                 // If this is UPDATEMODE_THROTTLED, the event mask changes on every event.
00320                                 updateObject(ev->obj, ev->event);
00321                         }
00322                 }

00470                 flightStats.RxDataRate = 0;

00475                 txErrors = 0;
