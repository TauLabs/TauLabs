00114 int32_t PIOS_EXTI_Init(const struct pios_exti_cfg * cfg)
00115 {
00116         PIOS_Assert(cfg);
00117         PIOS_Assert(&__start__exti);
00118         PIOS_Assert(cfg >= &__start__exti);
00119         PIOS_Assert(cfg < &__stop__exti);
00120 
00121         uint8_t cfg_index = cfg - &__start__exti;
00122 
00123         /* Connect this config to the requested vector */
00124         uint8_t line_index = PIOS_EXTI_line_to_index(cfg->line);
00125 
00126         if (pios_exti_line_to_cfg_map[line_index] != PIOS_EXTI_INVALID) {
00127                 /* Someone else already has this mapped */
00128                 goto out_fail;
00129         }
00130 
00131         /* Bind the config to the exti line */
00132         pios_exti_line_to_cfg_map[line_index] = cfg_index;
00133 
00134         /* Initialize the GPIO pin */
00135         GPIO_Init(cfg->pin.gpio, &cfg->pin.init);
00136 
00137         /* Set up the EXTI interrupt source */
00138         uint8_t exti_source_port = PIOS_EXTI_gpio_port_to_exti_source_port(cfg->pin.gpio);
00139         uint8_t exti_source_pin = PIOS_EXTI_gpio_pin_to_exti_source_pin(cfg->pin.init.GPIO_Pin);
00140         GPIO_EXTILineConfig(exti_source_port, exti_source_pin);
00141         EXTI_Init(&cfg->exti.init);
00142 
00143         /* Enable the interrupt channel */
00144         NVIC_Init(&cfg->irq.init);
00145 
00146         return 0;
00147 
00148 out_fail:
00149         return -1;
00150 }

00152 static bool PIOS_EXTI_generic_irq_handler(uint8_t line_index)
00153 {
00154         uint8_t cfg_index = pios_exti_line_to_cfg_map[line_index];
00155 
00156         PIOS_Assert(&__start__exti);
00157 
00158         if (cfg_index > NELEMENTS(pios_exti_line_to_cfg_map) ||
00159                 cfg_index == PIOS_EXTI_INVALID) {
00160                 /* Unconfigured interrupt just fired! */
00161                 return false;
00162         }
00163 
00164         struct pios_exti_cfg * cfg = &__start__exti + cfg_index;
00165         return cfg->vector();
00166 }
