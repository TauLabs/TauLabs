00040 // ****** convert Lat,Lon,Alt to ECEF  ************
00041 void LLA2ECEF(float LLA[3], float ECEF[3])
00042 {
00043         const float a = 6378137.0f;     // Equatorial Radius
00044         const float e = 8.1819190842622e-2f;    // Eccentricity
00045         float sinLat, sinLon, cosLat, cosLon;
00046         float N;
00047 
00048         sinLat = sinf(DEG2RAD * LLA[0]);
00049         sinLon = sinf(DEG2RAD * LLA[1]);
00050         cosLat = cosf(DEG2RAD * LLA[0]);
00051         cosLon = cosf(DEG2RAD * LLA[1]);
00052 
00053         N = a / sqrtf(1.0f - e * e * sinLat * sinLat);  //prime vertical radius of curvature
00054 
00055         ECEF[0] = (N + LLA[2]) * cosLat * cosLon;
00056         ECEF[1] = (N + LLA[2]) * cosLat * sinLon;
00057         ECEF[2] = ((1 - e * e) * N + LLA[2]) * sinLat;
00058 }

00060 // ****** convert ECEF to Lat,Lon,Alt (ITERATIVE!) *********
00061 uint16_t ECEF2LLA(float ECEF[3], float LLA[3])
00062 {
00072         const float a = 6378137.0f;     // Equatorial Radius
00073         const float e = 8.1819190842622e-2f;    // Eccentricity
00074         float x = ECEF[0], y = ECEF[1], z = ECEF[2];
00075         float Lat, N, NplusH, delta, esLat;
00076         uint16_t iter;
00077 #define MAX_ITER 10             // should not take more than 5 for valid coordinates
00078 #define ACCURACY 1.0e-11f       // used to be e-14, but we don't need sub micrometer exact calculations
00079 
00080         LLA[1] = RAD2DEG * atan2f(y, x);
00081         Lat = DEG2RAD * LLA[0];
00082         esLat = e * sinf(Lat);
00083         N = a / sqrtf(1 - esLat * esLat);
00084         NplusH = N + LLA[2];
00085         delta = 1;
00086         iter = 0;
00087 
00088         while (((delta > ACCURACY) || (delta < -ACCURACY))
00089                && (iter < MAX_ITER)) {
00090                 delta = Lat - atanf(z / (sqrtf(x * x + y * y) * (1 - (N * e * e / NplusH))));
00091                 Lat = Lat - delta;
00092                 esLat = e * sinf(Lat);
00093                 N = a / sqrtf(1 - esLat * esLat);
00094                 NplusH = sqrtf(x * x + y * y) / cosf(Lat);
00095                 iter += 1;
00096         }
00097 
00098         LLA[0] = RAD2DEG * Lat;
00099         LLA[2] = NplusH - N;
00100 
00101         return (iter < MAX_ITER);
00102 }

00126 void Quaternion2RPY(const float q[4], float rpy[3])
00127 {
00128         float R13, R11, R12, R23, R33;
00129         float q0s = q[0] * q[0];
00130         float q1s = q[1] * q[1];
00131         float q2s = q[2] * q[2];
00132         float q3s = q[3] * q[3];
00133 
00134         R13 = 2.0f * (q[1] * q[3] - q[0] * q[2]);
00135         R11 = q0s + q1s - q2s - q3s;
00136         R12 = 2.0f * (q[1] * q[2] + q[0] * q[3]);
00137         R23 = 2.0f * (q[2] * q[3] + q[0] * q[1]);
00138         R33 = q0s - q1s - q2s + q3s;

00140         rpy[1] = RAD2DEG * asinf(-R13); // pitch always between -pi/2 to pi/2
00141         rpy[2] = RAD2DEG * atan2f(R12, R11);
00142         rpy[0] = RAD2DEG * atan2f(R23, R33);

00148 void RPY2Quaternion(const float rpy[3], float q[4])
00149 {
00150         float phi, theta, psi;
00151         float cphi, sphi, ctheta, stheta, cpsi, spsi;
00152 
00153         phi = DEG2RAD * rpy[0] / 2;
00154         theta = DEG2RAD * rpy[1] / 2;
00155         psi = DEG2RAD * rpy[2] / 2;
00156         cphi = cosf(phi);
00157         sphi = sinf(phi);
00158         ctheta = cosf(theta);
00159         stheta = sinf(theta);
00160         cpsi = cosf(psi);
00161         spsi = sinf(psi);
00162 
00163         q[0] = cphi * ctheta * cpsi + sphi * stheta * spsi;
00164         q[1] = sphi * ctheta * cpsi - cphi * stheta * spsi;
00165         q[2] = cphi * stheta * cpsi + sphi * ctheta * spsi;
00166         q[3] = cphi * ctheta * spsi - sphi * stheta * cpsi;
00167 
00168         if (q[0] < 0) {         // q0 always positive for uniqueness
00169                 q[0] = -q[0];
00170                 q[1] = -q[1];
00171                 q[2] = -q[2];

00193 //** Find Rbe, that rotates a vector from earth fixed to body frame, from euler angles, using Tait-Bryan
00194 //**  convention, i.e. Rbe=Rot_x(roll)*Rot_y(pitch)*Rot_z(yaw) **
00195 void Euler2R(float rpy[3], float Rbe[3][3])
00196 {
00197         
00198         float sF = sinf(rpy[0]), cF = cosf(rpy[0]);
00199         float sT = sinf(rpy[1]), cT = cosf(rpy[1]);
00200         float sP = sinf(rpy[2]), cP = cosf(rpy[2]);
00201         
00202         Rbe[0][0] = cT*cP;
00203         Rbe[0][1] = cT*sP;
00204         Rbe[0][2] = -sT;
00205         Rbe[1][0] = sF*sT*cP - cF*sP;
00206         Rbe[1][1] = sF*sT*sP + cF*cP;
00207         Rbe[1][2] = cT*sF;
00208         Rbe[2][0] = cF*sT*cP + sF*sP;
00209         Rbe[2][1] = cF*sT*sP - sF*cP;
00210         Rbe[2][2] = cT*cF;
00211 }
