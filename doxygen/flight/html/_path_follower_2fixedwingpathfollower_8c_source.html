00513 float followStraightLine(float r[3], float q[3], float p[3], float psi,
00514                          float chi_inf, float k_path, float k_psi_int,
00515                          float delT)
00516 {
00517         float chi_q = atan2f(q[1], q[0]);
00518         while (chi_q - psi < -PI) {
00519                 chi_q += 2.0f * PI;
00520         }
00521         while (chi_q - psi > PI) {
00522                 chi_q -= 2.0f * PI;
00523         }
00524 
00525         float err_p = -sinf(chi_q) * (p[0] - r[0]) + cosf(chi_q) * (p[1] - r[1]);
00526         integral->lineError += delT * err_p;
00527         float psi_command = chi_q - chi_inf * 2.0f / PI * atanf(k_path * err_p) -
00528             k_psi_int * integral->lineError;

00530         return psi_command;
