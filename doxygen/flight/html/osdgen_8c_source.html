394 void write_hline(uint8_t *buff, unsigned int x0, unsigned int x1, unsigned int y, int mode)
395 {
396  CLIP_COORDS(x0, y);
397  CLIP_COORDS(x1, y);
398  if(x0 > x1)
399  {
400  SWAP(x0, x1);
401  }
402  if(x0 == x1) return;
403  /* This is an optimised algorithm for writing horizontal lines.
404  * We begin by finding the addresses of the x0 and x1 points. */
405  int addr0 = CALC_BUFF_ADDR(x0, y);
406  int addr1 = CALC_BUFF_ADDR(x1, y);
407  int addr0_bit = CALC_BIT_IN_WORD(x0);
408  int addr1_bit = CALC_BIT_IN_WORD(x1);
409  int mask, mask_l, mask_r, i;
410  /* If the addresses are equal, we only need to write one word
411  * which is an island. */
412  if(addr0 == addr1)
413  {
414  mask = COMPUTE_HLINE_ISLAND_MASK(addr0_bit, addr1_bit);
415  WRITE_WORD_MODE(buff, addr0, mask, mode);
416  }
417  /* Otherwise we need to write the edges and then the middle. */
418  else
419  {
420  mask_l = COMPUTE_HLINE_EDGE_L_MASK(addr0_bit);
421  mask_r = COMPUTE_HLINE_EDGE_R_MASK(addr1_bit);
422  WRITE_WORD_MODE(buff, addr0, mask_l, mode);
423  WRITE_WORD_MODE(buff, addr1, mask_r, mode);
424  // Now write 0xffff words from start+1 to end-1.
425  for(i = addr0 + 1; i <= addr1 - 1; i++)
426  {
427  uint8_t m=0xff;
428  WRITE_WORD_MODE(buff, i, m, mode);
429  }
430  }
431 }

573 void write_filled_rectangle(uint8_t *buff, unsigned int x, unsigned int y, unsigned int width, unsigned int height, int mode)
574 {
575  int yy, addr0_old, addr1_old;
576  CHECK_COORDS(x, y);
577  CHECK_COORD_X(x + width);
578  CHECK_COORD_Y(y + height);
579  if(width <= 0 || height <= 0) return;
580  // Calculate as if the rectangle was only a horizontal line. We then
581  // step these addresses through each row until we iterate `height` times.
582  int addr0 = CALC_BUFF_ADDR(x, y);
583  int addr1 = CALC_BUFF_ADDR(x + width, y);
584  int addr0_bit = CALC_BIT_IN_WORD(x);
585  int addr1_bit = CALC_BIT_IN_WORD(x + width);
586  int mask, mask_l, mask_r, i;
587  // If the addresses are equal, we need to write one word vertically.
588  if(addr0 == addr1)
589  {
590  mask = COMPUTE_HLINE_ISLAND_MASK(addr0_bit, addr1_bit);
591  while(height--)
592  {
593  WRITE_WORD_MODE(buff, addr0, mask, mode);
594  addr0 += GRAPHICS_WIDTH_REAL / 8;
595  }
596  }
597  // Otherwise we need to write the edges and then the middle repeatedly.
598  else
599  {
600  mask_l = COMPUTE_HLINE_EDGE_L_MASK(addr0_bit);
601  mask_r = COMPUTE_HLINE_EDGE_R_MASK(addr1_bit);
602  // Write edges first.
603  yy = 0;
604  addr0_old = addr0;
605  addr1_old = addr1;
606  while(yy < height)
607  {
608  WRITE_WORD_MODE(buff, addr0, mask_l, mode);
609  WRITE_WORD_MODE(buff, addr1, mask_r, mode);
610  addr0 += GRAPHICS_WIDTH_REAL / 8;
611  addr1 += GRAPHICS_WIDTH_REAL / 8;
612  yy++;
613  }
614  // Now write 0xffff words from start+1 to end-1 for each row.
615  yy = 0;
616  addr0 = addr0_old;
617  addr1 = addr1_old;
618  while(yy < height)
619  {
620  for(i = addr0 + 1; i <= addr1 - 1; i++)
621  {
622  uint8_t m=0xff;
623  WRITE_WORD_MODE(buff, i, m, mode);
624  }
625  addr0 += GRAPHICS_WIDTH_REAL / 8;
626  addr1 += GRAPHICS_WIDTH_REAL / 8;
627  yy++;
628  }
629  }

2092 void calcHomeArrow(int16_t m_yaw)
2093 {
2094  HomeLocationData home;
2095  HomeLocationGet (&home);
2096  GPSPositionData gpsData;
2097  GPSPositionGet (&gpsData);
2098 
2100  float lat1, lat2, lon1, lon2, a, c, d, x, y, brng, u2g;
2101  float elevation;
2102  float gcsAlt=home.Altitude; // Home MSL altitude
2103  float uavAlt=gpsData.Altitude; // UAV MSL altitude
2104  float dAlt=uavAlt-gcsAlt; // Altitude difference
2105 
2106  // Convert to radians
2107  lat1 = home.Latitude/10000000.0f * DEG2RAD; // Home lat
2108  lon1 = home.Longitude/10000000.0f * DEG2RAD; // Home lon
2109  lat2 = gpsData.Latitude/10000000.0f * DEG2RAD; // UAV lat
2110  lon2 = gpsData.Longitude/10000000.0f * DEG2RAD; // UAV lon
2111 
2112  // Bearing
2119  y = sinf(lon2-lon1) * cosf(lat2);
2120  x = cosf(lat1) * sinf(lat2) - sinf(lat1) * cosf(lat2) * cosf(lon2-lon1);
2121  brng = atan2f(y,x) * RAD2DEG;
2122  if(brng<0)
2123  brng+=360;
2124 
2125  // yaw corrected bearing, needs compass
2126  u2g=brng-180-m_yaw;
2127  if(u2g<0)
2128  u2g+=360;
2129 
2130  // Haversine formula for distance
2141  a = sinf((lat2-lat1)/2) * sinf((lat2-lat1)/2) +
2142  cosf(lat1) * cosf(lat2) *
2143  sinf((lon2-lon1)/2) * sinf((lon2-lon1)/2);
2144  c = 2 * atan2f(sqrtf(a), sqrtf(1-a));
2145  d = WGS84_RADIUS_EARTH_KM * 1000 * c;
2146 
2147  // Elevation v depends servo direction
2148  if(d!=0)
2149  elevation = 90-(atanf(dAlt/d) * RAD2DEG);
2150  else
2151  elevation = 0;
