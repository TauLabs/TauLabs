00052 /* models crc hardware (minor variation on polynomial division algorithm) */
00053 BIT16 crchware(BIT16 data, BIT16 genpoly, BIT16 accum)
00054 {
00055         static BIT16 i;
00056         data <<= 8;
00057         for (i = 8; i > 0; i--) {
00058                 if ((data ^ accum) & 0x8000)
00059                         accum = ((accum << 1) ^ genpoly) & 0xFFFF;
00060                 else
00061                         accum = (accum<<1) & 0xFFFF;
00062                 data = (data<<1) & 0xFFFF;
00063         }
00064         return (accum);
