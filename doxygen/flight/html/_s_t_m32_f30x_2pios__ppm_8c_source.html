122 extern int32_t PIOS_PPM_Init(uint32_t * ppm_id, const struct pios_ppm_cfg * cfg)
123 {
124  PIOS_DEBUG_Assert(ppm_id);
125  PIOS_DEBUG_Assert(cfg);
126 
127  struct pios_ppm_dev * ppm_dev;
128 
129  ppm_dev = (struct pios_ppm_dev *) PIOS_PPM_alloc();
130  if (!ppm_dev) goto out_fail;
131 
132  /* Bind the configuration to the device instance */
133  ppm_dev->cfg = cfg;
134 
135  /* Set up the state variables */
136  ppm_dev->PulseIndex = 0;
137  ppm_dev->PreviousTime = 0;
138  ppm_dev->CurrentTime = 0;
139  ppm_dev->DeltaTime = 0;
140  ppm_dev->LargeCounter = 0;
141  ppm_dev->NumChannels = -1;
142  ppm_dev->NumChannelsPrevFrame = -1;
143  ppm_dev->NumChannelCounter = 0;
144  ppm_dev->Tracking = false;
145  ppm_dev->Fresh = false;
146 
147  for (uint8_t i = 0; i < PIOS_PPM_IN_MAX_NUM_CHANNELS; i++) {
148  /* Flush counter variables */
149  ppm_dev->CaptureValue[i] = PIOS_RCVR_TIMEOUT;
150  ppm_dev->CaptureValueNewFrame[i] = PIOS_RCVR_TIMEOUT;
151 
152  }
153 
154  uint32_t tim_id;
155  if (PIOS_TIM_InitChannels(&tim_id, cfg->channels, cfg->num_channels, &tim_callbacks, (uint32_t)ppm_dev)) {
156  return -1;
157  }
158 
159  /* Configure the channels to be in capture/compare mode */
160  for (uint8_t i = 0; i < cfg->num_channels; i++) {
161  const struct pios_tim_channel * chan = &cfg->channels[i];
162 
163  /* Configure timer for input capture */
164  TIM_ICInitTypeDef TIM_ICInitStructure = cfg->tim_ic_init;
165  TIM_ICInitStructure.TIM_Channel = chan->timer_chan;
166  TIM_ICInit(chan->timer, &TIM_ICInitStructure);
167 
168  /* Enable the Capture Compare Interrupt Request */
169  switch (chan->timer_chan) {
170  case TIM_Channel_1:
171  TIM_ITConfig(chan->timer, TIM_IT_CC1 | TIM_IT_Update, ENABLE);
172  break;
173  case TIM_Channel_2:
174  TIM_ITConfig(chan->timer, TIM_IT_CC2 | TIM_IT_Update, ENABLE);
175  break;
176  case TIM_Channel_3:
177  TIM_ITConfig(chan->timer, TIM_IT_CC3 | TIM_IT_Update, ENABLE);
178  break;
179  case TIM_Channel_4:
180  TIM_ITConfig(chan->timer, TIM_IT_CC4 | TIM_IT_Update, ENABLE);
181  break;
182  }
183  }
184 
185  /* Setup local variable which stays in this scope */
186  /* Doing this here and using a local variable saves doing it in the ISR */
187  TIM_ICInitStructure.TIM_ICSelection = TIM_ICSelection_DirectTI;
188  TIM_ICInitStructure.TIM_ICPrescaler = TIM_ICPSC_DIV1;
189  TIM_ICInitStructure.TIM_ICFilter = 0x0;
190 
191  if (!PIOS_RTC_RegisterTickCallback(PIOS_PPM_Supervisor, (uint32_t)ppm_dev)) {
192  PIOS_DEBUG_Assert(0);
193  }
194 
195  *ppm_id = (uint32_t)ppm_dev;
196 
197  return(0);
198 
199 out_fail:
200  return(-1);
201 }

210 static int32_t PIOS_PPM_Get(uint32_t rcvr_id, uint8_t channel)
211 {
212  struct pios_ppm_dev * ppm_dev = (struct pios_ppm_dev *)rcvr_id;
213 
214  if (!PIOS_PPM_validate(ppm_dev)) {
215  /* Invalid device specified */
216  return PIOS_RCVR_INVALID;
217  }
218 
219  if (channel >= PIOS_PPM_IN_MAX_NUM_CHANNELS) {
220  /* Channel out of range */
221  return PIOS_RCVR_INVALID;
222  }
223  return ppm_dev->CaptureValue[channel];
224 }

241 static void PIOS_PPM_tim_edge_cb (uint32_t tim_id, uint32_t context, uint8_t chan_idx, uint16_t count)
242 {
243  /* Recover our device context */
244  struct pios_ppm_dev * ppm_dev = (struct pios_ppm_dev *)context;
245 
246  if (!PIOS_PPM_validate(ppm_dev)) {
247  /* Invalid device specified */
248  return;
249  }
250 
251  if (chan_idx >= ppm_dev->cfg->num_channels) {
252  /* Channel out of range */
253  return;
254  }
255 
256  /* Shift the last measurement out */
257  ppm_dev->PreviousTime = ppm_dev->CurrentTime;
258 
259  /* Grab the new count */
260  ppm_dev->CurrentTime = count;
261 
262  /* Convert to 32-bit timer result */
263  ppm_dev->CurrentTime += ppm_dev->LargeCounter;
264 
265  /* Capture computation */
266  ppm_dev->DeltaTime = ppm_dev->CurrentTime - ppm_dev->PreviousTime;
267 
268  ppm_dev->PreviousTime = ppm_dev->CurrentTime;
269 
270  /* Sync pulse detection */
271  if (ppm_dev->DeltaTime > PIOS_PPM_IN_MIN_SYNC_PULSE_US) {
272  if (ppm_dev->PulseIndex == ppm_dev->NumChannelsPrevFrame
273  && ppm_dev->PulseIndex >= PIOS_PPM_IN_MIN_NUM_CHANNELS
274  && ppm_dev->PulseIndex <= PIOS_PPM_IN_MAX_NUM_CHANNELS)
275  {
276  /* If we see n simultaneous frames of the same
277  number of channels we save it as our frame size */
278  if (ppm_dev->NumChannelCounter < PIOS_PPM_STABLE_CHANNEL_COUNT)
279  ppm_dev->NumChannelCounter++;
280  else
281  ppm_dev->NumChannels = ppm_dev->PulseIndex;
282  } else {
283  ppm_dev->NumChannelCounter = 0;
284  }
285 
286  /* Check if the last frame was well formed */
287  if (ppm_dev->PulseIndex == ppm_dev->NumChannels && ppm_dev->Tracking) {
288  /* The last frame was well formed */
289  for (uint32_t i = 0; i < ppm_dev->NumChannels; i++) {
290  ppm_dev->CaptureValue[i] = ppm_dev->CaptureValueNewFrame[i];
291  }
292  for (uint32_t i = ppm_dev->NumChannels;
293  i < PIOS_PPM_IN_MAX_NUM_CHANNELS; i++) {
294  ppm_dev->CaptureValue[i] = PIOS_RCVR_TIMEOUT;
295  }
296  }
297 
298  ppm_dev->Fresh = true;
299  ppm_dev->Tracking = true;
300  ppm_dev->NumChannelsPrevFrame = ppm_dev->PulseIndex;
301  ppm_dev->PulseIndex = 0;
302 
303  /* We rely on the supervisor to set CaptureValue to invalid
304  if no valid frame is found otherwise we ride over it */
305 
306  } else if (ppm_dev->Tracking) {
307  /* Valid pulse duration 0.75 to 2.5 ms*/
308  if (ppm_dev->DeltaTime > PIOS_PPM_IN_MIN_CHANNEL_PULSE_US
309  && ppm_dev->DeltaTime < PIOS_PPM_IN_MAX_CHANNEL_PULSE_US
310  && ppm_dev->PulseIndex < PIOS_PPM_IN_MAX_NUM_CHANNELS) {
311 
312  ppm_dev->CaptureValueNewFrame[ppm_dev->PulseIndex] = ppm_dev->DeltaTime;
313  ppm_dev->PulseIndex++;
314  } else {
315  /* Not a valid pulse duration */
316  ppm_dev->Tracking = false;
317  for (uint32_t i = 0; i < PIOS_PPM_IN_MAX_NUM_CHANNELS ; i++) {
318  ppm_dev->CaptureValueNewFrame[i] = PIOS_RCVR_TIMEOUT;
319  }
320  }
321  }
322 }

324 static void PIOS_PPM_Supervisor(uint32_t ppm_id) {
325  /* Recover our device context */
326  struct pios_ppm_dev * ppm_dev = (struct pios_ppm_dev *)ppm_id;
327 
328  if (!PIOS_PPM_validate(ppm_dev)) {
329  /* Invalid device specified */
330  return;
331  }
332 
333  /*
334  * RTC runs at 625Hz so divide down the base rate so
335  * that this loop runs at 25Hz.
336  */
337  if(++(ppm_dev->supv_timer) < 25) {
338  return;
339  }
340  ppm_dev->supv_timer = 0;
341 
342  if (!ppm_dev->Fresh) {
343  ppm_dev->Tracking = false;
344 
345  for (int32_t i = 0; i < PIOS_PPM_IN_MAX_NUM_CHANNELS ; i++) {
346  ppm_dev->CaptureValue[i] = PIOS_RCVR_TIMEOUT;
347  ppm_dev->CaptureValueNewFrame[i] = PIOS_RCVR_TIMEOUT;
348  }
349  }
350 
351  ppm_dev->Fresh = false;
352 }
