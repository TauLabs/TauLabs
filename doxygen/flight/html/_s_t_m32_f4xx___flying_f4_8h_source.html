00170 //      DAC                     |PWR                            |CAN1,2
00171 //      I2C1,2,3        |UART4,5                        |USART3,2

00174 //      RTC/BKP reg     
00175 // TIM2,3,4,5,6,7,12,13,14
00176 
00177 // Calculated as SYSCLK / APBPresc * (APBPre == 1 ? 1 : 2)   
00178 // Default APB1 Prescaler = 4
