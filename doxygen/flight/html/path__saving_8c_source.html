00052         int32_t  waypoint_size = UAVObjGetNumBytes(WaypointHandle());
00053         int32_t  erase_retval  = 0;
00054         int32_t  last_save_id  = -1;
00055         int32_t  retval        = 0;

00057         // Save all elements
00058         for (int32_t i = 0; i < num_waypoints && retval == 0; i++) {
00059                 WaypointInstGet(i, &waypoint);
00060 
00061                 // Stop saving when get to invalid waypoint.  Nothing after or including is valid
00062                 if (waypoint.Mode == WAYPOINT_MODE_INVALID || waypoint.Mode == WAYPOINT_MODE_STOP)
00063                         break;
00064 
00065                 retval = PIOS_FLASHFS_ObjSave(pios_waypoints_settings_fs_id, path_id, i, (uint8_t *) &waypoint, waypoint_size);
00066                 last_save_id = i; // Track the last valid waypoint id
00067         }
00068 
00069         // Use an explicit indication of the end of path
00070         waypoint.Mode = WAYPOINT_MODE_STOP;
00071         retval = PIOS_FLASHFS_ObjSave(pios_waypoints_settings_fs_id, path_id, ++last_save_id,
00072                                       (uint8_t *) &waypoint, waypoint_size);
00073 
00074         // Check for any waypoints after the saved end of the path and erase them
00075         for (int32_t i = last_save_id + 1; erase_retval == 0; i++) {
00076                 erase_retval = PIOS_FLASHFS_ObjLoad(pios_waypoints_settings_fs_id, path_id, i, (uint8_t *) &waypoint, waypoint_size);
00077                 if (erase_retval == 0) {
00078                         PIOS_FLASHFS_ObjDelete(pios_waypoints_settings_fs_id, path_id, i);
00079                 }
00080         }
