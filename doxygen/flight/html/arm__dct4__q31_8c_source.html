68  /* DCT4 computation involves DCT2 (which is calculated using RFFT)
69  * along with some pre-processing and post-processing.
70  * Computational procedure is explained as follows:
71  * (a) Pre-processing involves multiplying input with cos factor,
72  * r(n) = 2 * u(n) * cos(pi*(2*n+1)/(4*n))
73  * where,
74  * r(n) -- output of preprocessing
75  * u(n) -- input to preprocessing(actual Source buffer)
76  * (b) Calculation of DCT2 using FFT is divided into three steps:
77  * Step1: Re-ordering of even and odd elements of input.
78  * Step2: Calculating FFT of the re-ordered input.
79  * Step3: Taking the real part of the product of FFT output and weights.
80  * (c) Post-processing - DCT4 can be obtained from DCT2 output using the following equation:
81  * Y4(k) = Y2(k) - Y4(k-1) and Y4(-1) = Y4(0)
82  * where,
83  * Y4 -- DCT4 output, Y2 -- DCT2 output
84  * (d) Multiplying the output with the normalizing factor sqrt(2/N).
85  */
86 
87  /*-------- Pre-processing ------------*/
88  /* Multiplying input with cos factor i.e. r(n) = 2 * x(n) * cos(pi*(2*n+1)/(4*n)) */
89  arm_mult_q31(pInlineBuffer, cosFact, pInlineBuffer, S->N);
90  arm_shift_q31(pInlineBuffer, 1, pInlineBuffer, S->N);
91 
92  /* ----------------------------------------------------------------
93  * Step1: Re-ordering of even and odd elements as
94  * pState[i] = pInlineBuffer[2*i] and
95  * pState[N-i-1] = pInlineBuffer[2*i+1] where i = 0 to N/2
96  ---------------------------------------------------------------------*/
97

99  pS1 = pState;
100 
101  /* pS2 initialized to pState+N-1, so that it points to the end of the state buffer */
102  pS2 = pState + (S->N - 1u);
103 
104  /* pbuff initialized to input buffer */
105  pbuff = pInlineBuffer;

112  i = S->Nby2 >> 2u;

114  /* First part of the processing with loop unrolling. Compute 4 outputs at a time.
115  ** a second loop below computes the remaining 1 to 3 samples. */
116  do
117  {
118  /* Re-ordering of even and odd elements */
119  /* pState[i] = pInlineBuffer[2*i] */
120  *pS1++ = *pbuff++;
121  /* pState[N-i-1] = pInlineBuffer[2*i+1] */
122  *pS2-- = *pbuff++;

173  * Hence changing the format of N (i.e. 2*N elements) complex numbers to 1.31 format by shifting left by 2 bits. */
174  arm_shift_q31(pState, 2, pState, S->N * 2);

178  * Y4(k) = Y2(k) - Y4(k-1) and Y4(-1) = Y4(0)
179  * Hence, Y4(0) = Y2(0)/2 */
