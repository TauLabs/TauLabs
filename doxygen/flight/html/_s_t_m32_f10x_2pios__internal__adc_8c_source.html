00322                 return;
00323         }

00325         uint16_t sample;

00328         for (chan = 0; chan < PIOS_ADC_NUM_CHANNELS; chan++) {

00330                 for (sample = 0; sample < adc_dev->adc_oversample; sample++) {
00331                         sum += adc_dev->valid_data_buffer[chan + sample * adc_dev->dma_block_size] * adc_dev->fir_coeffs[sample];
00332                 }
00333                 downsampled_buffer[chan] = (float) sum / adc_dev->fir_coeffs[adc_dev->adc_oversample];

00337         if(adc_dev->data_queue) {

00341         }
00342 #endif
