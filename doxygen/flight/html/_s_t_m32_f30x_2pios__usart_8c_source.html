00193                 USART_InvPinCmd(usart_dev->cfg->regs, USART_InvPin_Rx, ENABLE);

00195                 USART_InvPinCmd(usart_dev->cfg->regs, USART_InvPin_Rx, DISABLE);
00196 
00197         if (usart_dev->cfg->tx_invert == true)
00198                 USART_InvPinCmd(usart_dev->cfg->regs, USART_InvPin_Tx, ENABLE);

00323 static void PIOS_USART_generic_irq_handler(uint32_t usart_id)
00324 {
00325         struct pios_usart_dev * usart_dev = (struct pios_usart_dev *)usart_id;
00326 
00327         bool rx_need_yield = false;
00328         bool tx_need_yield = false;
00329 
00330         bool valid = PIOS_USART_validate(usart_dev);
00331         PIOS_Assert(valid);
00332         
00333         /* Check if RXNE flag is set */
00334         if (USART_GetITStatus(usart_dev->cfg->regs, USART_IT_RXNE)) {
00335                 uint8_t byte = (uint8_t)USART_ReceiveData(usart_dev->cfg->regs);
00336                 if (usart_dev->rx_in_cb) {
00337                         (void) (usart_dev->rx_in_cb)(usart_dev->rx_in_context, &byte, 1, NULL, &rx_need_yield);
00338                 }
00339         }
00340         /* Check if TXE flag is set */
00341         if (USART_GetITStatus(usart_dev->cfg->regs, USART_IT_TXE)) {
00342                 if (usart_dev->tx_out_cb) {
00343                         uint8_t b;
00344                         uint16_t bytes_to_send;
00345                         
00346                         bytes_to_send = (usart_dev->tx_out_cb)(usart_dev->tx_out_context, &b, 1, NULL, &tx_need_yield);
00347                         
00348                         if (bytes_to_send > 0) {
00349                                 /* Send the byte we've been given */
00350                                 USART_SendData(usart_dev->cfg->regs, b);
00351                         } else {
00352                                 /* No bytes to send, disable TXE interrupt */
00353                                 USART_ITConfig(usart_dev->cfg->regs, USART_IT_TXE, DISABLE);
00354                         }
00355                 } else {
00356                         /* No bytes to send, disable TXE interrupt */
00357                         USART_ITConfig(usart_dev->cfg->regs, USART_IT_TXE, DISABLE);
00358                 }
00359         }
00360         /* Check for overrun condition
00361          * Note i really wanted to use USART_GetITStatus but it fails on getting the
00362          * ORE flag although RXNE interrupt is enabled.
00363          * Probably a bug in the ST library...
00364          */
00365         if (USART_GetFlagStatus(usart_dev->cfg->regs, USART_FLAG_ORE)) {
00366                 USART_ClearITPendingBit(usart_dev->cfg->regs, USART_IT_ORE);
00367                 ++usart_dev->error_overruns;
00368         }
00369         
00370 #if defined(PIOS_INCLUDE_FREERTOS)
00371         portEND_SWITCHING_ISR(rx_need_yield || tx_need_yield);
00372 #endif  /* PIOS_INCLUDE_FREERTOS */
00373 }
