00059 static void PIOS_RFM22B_COM_ChangeBaud(uint32_t rfm22b_id, uint32_t baud)
00060 {
00061         struct pios_rfm22b_dev *rfm22b_dev = (struct pios_rfm22b_dev *)rfm22b_id;
00062         if (!PIOS_RFM22B_validate(rfm22b_dev))
00063                 return;
00064         // This is only allowed for coordinators.
00065         if (!rfm22b_dev->coordinator)
00066                 return;
00067         // Set the RF data rate on the modem to ~2X the selected buad rate because the modem is half duplex.
00068         enum rfm22b_datarate datarate = RFM22_datarate_64000;
00069         if (baud <= 1024)
00070                 datarate = RFM22_datarate_500;
00071         else if (baud <= 2048)
00072                 datarate = RFM22_datarate_1000;
00073         else if (baud <= 4096)
00074                 datarate = RFM22_datarate_8000;
00075         else if (baud <= 9600)
00076                 datarate = RFM22_datarate_16000;
00077         else if (baud <= 19200)
00078                 datarate = RFM22_datarate_32000;
00079         else if (baud <= 38400)
00080                 datarate = RFM22_datarate_64000;
00081         else if (baud <= 57600)
00082                 datarate = RFM22_datarate_128000;
00083         else if (baud <= 115200)
00084                 datarate = RFM22_datarate_192000;
00085         PIOS_RFM22B_SetDatarate(rfm22b_id, datarate, true);
00086 }
