31 /*
32  * @note This is a stripped-down ADC driver intended primarily for sampling
33  * voltage and current values. Samples are averaged over the period between
34  * fetches so that relatively accurate measurements can be obtained without
35  * forcing higher-level logic to poll aggressively.
36  *
37  * @todo This module needs more work to be more generally useful. It should
38  * almost certainly grow callback support so that e.g. voltage and current readings
39  * can be shipped out for coulomb counting purposes. The F1xx interface presumes
40  * use with analog sensors, but that implementation largely dominates the ADC
41  * resources. Rather than commit to a new API without a defined use case, we
42  * should stick to our lightweight subset until we have a better idea of what's needed.
43  */

139 static void
140 init_dma(void)
141 {
142  /* Disable interrupts */
143  DMA_ITConfig(pios_adc_dev->cfg->dma.rx.channel, pios_adc_dev->cfg->dma.irq.flags, DISABLE);
144 
145  /* Configure DMA channel */
146  DMA_DeInit(pios_adc_dev->cfg->dma.rx.channel);
147  DMA_InitTypeDef DMAInit = pios_adc_dev->cfg->dma.rx.init;
148  DMAInit.DMA_Memory0BaseAddr = (uint32_t)&adc_raw_buffer[0];
149  DMAInit.DMA_BufferSize = sizeof(adc_raw_buffer[0]) / sizeof(uint16_t);
150  DMAInit.DMA_DIR = DMA_DIR_PeripheralToMemory;
151  DMAInit.DMA_PeripheralInc = DMA_PeripheralInc_Disable;
152  DMAInit.DMA_MemoryInc = DMA_MemoryInc_Enable;
153  DMAInit.DMA_PeripheralDataSize = DMA_PeripheralDataSize_HalfWord;
154  DMAInit.DMA_MemoryDataSize = DMA_MemoryDataSize_HalfWord;
155  DMAInit.DMA_Mode = DMA_Mode_Circular;
156  DMAInit.DMA_Priority = DMA_Priority_Low;
157  DMAInit.DMA_FIFOMode = DMA_FIFOMode_Disable;
158  DMAInit.DMA_FIFOThreshold = DMA_FIFOThreshold_HalfFull;
159  DMAInit.DMA_MemoryBurst = DMA_MemoryBurst_Single;
160  DMAInit.DMA_PeripheralBurst = DMA_PeripheralBurst_Single;
161 
162  DMA_Init(pios_adc_dev->cfg->dma.rx.channel, &DMAInit); /* channel is actually stream ... */
163 
164  /* configure for double-buffered mode and interrupt on every buffer flip */
165  DMA_DoubleBufferModeConfig(pios_adc_dev->cfg->dma.rx.channel, (uint32_t)&adc_raw_buffer[1], DMA_Memory_0);
166  DMA_DoubleBufferModeCmd(pios_adc_dev->cfg->dma.rx.channel, ENABLE);
167  DMA_ITConfig(pios_adc_dev->cfg->dma.rx.channel, DMA_IT_TC, ENABLE);
168  //DMA_ITConfig(pios_adc_dev->cfg->dma.rx.channel, DMA_IT_HT, ENABLE);
169 
170  /* enable DMA */
171  DMA_Cmd(pios_adc_dev->cfg->dma.rx.channel, ENABLE);
172 
173  /* Configure DMA interrupt */
174  NVIC_InitTypeDef NVICInit = pios_adc_dev->cfg->dma.irq.init;
175  NVIC_Init(&NVICInit);
176 }

178 static void
179 init_adc(void)
180 {
181  RCC_APB2PeriphClockCmd(RCC_APB2Periph_ADC1, ENABLE);
182 
183  ADC_DeInit();
184 
185  /* turn on VREFInt in case we need it */
186  ADC_TempSensorVrefintCmd(ENABLE);
187 
188  /* Do common ADC init */
189  ADC_CommonInitTypeDef ADC_CommonInitStructure;
190  ADC_CommonStructInit(&ADC_CommonInitStructure);
191  ADC_CommonInitStructure.ADC_Mode = ADC_Mode_Independent;
192  ADC_CommonInitStructure.ADC_Prescaler = ADC_Prescaler_Div8;
193  ADC_CommonInitStructure.ADC_DMAAccessMode = ADC_DMAAccessMode_Disabled;
194  ADC_CommonInitStructure.ADC_TwoSamplingDelay = ADC_TwoSamplingDelay_5Cycles;
195  ADC_CommonInit(&ADC_CommonInitStructure);
196 
197  ADC_InitTypeDef ADC_InitStructure;
198  ADC_StructInit(&ADC_InitStructure);
199  ADC_InitStructure.ADC_Resolution = ADC_Resolution_12b;
200  ADC_InitStructure.ADC_ScanConvMode = ENABLE;
201  ADC_InitStructure.ADC_ContinuousConvMode = ENABLE;
202  ADC_InitStructure.ADC_ExternalTrigConvEdge = ADC_ExternalTrigConvEdge_None;
203  ADC_InitStructure.ADC_DataAlign = ADC_DataAlign_Right;
204  ADC_InitStructure.ADC_NbrOfConversion = ((PIOS_ADC_NUM_PINS)/* >> 1*/);
205  ADC_Init(pios_adc_dev->cfg->adc_dev, &ADC_InitStructure);
206 
207  /* Enable DMA request */
208  ADC_DMACmd(pios_adc_dev->cfg->adc_dev, ENABLE);
209 
210  /* Configure input scan */
211  for (int32_t i = 0; i < PIOS_ADC_NUM_PINS; i++) {
212  ADC_RegularChannelConfig(pios_adc_dev->cfg->adc_dev,
213  config[i].channel,
214  i+1,
215  ADC_SampleTime_56Cycles); /* XXX this is totally arbitrary... */
216  }
217 
218  ADC_DMARequestAfterLastTransferCmd(pios_adc_dev->cfg->adc_dev, ENABLE);
219 
220  /* Finally start initial conversion */
221  ADC_Cmd(pios_adc_dev->cfg->adc_dev, ENABLE);
222  ADC_ContinuousModeCmd(pios_adc_dev->cfg->adc_dev, ENABLE);
223  ADC_SoftwareStartConv(pios_adc_dev->cfg->adc_dev);
224 }

298 int32_t PIOS_ADC_PinGet(uint32_t pin)
299 {
300 #if defined(PIOS_INCLUDE_ADC)
301  int32_t result;
302 
303  /* Check if pin exists */
304  if (pin >= PIOS_ADC_NUM_PINS) {
305  return -1;
306  }
307 
308  if (accumulator[pin].accumulator <= 0)
309  return -2;
310 
311  /* return accumulated result and clear accumulator */
312  result = accumulator[pin].accumulator / (accumulator[pin].count ?: 1);
313  accumulator[pin].accumulator = result;
314  accumulator[pin].count = 1;
315 
316  return result;
317 #endif
318  return -1;
319 }

385 void accumulate(uint16_t *buffer, uint32_t count)
386 {
387 #if defined(PIOS_INCLUDE_ADC)
388  uint16_t *sp = buffer;
389 
390  /*
391  * Accumulate sampled values.
392  */
393  while (count--) {
394  for (int i = 0; i < PIOS_ADC_NUM_PINS; i++) {
395  accumulator[i].accumulator += *sp++;
396  accumulator[i].count++;
397  /*
398  * If the accumulator reaches half-full, rescale in order to
399  * make more space.
400  */
401  if (accumulator[i].accumulator >= (1 << 31)) {
402  accumulator[i].accumulator /= 2;
403  accumulator[i].count /= 2;
404  }
405  }
406  }
407 
408 #if defined(PIOS_INCLUDE_FREERTOS)
409  // XXX should do something with this
410  if (pios_adc_dev->data_queue) {
411  static portBASE_TYPE xHigherPriorityTaskWoken;
412 // xQueueSendFromISR(pios_adc_dev->data_queue, pios_adc_dev->downsampled_buffer, &xHigherPriorityTaskWoken);
413  portEND_SWITCHING_ISR(xHigherPriorityTaskWoken);
414  }

429 void PIOS_ADC_DMA_Handler(void)
430 {
431  if (!PIOS_ADC_validate(pios_adc_dev))
432  return;
433 
434 #if defined(PIOS_INCLUDE_ADC)
435  /* terminal count, buffer has flipped */
436  if (DMA_GetITStatus(pios_adc_dev->cfg->dma.rx.channel, pios_adc_dev->cfg->full_flag)) {
437  DMA_ClearITPendingBit(pios_adc_dev->cfg->dma.rx.channel, pios_adc_dev->cfg->full_flag);
438 
439  /* accumulate results from the buffer that was just completed */
440  accumulate(&adc_raw_buffer[DMA_GetCurrentMemoryTarget(pios_adc_dev->cfg->dma.rx.channel) ? 0 : 1][0][0],
441  PIOS_ADC_MAX_SAMPLES);
442 
443  }
