381 {
382 #ifdef USE_FREERTOS

404 static void go_starting(struct pios_i2c_adapter *i2c_adapter)
405 {
406  PIOS_DEBUG_Assert(i2c_adapter->active_txn);
407  PIOS_DEBUG_Assert(i2c_adapter->active_txn >= i2c_adapter->first_txn);
408  PIOS_DEBUG_Assert(i2c_adapter->active_txn <= i2c_adapter->last_txn);
409 
410  i2c_adapter->active_byte = &(i2c_adapter->active_txn->buf[0]);
411  i2c_adapter->last_byte = &(i2c_adapter->active_txn->buf[i2c_adapter->active_txn->len - 1]);
412 
413  I2C_GenerateSTART(i2c_adapter->cfg->regs, ENABLE);
414  if (i2c_adapter->active_txn->rw == PIOS_I2C_TXN_READ) {
415  I2C_ITConfig(i2c_adapter->cfg->regs, I2C_IT_EVT | I2C_IT_BUF | I2C_IT_ERR, ENABLE);
416  } else {
417  // For write operations, do not enable the IT_BUF events.
418  // The current driver does not act when the TX data register is not full, only when the complete byte is sent.
419  // With the IT_BUF enabled, we constantly get IRQs, See OP-326
420  I2C_ITConfig(i2c_adapter->cfg->regs, I2C_IT_EVT | I2C_IT_ERR, ENABLE);
421  }
422 }

454 /* Common to 'more' and 'last' transaction */
455 static void go_r_any_txn_pre_middle(struct pios_i2c_adapter *i2c_adapter)
456 {
457  PIOS_DEBUG_Assert(i2c_adapter->active_byte);
458  PIOS_DEBUG_Assert(i2c_adapter->active_byte <= i2c_adapter->last_byte);
459 
460  *(i2c_adapter->active_byte) = I2C_ReceiveData(i2c_adapter->cfg->regs);
461 
462  /* Move to the next byte */
463  i2c_adapter->active_byte++;
464  PIOS_DEBUG_Assert(i2c_adapter->active_byte <= i2c_adapter->last_byte);
465 }

591 static void i2c_adapter_inject_event(struct pios_i2c_adapter *i2c_adapter, enum i2c_adapter_event event)
592 {
593  PIOS_IRQ_Disable();
594 
595 #if defined(PIOS_I2C_DIAGNOSTICS)
596  i2c_state_event_history[i2c_state_event_history_pointer] = event;
597  i2c_state_event_history_pointer = (i2c_state_event_history_pointer + 1) % I2C_LOG_DEPTH;
598 
599  i2c_state_history[i2c_state_history_pointer] = i2c_adapter->curr_state;
600  i2c_state_history_pointer = (i2c_state_history_pointer + 1) % I2C_LOG_DEPTH;
601 
602  if(i2c_adapter_transitions[i2c_adapter->curr_state].next_state[event] == I2C_STATE_FSM_FAULT)
603  i2c_adapter_log_fault(PIOS_I2C_ERROR_FSM);
604 #endif
605  /*
606  * Move to the next state
607  *
608  * This is done prior to calling the new state's entry function to
609  * guarantee that the entry function never depends on the previous
610  * state. This way, it cannot ever know what the previous state was.
611  */
612  enum i2c_adapter_state prev_state = i2c_adapter->curr_state;
613  if (prev_state) ;
614 
615  i2c_adapter->curr_state = i2c_adapter_transitions[i2c_adapter->curr_state].next_state[event];
616 
617  /* Call the entry function (if any) for the next state. */
618  if (i2c_adapter_transitions[i2c_adapter->curr_state].entry_fn) {
619  i2c_adapter_transitions[i2c_adapter->curr_state].entry_fn(i2c_adapter);
620  }
621 
622  /* Process any AUTO transitions in the FSM */
623  i2c_adapter_process_auto(i2c_adapter);
624 
625  PIOS_IRQ_Enable();
626 }

628 static void i2c_adapter_process_auto(struct pios_i2c_adapter *i2c_adapter)
629 {
630  PIOS_IRQ_Disable();
631 
632  enum i2c_adapter_state prev_state = i2c_adapter->curr_state;
633  if (prev_state) ;
634 
635  while (i2c_adapter_transitions[i2c_adapter->curr_state].next_state[I2C_EVENT_AUTO]) {
636  i2c_adapter->curr_state = i2c_adapter_transitions[i2c_adapter->curr_state].next_state[I2C_EVENT_AUTO];
637 
638  /* Call the entry function (if any) for the next state. */
639  if (i2c_adapter_transitions[i2c_adapter->curr_state].entry_fn) {
640  i2c_adapter_transitions[i2c_adapter->curr_state].entry_fn(i2c_adapter);
641  }
642  }
643 
644  PIOS_IRQ_Enable();
645 }

653 static bool i2c_adapter_wait_for_stopped(struct pios_i2c_adapter *i2c_adapter)
654 {
655  uint32_t guard;
656 
657  /*
658  * Wait for the bus to return to the stopped state.
659  * This was pulled out of the FSM due to occasional
660  * failures at this transition which previously resulted
661  * in spinning on this bit in the ISR forever.
662  */
663 #define I2C_CR1_STOP_REQUESTED 0x0200
664  for (guard = 1e6; /* FIXME: should use the configured bus timeout */
665  guard && (i2c_adapter->cfg->regs->CR1 & I2C_CR1_STOP_REQUESTED); guard--)
666  continue;
667  if (!guard) {
668  /* We timed out waiting for the stop condition */
669  return false;
670  }
671 
672  return true;
673 }

675 static void i2c_adapter_reset_bus(struct pios_i2c_adapter *i2c_adapter)
676 {
677  static uint8_t retry_count = 0;
678  static uint8_t retry_count_clk = 0;
679 
680  /* Reset the I2C block */
681  I2C_DeInit(i2c_adapter->cfg->regs);
682 
683  /* Make sure the bus is free by clocking it until any slaves release the bus. */
684  GPIO_InitTypeDef scl_gpio_init;
685  scl_gpio_init = i2c_adapter->cfg->scl.init;
686  scl_gpio_init.GPIO_Mode = GPIO_Mode_Out_OD;
687  GPIO_SetBits(i2c_adapter->cfg->scl.gpio, i2c_adapter->cfg->scl.init.GPIO_Pin);
688  GPIO_Init(i2c_adapter->cfg->scl.gpio, &scl_gpio_init);
689 
690  GPIO_InitTypeDef sda_gpio_init;
691  sda_gpio_init = i2c_adapter->cfg->sda.init;
692  sda_gpio_init.GPIO_Mode = GPIO_Mode_Out_OD;
693  GPIO_SetBits(i2c_adapter->cfg->sda.gpio, i2c_adapter->cfg->sda.init.GPIO_Pin);
694  GPIO_Init(i2c_adapter->cfg->sda.gpio, &sda_gpio_init);
695 
696  /* Check SDA line to determine if slave is asserting bus and clock out if so, this may */
697  /* have to be repeated (due to futher bus errors) but better than clocking 0xFF into an */
698  /* ESC */
699  //bool sda_hung = GPIO_ReadInputDataBit(i2c_adapter->cfg->sda.gpio, i2c_adapter->cfg->sda.init.GPIO_Pin) == Bit_RESET;
700  retry_count_clk = 0;
701  while(GPIO_ReadInputDataBit(i2c_adapter->cfg->sda.gpio, i2c_adapter->cfg->sda.init.GPIO_Pin) == Bit_RESET &&
702  (retry_count_clk++ < MAX_I2C_RETRY_COUNT))
703  {
704  retry_count = 0;
705  /* Set clock high and wait for any clock stretching to finish. */
706  GPIO_SetBits(i2c_adapter->cfg->scl.gpio, i2c_adapter->cfg->scl.init.GPIO_Pin);
707  while (GPIO_ReadInputDataBit(i2c_adapter->cfg->scl.gpio, i2c_adapter->cfg->scl.init.GPIO_Pin) == Bit_RESET &&
708  retry_count++ < MAX_I2C_RETRY_COUNT)
709  PIOS_DELAY_WaituS(1);
710  PIOS_DELAY_WaituS(2);
711 
712  /* Set clock low */
713  GPIO_ResetBits(i2c_adapter->cfg->scl.gpio, i2c_adapter->cfg->scl.init.GPIO_Pin);
714  PIOS_DELAY_WaituS(2);
715 
716  /* Clock high again */
717  GPIO_SetBits(i2c_adapter->cfg->scl.gpio, i2c_adapter->cfg->scl.init.GPIO_Pin);
718  PIOS_DELAY_WaituS(2);
719  }
720 
721  /* Generate a start then stop condition */
722  GPIO_SetBits(i2c_adapter->cfg->scl.gpio, i2c_adapter->cfg->scl.init.GPIO_Pin);
723  PIOS_DELAY_WaituS(2);
724  GPIO_ResetBits(i2c_adapter->cfg->sda.gpio, i2c_adapter->cfg->sda.init.GPIO_Pin);
725  PIOS_DELAY_WaituS(2);
726  GPIO_SetBits(i2c_adapter->cfg->sda.gpio, i2c_adapter->cfg->sda.init.GPIO_Pin);
727  PIOS_DELAY_WaituS(2);
728 
729  /* Set data and clock high and wait for any clock stretching to finish. */
730  GPIO_SetBits(i2c_adapter->cfg->sda.gpio, i2c_adapter->cfg->sda.init.GPIO_Pin);
731  GPIO_SetBits(i2c_adapter->cfg->scl.gpio, i2c_adapter->cfg->scl.init.GPIO_Pin);
732 
733  retry_count = 0;
734  while (GPIO_ReadInputDataBit(i2c_adapter->cfg->scl.gpio, i2c_adapter->cfg->scl.init.GPIO_Pin) == Bit_RESET &&
735  retry_count++ < MAX_I2C_RETRY_COUNT)
736  PIOS_DELAY_WaituS(1);
737  /* Wait for data to be high */
738  retry_count = 0;
739  while (GPIO_ReadInputDataBit(i2c_adapter->cfg->sda.gpio, i2c_adapter->cfg->sda.init.GPIO_Pin) != Bit_SET &&
740  retry_count++ < MAX_I2C_RETRY_COUNT)
741  PIOS_DELAY_WaituS(1);
742 
743 
744  /* Bus signals are guaranteed to be high (ie. free) after this point */
745  /* Initialize the GPIO pins to the peripheral function */
746  GPIO_Init(i2c_adapter->cfg->scl.gpio, &(i2c_adapter->cfg->scl.init));
747  GPIO_Init(i2c_adapter->cfg->sda.gpio, &(i2c_adapter->cfg->sda.init));
748 
749  /* Reset the I2C block */
750  I2C_DeInit(i2c_adapter->cfg->regs);
751 
752  /* Initialize the I2C block */
753  I2C_Init(i2c_adapter->cfg->regs, &(i2c_adapter->cfg->init));
754 
755 #define I2C_BUSY 0x20
756  if (i2c_adapter->cfg->regs->SR2 & I2C_BUSY) {
757  /* Reset the I2C block */
758  I2C_SoftwareResetCmd(i2c_adapter->cfg->regs, ENABLE);
759  I2C_SoftwareResetCmd(i2c_adapter->cfg->regs, DISABLE);
760  }
761 }

876 int32_t PIOS_I2C_Init(uint32_t * i2c_id, const struct pios_i2c_adapter_cfg * cfg)
877 {
878  PIOS_DEBUG_Assert(i2c_id);
879  PIOS_DEBUG_Assert(cfg);
880 
881  struct pios_i2c_adapter * i2c_adapter;
882 
883  i2c_adapter = (struct pios_i2c_adapter *) PIOS_I2C_alloc();
884  if (!i2c_adapter) goto out_fail;
885 
886  /* Bind the configuration to the device instance */
887  i2c_adapter->cfg = cfg;
888 
889 #ifdef USE_FREERTOS
890  /*
891  * Must be done prior to calling i2c_adapter_fsm_init()
892  * since the sem_ready mutex is used in the initial state.
893  */
894  vSemaphoreCreateBinary(i2c_adapter->sem_ready);
895  i2c_adapter->sem_busy = xSemaphoreCreateMutex();
896 #endif // USE_FREERTOS
897 
898  /* Enable the associated peripheral clock */
899  switch ((uint32_t) i2c_adapter->cfg->regs) {
900  case (uint32_t) I2C1:
901  /* Enable I2C peripheral clock (APB1 == slow speed) */
902  RCC_APB1PeriphClockCmd(RCC_APB1Periph_I2C1, ENABLE);
903  break;
904  case (uint32_t) I2C2:
905  /* Enable I2C peripheral clock (APB1 == slow speed) */
906  RCC_APB1PeriphClockCmd(RCC_APB1Periph_I2C2, ENABLE);
907  break;
908  }
909 
910  if (i2c_adapter->cfg->remap) {
911  GPIO_PinRemapConfig(i2c_adapter->cfg->remap, ENABLE);
912  }
913 
914  /* Initialize the state machine */
915  i2c_adapter_fsm_init(i2c_adapter);
916 
917  *i2c_id = (uint32_t)i2c_adapter;
918 
919  /* Configure and enable I2C interrupts */
920  NVIC_Init(&(i2c_adapter->cfg->event.init));
921  NVIC_Init(&(i2c_adapter->cfg->error.init));
922 
923  /* No error */
924  return 0;
925 
926 out_fail:
927  return(-1);
928 }

937 {

994  struct pios_i2c_adapter * i2c_adapter = (struct pios_i2c_adapter *)i2c_id;
995 
996  bool valid = PIOS_I2C_validate(i2c_adapter);
997  PIOS_Assert(valid)
998 
999  PIOS_DEBUG_Assert(txn_list);
1000  PIOS_DEBUG_Assert(num_txns);
1001 
1002  bool semaphore_success = true;
1003 
1004 #ifdef USE_FREERTOS
1005  /* Lock the bus */
1006  portTickType timeout;
1007  timeout = i2c_adapter->cfg->transfer_timeout_ms / portTICK_RATE_MS;
1008  if (xSemaphoreTake(i2c_adapter->sem_busy, timeout) == pdFALSE)
1009  return -2;
1010 #else
1011  uint32_t timeout = 0xfff;
1012  while(i2c_adapter->busy == 1 && --timeout);
1013  if(timeout == 0) //timed out
1014  return false;
1015 
1016  PIOS_IRQ_Disable();
1017  if(i2c_adapter->busy == 1) {
1018  PIOS_IRQ_Enable();
1019  return false;
1020  }
1021  i2c_adapter->busy = 1;
1022  PIOS_IRQ_Enable();
1023 #endif /* USE_FREERTOS */
1024 
1025  PIOS_DEBUG_Assert(i2c_adapter->curr_state == I2C_STATE_STOPPED);
1026 
1027  i2c_adapter->first_txn = &txn_list[0];
1028  i2c_adapter->last_txn = &txn_list[num_txns - 1];
1029  i2c_adapter->active_txn = i2c_adapter->first_txn;
1030 
1031 #ifdef USE_FREERTOS
1032  /* Make sure the done/ready semaphore is consumed before we start */
1033  semaphore_success &= (xSemaphoreTake(i2c_adapter->sem_ready, timeout) == pdTRUE);
1034 #endif
1035 
1036  i2c_adapter->bus_error = false;
1037  i2c_adapter_inject_event(i2c_adapter, I2C_EVENT_START);
1038 
1039  /* Wait for the transfer to complete */
1040 #ifdef USE_FREERTOS
1041  semaphore_success &= (xSemaphoreTake(i2c_adapter->sem_ready, timeout) == pdTRUE);
1042  xSemaphoreGive(i2c_adapter->sem_ready);
1043 #else
1044  i2c_adapter->busy = 0;
1045 #endif /* USE_FREERTOS */
1046 
1047  /* Spin waiting for the transfer to finish */
1048  while (!i2c_adapter_fsm_terminated(i2c_adapter)) ;
1049 
1050  if (i2c_adapter_wait_for_stopped(i2c_adapter)) {
1051  i2c_adapter_inject_event(i2c_adapter, I2C_EVENT_STOPPED);
1052  } else {
1053  i2c_adapter_fsm_init(i2c_adapter);
1054  }
1055 
1056 #ifdef USE_FREERTOS
1057  /* Unlock the bus */
1058  xSemaphoreGive(i2c_adapter->sem_busy);
1059  if(!semaphore_success)
1060  i2c_timeout_counter++;
1061 #endif /* USE_FREERTOS */
1062 
1063  return !semaphore_success ? -2 :
1064  i2c_adapter->bus_error ? -1 :
1065  0;
1066 }

1069 void PIOS_I2C_EV_IRQ_Handler(uint32_t i2c_id)
1070 {
1071  struct pios_i2c_adapter * i2c_adapter = (struct pios_i2c_adapter *)i2c_id;
1072 
1073  bool valid = PIOS_I2C_validate(i2c_adapter);
1074  PIOS_Assert(valid)
1075 
1076  uint32_t event = I2C_GetLastEvent(i2c_adapter->cfg->regs);
1077 
1078 #if defined(PIOS_I2C_DIAGNOSTICS)
1079  /* Store event for diagnostics */
1080  i2c_evirq_history[i2c_evirq_history_pointer] = event;
1081  i2c_evirq_history_pointer = (i2c_evirq_history_pointer + 1) % I2C_LOG_DEPTH;
1082 #endif
1083 
1084 #define EVENT_MASK 0x000700FF
1085  event &= EVENT_MASK;
1086 
1087 
1088  switch (event) { /* Mask out all the bits we don't care about */
1089  case (I2C_EVENT_MASTER_MODE_SELECT | 0x40):
1090  /* Unexplained event: EV5 + RxNE : Extraneous Rx. Probably a late NACK from previous read. */
1091  /* Clean up the extra Rx until the root cause is identified and just keep going */
1092  (void)I2C_ReceiveData(i2c_adapter->cfg->regs);
1093  /* Fall through */
1094  case I2C_EVENT_MASTER_MODE_SELECT: /* EV5 */
1095  switch (i2c_adapter->active_txn->rw) {
1096  case PIOS_I2C_TXN_READ:
1097  if (i2c_adapter->active_txn == i2c_adapter->last_txn) {
1098  /* Final transaction */
1099  i2c_adapter_inject_event(i2c_adapter, I2C_EVENT_STARTED_LAST_TXN_READ);
1100  } else if (i2c_adapter->active_txn < i2c_adapter->last_txn) {
1101  /* More transactions follow */
1102  i2c_adapter_inject_event(i2c_adapter, I2C_EVENT_STARTED_MORE_TXN_READ);
1103  } else {
1104  PIOS_DEBUG_Assert(0);
1105  }
1106  break;
1107  case PIOS_I2C_TXN_WRITE:
1108  if (i2c_adapter->active_txn == i2c_adapter->last_txn) {
1109  /* Final transaction */
1110  i2c_adapter_inject_event(i2c_adapter, I2C_EVENT_STARTED_LAST_TXN_WRITE);
1111  } else if (i2c_adapter->active_txn < i2c_adapter->last_txn) {
1112  /* More transactions follow */
1113  i2c_adapter_inject_event(i2c_adapter, I2C_EVENT_STARTED_MORE_TXN_WRITE);
1114  } else {
1115  PIOS_DEBUG_Assert(0);
1116  }
1117  break;
1118  default:
1119  PIOS_DEBUG_Assert(0);
1120  break;
1121  }
1122  break;
1123  case I2C_EVENT_MASTER_TRANSMITTER_MODE_SELECTED: /* EV6 */
1124  case I2C_EVENT_MASTER_RECEIVER_MODE_SELECTED: /* EV6 */
1125  switch (i2c_adapter->last_byte - i2c_adapter->active_byte + 1) {
1126  case 0:
1127  i2c_adapter_inject_event(i2c_adapter, I2C_EVENT_ADDR_SENT_LEN_EQ_0);
1128  break;
1129  case 1:
1130  i2c_adapter_inject_event(i2c_adapter, I2C_EVENT_ADDR_SENT_LEN_EQ_1);
1131  break;
1132  case 2:
1133  i2c_adapter_inject_event(i2c_adapter, I2C_EVENT_ADDR_SENT_LEN_EQ_2);
1134  break;
1135  default:
1136  i2c_adapter_inject_event(i2c_adapter, I2C_EVENT_ADDR_SENT_LEN_GT_2);
1137  break;
1138  }
1139  break;
1140  case 0x80: /* TxE only. TRA + MSL + BUSY have been cleared before we got here. */
1141  /* Ignore */
1142  {
1143  static volatile bool halt = FALSE;
1144  while (halt) ;
1145  }
1146  break;
1147  case 0: /* This triggers an FSM fault sometimes, but not having it stops things working */
1148  case 0x40: /* RxNE only. MSL + BUSY have already been cleared by HW. */
1149  case 0x44: /* RxNE + BTF. MSL + BUSY have already been cleared by HW. */
1150  case I2C_EVENT_MASTER_BYTE_RECEIVED: /* EV7 */
1151  case (I2C_EVENT_MASTER_BYTE_RECEIVED | 0x4): /* EV7 + BTF */
1152  case I2C_EVENT_MASTER_BYTE_TRANSMITTED: /* EV8_2 */
1153  case 0x84: /* TxE + BTF. EV8_2 but TRA + MSL + BUSY have already been cleared by HW. */
1154  switch (i2c_adapter->last_byte - i2c_adapter->active_byte + 1) {
1155  case 0:
1156  i2c_adapter_inject_event(i2c_adapter, I2C_EVENT_TRANSFER_DONE_LEN_EQ_0);
1157  break;
1158  case 1:
1159  i2c_adapter_inject_event(i2c_adapter, I2C_EVENT_TRANSFER_DONE_LEN_EQ_1);
1160  break;
1161  case 2:
1162  i2c_adapter_inject_event(i2c_adapter, I2C_EVENT_TRANSFER_DONE_LEN_EQ_2);
1163  break;
1164  default:
1165  i2c_adapter_inject_event(i2c_adapter, I2C_EVENT_TRANSFER_DONE_LEN_GT_2);
1166  break;
1167  }
1168  break;
1169  case I2C_EVENT_MASTER_BYTE_TRANSMITTING: /* EV8 */
1170  /* Ignore this event and wait for TRANSMITTED in case we can't keep up */
1171  goto skip_event;
1172  break;
1173  case 0x30084: /* Occurs between byte tranmistted and master mode selected */
1174  case 0x30000: /* Need to throw away this spurious event */
1175  case 0x30403 & EVENT_MASK: /* Detected this after got a NACK, probably stop bit */
1176  goto skip_event;
1177  break;
1178  default:
1179  i2c_adapter_log_fault(PIOS_I2C_ERROR_EVENT);
1180 #if defined(I2C_HALT_ON_ERRORS)
1181  PIOS_DEBUG_Assert(0);
1182 #endif
1183  i2c_adapter_inject_event(i2c_adapter, I2C_EVENT_BUS_ERROR);
1184  break;
1185  }
1186 
1187 skip_event:
1188  ;
1189 }
