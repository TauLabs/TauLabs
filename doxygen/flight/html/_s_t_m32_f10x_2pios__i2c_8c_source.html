00404 static void go_starting(struct pios_i2c_adapter *i2c_adapter)
00405 {
00406         PIOS_DEBUG_Assert(i2c_adapter->active_txn);
00407         PIOS_DEBUG_Assert(i2c_adapter->active_txn >= i2c_adapter->first_txn);
00408         PIOS_DEBUG_Assert(i2c_adapter->active_txn <= i2c_adapter->last_txn);
00409 
00410         i2c_adapter->active_byte = &(i2c_adapter->active_txn->buf[0]);
00411         i2c_adapter->last_byte = &(i2c_adapter->active_txn->buf[i2c_adapter->active_txn->len - 1]);
00412 
00413         I2C_GenerateSTART(i2c_adapter->cfg->regs, ENABLE);
00414         if (i2c_adapter->active_txn->rw == PIOS_I2C_TXN_READ) {
00415                 I2C_ITConfig(i2c_adapter->cfg->regs, I2C_IT_EVT | I2C_IT_BUF | I2C_IT_ERR, ENABLE);
00416         } else {
00417                 // For write operations, do not enable the IT_BUF events.
00418                 // The current driver does not act when the TX data register is not full, only when the complete byte is sent.
00419                 // With the IT_BUF enabled, we constantly get IRQs, See OP-326
00420                 I2C_ITConfig(i2c_adapter->cfg->regs, I2C_IT_EVT | I2C_IT_ERR, ENABLE);
00421         }
00422 }

00702                 (retry_count_clk++ < MAX_I2C_RETRY_COUNT))

00704                 retry_count = 0;

00706                 GPIO_SetBits(i2c_adapter->cfg->scl.gpio, i2c_adapter->cfg->scl.init.GPIO_Pin);
00707                 while (GPIO_ReadInputDataBit(i2c_adapter->cfg->scl.gpio, i2c_adapter->cfg->scl.init.GPIO_Pin) == Bit_RESET && 
00708                         retry_count++ < MAX_I2C_RETRY_COUNT)

00714                 PIOS_DELAY_WaituS(2);
00715

00717                 GPIO_SetBits(i2c_adapter->cfg->scl.gpio, i2c_adapter->cfg->scl.init.GPIO_Pin);

00725         PIOS_DELAY_WaituS(2);

00727         PIOS_DELAY_WaituS(2);
00728 
00729         /* Set data and clock high and wait for any clock stretching to finish. */
00730         GPIO_SetBits(i2c_adapter->cfg->sda.gpio, i2c_adapter->cfg->sda.init.GPIO_Pin);
00731         GPIO_SetBits(i2c_adapter->cfg->scl.gpio, i2c_adapter->cfg->scl.init.GPIO_Pin);
00732 
00733         retry_count = 0;

00735             retry_count++ < MAX_I2C_RETRY_COUNT)
00736                 PIOS_DELAY_WaituS(1);
00737         /* Wait for data to be high */
00738         retry_count = 0;
00739         while (GPIO_ReadInputDataBit(i2c_adapter->cfg->sda.gpio, i2c_adapter->cfg->sda.init.GPIO_Pin) != Bit_SET &&

00749         /* Reset the I2C block */

00751

01022         PIOS_IRQ_Enable();

01024

01026

01033         semaphore_success &= (xSemaphoreTake(i2c_adapter->sem_ready, timeout) == pdTRUE);

01041         semaphore_success &= (xSemaphoreTake(i2c_adapter->sem_ready, timeout) == pdTRUE);
01042         xSemaphoreGive(i2c_adapter->sem_ready);

01050         if (i2c_adapter_wait_for_stopped(i2c_adapter)) {
01051                 i2c_adapter_inject_event(i2c_adapter, I2C_EVENT_STOPPED);
01052         } else {
01053                 i2c_adapter_fsm_init(i2c_adapter);
01054         }
01055 
01056 #ifdef USE_FREERTOS
01057         /* Unlock the bus */
01058         xSemaphoreGive(i2c_adapter->sem_busy);
01059         if(!semaphore_success)
01060                 i2c_timeout_counter++;
01061 #endif /* USE_FREERTOS */
01062 
01063         return !semaphore_success ? -2 :
01064                 i2c_adapter->bus_error ? -1 :
01065                 0;

01072

01077

01081         i2c_evirq_history_pointer = (i2c_evirq_history_pointer + 1) % I2C_LOG_DEPTH;
01082 #endif
01083         
01084 #define EVENT_MASK 0x000700FF
01085         event &= EVENT_MASK;
01086         
01087         
01088         switch (event) { /* Mask out all the bits we don't care about */
01089         case (I2C_EVENT_MASTER_MODE_SELECT | 0x40):
01090                 /* Unexplained event: EV5 + RxNE : Extraneous Rx.  Probably a late NACK from previous read. */
01091                 /* Clean up the extra Rx until the root cause is identified and just keep going */
01092                 (void)I2C_ReceiveData(i2c_adapter->cfg->regs);
01093                 /* Fall through */
01094         case I2C_EVENT_MASTER_MODE_SELECT:      /* EV5 */
01095                 switch (i2c_adapter->active_txn->rw) {
01096                 case PIOS_I2C_TXN_READ:
01097                         if (i2c_adapter->active_txn == i2c_adapter->last_txn) {
01098                                 /* Final transaction */
01099                                 i2c_adapter_inject_event(i2c_adapter, I2C_EVENT_STARTED_LAST_TXN_READ);
01100                         } else if (i2c_adapter->active_txn < i2c_adapter->last_txn) {
01101                                 /* More transactions follow */
01102                                 i2c_adapter_inject_event(i2c_adapter, I2C_EVENT_STARTED_MORE_TXN_READ);
01103                         } else {
01104                                 PIOS_DEBUG_Assert(0);
01105                         }
01106                         break;
01107                 case PIOS_I2C_TXN_WRITE:
01108                         if (i2c_adapter->active_txn == i2c_adapter->last_txn) {
01109                                 /* Final transaction */
01110                                 i2c_adapter_inject_event(i2c_adapter, I2C_EVENT_STARTED_LAST_TXN_WRITE);
01111                         } else if (i2c_adapter->active_txn < i2c_adapter->last_txn) {
01112                                 /* More transactions follow */
01113                                 i2c_adapter_inject_event(i2c_adapter, I2C_EVENT_STARTED_MORE_TXN_WRITE);
01114                         } else {
01115                                 PIOS_DEBUG_Assert(0);
01116                         }
01117                         break;
01118                 default:
01119                         PIOS_DEBUG_Assert(0);
01120                         break;
01121                 }
01122                 break;
01123         case I2C_EVENT_MASTER_TRANSMITTER_MODE_SELECTED:        /* EV6 */
01124         case I2C_EVENT_MASTER_RECEIVER_MODE_SELECTED:   /* EV6 */
01125                 switch (i2c_adapter->last_byte - i2c_adapter->active_byte + 1) {
01126                 case 0:
01127                         i2c_adapter_inject_event(i2c_adapter, I2C_EVENT_ADDR_SENT_LEN_EQ_0);

01130                         i2c_adapter_inject_event(i2c_adapter, I2C_EVENT_ADDR_SENT_LEN_EQ_1);

01137                         break;
01138                 }
01139                 break;
01140         case 0x80:              /* TxE only.  TRA + MSL + BUSY have been cleared before we got here. */

01145                 }
01146                 break;

01149         case 0x44:              /* RxNE + BTF.  MSL + BUSY have already been cleared by HW. */
01150         case I2C_EVENT_MASTER_BYTE_RECEIVED:    /* EV7 */

01152         case I2C_EVENT_MASTER_BYTE_TRANSMITTED: /* EV8_2 */
01153         case 0x84:              /* TxE + BTF. EV8_2 but TRA + MSL + BUSY have already been cleared by HW. */

01155                 case 0:
