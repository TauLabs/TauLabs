00413 static void go_starting(struct pios_i2c_adapter *i2c_adapter)
00414 {
00415         PIOS_DEBUG_Assert(i2c_adapter->active_txn);
00416         PIOS_DEBUG_Assert(i2c_adapter->active_txn >= i2c_adapter->first_txn);
00417         PIOS_DEBUG_Assert(i2c_adapter->active_txn <= i2c_adapter->last_txn);
00418 
00419         i2c_adapter->active_byte = &(i2c_adapter->active_txn->buf[0]);
00420         i2c_adapter->last_byte = &(i2c_adapter->active_txn->buf[i2c_adapter->active_txn->len - 1]);
00421 
00422         I2C_GenerateSTART(i2c_adapter->cfg->regs, ENABLE);
00423         if (i2c_adapter->active_txn->rw == PIOS_I2C_TXN_READ) {
00424                 I2C_ITConfig(i2c_adapter->cfg->regs, I2C_IT_EVT | I2C_IT_BUF | I2C_IT_ERR, ENABLE);
00425         } else {
00426                 // For write operations, do not enable the IT_BUF events.
00427                 // The current driver does not act when the TX data register is not full, only when the complete byte is sent.
00428                 // With the IT_BUF enabled, we constantly get IRQs, See OP-326
00429                 I2C_ITConfig(i2c_adapter->cfg->regs, I2C_IT_EVT | I2C_IT_ERR, ENABLE);
00430         }
00431 }

00601 static void i2c_adapter_inject_event(struct pios_i2c_adapter *i2c_adapter, enum i2c_adapter_event event)
00602 {
00603         PIOS_IRQ_Disable();
00604 
00605 #if defined(PIOS_I2C_DIAGNOSTICS)       
00606         i2c_state_event_history[i2c_state_event_history_pointer] = event;
00607         i2c_state_event_history_pointer = (i2c_state_event_history_pointer + 1) % I2C_LOG_DEPTH;
00608 
00609         i2c_state_history[i2c_state_history_pointer] = i2c_adapter->curr_state;
00610         i2c_state_history_pointer = (i2c_state_history_pointer + 1) % I2C_LOG_DEPTH;

00627         /* Call the entry function (if any) for the next state. */
00628         if (i2c_adapter_transitions[i2c_adapter->curr_state].entry_fn) {
00629                 i2c_adapter_transitions[i2c_adapter->curr_state].entry_fn(i2c_adapter);
00630         }
00631 
00632         /* Process any AUTO transitions in the FSM */
00633         i2c_adapter_process_auto(i2c_adapter);

00684 static void i2c_adapter_reset_bus(struct pios_i2c_adapter *i2c_adapter)
00685 {
00686         static uint8_t retry_count = 0;
00687         static uint8_t retry_count_clk = 0;
00688         /* Reset the I2C block */
00689         I2C_DeInit(i2c_adapter->cfg->regs);
00690 
00691         /* Make sure the bus is free by clocking it until any slaves release the bus. */
00692         GPIO_InitTypeDef scl_gpio_init;
00693         scl_gpio_init = i2c_adapter->cfg->scl.init;
00694         scl_gpio_init.GPIO_Mode  = GPIO_Mode_OUT;
00695         GPIO_SetBits(i2c_adapter->cfg->scl.gpio, i2c_adapter->cfg->scl.init.GPIO_Pin);
00696         GPIO_Init(i2c_adapter->cfg->scl.gpio, &scl_gpio_init);
00697 
00698         GPIO_InitTypeDef sda_gpio_init;
00699         sda_gpio_init = i2c_adapter->cfg->sda.init;
00700         sda_gpio_init.GPIO_Mode  = GPIO_Mode_OUT;
00701         GPIO_SetBits(i2c_adapter->cfg->sda.gpio, i2c_adapter->cfg->sda.init.GPIO_Pin);
00702         GPIO_Init(i2c_adapter->cfg->sda.gpio, &sda_gpio_init);
00703 
00704         /* Check SDA line to determine if slave is asserting bus and clock out if so, this may  */
00705         /* have to be repeated (due to futher bus errors) but better than clocking 0xFF into an */
00706         /* ESC */
00707         //bool sda_hung = GPIO_ReadInputDataBit(i2c_adapter->cfg->sda.gpio, i2c_adapter->cfg->sda.init.GPIO_Pin) == Bit_RESET;
00708         retry_count_clk = 0;
00709         while(GPIO_ReadInputDataBit(i2c_adapter->cfg->sda.gpio, i2c_adapter->cfg->sda.init.GPIO_Pin) == Bit_RESET && 
00710                 (retry_count_clk++ < MAX_I2C_RETRY_COUNT)) 
00711         {
00712                 retry_count = 0;
00713                 /* Set clock high and wait for any clock stretching to finish. */
00714                 GPIO_SetBits(i2c_adapter->cfg->scl.gpio, i2c_adapter->cfg->scl.init.GPIO_Pin);
00715                 while (GPIO_ReadInputDataBit(i2c_adapter->cfg->scl.gpio, i2c_adapter->cfg->scl.init.GPIO_Pin) == Bit_RESET && 
00716                         retry_count++ < MAX_I2C_RETRY_COUNT)
00717                     PIOS_DELAY_WaituS(1);
00718                 PIOS_DELAY_WaituS(2);
00719                 
00720                 /* Set clock low */
00721                 GPIO_ResetBits(i2c_adapter->cfg->scl.gpio, i2c_adapter->cfg->scl.init.GPIO_Pin);
00722                 PIOS_DELAY_WaituS(2);
00723                 
00724                 /* Clock high again */
00725                 GPIO_SetBits(i2c_adapter->cfg->scl.gpio, i2c_adapter->cfg->scl.init.GPIO_Pin);
00726                 PIOS_DELAY_WaituS(2);
00727         }
00728                 
00729         /* Generate a start then stop condition */
00730         GPIO_SetBits(i2c_adapter->cfg->scl.gpio, i2c_adapter->cfg->scl.init.GPIO_Pin);
00731         PIOS_DELAY_WaituS(2);
00732         GPIO_ResetBits(i2c_adapter->cfg->sda.gpio, i2c_adapter->cfg->sda.init.GPIO_Pin);
00733         PIOS_DELAY_WaituS(2);
00734         GPIO_SetBits(i2c_adapter->cfg->sda.gpio, i2c_adapter->cfg->sda.init.GPIO_Pin);
00735         PIOS_DELAY_WaituS(2);
00736 
00737         /* Set data and clock high and wait for any clock stretching to finish. */
00738         GPIO_SetBits(i2c_adapter->cfg->sda.gpio, i2c_adapter->cfg->sda.init.GPIO_Pin);
00739         GPIO_SetBits(i2c_adapter->cfg->scl.gpio, i2c_adapter->cfg->scl.init.GPIO_Pin);
00740         
00741         retry_count = 0;
00742         while (GPIO_ReadInputDataBit(i2c_adapter->cfg->scl.gpio, i2c_adapter->cfg->scl.init.GPIO_Pin) == Bit_RESET && 
00743             retry_count++ < MAX_I2C_RETRY_COUNT)
00744                 PIOS_DELAY_WaituS(1);
00745 
00746         /* Wait for data to be high */
00747         retry_count = 0;
00748         while (GPIO_ReadInputDataBit(i2c_adapter->cfg->sda.gpio, i2c_adapter->cfg->sda.init.GPIO_Pin) != Bit_SET && 
00749             retry_count++ < MAX_I2C_RETRY_COUNT)
00750                 PIOS_DELAY_WaituS(1);
00751 
00752         
00753         /* Bus signals are guaranteed to be high (ie. free) after this point */
00754         /* Initialize the GPIO pins to the peripheral function */

00756                 GPIO_PinAFConfig(i2c_adapter->cfg->scl.gpio,
00757                                 __builtin_ctz(i2c_adapter->cfg->scl.init.GPIO_Pin),
00758                                 i2c_adapter->cfg->remap);
00759                 GPIO_PinAFConfig(i2c_adapter->cfg->sda.gpio,
00760                                 __builtin_ctz(i2c_adapter->cfg->sda.init.GPIO_Pin),
00761                                 i2c_adapter->cfg->remap);
00762         }
00763         GPIO_Init(i2c_adapter->cfg->scl.gpio, (GPIO_InitTypeDef*)&(i2c_adapter->cfg->scl.init)); // Struct is const, function signature not
00764         GPIO_Init(i2c_adapter->cfg->sda.gpio, (GPIO_InitTypeDef*)&(i2c_adapter->cfg->sda.init));
00765 
00766         /* Reset the I2C block */

00768 
00769         /* Initialize the I2C block */
00770         I2C_Init(i2c_adapter->cfg->regs, (I2C_InitTypeDef*)&(i2c_adapter->cfg->init));
00771 
00772         if (i2c_adapter->cfg->regs->SR2 & I2C_FLAG_BUSY) {

00794 static bool i2c_adapter_callback_handler(struct pios_i2c_adapter * i2c_adapter) 
00795 {

00798 #ifdef USE_FREERTOS

00812         }

00814         // Execute user supplied function
00815         i2c_adapter->callback();
00816         
00817         i2c_cb_count++;
00818 
00819 #ifdef USE_FREERTOS
00820         /* Unlock the bus */
00821         xSemaphoreGive(i2c_adapter->sem_busy);
00822         if(!semaphore_success)
00823                 i2c_timeout_counter++;
00824 #else
00825         i2c_adapter->busy = 0;

01032                 return -1;

01034         bool semaphore_success = true;
01035

01037         /* Lock the bus */

01070         semaphore_success &= (xSemaphoreTake(i2c_adapter->sem_ready, timeout) == pdTRUE);
01071         xSemaphoreGive(i2c_adapter->sem_ready);
01072 #endif /* USE_FREERTOS */

01074         /* Spin waiting for the transfer to finish */

01088 #else

01143 {

01145 
01146         bool valid = PIOS_I2C_validate(i2c_adapter);
01147         PIOS_Assert(valid)
01148 
01149         uint32_t event = I2C_GetLastEvent(i2c_adapter->cfg->regs);
01150 
01151 #if defined(PIOS_I2C_DIAGNOSTICS)       
01152         /* Store event for diagnostics */
01153         i2c_evirq_history[i2c_evirq_history_pointer] = event;
01154         i2c_evirq_history_pointer = (i2c_evirq_history_pointer + 1) % I2C_LOG_DEPTH;
01155 #endif
01156         
01157 #define EVENT_MASK 0x000700FF
01158         event &= EVENT_MASK;
01159         
01160         // This is very poor and inconsistent practice with the FSM since no other 
01161         // throw event depends on the current state.  However when accelerated (-Os)
01162         // we definitely catch this event twice and there is no clean way to do deal
01163         // with that in the FMS short of a special state for it
01164         if(i2c_adapter->curr_state == I2C_STATE_STARTING && event == 0x70084)
01165                 return;

01168         case (I2C_EVENT_MASTER_MODE_SELECT | 0x40):

01171                 (void)I2C_ReceiveData(i2c_adapter->cfg->regs);
01172                 /* Fall through */
01173         case I2C_EVENT_MASTER_MODE_SELECT:      /* EV5 */
01174                 switch (i2c_adapter->active_txn->rw) {
01175                 case PIOS_I2C_TXN_READ:
01176                         if (i2c_adapter->active_txn == i2c_adapter->last_txn) {
01177                                 /* Final transaction */
01178                                 i2c_adapter_inject_event(i2c_adapter, I2C_EVENT_STARTED_LAST_TXN_READ);
01179                         } else if (i2c_adapter->active_txn < i2c_adapter->last_txn) {
01180                                 /* More transactions follow */

01184                         }

01199                         break;
01200                 }
