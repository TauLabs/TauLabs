106 static bool PIOS_Flash_Internal_GetSectorInfo(uint32_t address, uint8_t * sector_number, uint32_t *sector_start, uint32_t *sector_size)
107 {
108  for (uint8_t i = 0; i < NELEMENTS(flash_sectors); i++) {
109  struct device_flash_sector * sector = &flash_sectors[i];
110  if ((address >= sector->start) &&
111  (address < (sector->start + sector->size))) {
112  /* address lies within this sector */
113  *sector_number = sector->st_sector;
114  *sector_start = sector->start;
115  *sector_size = sector->size;
116  return (true);
117  }
118  }
119 
120  return (false);
121 }

171 {

175  if (flash_dev == NULL)

194 static int32_t PIOS_Flash_Internal_StartTransaction(uintptr_t flash_id)
195 {
196  struct pios_internal_flash_dev * flash_dev = (struct pios_internal_flash_dev *)flash_id;
197 
198  if (!PIOS_Flash_Internal_Validate(flash_dev))
199  return -1;
200 
201 #if defined(PIOS_INCLUDE_FREERTOS)
202  if (xSemaphoreTake(flash_dev->transaction_lock, portMAX_DELAY) != pdTRUE)
203  return -2;
204 #endif /* defined(PIOS_INCLUDE_FREERTOS) */
205 
206  /* Unlock the internal flash so we can write to it */
207  FLASH_Unlock();
208  return 0;
209 }

211 static int32_t PIOS_Flash_Internal_EndTransaction(uintptr_t flash_id)
212 {
213  struct pios_internal_flash_dev * flash_dev = (struct pios_internal_flash_dev *)flash_id;
214 
215  if (!PIOS_Flash_Internal_Validate(flash_dev))
216  return -1;
217 
218 #if defined(PIOS_INCLUDE_FREERTOS)
219  if (xSemaphoreGive(flash_dev->transaction_lock) != pdTRUE)
220  return -2;
221 #endif /* defined(PIOS_INCLUDE_FREERTOS) */
222 
223  /* Lock the internal flash again so we can no longer write to it */
224  FLASH_Lock();
225 
226  return 0;
227 }

257 static int32_t PIOS_Flash_Internal_WriteData(uintptr_t flash_id, uint32_t addr, uint8_t * data, uint16_t len)
258 {
259  PIOS_Assert(data);
260 
261  struct pios_internal_flash_dev * flash_dev = (struct pios_internal_flash_dev *)flash_id;
262 
263  if (!PIOS_Flash_Internal_Validate(flash_dev))
264  return -1;
265 
266  uint8_t sector_number;
267  uint32_t sector_start;
268  uint32_t sector_size;
269 
270  /* Ensure that the base address is in a valid sector */
271  if (!PIOS_Flash_Internal_GetSectorInfo(addr,
272  &sector_number,
273  &sector_start,
274  &sector_size)) {
275  /* We're asking for an invalid flash address */
276  return -2;
277  }
278 
279  /* Ensure that the entire write occurs within the same sector */
280  if ((uintptr_t)addr + len > sector_start + sector_size) {
281  /* Write crosses the end of the sector */
282  return -3;
283  }
284 
285  /* Write the data */
286  for (uint16_t i = 0; i < len; i++) {
287  FLASH_Status status;
288  /*
289  * This is inefficient. Should try to do word writes.
290  * Not sure if word writes need to be aligned though.
291  */
292  status = FLASH_ProgramByte(addr + i, data[i]);
293  PIOS_Assert(status == FLASH_COMPLETE);
294  }
295 
296  return 0;
297 }
