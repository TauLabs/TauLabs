00055 int32_t failsafe_control_select(bool reset_controller)
00056 {
00057         uint8_t flight_status;
00058         FlightStatusFlightModeGet(&flight_status);
00059         if (flight_status != FLIGHTSTATUS_FLIGHTMODE_STABILIZED1 || reset_controller) {
00060                 flight_status = FLIGHTSTATUS_FLIGHTMODE_STABILIZED1;
00061                 FlightStatusFlightModeSet(&flight_status);
00062         }
00063 
00064         // Pick default values that will roughly cause a plane to circle down
00065         // and a quad to fall straight down
00066         StabilizationDesiredData stabilization_desired;
00067         StabilizationDesiredGet(&stabilization_desired);
00068         stabilization_desired.Throttle = -1;
00069         stabilization_desired.Roll = -10;
00070         stabilization_desired.Pitch = 0;
00071         stabilization_desired.Yaw = -5;
00072         stabilization_desired.StabilizationMode[STABILIZATIONDESIRED_STABILIZATIONMODE_ROLL] = STABILIZATIONDESIRED_STABILIZATIONMODE_ATTITUDE;
00073         stabilization_desired.StabilizationMode[STABILIZATIONDESIRED_STABILIZATIONMODE_PITCH] = STABILIZATIONDESIRED_STABILIZATIONMODE_ATTITUDE;
00074         stabilization_desired.StabilizationMode[STABILIZATIONDESIRED_STABILIZATIONMODE_YAW] = STABILIZATIONDESIRED_STABILIZATIONMODE_AXISLOCK;
00075         StabilizationDesiredSet(&stabilization_desired);
00076 
00077         return 0;
00078 }
