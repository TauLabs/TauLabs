00452 static bool stream_trigger(enum MAV_DATA_STREAM stream_num) {
00453         uint8_t rate = (uint8_t) mav_rates[stream_num];
00454 
00455         if (rate == 0) {
00456                 return false;
00457         }
00458 
00459         if (stream_ticks[stream_num] == 0) {
00460                 // we're triggering now, setup the next trigger point
00461                 if (rate > TASK_RATE_HZ) {
00462                         rate = TASK_RATE_HZ;
00463                 }
00464                 stream_ticks[stream_num] = (TASK_RATE_HZ / rate);
00465                 return true;
00466         }
00467 
00468         // count down at 50Hz
00469         stream_ticks[stream_num]--;
00470         return false;
