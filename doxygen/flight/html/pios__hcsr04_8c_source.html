00158 void TIM3_IRQHandler(void)
00159 {
00160         /* Zero value always will be changed but this prevents compiler warning */
00161         int32_t i = 0;
00162 
00163         /* Do this as it's more efficient */
00164         if (TIM_GetITStatus(TIM3, TIM_IT_CC2) == SET) {
00165                 i = 7;
00166                 if (CaptureState == 0) {
00167                         RiseValue = TIM_GetCapture2(TIM3);
00168                 } else {
00169                         FallValue = TIM_GetCapture2(TIM3);
00170                 }
00171         }
00172 
00173         /* Clear TIM3 Capture compare interrupt pending bit */
00174         TIM_ClearITPendingBit(TIM3, TIM_IT_CC2);
00175 
00176         /* Simple rise or fall state machine */
00177         if (CaptureState == 0) {
00178                 /* Switch states */
00179                 CaptureState = 1;
00180 
00181                 /* Switch polarity of input capture */
00182                 TIM_ICInitStructure.TIM_ICPolarity = TIM_ICPolarity_Falling;
00183                 TIM_ICInitStructure.TIM_Channel = TIM_Channel_2;
00184                 TIM_ICInit(TIM3, &TIM_ICInitStructure);
00185 
00186         } else {
00187                 /* Capture computation */
00188                 if (FallValue > RiseValue) {
00189                         CaptureValue = (FallValue - RiseValue);
00190                 } else {
00191                         CaptureValue = ((0xFFFF - RiseValue) + FallValue);
00192                 }
00193 
00194                 /* Switch states */
00195                 CaptureState = 0;
00196 
00197                 /* Increase supervisor counter */
00198                 CapCounter++;
00199                 TIM_ITConfig(TIM3, TIM_IT_CC2, DISABLE);
00200 
00201                 /* Switch polarity of input capture */
00202                 TIM_ICInitStructure.TIM_ICPolarity = TIM_ICPolarity_Rising;
00203                 TIM_ICInitStructure.TIM_Channel = TIM_Channel_2;
00204                 TIM_ICInit(TIM3, &TIM_ICInitStructure);
