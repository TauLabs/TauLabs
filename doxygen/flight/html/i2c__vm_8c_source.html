00750 /* Run virtual machine. This is the code that loops through and interprets all the instructions
00751  *
00752  * @param[in] code pointer to program to execute
00753  * @param[in] code_len number of 32-bit instructions contained in the program
00754  * @param[in] i2c_adapter opaque I2C adapter handle to use for i2c transactions
00755  *
00756  */
00757 bool i2c_vm_run (const uint32_t * code, uint8_t code_len, uintptr_t i2c_adapter)
00758 {
00759         if (code == NULL || code_len == 0)
00760                 return false;
00761 
00762         static struct i2c_vm_regs vm;
00763 
00764         i2c_vm_reboot (&vm, i2c_adapter);
00765 
00766         while (!vm.halted) {
00767                 if (vm.uavo.pc > code_len) {
00768                         /* PC is entirely out of range */
00769                         vm.fault  = true;
00770                         vm.halted = true;
00771                         continue;
00772                 }
00773                 if (vm.uavo.pc == code_len) {
00774                         /* PC is just past the end of the code, assume program is completed */
00775                         vm.halted = true;
00776                         continue;
00777                 }
00778                 /* Fetch */
00779                 uint32_t instruction = code[vm.uavo.pc];
00780 
00781                 /* Decode */
00782                 uint8_t operator = (instruction & 0xFF000000) >> 24;
00783                 uint8_t op1      = (instruction & 0x00FF0000) >> 16;
00784                 uint8_t op2      = (instruction & 0x0000FF00) >>  8;
00785                 uint8_t op3      = (instruction & 0x000000FF);
00786 
00787                 if (operator > NELEMENTS(i2c_vm_handlers)) {
00788                         vm.fault = true;
00789                         vm.halted = true;
00790                         continue;
00791                 }
00792                 i2c_vm_inst_handler f = i2c_vm_handlers[operator];
00793 
00794                 /* Execute + Writeback */
00795                 if (!f || !f(&vm, op1, op2, op3)) {
00796                         vm.fault  = true;
00797                         vm.halted = true;
00798                         continue;
00799                 }
00800         }
00801 
00802         return (!vm.fault);
00803 }
