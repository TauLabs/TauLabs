00212 static int32_t packData(uint8_t * data, int32_t length)
00213 {
00214         // Get the lock for manipulating the buffer
00215         xSemaphoreTake(overosync->buffer_lock, portMAX_DELAY);
00216 
00217         portTickType tickTime = xTaskGetTickCount();
00218         uint64_t packetSize = data[2] + (data[3] << 8);
00219         fwrite((void *) &tickTime, 1, sizeof(tickTime), fid);
00220         fwrite((void *) &packetSize, sizeof(packetSize), 1, fid);
00221         fwrite((void *) data, 1, length, fid);
00222         overosync->sent_bytes += length;
00223         overosync->sent_objects++;
00224 
00225         xSemaphoreGive(overosync->buffer_lock);
00226
