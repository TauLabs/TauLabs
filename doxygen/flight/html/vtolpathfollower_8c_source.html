00252 static void updatePathVelocity()
00253 {
00254         float dT = guidanceSettings.UpdatePeriod / 1000.0f;
00255 
00256         PositionActualData positionActual;
00257         PositionActualGet(&positionActual);
00258         
00259         float cur[3] = {positionActual.North, positionActual.East, positionActual.Down};
00260         struct path_status progress;
00261         
00262         path_progress(&pathDesired, cur, &progress);
00263         
00264         // Update the path status UAVO
00265         PathStatusData pathStatus;
00266         PathStatusGet(&pathStatus);
00267         pathStatus.fractional_progress = progress.fractional_progress;
00268         if (pathStatus.fractional_progress < 1)
00269                 pathStatus.Status = PATHSTATUS_STATUS_INPROGRESS;
00270         else
00271                 pathStatus.Status = PATHSTATUS_STATUS_COMPLETED;
00272         PathStatusSet(&pathStatus);
00273 
00274         float groundspeed = pathDesired.StartingVelocity + 
00275             (pathDesired.EndingVelocity - pathDesired.StartingVelocity) * progress.fractional_progress;
00276         if(progress.fractional_progress > 1)
00277                 groundspeed = 0;
00278         
00279         VelocityDesiredData velocityDesired;
00280         velocityDesired.North = progress.path_direction[0] * groundspeed;
00281         velocityDesired.East = progress.path_direction[1] * groundspeed;
00282         
00283         float error_speed = progress.error * guidanceSettings.HorizontalPosPI[VTOLPATHFOLLOWERSETTINGS_HORIZONTALPOSPI_KP];
00284         float correction_velocity[2] = {progress.correction_direction[0] * error_speed, 
00285             progress.correction_direction[1] * error_speed};
00286         
00287         float total_vel = sqrtf(powf(correction_velocity[0],2) + powf(correction_velocity[1],2));
00288         float scale = 1;
00289         if(total_vel > guidanceSettings.HorizontalVelMax)
00290                 scale = guidanceSettings.HorizontalVelMax / total_vel;
00291 
00292         // Currently not apply a PID loop for horizontal corrections
00293         velocityDesired.North += progress.correction_direction[0] * error_speed * scale;
00294         velocityDesired.East += progress.correction_direction[1] * error_speed * scale;
00295         
00296         // Interpolate desired velocity along the path
00297         float altitudeSetpoint = pathDesired.Start[2] + (pathDesired.End[2] - pathDesired.Start[2]) *
00298             bound_min_max(progress.fractional_progress,0,1);
00299 
00300         float downError = altitudeSetpoint - positionActual.Down;
00301         velocityDesired.Down = pid_apply_antiwindup(&vtol_pids[DOWN_POSITION], downError,
00302                 -guidanceSettings.VerticalVelMax, guidanceSettings.VerticalVelMax, dT);
00303 
00304         VelocityDesiredSet(&velocityDesired);
00305 }

00313 void updateEndpointVelocity()
00314 {
00315         float dT = guidanceSettings.UpdatePeriod / 1000.0f;
00316 
00317         PositionActualData positionActual;
00318         VelocityDesiredData velocityDesired;
00319         
00320         PositionActualGet(&positionActual);
00321         VelocityDesiredGet(&velocityDesired);
00322         
00323         float northError;
00324         float eastError;
00325         float downError;
00326         float northCommand;
00327         float eastCommand;
00328         
00329         float northPos = positionActual.North;
00330         float eastPos = positionActual.East;
00331         float downPos = positionActual.Down;
00332 
00333         // Compute desired north command velocity from position error
00334         northError = pathDesired.End[PATHDESIRED_END_NORTH] - northPos;
00335         northCommand = pid_apply_antiwindup(&vtol_pids[NORTH_POSITION], northError,
00336             -guidanceSettings.HorizontalVelMax, guidanceSettings.HorizontalVelMax, dT);
00337 
00338         // Compute desired east command velocity from position error
00339         eastError = pathDesired.End[PATHDESIRED_END_EAST] - eastPos;
00340         eastCommand = pid_apply_antiwindup(&vtol_pids[EAST_POSITION], eastError,
00341             -guidanceSettings.HorizontalVelMax, guidanceSettings.HorizontalVelMax, dT);
00342 
00343         // Limit the maximum velocity any direction (not north and east separately)
00344         float total_vel = sqrtf(powf(northCommand,2) + powf(eastCommand,2));
00345         float scale = 1;
00346         if(total_vel > guidanceSettings.HorizontalVelMax)
00347                 scale = guidanceSettings.HorizontalVelMax / total_vel;
00348 
00349         velocityDesired.North = northCommand * scale;
00350         velocityDesired.East = eastCommand * scale;
00351 
00352         // Compute the desired velocity from the position difference
00353         downError = pathDesired.End[PATHDESIRED_END_DOWN] - downPos;
00354         velocityDesired.Down = pid_apply_antiwindup(&vtol_pids[DOWN_POSITION], downError,
00355             -guidanceSettings.VerticalVelMax, guidanceSettings.VerticalVelMax, dT);
00356         
00357         VelocityDesiredSet(&velocityDesired);   
00358 
00359         // Indicate whether we are in radius of this endpoint
00360         uint8_t path_status = PATHSTATUS_STATUS_INPROGRESS;
00361         float distance2 = powf(northError, 2) + powf(eastError, 2);
00362         if (distance2 < (guidanceSettings.EndpointRadius * guidanceSettings.EndpointRadius)) {
00363                 path_status = PATHSTATUS_STATUS_COMPLETED;
00364         }
00365         PathStatusStatusSet(&path_status);

00379         VelocityDesiredData velocityDesired;
00380         VelocityActualData velocityActual;
00381         StabilizationDesiredData stabDesired;
00382         AttitudeActualData attitudeActual;

00391 
00392         float downError;

00394                 
00395         VtolPathFollowerSettingsGet(&guidanceSettings);
00396         
00397         VelocityActualGet(&velocityActual);
00398         VelocityDesiredGet(&velocityDesired);
00399         StabilizationDesiredGet(&stabDesired);
00400         VelocityDesiredGet(&velocityDesired);
00401         AttitudeActualGet(&attitudeActual);
00402         StabilizationSettingsGet(&stabSettings);
00403         NedAccelGet(&nedAccel);
00404         
00405         float northVel = velocityActual.North;
00406         float eastVel = velocityActual.East;
00407         float downVel = velocityActual.Down;
00408 
00409         // Compute desired north command from velocity error
00410         northError = velocityDesired.North - northVel;
00411         northCommand = pid_apply_antiwindup(&vtol_pids[NORTH_VELOCITY], northError, 
00412             -guidanceSettings.MaxRollPitch, guidanceSettings.MaxRollPitch, dT) + velocityDesired.North * guidanceSettings.VelocityFeedforward;
00413         
00414         // Compute desired east command from velocity error
00415         eastError = velocityDesired.East - eastVel;
00416         eastCommand = pid_apply_antiwindup(&vtol_pids[NORTH_VELOCITY], eastError,
00417             -guidanceSettings.MaxRollPitch, guidanceSettings.MaxRollPitch, dT) + velocityDesired.East * guidanceSettings.VelocityFeedforward;
00418         
00419         // Compute desired down command.  Using NED accel as the damping term
00420         downError = velocityDesired.Down - downVel;
00421         // Negative is critical here since throttle is negative with down
00422         downCommand = -pid_apply_antiwindup(&vtol_pids[DOWN_VELOCITY], downError, -1, 1, dT) +
00423             nedAccel.Down * guidanceSettings.VerticalVelPID[VTOLPATHFOLLOWERSETTINGS_VERTICALVELPID_KD];
00424 
00425         // If this setting is zero then the throttle level available when enabled is used for hover:wq
00426         float used_throttle_offset = (guidanceSettings.HoverThrottle == 0) ? throttleOffset : guidanceSettings.HoverThrottle;
00427         stabDesired.Throttle = bound_min_max(downCommand + used_throttle_offset, 0, 1);
00428         
00429         // Project the north and east command signals into the pitch and roll based on yaw.
00430         // For this to behave well the craft should move similarly for 5 deg roll versus 5 deg pitch.
00431         // Notice the inputs are crudely bounded by the anti-winded but if both N and E were
00432         // saturated and the craft were at 45 degrees that would result in a value greater than
00433         // the limit, so apply limit again here.
00434         stabDesired.Pitch = bound_min_max(-northCommand * cosf(attitudeActual.Yaw * M_PI / 180) + 
00435                                       -eastCommand * sinf(attitudeActual.Yaw * M_PI / 180),
00436                                       -guidanceSettings.MaxRollPitch, guidanceSettings.MaxRollPitch);
00437         stabDesired.Roll = bound_min_max(-northCommand * sinf(attitudeActual.Yaw * DEG2RAD) + 
00438                                      eastCommand * cosf(attitudeActual.Yaw * DEG2RAD),
00439                                      -guidanceSettings.MaxRollPitch, guidanceSettings.MaxRollPitch);
00440         
00441         if(guidanceSettings.ThrottleControl == VTOLPATHFOLLOWERSETTINGS_THROTTLECONTROL_FALSE) {
00442                 // For now override throttle with manual control.  Disable at your risk, quad goes to China.
00443                 ManualControlCommandData manualControl;
00444                 ManualControlCommandGet(&manualControl);
00445                 stabDesired.Throttle = manualControl.Throttle;
00446         }
00447         
00448         stabDesired.StabilizationMode[STABILIZATIONDESIRED_STABILIZATIONMODE_ROLL] = STABILIZATIONDESIRED_STABILIZATIONMODE_ATTITUDE;
00449         stabDesired.StabilizationMode[STABILIZATIONDESIRED_STABILIZATIONMODE_PITCH] = STABILIZATIONDESIRED_STABILIZATIONMODE_ATTITUDE;
00450 
00451         float yaw;
00452         switch(guidanceSettings.YawMode) {
00453         case VTOLPATHFOLLOWERSETTINGS_YAWMODE_RATE:
00454                 /* This is awkward.  This allows the transmitter to control the yaw while flying navigation */

00457                 stabDesired.StabilizationMode[STABILIZATIONDESIRED_STABILIZATIONMODE_YAW] = STABILIZATIONDESIRED_STABILIZATIONMODE_RATE;
00458                 break;
00459         case VTOLPATHFOLLOWERSETTINGS_YAWMODE_AXISLOCK:
00460                 ManualControlCommandYawGet(&yaw);

00463                 break;
00464         case VTOLPATHFOLLOWERSETTINGS_YAWMODE_ATTITUDE:
00465                 ManualControlCommandYawGet(&yaw);
00466                 stabDesired.Yaw = stabSettings.YawMax * yaw;      
00467                 stabDesired.StabilizationMode[STABILIZATIONDESIRED_STABILIZATIONMODE_YAW] = STABILIZATIONDESIRED_STABILIZATIONMODE_ATTITUDE;

00508         
00509         NedAccelData accelData;
00510         NedAccelGet(&accelData);
00511         accelData.North = accel_ned[0];
