117  // TODO: How do we protect against the DMA buffer swapping midway through adding data

121  int32_t max_bytes = PACKET_SIZE - overo_dev->writing_offset;

123  if (max_bytes > 0) {
124  uint16_t bytes_added;
125  bool tx_need_yield = false;

202 int32_t PIOS_OVERO_Init(uint32_t * overo_id, const struct pios_overo_cfg * cfg)
203 {
204  PIOS_DEBUG_Assert(overo_id);
205  PIOS_DEBUG_Assert(cfg);
206 
207  struct pios_overo_dev *overo_dev;
208 
209  overo_dev = (struct pios_overo_dev *) PIOS_OVERO_alloc();
210  if (!overo_dev) goto out_fail;
211 
212  /* Bind the configuration to the device instance */
213  overo_dev->cfg = cfg;
214  overo_dev->writing_buffer = 1; // First writes to second buffer
215 
216  /* Put buffers to a known state */
217  memset(&overo_dev->tx_buffer[0][0], 0xFF, PACKET_SIZE);
218  memset(&overo_dev->tx_buffer[1][0], 0xFF, PACKET_SIZE);
219  memset(&overo_dev->rx_buffer[0][0], 0xFF, PACKET_SIZE);
220  memset(&overo_dev->rx_buffer[1][0], 0xFF, PACKET_SIZE);
221 
222  /*
223  * Enable the SPI device
224  *
225  * 1. Enable the SPI port
226  * 2. Enable DMA with circular buffered DMA (validate config)
227  * 3. Enable the DMA Tx IRQ
228  */
229 
230  //PIOS_Assert(overo_dev->cfg->dma.tx-> == CIRCULAR);
231  //PIOS_Assert(overo_dev->cfg->dma.rx-> == CIRCULAR);
232 
233  /* only legal for single-slave config */
234  PIOS_Assert(overo_dev->cfg->slave_count == 1);
235  SPI_SSOutputCmd(overo_dev->cfg->regs, DISABLE);
236 
237  /* Initialize the GPIO pins */
238  /* note __builtin_ctz() due to the difference between GPIO_PinX and GPIO_PinSourceX */
239  GPIO_PinAFConfig(overo_dev->cfg->sclk.gpio,
240  __builtin_ctz(overo_dev->cfg->sclk.init.GPIO_Pin),
241  overo_dev->cfg->remap);
242  GPIO_PinAFConfig(overo_dev->cfg->mosi.gpio,
243  __builtin_ctz(overo_dev->cfg->mosi.init.GPIO_Pin),
244  overo_dev->cfg->remap);
245  GPIO_PinAFConfig(overo_dev->cfg->miso.gpio,
246  __builtin_ctz(overo_dev->cfg->miso.init.GPIO_Pin),
247  overo_dev->cfg->remap);
248  GPIO_PinAFConfig(overo_dev->cfg->ssel[0].gpio,
249  __builtin_ctz(overo_dev->cfg->ssel[0].init.GPIO_Pin),
250  overo_dev->cfg->remap);
251 
252  GPIO_Init(overo_dev->cfg->sclk.gpio, (GPIO_InitTypeDef*)&(overo_dev->cfg->sclk.init));
253  GPIO_Init(overo_dev->cfg->mosi.gpio, (GPIO_InitTypeDef*)&(overo_dev->cfg->mosi.init));
254  GPIO_Init(overo_dev->cfg->miso.gpio, (GPIO_InitTypeDef*)&(overo_dev->cfg->miso.init));
255 
256  /* Configure circular buffer targets. Configure 0 to be initially active */
257  DMA_InitTypeDef dma_init;
258 
259  DMA_DeInit(overo_dev->cfg->dma.rx.channel);
260  dma_init = overo_dev->cfg->dma.rx.init;
261  dma_init.DMA_Memory0BaseAddr = (uint32_t) overo_dev->rx_buffer[0];
262  dma_init.DMA_MemoryInc = DMA_MemoryInc_Enable;
263  dma_init.DMA_BufferSize = PACKET_SIZE;
264  DMA_Init(overo_dev->cfg->dma.rx.channel, &dma_init);
265  DMA_DoubleBufferModeConfig(overo_dev->cfg->dma.rx.channel, (uint32_t) overo_dev->rx_buffer[1], DMA_Memory_0);
266  DMA_DoubleBufferModeCmd(overo_dev->cfg->dma.rx.channel, ENABLE);
267 
268  DMA_DeInit(overo_dev->cfg->dma.tx.channel);
269  dma_init = overo_dev->cfg->dma.tx.init;
270  dma_init.DMA_Memory0BaseAddr = (uint32_t) overo_dev->tx_buffer[0];
271  dma_init.DMA_MemoryInc = DMA_MemoryInc_Enable;
272  dma_init.DMA_BufferSize = PACKET_SIZE;
273  DMA_Init(overo_dev->cfg->dma.tx.channel, &dma_init);
274  DMA_DoubleBufferModeConfig(overo_dev->cfg->dma.tx.channel, (uint32_t) overo_dev->tx_buffer[1], DMA_Memory_0);
275  DMA_DoubleBufferModeCmd(overo_dev->cfg->dma.tx.channel, ENABLE);
276 
277  /* Set the packet size */
278  DMA_SetCurrDataCounter(overo_dev->cfg->dma.rx.channel, PACKET_SIZE);
279  DMA_SetCurrDataCounter(overo_dev->cfg->dma.tx.channel, PACKET_SIZE);
280 
281  /* Initialize the SPI block */
282  SPI_DeInit(overo_dev->cfg->regs);
283  SPI_Init(overo_dev->cfg->regs, (SPI_InitTypeDef*)&(overo_dev->cfg->init));
284 
285  SPI_CalculateCRC(overo_dev->cfg->regs, DISABLE);
286 
287  /* Enable SPI interrupts to DMA */
288  SPI_I2S_DMACmd(overo_dev->cfg->regs, SPI_I2S_DMAReq_Tx | SPI_I2S_DMAReq_Rx, ENABLE);
289 
290  /* Configure DMA interrupt */
291  NVIC_Init((NVIC_InitTypeDef*)&(overo_dev->cfg->dma.irq.init));
292  DMA_ITConfig(overo_dev->cfg->dma.tx.channel, DMA_IT_TC, ENABLE);
293 
294  /* Enable the DMA channels */
295  DMA_Cmd(overo_dev->cfg->dma.tx.channel, ENABLE);
296  DMA_Cmd(overo_dev->cfg->dma.rx.channel, ENABLE);
297 
298  *overo_id = (uint32_t) overo_dev;
299 
300  return(0);
301 
302 out_fail:
303  return(-1);
304 }
