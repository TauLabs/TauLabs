00299 void PIOS_MPU6000_SetSampleRate(uint16_t samplerate_hz)
00300 {
00301         uint16_t filter_frequency = 8000;
00302 
00303         if (pios_mpu6000_dev->filter != PIOS_MPU60X0_LOWPASS_256_HZ)
00304                 filter_frequency = 1000;
00305 
00306         // limit samplerate to filter frequency
00307         if (samplerate_hz > filter_frequency)
00308                 samplerate_hz = filter_frequency;
00309 
00310         // calculate divisor, round to nearest integeter
00311         int32_t divisor = (int32_t)(((float)filter_frequency / samplerate_hz) + 0.5f) - 1;
00312 
00313         // limit resulting divisor to register value range
00314         if (divisor < 0)
00315                 divisor = 0;
00316 
00317         if (divisor > 0xff)
00318                 divisor = 0xff;
00319 
00320         PIOS_MPU6000_SetReg(PIOS_MPU60X0_SMPLRT_DIV_REG, (uint8_t)divisor);
00321 }
