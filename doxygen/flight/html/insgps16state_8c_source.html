353 // ************* CovariancePrediction *************
354 // Does the prediction step of the Kalman filter for the covariance matrix
355 // Output, Pnew, overwrites P, the input covariance
356 // Pnew = (I+F*T)*P*(I+F*T)' + T^2*G*Q*G'
357 // Q is the discrete time covariance of process noise
358 // Q is vector of the diagonal for a square matrix with
359 // dimensions equal to the number of disturbance noise variables
360 // The General Method is very inefficient,not taking advantage of the sparse F and G
361 // The first Method is very specific to this implementation
362 // ************************************************
363

366 void CovariancePrediction(float F[NUMX][NUMX], float G[NUMX][NUMW],
367  float Q[NUMW], float dT, float P[NUMX][NUMX])
368 {
369  float Dummy[NUMX][NUMX], dTsq;
370  uint8_t i, j, k;
371 
372  // Pnew = (I+F*T)*P*(I+F*T)' + T^2*G*Q*G' = T^2[(P/T + F*P)*(I/T + F') + G*Q*G')]
373 
374  dTsq = dT * dT;
375 
376  for (i = 0; i < NUMX; i++) // Calculate Dummy = (P/T +F*P)
377  for (j = 0; j < NUMX; j++) {
378  Dummy[i][j] = P[i][j] / dT;
379  for (k = 0; k < NUMX; k++)
380  Dummy[i][j] += F[i][k] * P[k][j];
381  }
382  for (i = 0; i < NUMX; i++) // Calculate Pnew = Dummy/T + Dummy*F' + G*Qw*G'
383  for (j = i; j < NUMX; j++) { // Use symmetry, ie only find upper triangular
384  P[i][j] = Dummy[i][j] / dT;
385  for (k = 0; k < NUMX; k++)
386  P[i][j] += Dummy[i][k] * F[j][k]; // P = Dummy/T + Dummy*F'
387  for (k = 0; k < NUMW; k++)
388  P[i][j] += Q[k] * G[i][k] * G[j][k]; // P = Dummy/T + Dummy*F' + G*Q*G'
389  P[j][i] = P[i][j] = P[i][j] * dTsq; // Pnew = T^2*P and fill in lower triangular;
390  }
391 }

396  float Q[NUMW], float dT, float P[NUMX][NUMX])

398  float D[NUMX][NUMX], T, Tsq;
399  uint8_t i, j;
400 
401  // Pnew = (I+F*T)*P*(I+F*T)' + T^2*G*Q*G' = scalar expansion from symbolic manipulator
402 
403  T = dT;
404  Tsq = dT * dT;
405 
406  for (i = 0; i < NUMX; i++) // Create a copy of the upper triangular of P
407  for (j = i; j < NUMX; j++)
408  D[i][j] = P[i][j];

550 // ************* SerialUpdate *******************
551 // Does the update step of the Kalman filter for the covariance and estimate
552 // Outputs are Xnew & Pnew, and are written over P and X
553 // Z is actual measurement, Y is predicted measurement
554 // Xnew = X + K*(Z-Y), Pnew=(I-K*H)*P,
555 // where K=P*H'*inv[H*P*H'+R]
556 // NOTE the algorithm assumes R (measurement covariance matrix) is diagonal
557 // i.e. the measurment noises are uncorrelated.
558 // It therefore uses a serial update that requires no matrix inversion by
559 // processing the measurements one at a time.
560 // Algorithm - see Grewal and Andrews, "Kalman Filtering,2nd Ed" p.121 & p.253
561 // - or see Simon, "Optimal State Estimation," 1st Ed, p.150
562 // The SensorsUsed variable is a bitwise mask indicating which sensors
563 // should be used in the update.
564 // ************************************************
565 
566 void SerialUpdate(float H[NUMV][NUMX], float R[NUMV], float Z[NUMV],
567  float Y[NUMV], float P[NUMX][NUMX], float X[NUMX],
568  uint16_t SensorsUsed)
569 {
570  float HP[NUMX], HPHR, Error;
571  uint8_t i, j, k, m;
572 
573  for (m = 0; m < NUMV; m++) {
574 
575  if (SensorsUsed & (0x01 << m)) { // use this sensor for update
576 
577  for (j = 0; j < NUMX; j++) { // Find Hp = H*P
578  HP[j] = 0.0f;
579  for (k = 0; k < NUMX; k++)
580  HP[j] += H[m][k] * P[k][j];
581  }
582  HPHR = R[m]; // Find HPHR = H*P*H' + R
583  for (k = 0; k < NUMX; k++)
584  HPHR += HP[k] * H[m][k];
585 
586  for (k = 0; k < NUMX; k++)
587  K[k][m] = HP[k] / HPHR; // find K = HP/HPHR
588 
589  for (i = 0; i < NUMX; i++) { // Find P(m)= P(m-1) + K*HP
590  for (j = i; j < NUMX; j++)
591  P[i][j] = P[j][i] =
592  P[i][j] - K[i][m] * HP[j];
593  }
594 
595  Error = Z[m] - Y[m];
596  for (i = 0; i < NUMX; i++) // Find X(m)= X(m-1) + K*Error
597  X[i] = X[i] + K[i][m] * Error;
598 
599  }
600  }
601 }

603 // ************* RungeKutta **********************
604 // Does a 4th order Runge Kutta numerical integration step
605 // Output, Xnew, is written over X
606 // NOTE the algorithm assumes time invariant state equations and
607 // constant inputs over integration step
608 // ************************************************
609 
610 void RungeKutta(float X[NUMX], float U[NUMU], float dT)
611 {
612 
613  float dT2 =
614  dT / 2.0f, K1[NUMX], K2[NUMX], K3[NUMX], K4[NUMX], Xlast[NUMX];
615  uint8_t i;
616 
617  for (i = 0; i < NUMX; i++)
618  Xlast[i] = X[i]; // make a working copy
619 
620  StateEq(X, U, K1); // k1 = f(x,u)
621  for (i = 0; i < NUMX; i++)
622  X[i] = Xlast[i] + dT2 * K1[i];
623  StateEq(X, U, K2); // k2 = f(x+0.5*dT*k1,u)
624  for (i = 0; i < NUMX; i++)
625  X[i] = Xlast[i] + dT2 * K2[i];
626  StateEq(X, U, K3); // k3 = f(x+0.5*dT*k2,u)
627  for (i = 0; i < NUMX; i++)
628  X[i] = Xlast[i] + dT * K3[i];
629  StateEq(X, U, K4); // k4 = f(x+dT*k3,u)
630

632  for (i = 0; i < NUMX; i++)
633  X[i] =
634  Xlast[i] + dT * (K1[i] + 2.0f * K2[i] + 2.0f * K3[i] +
635  K4[i]) / 6.0f;
