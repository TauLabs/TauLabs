00269         VelocityActualData velocityActual;

00271 
00272         float cur[3] = {positionActual.North, positionActual.East, positionActual.Down};
00273         struct path_status progress;
00274 
00275         path_progress(&pathDesired, cur, &progress);
00276         
00277         float groundspeed = 0;
00278         float altitudeSetpoint = 0;
00279         switch (pathDesired.Mode) {
00280                 case PATHDESIRED_MODE_FLYCIRCLERIGHT:
00281                 case PATHDESIRED_MODE_DRIVECIRCLERIGHT:
00282                 case PATHDESIRED_MODE_FLYCIRCLELEFT:
00283                 case PATHDESIRED_MODE_DRIVECIRCLELEFT:
00284                         groundspeed = pathDesired.EndingVelocity;
00285                         altitudeSetpoint = pathDesired.End[2];
00286                         break;
00287                 case PATHDESIRED_MODE_FLYENDPOINT:
00288                 case PATHDESIRED_MODE_DRIVEENDPOINT:
00289                 case PATHDESIRED_MODE_FLYVECTOR:
00290                 case PATHDESIRED_MODE_DRIVEVECTOR:
00291                 default:
00292                         groundspeed = pathDesired.StartingVelocity + (pathDesired.EndingVelocity - pathDesired.StartingVelocity) *
00293                                 bound_min_max(progress.fractional_progress,0,1);
00294                         altitudeSetpoint = pathDesired.Start[2] + (pathDesired.End[2] - pathDesired.Start[2]) *
00295                                 bound_min_max(progress.fractional_progress,0,1);
00296                         break;
00297         }
00298         // this ensures a significant forward component at least close to the real trajectory
00299         if (groundspeed<fixedWingAirspeeds.BestClimbRateSpeed/10.)
00300                 groundspeed=fixedWingAirspeeds.BestClimbRateSpeed/10.;
00301         
00302         // calculate velocity - can be zero if waypoints are too close
00303         VelocityDesiredData velocityDesired;
00304         velocityDesired.North = progress.path_direction[0] * groundspeed;
00305         velocityDesired.East = progress.path_direction[1] * groundspeed;
00306         
00307         float error_speed = progress.error * fixedwingpathfollowerSettings.HorizontalPosP;
00308 
00309         // calculate correction - can also be zero if correction vector is 0 or no error present
00310         velocityDesired.North += progress.correction_direction[0] * error_speed;
00311         velocityDesired.East += progress.correction_direction[1] * error_speed;
00312         
00313         float downError = altitudeSetpoint - positionActual.Down;
00314         velocityDesired.Down = downError * fixedwingpathfollowerSettings.VerticalPosP;
00315 
00316         // update pathstatus
00317         pathStatus.error = progress.error;
00318         pathStatus.fractional_progress = progress.fractional_progress;
00319 
00320         pathStatus.fractional_progress = progress.fractional_progress;
00321         if (pathStatus.fractional_progress < 1)
00322                 pathStatus.Status = PATHSTATUS_STATUS_INPROGRESS;
00323         else
00324                 pathStatus.Status = PATHSTATUS_STATUS_COMPLETED;
00325         VelocityDesiredSet(&velocityDesired);
00326 }
