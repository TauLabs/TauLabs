122 int32_t PIOS_SDCARD_PowerOn(void)
123 {
124  int32_t status;
125  int i;
126 
127  SDCARD_MUTEX_TAKE;
128  /* Ensure that chip select line deactivated */
129  PIOS_SPI_RC_PinSet(PIOS_SDCARD_SPI, 0, 1); /* spi, pin_value */
130 
131  /* Init SPI port for slow frequency access (ca. 0.3 MBit/s) */
132  PIOS_SPI_SetClockSpeed(PIOS_SDCARD_SPI, PIOS_SPI_PRESCALER_256);
133 
134  /* Send 80 clock cycles to start up */
135  for (i = 0; i < 10; ++i) {
136  PIOS_SPI_TransferByte(PIOS_SDCARD_SPI, 0xff);
137  }
138 
139  /* Activate chip select */
140  PIOS_SPI_RC_PinSet(PIOS_SDCARD_SPI, 0, 0); /* spi, pin_value */
141 
142  /* wait for 1 mS */
143  PIOS_DELAY_WaituS(1000);
144 
145  /* Send CMD0 to reset the media */
146  if ((status = PIOS_SDCARD_SendSDCCmd(SDCMD_GO_IDLE_STATE, 0, SDCMD_GO_IDLE_STATE_CRC)) < 0) {
147  goto error;
148  }
149 
150  CardType = 0;
151 
152  /* A card is detected, what type is it? Use SEND_IF_COND (CMD8) to find out */
153  if ((status = PIOS_SDCARD_SendSDCCmd(SDCMD_SEND_IF_COND, 0x1AA, SDCMD_SEND_IF_COND_CRC)) == 0x01aa) {
154  /* SDHC Card Detected. */
155 
156  /* We now check to see if we should use block mode or byte mode. Command is SEND_OP_COND_SDC (ACMD41) with HCS (bit 30) set */
157  for (i = 0; i < 16384; i++)
158  if ((status = PIOS_SDCARD_SendSDCCmd(SDCMD_SEND_OP_COND_SDC, 0x01 << 30, SDCMD_SEND_OP_COND_SDC_CRC)) == 0)
159  break;
160 
161  if (i < 16384) {
162  status = PIOS_SDCARD_SendSDCCmd(SDCMD_READ_OCR, 0, SDCMD_READ_OCR_CRC);
163  CardType = ((status >> 24) & 0x40) ? CT_SD2 | CT_BLOCK : CT_SD2;
164  status = 0;
165  } else {
166  /* We waited long enough! */
167  status = -2;
168  }
169  } else {
170  /* Card is SDv1 or MMC. */
171 
172  /* MMC will accept ACMD41 (SDv1 won't and requires CMD1) so send ACMD41 first and then CMD1 if that fails */
173  if ((status = PIOS_SDCARD_SendSDCCmd(SDCMD_SEND_OP_COND_SDC, 0, SDCMD_SEND_OP_COND_SDC_CRC)) <= 1) {
174  CardType = CT_SD1;
175  } else {
176  CardType = CT_MMC;
177  }
178 
179  for (i = 0; i < 16384; i++) {
180  if (CardType == CT_SD1) {
181  status = PIOS_SDCARD_SendSDCCmd(SDCMD_SEND_OP_COND_SDC, 0, SDCMD_SEND_OP_COND_SDC_CRC);
182  } else {
183  status = PIOS_SDCARD_SendSDCCmd(SDCMD_SEND_OP_COND, 0, SDCMD_SEND_OP_COND_CRC);
184  }
185 
186  if (status < 0) {
187  break;
188  }
189  if (status == 0) {
190  break;
191  }
192  }
193 
194  /* The block size should already be 512 bytes but re-initialize just in case (ignore if it fails) */
195  PIOS_SDCARD_SendSDCCmd(SDCMD_SET_BLOCKLEN, 512, SDCMD_SEND_OP_COND_CRC);
196  }
197 
198  if (i == 16384 || CardType == 0) {
199  /* The last loop timed out or the cardtype was not detected... */
200  status = -2;
201  }
202 
203 error:
204  /* Deactivate chip select */
205  PIOS_SPI_RC_PinSet(PIOS_SDCARD_SPI, 0, 1); /* spi, pin_value */
206 
207  /* Send dummy byte once deactivated to drop cards DO */
208  PIOS_SPI_TransferByte(PIOS_SDCARD_SPI, 0xff);
209  SDCARD_MUTEX_GIVE;
210  return status; /* Status should be 0 if nothing went wrong! */
211 }

264 int32_t PIOS_SDCARD_CheckAvailable(uint8_t was_available)
265 {
266  int32_t ret;
267  SDCARD_MUTEX_TAKE;
268 
269  if (was_available) {
270  /* Init SPI port for fast frequency access (ca. 18 MBit/s) */
271  /* This is required for the case that the SPI port is shared with other devices */
272  PIOS_SPI_SetClockSpeed(PIOS_SDCARD_SPI, PIOS_SPI_PRESCALER_4);
273 
274  /* Activate chip select */
275  PIOS_SPI_RC_PinSet(PIOS_SDCARD_SPI, 0, 0); /* spi, pin_value */
276 
277  /* Send STATUS command to check if media is available */
278  ret = PIOS_SDCARD_SendSDCCmd(SDCMD_SEND_STATUS, 0, SDCMD_SEND_STATUS_CRC);
279  } else {
280  /* Ensure that SPI interface is clocked at low speed */
281  PIOS_SPI_SetClockSpeed(PIOS_SDCARD_SPI, PIOS_SPI_PRESCALER_256);
282 
283  /* Deactivate chip select */
284  PIOS_SPI_RC_PinSet(PIOS_SDCARD_SPI, 0, 1); /* spi, pin_value */
285  /* send 80 clock cycles to start up */
286  uint8_t i;
287  for (i = 0; i < 10; ++i) {
288  PIOS_SPI_TransferByte(PIOS_SDCARD_SPI, 0xff);
289  }
290 
291  /* Activate chip select */
292  PIOS_SPI_RC_PinSet(PIOS_SDCARD_SPI, 0, 0); /* spi, pin_value */
293 
294  /* Send CMD0 to reset the media */
295  if ((ret = (PIOS_SDCARD_SendSDCCmd(SDCMD_GO_IDLE_STATE, 0, SDCMD_GO_IDLE_STATE_CRC))) < 0) {
296  goto not_available;
297  }
298 
299  /* Deactivate chip select */
300  PIOS_SPI_RC_PinSet(PIOS_SDCARD_SPI, 0, 1); /* spi, pin_value */
301 
302  SDCARD_MUTEX_GIVE;
303  /* Run power-on sequence (negative return = not available) */
304  ret = PIOS_SDCARD_PowerOn();
305  }
306 
307 not_available:
308  /* Deactivate chip select */
309  PIOS_SPI_RC_PinSet(PIOS_SDCARD_SPI, 0, 1); /* spi, pin_value */
310  /* Send dummy byte once deactivated to drop cards DO */
311  PIOS_SPI_TransferByte(PIOS_SDCARD_SPI, 0xff);
312  SDCARD_MUTEX_GIVE;
313 
314  return (ret == 0) ? 1 : 0; /* 1 = available, 0 = not available. */
315 }

325 int32_t PIOS_SDCARD_SendSDCCmd(uint8_t cmd, uint32_t addr, uint8_t crc)
326 {
327  int i;
328  int32_t ret;
329 
330  if (cmd & 0x80) { /* ACMD<n> is the command sequence of CMD55-CMD<n> */
331  cmd &= 0x7F;
332  ret = PIOS_SDCARD_SendSDCCmd(SDCMD_APP_CMD, 0, SDCMD_APP_CMD_CRC);
333  if (ret > 1) {
334  return ret;
335  }
336  }
337 
338  /* Activate chip select */
339  PIOS_SPI_RC_PinSet(PIOS_SDCARD_SPI, 0, 0); /* spi, pin_value */
340 
341  /* Transfer to card */
342  PIOS_SPI_TransferByte(PIOS_SDCARD_SPI, (uint8_t) cmd);
343  PIOS_SPI_TransferByte(PIOS_SDCARD_SPI, (addr >> 24) & 0xff);
344  PIOS_SPI_TransferByte(PIOS_SDCARD_SPI, (addr >> 16) & 0xff);
345  PIOS_SPI_TransferByte(PIOS_SDCARD_SPI, (addr >> 8) & 0xff);
346  PIOS_SPI_TransferByte(PIOS_SDCARD_SPI, (addr >> 0) & 0xff);
347  PIOS_SPI_TransferByte(PIOS_SDCARD_SPI, crc);
348 
349  uint8_t timeout = 0;
350 
351  if (cmd == SDCMD_SEND_STATUS) {
352  /* One dummy read */
353  PIOS_SPI_TransferByte(PIOS_SDCARD_SPI, 0xff);
354 
355  /* Read two bytes (only last value will be returned) */
356  ret = PIOS_SPI_TransferByte(PIOS_SDCARD_SPI, 0xff);
357  ret = PIOS_SPI_TransferByte(PIOS_SDCARD_SPI, 0xff);
358 
359  /* All-one read: we expect that SD card is not connected: notify timeout! */
360  if (ret == 0xff) {
361  timeout = 1;
362  }
363  } else {
364  /* Wait for standard R1 response */
365  for (i = 0; i < 8; ++i) {
366  if (!((ret = PIOS_SPI_TransferByte(PIOS_SDCARD_SPI, 0xff)) & 0x80)) {
367  break;
368  }
369  }
370  if (i == 8) {
371  timeout = 1;
372  }
373  }
374 
375  if ((cmd == SDCMD_SEND_IF_COND || cmd == SDCMD_READ_OCR) && timeout != 1) {
376  /* This is a 4 byte R3 or R7 response. */
377  ret = (PIOS_SPI_TransferByte(PIOS_SDCARD_SPI, 0xff) << 24);
378  ret |= (PIOS_SPI_TransferByte(PIOS_SDCARD_SPI, 0xff) << 16);
379  ret |= (PIOS_SPI_TransferByte(PIOS_SDCARD_SPI, 0xff) << 8);
380  ret |= (PIOS_SPI_TransferByte(PIOS_SDCARD_SPI, 0xff) << 0);
381  }
382 
383  /* Required clocking (see spec) */
384  PIOS_SPI_TransferByte(PIOS_SDCARD_SPI, 0xff);
385 
386  /* Deactivate chip-select on timeout, and return error code */
387  if (timeout) {
388  PIOS_SPI_RC_PinSet(PIOS_SDCARD_SPI, 0, 1); /* spi, pin_value */
389 
390  /* Send dummy byte once deactivated to drop cards DO */
391  PIOS_SPI_TransferByte(PIOS_SDCARD_SPI, 0xff);
392  return -1;
393  }
394 
395  /* Else return received value - don't deactivate chip select or mutex (for continuous access) */
396  return ret;
397 }

706  csd->RdBlockMisalign = (csd_buffer[6] & 0x20) >> 5;
707  csd->DSRImpl = (csd_buffer[6] & 0x10) >> 4;
708  csd->Reserved2 = 0; /* Reserved */
709  csd->DeviceSize = (csd_buffer[6] & 0x03) << 10;
710  /* Byte 7 */
711  csd->DeviceSize |= (csd_buffer[7]) << 2;
712  /* Byte 8 */
713  csd->DeviceSize |= (csd_buffer[8] & 0xC0) >> 6;
714  csd->MaxRdCurrentVDDMin = (csd_buffer[8] & 0x38) >> 3;
715  csd->MaxRdCurrentVDDMax = (csd_buffer[8] & 0x07);
716  /* Byte 9 */
717  csd->MaxWrCurrentVDDMin = (csd_buffer[9] & 0xE0) >> 5;
718  csd->MaxWrCurrentVDDMax = (csd_buffer[9] & 0x1C) >> 2;
719  csd->DeviceSizeMul = (csd_buffer[9] & 0x03) << 1;
720  /* Byte 10 */
721  csd->DeviceSizeMul |= (csd_buffer[10] & 0x80) >> 7;
722  csd->EraseGrSize = (csd_buffer[10] & 0x7C) >> 2;
723  csd->EraseGrMul = (csd_buffer[10] & 0x03) << 3;
724  /* Byte 11 */
725  csd->EraseGrMul |= (csd_buffer[11] & 0xE0) >> 5;
726  csd->WrProtectGrSize = (csd_buffer[11] & 0x1F);
727  /* Byte 12 */
728  csd->WrProtectGrEnable = (csd_buffer[12] & 0x80) >> 7;
729  csd->ManDeflECC = (csd_buffer[12] & 0x60) >> 5;
730  csd->WrSpeedFact = (csd_buffer[12] & 0x1C) >> 2;
731  csd->MaxWrBlockLen = (csd_buffer[12] & 0x03) << 2;
732  /* Byte 13 */
733  csd->MaxWrBlockLen |= (csd_buffer[13] & 0xc0) >> 6;
734  csd->WriteBlockPaPartial = (csd_buffer[13] & 0x20) >> 5;
735  csd->Reserved3 = 0;
736  csd->ContentProtectAppli = (csd_buffer[13] & 0x01);
737  /* Byte 14 */
738  csd->FileFormatGrouop = (csd_buffer[14] & 0x80) >> 7;
739  csd->CopyFlag = (csd_buffer[14] & 0x40) >> 6;
740  csd->PermWrProtect = (csd_buffer[14] & 0x20) >> 5;
741  csd->TempWrProtect = (csd_buffer[14] & 0x10) >> 4;
742  csd->FileFormat = (csd_buffer[14] & 0x0C) >> 2;
743  csd->ECC = (csd_buffer[14] & 0x03);
744  /* Byte 15 */
745  csd->msd_CRC = (csd_buffer[15] & 0xFE) >> 1;
746  csd->Reserved4 = 1;
747 
748 error:
749  /* Deactivate chip select */
750  PIOS_SPI_RC_PinSet(PIOS_SDCARD_SPI, 0, 1); /* spi, pin_value */
751  /* Send dummy byte once deactivated to drop cards DO */
752  PIOS_SPI_TransferByte(PIOS_SDCARD_SPI, 0xff);
753  SDCARD_MUTEX_GIVE;
754 
755  return status;
756 }

863 int32_t PIOS_SDCARD_GetFree(void)
864 {
865  uint32_t VolFreeBytes = 0xffffffff;
866  uint32_t ScratchCache = 0;
867 
868  /* This takes very long, since it scans ALL clusters */
869  /* It takes ca. 7 seconds to determine free clusters out of ~47000 (2Gb) */
870 
871  /* Scan all the clusters */
872  for (uint32_t i = 2; i < PIOS_SDCARD_VolInfo.numclusters; ++i) {
873  if (!DFS_GetFAT(&PIOS_SDCARD_VolInfo, PIOS_SDCARD_Sector, &ScratchCache, i)) {
874  VolFreeBytes += PIOS_SDCARD_VolInfo.secperclus * SECTOR_SIZE;
875  }
876  }
877 
878  return VolFreeBytes;

948 int32_t PIOS_SDCARD_FileCopy(char *Source, char *Destination)
949 {
950  FILEINFO SourceFile, DestFile;
951 
952  /* Disable caching to avoid file inconsistencies while using different sector buffers! */
953  DFS_CachingEnabledSet(0);
954 
955  if (DFS_OpenFile(&PIOS_SDCARD_VolInfo, (uint8_t *) Source, DFS_READ, PIOS_SDCARD_Sector, &SourceFile)) {
956  /* Source file doesn't exist */
957  return -1;
958  } else {
959  /* Delete destination file if it already exists - ignore errors */
960  DFS_UnlinkFile(&PIOS_SDCARD_VolInfo, (uint8_t *) Destination, PIOS_SDCARD_Sector);
961 
962  if (DFS_OpenFile(&PIOS_SDCARD_VolInfo, (uint8_t *) Destination, DFS_WRITE, PIOS_SDCARD_Sector, &DestFile)) {
963  /* Failed to create destination file */
964  return -2;
965  }
966  }
967 
968  /* Copy operation */
969  uint8_t WriteBuffer[SECTOR_SIZE];
970  uint32_t SuccessCountRead;
971  uint32_t SuccessCountWrite;
972  do {
973  if (DFS_ReadFile(&SourceFile, PIOS_SDCARD_Sector, WriteBuffer, &SuccessCountRead, SECTOR_SIZE)) {
974  /* DFS_ReadFile failed */
975  return -3;
976  } else if (DFS_WriteFile(&DestFile, PIOS_SDCARD_Sector, WriteBuffer, &SuccessCountWrite, SuccessCountRead)) {
977  /* DFS_WriteFile failed */
978  return -4;
979  }
980  } while (SuccessCountRead > 0);
981 
982  /* No errors */
983  return 0;
984 }
