00050 /*
00051  * Correct sensor drift, using the 3C approach by J. Cotton
00052  */
00053 void CottonComplementaryCorrection(float *accels, float *gyros, const float delT, GlobalAttitudeVariables *glblAtt, float *accel_err_b)
00054 {
00055         float grav_b[3];
00056         
00057         // Rotate normalized gravity reference vector to body frame and cross with measured acceleration
00058         grav_b[0] = -(2 * (glblAtt->q[1] * glblAtt->q[3] - glblAtt->q[0] * glblAtt->q[2]));
00059         grav_b[1] = -(2 * (glblAtt->q[2] * glblAtt->q[3] + glblAtt->q[0] * glblAtt->q[1]));
00060         grav_b[2] = -(glblAtt->q[0] * glblAtt->q[0] - glblAtt->q[1] * glblAtt->q[1] -
00061                                           glblAtt->q[2] * glblAtt->q[2] + glblAtt->q[3] * glblAtt->q[3]);
00062         CrossProduct((const float *)accels, (const float *)grav_b, accel_err_b);
00063         
00064         // Account for accel magnitude
00065         float accel_mag = VectorMagnitude(accels);
00066         if (accel_mag < 1.0e-3f)
00067                 return;
00068         
00069         // Normalize error vector
00070         accel_err_b[0] /= accel_mag;
00071         accel_err_b[1] /= accel_mag;
00072         accel_err_b[2] /= accel_mag;
00073         
00074         // Accumulate integral of error.  Scale here so that units are (deg/s) but accelKi has units of s
00075         glblAtt->gyro_correct_int[0] += accel_err_b[0] * glblAtt->accelKi;
00076         glblAtt->gyro_correct_int[1] += accel_err_b[1] * glblAtt->accelKi;
00077         
00078         // Because most crafts wont get enough information from gravity to zero yaw gyro, we try
00079         // and make it average zero (weakly)
00080         glblAtt->gyro_correct_int[2] += -gyros[2] * glblAtt->yawBiasRate;
00081         
00082         // In this step, correct rates based on proportional error. The integral component is applied in updateSensors
00083         gyros[0] += accel_err_b[0] * glblAtt->accelKp / delT;
00084         gyros[1] += accel_err_b[1] * glblAtt->accelKp / delT;
00085         gyros[2] += accel_err_b[2] * glblAtt->accelKp / delT;
00086 }
