61 /*
62  * Correct sensor drift, using the 3C approach by J. Cotton
63  */
64 void CottonComplementaryCorrection(float *accels, float *gyros, const float delT, GlobalAttitudeVariables *glblAtt, float *accel_err_b)
65 {
66  float grav_b[3];
67 
68  // Rotate normalized gravity reference vector to body frame and cross with measured acceleration
69  grav_b[0] = -(2 * (glblAtt->q[1] * glblAtt->q[3] - glblAtt->q[0] * glblAtt->q[2]));
70  grav_b[1] = -(2 * (glblAtt->q[2] * glblAtt->q[3] + glblAtt->q[0] * glblAtt->q[1]));
71  grav_b[2] = -(glblAtt->q[0] * glblAtt->q[0] - glblAtt->q[1] * glblAtt->q[1] -
72  glblAtt->q[2] * glblAtt->q[2] + glblAtt->q[3] * glblAtt->q[3]);
73  CrossProduct((const float *)accels, (const float *)grav_b, accel_err_b);
74 
75  // Account for accel magnitude
76  float accel_mag = VectorMagnitude(accels);
77  if (accel_mag < 1.0e-3f)
78  return;
79 
80  // Normalize error vector
81  accel_err_b[0] /= accel_mag;
82  accel_err_b[1] /= accel_mag;
83  accel_err_b[2] /= accel_mag;
84 
85  // Accumulate integral of error. Scale here so that units are (deg/s) but accelKi has units of s
86  glblAtt->gyro_correct_int[0] += accel_err_b[0] * glblAtt->accelKi;
87  glblAtt->gyro_correct_int[1] += accel_err_b[1] * glblAtt->accelKi;
88 
89  // Because most crafts wont get enough information from gravity to zero yaw gyro, we try
90  // and make it average zero (weakly)
91  glblAtt->gyro_correct_int[2] += -gyros[2] * glblAtt->yawBiasRate;
92 
93  // In this step, correct rates based on proportional error. The integral component is applied in updateSensors
94  gyros[0] += accel_err_b[0] * glblAtt->accelKp / delT;
95  gyros[1] += accel_err_b[1] * glblAtt->accelKp / delT;
96  gyros[2] += accel_err_b[2] * glblAtt->accelKp / delT;
97 }
