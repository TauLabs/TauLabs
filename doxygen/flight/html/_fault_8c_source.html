00040 static int32_t fault_initialize(void)
00041 {
00042 #ifdef MODULE_Fault_BUILTIN
00043         module_enabled = true;
00044 #else
00045         uint8_t module_state[MODULESETTINGS_STATE_NUMELEM];
00046         ModuleSettingsEnabledGet(module_state);
00047         if (module_state[MODULESETTINGS_STATE_FAULT] == MODULESETTINGS_STATE_ENABLED) {
00048                 module_enabled = true;
00049         } else {
00050                 module_enabled = false;
00051         }
00052 #endif
00053 
00054         /* Do this outside the module_enabled test so that it
00055          * can be changed even when the module has been disabled.
00056          * This is important so we can remove faults even when
00057          * we've booted in BootFault recovery mode with all optional
00058          * modules disabled.
00059          */
00060         FaultSettingsInitialize();
00061 
00062         if (module_enabled) {
00063                 FaultSettingsActivateFaultGet(&active_fault);
00064 
00065                 switch (active_fault) {
00066                 case FAULTSETTINGS_ACTIVATEFAULT_MODULEINITASSERT:
00067                         /* Simulate an assert during module init */
00068                         PIOS_Assert(0);
00069                         break;
00070                 case FAULTSETTINGS_ACTIVATEFAULT_INITOUTOFMEMORY:
00071                         /* Leak all available memory */
00072                         while (pvPortMalloc(10)) ;
00073                         break;
00074                 case FAULTSETTINGS_ACTIVATEFAULT_INITBUSERROR:
00075                         {
00076                                 /* Force a bad access */
00077                                 uint32_t * bad_ptr = (uint32_t *)0xFFFFFFFF;
00078                                 *bad_ptr = 0xAA55AA55;
00079                         }
00080                         break;
00081                 }
00082         }
00083 
00084         return 0;
00085 }
