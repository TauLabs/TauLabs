90 int32_t PIOS_TIM_InitClock(const struct pios_tim_clock_cfg * cfg)
91 {
92  PIOS_DEBUG_Assert(cfg);
93 
94  /* Configure the dividers for this timer */
95  TIM_TimeBaseInit(cfg->timer, (TIM_TimeBaseInitTypeDef*)cfg->time_base_init);
96 
97  /* Configure internal timer clocks */
98  TIM_InternalClockConfig(cfg->timer);
99 
100  /* Enable timers */
101  TIM_Cmd(cfg->timer, ENABLE);
102 
103  /* Enable Interrupts */
104  NVIC_Init((NVIC_InitTypeDef*)&cfg->irq.init);
105 
106  /* Check for optional second vector (dirty hack)
107  * This is needed for timers 1 and 8 when requiring more than one event
108  * to generate an interrupt. Actually up to 4 interrupts may be necessary.
109  */
110  if (cfg->irq2.init.NVIC_IRQChannel != 0)
111  NVIC_Init((NVIC_InitTypeDef*)&cfg->irq2.init);
112 
113  return 0;
114 }

116 int32_t PIOS_TIM_InitChannels(uint32_t * tim_id, const struct pios_tim_channel * channels, uint8_t num_channels, const struct pios_tim_callbacks * callbacks, uint32_t context)
117 {
118  PIOS_Assert(channels);
119  PIOS_Assert(num_channels);
120 
121  struct pios_tim_dev * tim_dev;
122  tim_dev = (struct pios_tim_dev *) PIOS_TIM_alloc();
123  if (!tim_dev) goto out_fail;
124 
125  /* Bind the configuration to the device instance */
126  tim_dev->channels = channels;
127  tim_dev->num_channels = num_channels;
128  tim_dev->callbacks = callbacks;
129  tim_dev->context = context;
130 
131  /* Configure the pins */
132  for (uint8_t i = 0; i < num_channels; i++) {
133  const struct pios_tim_channel * chan = &(channels[i]);
134 
135  GPIO_Init(chan->pin.gpio, (GPIO_InitTypeDef*)&chan->pin.init);
136 
137  PIOS_DEBUG_Assert(chan->remap);
138 
139  // Second parameter should technically be PinSource but they are numerically the same
140  GPIO_PinAFConfig(chan->pin.gpio, chan->pin.pin_source,chan->remap);
141  }
142 
143  *tim_id = (uint32_t)tim_dev;
144 
145  return(0);
146 
147 out_fail:
148  return(-1);
149 }

151 static void PIOS_TIM_generic_irq_handler(TIM_TypeDef * timer)
152 {
153  /* Iterate over all registered clients of the TIM layer to find channels on this timer */
154  for (uint8_t i = 0; i < pios_tim_num_devs; i++) {
155  const struct pios_tim_dev * tim_dev = &pios_tim_devs[i];
156 
157  if (!tim_dev->channels || tim_dev->num_channels == 0) {
158  /* No channels to process on this client */
159  continue;
160  }
161 
162  /* Check for an overflow event on this timer */
163  bool overflow_event;
164  uint16_t overflow_count;
165  if (TIM_GetITStatus(timer, TIM_IT_Update) == SET) {
166  TIM_ClearITPendingBit(timer, TIM_IT_Update);
167  overflow_count = timer->ARR;
168  overflow_event = true;
169  } else {
170  overflow_count = 0;
171  overflow_event = false;
172  }
173 
174  for (uint8_t j = 0; j < tim_dev->num_channels; j++) {
175  const struct pios_tim_channel * chan = &tim_dev->channels[j];
176 
177  if (chan->timer != timer) {
178  /* channel is not on this timer */
179  continue;
180  }
181 
182  /* Figure out which interrupt bit we should be looking at */
183  uint16_t timer_it;
184  switch (chan->timer_chan) {
185  case TIM_Channel_1:
186  timer_it = TIM_IT_CC1;
187  break;
188  case TIM_Channel_2:
189  timer_it = TIM_IT_CC2;
190  break;
191  case TIM_Channel_3:
192  timer_it = TIM_IT_CC3;
193  break;
194  case TIM_Channel_4:
195  timer_it = TIM_IT_CC4;
196  break;
197  default:
198  PIOS_Assert(0);
199  break;
200  }
201 
202  bool edge_event;
203  uint16_t edge_count;
204  if (TIM_GetITStatus(chan->timer, timer_it) == SET) {
205  TIM_ClearITPendingBit(chan->timer, timer_it);
206 
207  /* Read the current counter */
208  switch(chan->timer_chan) {
209  case TIM_Channel_1:
210  edge_count = TIM_GetCapture1(chan->timer);
211  break;
212  case TIM_Channel_2:
213  edge_count = TIM_GetCapture2(chan->timer);
214  break;
215  case TIM_Channel_3:
216  edge_count = TIM_GetCapture3(chan->timer);
217  break;
218  case TIM_Channel_4:
219  edge_count = TIM_GetCapture4(chan->timer);
220  break;
221  default:
222  PIOS_Assert(0);
223  break;
224  }
225  edge_event = true;
226  } else {
227  edge_event = false;
228  edge_count = 0;
229  }
230 
231  if (!tim_dev->callbacks) {
232  /* No callbacks registered, we're done with this channel */
233  continue;
234  }
235 
236  /* Generate the appropriate callbacks */
237  if (overflow_event & edge_event) {
238  /*
239  * When both edge and overflow happen in the same interrupt, we
240  * need a heuristic to determine the order of the edge and overflow
241  * events so that the callbacks happen in the right order. If we
242  * get the order wrong, our pulse width calculations could be off by up
243  * to ARR ticks. That could be bad.
244  *
245  * Heuristic: If the edge_count is < 16 ticks above zero then we assume the
246  * edge happened just after the overflow.
247  */
248 
249  if (edge_count < 16) {
250  /* Call the overflow callback first */
251  if (tim_dev->callbacks->overflow) {
252  (*tim_dev->callbacks->overflow)((uint32_t)tim_dev,
253  tim_dev->context,
254  j,
255  overflow_count);
256  }
257  /* Call the edge callback second */
258  if (tim_dev->callbacks->edge) {
259  (*tim_dev->callbacks->edge)((uint32_t)tim_dev,
260  tim_dev->context,
261  j,
262  edge_count);
263  }
264  } else {
265  /* Call the edge callback first */
266  if (tim_dev->callbacks->edge) {
267  (*tim_dev->callbacks->edge)((uint32_t)tim_dev,
268  tim_dev->context,
269  j,
270  edge_count);
271  }
272  /* Call the overflow callback second */
273  if (tim_dev->callbacks->overflow) {
274  (*tim_dev->callbacks->overflow)((uint32_t)tim_dev,
275  tim_dev->context,
276  j,
277  overflow_count);
278  }
279  }
280  } else if (overflow_event && tim_dev->callbacks->overflow) {
281  (*tim_dev->callbacks->overflow)((uint32_t)tim_dev,
282  tim_dev->context,
283  j,
284  overflow_count);
285  } else if (edge_event && tim_dev->callbacks->edge) {
286  (*tim_dev->callbacks->edge)((uint32_t)tim_dev,
287  tim_dev->context,
288  j,
289  edge_count);
290  }
291  }
292  }
293 }
