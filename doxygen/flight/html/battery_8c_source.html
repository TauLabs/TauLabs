78 int32_t BatteryInitialize(void)
79 {
80  ADCRoutingInitialize();
81 
82 #ifdef MODULE_Battery_BUILTIN
83  module_enabled = true;
84 #else
85  uint8_t module_state[MODULESETTINGS_ADMINSTATE_NUMELEM];
86  ModuleSettingsAdminStateGet(module_state);
87  if (module_state[MODULESETTINGS_ADMINSTATE_BATTERY] == MODULESETTINGS_ADMINSTATE_ENABLED) {
88  module_enabled = true;
89  } else {
90  module_enabled = false;
91  }
92 #endif
93 
94  uint8_t adc_channel_map[ADCROUTING_CHANNELMAP_NUMELEM];
95  ADCRoutingChannelMapGet(adc_channel_map);
96 
97  //Determine if the battery sensors are routed to ADC pins
98  for (int i = 0; i < ADCROUTING_CHANNELMAP_NUMELEM; i++) {
99  if (adc_channel_map[i] == ADCROUTING_CHANNELMAP_BATTERYVOLTAGE) {
100  voltageADCPin = i;
101  }
102  if (adc_channel_map[i] == ADCROUTING_CHANNELMAP_BATTERYCURRENT) {
103  currentADCPin = i;
104  }
105  }
106 
107  //Don't enable module if no ADC pins are routed to the sensors
108  if (voltageADCPin <0 && currentADCPin <0)
109  module_enabled = false;
110 
111  //Start module
112  if (module_enabled) {
113  FlightBatteryStateInitialize();
114  FlightBatterySettingsInitialize();
115 
116  static UAVObjEvent ev;
117 
118  memset(&ev,0,sizeof(UAVObjEvent));
119  EventPeriodicCallbackCreate(&ev, onTimer, SAMPLE_PERIOD_MS / portTICK_RATE_MS);
120  }
121 
122  return 0;
123 }

127 static void onTimer(UAVObjEvent* ev)
128 {
129  static FlightBatteryStateData flightBatteryData;
130  FlightBatterySettingsData batterySettings;
131 
132  FlightBatterySettingsGet(&batterySettings);
133 
134  static float dT = SAMPLE_PERIOD_MS / 1000.0f;
135  float energyRemaining;
136 
137  //calculate the battery parameters
138  if (voltageADCPin >=0) {
139  flightBatteryData.Voltage = ((float)PIOS_ADC_PinGet(voltageADCPin)) * batterySettings.SensorCalibrations[FLIGHTBATTERYSETTINGS_SENSORCALIBRATIONS_VOLTAGEFACTOR]; //in Volts
140  }
141  else {
142  flightBatteryData.Voltage=1234; //Dummy placeholder value. This is in case we get another source of battery current which is not from the ADC
143  }
144 
145  if (currentADCPin >=0) {
146  flightBatteryData.Current = ((float)PIOS_ADC_PinGet(currentADCPin)) * batterySettings.SensorCalibrations[FLIGHTBATTERYSETTINGS_SENSORCALIBRATIONS_CURRENTFACTOR]; //in Amps
147  if (flightBatteryData.Current > flightBatteryData.PeakCurrent)
148  flightBatteryData.PeakCurrent = flightBatteryData.Current; //in Amps
149  }
150  else { //If there's no current measurement, we still need to assign one. Make it negative, so it can never trigger an alarm
151  flightBatteryData.Current=-0.1234f; //Dummy placeholder value. This is in case we get another source of battery current which is not from the ADC
152  }
153 
154  flightBatteryData.ConsumedEnergy += (flightBatteryData.Current * dT * 1000.0f / 3600.0f) ;//in mAh
155 
156  //Apply a 2 second rise time low-pass filter to average the current
157  float alpha = 1.0f-dT/(dT+2.0f);
158  flightBatteryData.AvgCurrent=alpha*flightBatteryData.AvgCurrent+(1-alpha)*flightBatteryData.Current; //in Amps
159 
160  /*The motor could regenerate power. Or we could have solar cells.
161  In short, is there any likelihood of measuring negative current? If it's a bad current reading we want to check, then
162  it makes sense to saturate at max and min values, because a misreading could as easily be very large, as negative. The simple
163  sign check doesn't catch this.*/
164 // //sanity checks
165 // if (flightBatteryData.AvgCurrent<0) flightBatteryData.AvgCurrent=0.0f;
166 // if (flightBatteryData.PeakCurrent<0) flightBatteryData.PeakCurrent=0.0f;
167 // if (flightBatteryData.ConsumedEnergy<0) flightBatteryData.ConsumedEnergy=0.0f;
168 
169  energyRemaining = batterySettings.Capacity - flightBatteryData.ConsumedEnergy; // in mAh
170  if (flightBatteryData.AvgCurrent > 0)
171  flightBatteryData.EstimatedFlightTime = (energyRemaining / (flightBatteryData.AvgCurrent*1000.0f))*3600.0f;//in Sec
172  else
173  flightBatteryData.EstimatedFlightTime = 9999;
174 
175  //generate alarms where needed...
176  if ((flightBatteryData.Voltage<=0) && (flightBatteryData.Current<=0))
177  {
178  //FIXME: There's no guarantee that a floating ADC will give 0. So this
179  // check might fail, even when there's nothing attached.
180  AlarmsSet(SYSTEMALARMS_ALARM_BATTERY, SYSTEMALARMS_ALARM_ERROR);
181  AlarmsSet(SYSTEMALARMS_ALARM_FLIGHTTIME, SYSTEMALARMS_ALARM_ERROR);
182  }
183  else
184  {
185  // FIXME: should make the timer alarms user configurable
186  if (flightBatteryData.EstimatedFlightTime < 30)
187  AlarmsSet(SYSTEMALARMS_ALARM_FLIGHTTIME, SYSTEMALARMS_ALARM_CRITICAL);
188  else if (flightBatteryData.EstimatedFlightTime < 120)
189  AlarmsSet(SYSTEMALARMS_ALARM_FLIGHTTIME, SYSTEMALARMS_ALARM_WARNING);
190  else
191  AlarmsClear(SYSTEMALARMS_ALARM_FLIGHTTIME);
192 
193  // FIXME: should make the battery voltage detection dependent on battery type.
194  /*Not so sure. Some users will want to run their batteries harder than others, so it should be the user's choice. [KDS]*/
195  if (flightBatteryData.Voltage < batterySettings.VoltageThresholds[FLIGHTBATTERYSETTINGS_VOLTAGETHRESHOLDS_ALARM])
196  AlarmsSet(SYSTEMALARMS_ALARM_BATTERY, SYSTEMALARMS_ALARM_CRITICAL);
197  else if (flightBatteryData.Voltage < batterySettings.VoltageThresholds[FLIGHTBATTERYSETTINGS_VOLTAGETHRESHOLDS_WARNING])
198  AlarmsSet(SYSTEMALARMS_ALARM_BATTERY, SYSTEMALARMS_ALARM_WARNING);
199  else
200  AlarmsClear(SYSTEMALARMS_ALARM_BATTERY);
201  }
202 
203  FlightBatteryStateSet(&flightBatteryData);
204 }
