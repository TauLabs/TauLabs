247 static struct UAVOData * UAVObjAllocSingle(uint32_t num_bytes)
248 {
249  /* Compute the complete size of the object, including the data for a single embedded instance */
250  uint32_t object_size = sizeof(struct UAVOSingle) + num_bytes;
251 
252  /* Allocate the object from the heap */
253  struct UAVOSingle * uavo_single = (struct UAVOSingle *) pvPortMalloc(object_size);
254  if (!uavo_single)
255  return (NULL);
256 
257  /* Fill in the common part of the UAVO */
258  struct UAVOBase * uavo_base = &(uavo_single->uavo.base);
259  memset(uavo_base, 0, sizeof(*uavo_base));
260  uavo_base->flags.isSingle = true;
261  uavo_base->next_event = NULL;
262 
263  /* Clear the instance data carried in the UAVO */
264  memset(&(uavo_single->instance0), 0, num_bytes);
265 
266  /* Give back the generic UAVO part */

310 UAVObjHandle UAVObjRegister(uint32_t id,
311  int32_t isSingleInstance, int32_t isSettings,
312  uint32_t num_bytes,
313  UAVObjInitializeCallback initCb)
314 {
315  struct UAVOData * uavo_data = NULL;
316 
317  xSemaphoreTakeRecursive(mutex, portMAX_DELAY);
318 
319  /* Don't allow duplicate registrations */
320  if (UAVObjGetByID(id))
321  goto unlock_exit;
322 
323  /* Map the various flags to one of the UAVO types we understand */
324  if (isSingleInstance) {
325  uavo_data = UAVObjAllocSingle (num_bytes);
326  } else {
327  uavo_data = UAVObjAllocMulti (num_bytes);
328  }
329 
330  if (!uavo_data)
331  goto unlock_exit;
332 
333  /* Fill in the details about this UAVO */
334  uavo_data->id = id;
335  uavo_data->instance_size = num_bytes;
336  if (isSettings) {
337  uavo_data->base.flags.isSettings = true;
338  }
339 
340  /* Initialize the embedded meta UAVO */
341  UAVObjInitMetaData (&uavo_data->metaObj);
342 
343  /* Add the newly created object to the global list of objects */
344  LL_APPEND(uavo_list, uavo_data);
345 
346  /* Initialize object fields and metadata to default values */
347  if (initCb)
348  initCb((UAVObjHandle) uavo_data, 0);
349 
350  /* Always try to load the meta object from flash */
351  UAVObjLoad((UAVObjHandle) &(uavo_data->metaObj), 0);
352 
353  /* Attempt to load settings object from flash */
354  if (uavo_data->base.flags.isSettings)
355  UAVObjLoad((UAVObjHandle) uavo_data, 0);
356 
357  // fire events for outer object and its embedded meta object
358  UAVObjInstanceUpdated((UAVObjHandle) uavo_data, 0);
359  UAVObjInstanceUpdated((UAVObjHandle) &(uavo_data->metaObj), 0);
360 
361 unlock_exit:
362  xSemaphoreGiveRecursive(mutex);
363  return (UAVObjHandle) uavo_data;
364 }

371 UAVObjHandle UAVObjGetByID(uint32_t id)
372 {
373  UAVObjHandle * found_obj = (UAVObjHandle *) NULL;
374 
375  // Get lock
376  xSemaphoreTakeRecursive(mutex, portMAX_DELAY);
377 
378  // Look for object
379  struct UAVOData * tmp_obj;
380  LL_FOREACH(uavo_list, tmp_obj) {
381  if (tmp_obj->id == id) {
382  found_obj = (UAVObjHandle *)tmp_obj;
383  goto unlock_exit;
384  }
385  if (MetaObjectId(tmp_obj->id) == id) {
386  found_obj = (UAVObjHandle *)&(tmp_obj->metaObj);
387  goto unlock_exit;
388  }
389  }
390 
391 unlock_exit:
392  xSemaphoreGiveRecursive(mutex);
393  return found_obj;
394 }

584 int32_t UAVObjUnpack(UAVObjHandle obj_handle, uint16_t instId,
585  const uint8_t * dataIn)
586 {
587  PIOS_Assert(obj_handle);
588 
589  // Lock
590  xSemaphoreTakeRecursive(mutex, portMAX_DELAY);
591 
592  int32_t rc = -1;
593 
594  if (UAVObjIsMetaobject(obj_handle)) {
595  if (instId != 0) {
596  goto unlock_exit;
597  }
598  memcpy(MetaDataPtr((struct UAVOMeta *)obj_handle), dataIn, MetaNumBytes);
599  } else {
600  struct UAVOData *obj;
601  InstanceHandle instEntry;
602 
603  // Cast handle to object
604  obj = (struct UAVOData *) obj_handle;
605 
606  // Get the instance
607  instEntry = getInstance(obj, instId);
608 
609  // If the instance does not exist create it and any other instances before it
610  if (instEntry == NULL) {
611  instEntry = createInstance(obj, instId);
612  if (instEntry == NULL) {
613  goto unlock_exit;
614  }
615  }
616  // Set the data
617  memcpy(InstanceData(instEntry), dataIn, obj->instance_size);
618  }
619 
620  // Fire event
621  sendEvent((struct UAVOBase*)obj_handle, instId, EV_UNPACKED);
622  rc = 0;
623 
624 unlock_exit:
625  xSemaphoreGiveRecursive(mutex);
626  return rc;
627 }

683 int32_t UAVObjSaveToFile(UAVObjHandle obj_handle, uint16_t instId,
684  FILEINFO * file)
685 {
686  PIOS_Assert(obj_handle);
687 
688  uint32_t bytesWritten;
689  // Check for file system availability
690  if (PIOS_SDCARD_IsMounted() == 0) {
691  return -1;
692  }
693  // Lock
694  xSemaphoreTakeRecursive(mutex, portMAX_DELAY);
695 
696  if (UAVObjIsMetaobject(obj_handle)) {
697  // Get the instance information
698  if (instId != 0) {
699  xSemaphoreGiveRecursive(mutex);
700  return -1;
701  }
702  // Write the object ID
703  uint32_t objId = UAVObjGetID(obj_handle);
704  PIOS_FWRITE(file, &objId, sizeof(objId),
705  &bytesWritten);
706 
707  // Write the data and check that the write was successful
708  PIOS_FWRITE(file, MetaDataPtr((struct UAVOMeta *)obj_handle), MetaNumBytes,
709  &bytesWritten);
710  if (bytesWritten != MetaNumBytes) {
711  xSemaphoreGiveRecursive(mutex);
712  return -1;
713  }
714  } else {
715  struct UAVOData * uavo;
716  InstanceHandle instEntry;
717 
718  // Cast to object
719  uavo = (struct UAVOData *) obj_handle;
720 
721  // Get the instance information
722  instEntry = getInstance(uavo, instId);
723  if (instEntry == NULL) {
724  xSemaphoreGiveRecursive(mutex);
725  return -1;
726  }
727  // Write the object ID
728  PIOS_FWRITE(file, &uavo->id, sizeof(uavo->id),
729  &bytesWritten);
730 
731  // Write the instance ID
732  if (!UAVObjIsSingleInstance(obj_handle)) {
733  PIOS_FWRITE(file, &instId,
734  sizeof(instId), &bytesWritten);
735  }
736  // Write the data and check that the write was successful
737  PIOS_FWRITE(file, InstanceData(instEntry), uavo->instance_size,
738  &bytesWritten);
739  if (bytesWritten != uavo->instance_size) {
740  xSemaphoreGiveRecursive(mutex);
741  return -1;
742  }
743  }
744  // Done
745  xSemaphoreGiveRecursive(mutex);
746  return 0;
747 }

760 int32_t UAVObjSave(UAVObjHandle obj_handle, uint16_t instId)
761 {
762  PIOS_Assert(obj_handle);
763 
764 #if defined(PIOS_INCLUDE_FLASH_SECTOR_SETTINGS)
765  if (UAVObjIsMetaobject(obj_handle)) {
766  if (instId != 0)
767  return -1;
768 
769  if (PIOS_FLASHFS_ObjSave(pios_uavo_settings_fs_id, UAVObjGetID(obj_handle), instId, (uint8_t*) MetaDataPtr((struct UAVOMeta *)obj_handle), UAVObjGetNumBytes(obj_handle)) != 0)
770  return -1;
771  } else {
772  InstanceHandle instEntry = getInstance( (struct UAVOData *)obj_handle, instId);
773 
774  if (instEntry == NULL)
775  return -1;
776 
777  if (InstanceData(instEntry) == NULL)
778  return -1;
779 
780  if (PIOS_FLASHFS_ObjSave(pios_uavo_settings_fs_id, UAVObjGetID(obj_handle), instId, InstanceData(instEntry), UAVObjGetNumBytes(obj_handle)) != 0)
781  return -1;
782  }
783 #endif
784 #if defined(PIOS_INCLUDE_SDCARD)
785  FILEINFO file;
786  uint8_t filename[14];
787 
788  // Check for file system availability
789  if (PIOS_SDCARD_IsMounted() == 0) {
790  return -1;
791  }

794 
795  // Get filename
796  objectFilename(UAVObjGetID(obj_handle), filename);
797 
798  // Open file
799  if (PIOS_FOPEN_WRITE(filename, file)) {
800  xSemaphoreGiveRecursive(mutex);

802  }
803  // Append object
804  if (UAVObjSaveToFile(obj_handle, instId, &file) == -1) {
805  PIOS_FCLOSE(file);
806  xSemaphoreGiveRecursive(mutex);
807  return -1;
808  }
809  // Done, close file and unlock
810  PIOS_FCLOSE(file);
811  xSemaphoreGiveRecursive(mutex);
812 #endif /* PIOS_INCLUDE_SDCARD */
813  return 0;

822 UAVObjHandle UAVObjLoadFromFile(FILEINFO * file)
823 {
824  uint32_t bytesRead;
825  struct UAVOBase *objEntry;
826  InstanceHandle instEntry;
827  uint32_t objId;
828  uint16_t instId;
829  UAVObjHandle obj_handle;
830 
831  // Check for file system availability
832  if (PIOS_SDCARD_IsMounted() == 0) {
833  return NULL;
834  }
835  // Lock
836  xSemaphoreTakeRecursive(mutex, portMAX_DELAY);
837 
838  // Read the object ID
839  if (PIOS_FREAD(file, &objId, sizeof(objId), &bytesRead)) {
840  xSemaphoreGiveRecursive(mutex);
841  return NULL;
842  }
843  // Get the object
844  obj_handle = UAVObjGetByID(objId);
845  if (obj_handle == 0) {
846  xSemaphoreGiveRecursive(mutex);
847  return NULL;
848  }
849  objEntry = (struct UAVOBase *) obj_handle;
850 
851  // Get the instance ID
852  instId = 0;
853  if (!UAVObjIsSingleInstance(obj_handle)) {
854  if (PIOS_FREAD
855  (file, &instId, sizeof(instId), &bytesRead)) {
856  xSemaphoreGiveRecursive(mutex);
857  return NULL;
858  }
859  }
860 
861  if (UAVObjIsMetaobject(obj_handle)) {
862  // If the instance does not exist create it and any other instances before it
863  if (instId != 0) {
864  // Error, unlock and return
865  xSemaphoreGiveRecursive(mutex);
866  return NULL;
867  }
868  // Read the instance data
869  if (PIOS_FREAD
870  (file, MetaDataPtr((struct UAVOMeta *)obj_handle), MetaNumBytes, &bytesRead)) {
871  xSemaphoreGiveRecursive(mutex);
872  return NULL;
873  }
874  } else {
875 
876  // Get the instance information
877  instEntry = getInstance((struct UAVOData *)objEntry, instId);
878 
879  // If the instance does not exist create it and any other instances before it
880  if (instEntry == NULL) {
881  instEntry = createInstance((struct UAVOData *)objEntry, instId);
882  if (instEntry == NULL) {
883  // Error, unlock and return
884  xSemaphoreGiveRecursive(mutex);
885  return NULL;
886  }
887  }
888  // Read the instance data
889  if (PIOS_FREAD
890  (file, InstanceData(instEntry), ((struct UAVOData *)objEntry)->instance_size, &bytesRead)) {
891  xSemaphoreGiveRecursive(mutex);
892  return NULL;
893  }
894 
895  }
896 
897  // Fire event
898  sendEvent(objEntry, instId, EV_UNPACKED);
899 
900  // Unlock
901  xSemaphoreGiveRecursive(mutex);
902  return obj_handle;
903 }

1804 static int32_t sendEvent(struct UAVOBase * obj, uint16_t instId,
1805  UAVObjEventType triggered_event)
1806 {
1807  /* Set up the message that will be sent to all registered listeners */
1808  UAVObjEvent msg = {
1809  .obj = (UAVObjHandle) obj,
1810  .event = triggered_event,
1811  .instId = instId,
1812  };
1813 
1814  // Go through each object and push the event message in the queue (if event is activated for the queue)
1815  struct ObjectEventEntry *event;
1816  LL_FOREACH(obj->next_event, event) {
1817  if (event->eventMask == 0
1818  || (event->eventMask & triggered_event) != 0) {
1819  // Send to queue if a valid queue is registered
1820  if (event->queue) {
1821  // will not block
1822  if (xQueueSend(event->queue, &msg, 0) != pdTRUE) {
1823  stats.lastQueueErrorID = UAVObjGetID(obj);
1824  ++stats.eventQueueErrors;
1825  }
1826  }
1827 
1828  // Invoke callback (from event task) if a valid one is registered
1829  if (event->cb) {
1830  // invoke callback from the event task, will not block
1831  if (EventCallbackDispatch(&msg, event->cb) != pdTRUE) {
1832  ++stats.eventCallbackErrors;
1833  stats.lastCallbackErrorID = UAVObjGetID(obj);
1834  }
1835  }
1836  }
1837  }
1838 
1839  return 0;
1840 }

1845 static InstanceHandle createInstance(struct UAVOData * obj, uint16_t instId)
1846 {
1847  struct UAVOMultiInst *instEntry;
1848 
1849  /* Don't allow more than one instance for single instance objects */
1850  if (UAVObjIsSingleInstance(&(obj->base))) {
1851  PIOS_Assert(0);
1852  return NULL;
1853  }
1854 
1855  /* Don't create more than the allowed number of instances */
1856  if (instId >= UAVOBJ_MAX_INSTANCES) {
1857  return NULL;
1858  }
1859 
1860  /* Don't allow duplicate instances */
1861  if (instId < UAVObjGetNumInstances(&(obj->base))) {
1862  return NULL;
1863  }
1864 
1865  // Create any missing instances (all instance IDs must be sequential)
1866  for (uint16_t n = UAVObjGetNumInstances(&(obj->base)); n < instId; ++n) {
1867  if (createInstance(obj, n) == NULL) {
1868  return NULL;
1869  }
1870  }
1871 
1872  /* Create the actual instance */
1873  instEntry = (struct UAVOMultiInst *) pvPortMalloc(sizeof(struct UAVOMultiInst)+obj->instance_size);
1874  if (!instEntry)
1875  return NULL;
1876  memset(InstanceDataOffset(instEntry), 0, obj->instance_size);
1877  LL_APPEND(( (struct UAVOMulti*)obj )->instance0.next, instEntry);
1878 
1879  ( (struct UAVOMulti*)obj )->num_instances++;
1880 
1881  // Fire event
1882  UAVObjInstanceUpdated((UAVObjHandle) obj, instId);
1883 
1884  // Done
1885  return InstanceDataOffset(instEntry);
1886 }

1941 static int32_t connectObj(UAVObjHandle obj_handle, xQueueHandle queue,
1942  UAVObjEventCallback cb, uint8_t eventMask)
1943 {
1944  struct ObjectEventEntry *event;
1945  struct UAVOBase *obj;
1946 
1947  // Check that the queue is not already connected, if it is simply update event mask
1948  obj = (struct UAVOBase *) obj_handle;
1949  LL_FOREACH(obj->next_event, event) {
1950  if (event->queue == queue && event->cb == cb) {
1951  // Already connected, update event mask and return
1952  event->eventMask = eventMask;
1953  return 0;
1954  }
1955  }
1956 
1957  // Add queue to list
1958  event = (struct ObjectEventEntry *) pvPortMalloc(sizeof(struct ObjectEventEntry));
1959  if (event == NULL) {
1960  return -1;
1961  }
1962  event->queue = queue;
1963  event->cb = cb;
1964  event->eventMask = eventMask;
1965  LL_APPEND(obj->next_event, event);
1966 
1967  // Done
1968  return 0;
1969 }

1978 static int32_t disconnectObj(UAVObjHandle obj_handle, xQueueHandle queue,
1979  UAVObjEventCallback cb)
1980 {
1981  struct ObjectEventEntry *event;
1982  struct UAVOBase *obj;
1983 
1984  // Find queue and remove it
1985  obj = (struct UAVOBase *) obj_handle;
1986  LL_FOREACH(obj->next_event, event) {
1987  if ((event->queue == queue
1988  && event->cb == cb)) {
1989  LL_DELETE(obj->next_event, event);
1990  vPortFree(event);
1991  return 0;
1992  }
1993  }
1994 
1995  // If this point is reached the queue was not found
1996  return -1;
