54 /* initcalls are now grouped by functionality into separate
55  * subsections. Ordering inside the subsections is determined
56  * by link order.
57  *
58  * The `id' arg to __define_initcall() is needed so that multiple initcalls
59  * can point at the same handler without causing duplicate-symbol build errors.
60  */
61 
62 #define __define_initcall(level,fn,id) \
63  static initcall_t __initcall_##fn##id __attribute__((__used__)) \
64  __attribute__((__section__(".initcall" level ".init"))) = fn

66 #define __define_module_initcall(level, ifn, sfn) \
67  static initmodule_t __initcall_##fn __attribute__((__used__)) \
68  __attribute__((__section__(".initcall" level ".init"))) = { .fn_minit = ifn, .fn_tinit = sfn };
