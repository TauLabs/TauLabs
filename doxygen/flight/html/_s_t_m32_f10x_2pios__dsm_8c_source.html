00173 static int PIOS_DSM_UnrollChannels(struct pios_dsm_dev *dsm_dev)
00174 {
00175         struct pios_dsm_state *state = &(dsm_dev->state);
00176         uint8_t resolution;
00177 
00178 #ifdef DSM_LOST_FRAME_COUNTER
00179         /* increment the lost frame counter */
00180         uint8_t frames_lost = state->received_data[0];
00181         state->frames_lost += (frames_lost - state->frames_lost_last);
00182         state->frames_lost_last = frames_lost;
00183 #endif
00184 
00185         /* check the frame type assuming master satellite stream */
00186         uint8_t type = state->received_data[1];
00187         switch (type) {
00188         case 0x01:
00189         case 0x02:
00190         case 0x12:
00191                 /* DSM2, DSMJ stream */
00192                 if (dsm_dev->proto == PIOS_DSM_PROTO_DSM2) {
00193                         /* DSM2/DSMJ resolution is known from the header */
00194                         resolution = (type & DSM_DSM2_RES_MASK) ? 11 : 10;
00195                 } else {
00196                         /* DSMX resolution should explicitly be selected */
00197                         goto stream_error;
00198                 }
00199                 break;
00200         case 0xA2:
00201         case 0xB2:
00202                 /* DSMX stream */
00203                 if (dsm_dev->proto == PIOS_DSM_PROTO_DSMX10BIT) {
00204                         resolution = 10;
00205                 } else if (dsm_dev->proto == PIOS_DSM_PROTO_DSMX11BIT) {
00206                         resolution = 11;
00207                 } else {
00208                         /* DSMX resolution should explicitly be selected */
00209                         goto stream_error;
00210                 }
00211                 break;
00212         default:
00213                 /* unknown yet data stream */
00214                 goto stream_error;
00215         }
00216 
00217         /* unroll channels */
00218         uint8_t *s = &(state->received_data[2]);
00219         uint16_t mask = (resolution == 10) ? 0x03ff : 0x07ff;
00220 
00221         for (int i = 0; i < DSM_CHANNELS_PER_FRAME; i++) {
00222                 uint16_t word = ((uint16_t)s[0] << 8) | s[1];
00223                 s += 2;
00224 
00225                 /* skip empty channel slot */
00226                 if (word == 0xffff)
00227                         continue;
00228 
00229                 /* minimal data validation */
00230                 if ((i > 0) && (word & DSM_2ND_FRAME_MASK)) {
00231                         /* invalid frame data, ignore rest of the frame */
00232                         goto stream_error;
00233                 }
00234 
00235                 /* extract and save the channel value */
00236                 uint8_t channel_num = (word >> resolution) & 0x0f;
00237                 if (channel_num < PIOS_DSM_NUM_INPUTS)
00238                         state->channel_data[channel_num] = (word & mask);
00239         }
00240 
00241 #ifdef DSM_LOST_FRAME_COUNTER
00242         /* put lost frames counter into the last channel for debugging */
00243         state->channel_data[PIOS_DSM_NUM_INPUTS-1] = state->frames_lost;
00244 #endif
00245 
00246         /* all channels processed */
00247         return 0;
00248 
00249 stream_error:
00250         /* either DSM2 selected with DSMX stream found, or vice-versa */
00251         return -1;
00252 }
