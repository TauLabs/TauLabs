00274                 groundspeed = 0;

00277         velocityDesired.North = progress.path_direction[0] * groundspeed;
00278         velocityDesired.East = progress.path_direction[1] * groundspeed;
00279 
00280         float error_speed = progress.error * guidanceSettings.HorizontalPosPI[GROUNDPATHFOLLOWERSETTINGS_HORIZONTALPOSPI_KP];
00281         float correction_velocity[2] = {progress.correction_direction[0] * error_speed,
00282             progress.correction_direction[1] * error_speed};
00283 
00284         float total_vel = sqrtf(powf(correction_velocity[0],2) + powf(correction_velocity[1],2));
00285         float scale = 1;

00287                 scale = guidanceSettings.HorizontalVelMax / total_vel;
00288 
00289         // Currently not apply a PID loop for horizontal corrections
00290         velocityDesired.North += progress.correction_direction[0] * error_speed * scale;
00291         velocityDesired.East += progress.correction_direction[1] * error_speed * scale;
