51 int stabilization_relay_rate(float error, float *output, int axis, bool reinit)
52 {
53  RelayTuningData relay;
54  RelayTuningGet(&relay);
55 
56  static portTickType lastHighTime;
57  static portTickType lastLowTime;
58 
59  static float accum_sin, accum_cos;
60  static uint32_t accumulated = 0;
61 
62  const uint16_t DEGLITCH_TIME = 20; // ms
63  const float AMPLITUDE_ALPHA = 0.95;
64  const float PERIOD_ALPHA = 0.95;
65 
66  portTickType thisTime = xTaskGetTickCount();
67 
68  static bool rateRelayRunning[MAX_AXES];
69 
70  // This indicates the current estimate of the smoothed error. So when it is high
71  // we are waiting for it to go low.
72  static bool high = false;
73 
74  // On first run initialize estimates to something reasonable
75  if(reinit) {
76  rateRelayRunning[axis] = false;
77  relay.Period[axis] = 200;
78  relay.Gain[axis] = 0;
79 
80  accum_sin = 0;
81  accum_cos = 0;
82  accumulated = 0;
83 
84  // These should get reinitialized anyway
85  high = true;
86  lastHighTime = thisTime;
87  lastLowTime = thisTime;
88  RelayTuningSet(&relay);
89  }
90 
91 
92  RelayTuningSettingsData relaySettings;
93  RelayTuningSettingsGet(&relaySettings);
94 
95  // Compute output, simple threshold on error
96  *output = high ? relaySettings.Amplitude : -relaySettings.Amplitude;
97 
98  /**** The code below here is to estimate the properties of the oscillation ****/
99 
100  // Make sure the period can't go below limit
101  if (relay.Period[axis] < DEGLITCH_TIME)
102  relay.Period[axis] = DEGLITCH_TIME;
103 
104  // Project the error onto a sine and cosine of the same frequency
105  // to accumulate the average amplitude
106  int32_t dT = thisTime - lastHighTime;
107  float phase = ((float)360 * (float)dT) / relay.Period[axis];
108  if(phase >= 360)
109  phase = 0;
110  accum_sin += sin_lookup_deg(phase) * error;
111  accum_cos += sin_lookup_deg(phase + 90) * error;
112  accumulated ++;
113 
114  // Make sure we've had enough time since last transition then check for a change in the output
115  bool time_hysteresis = (high ? (thisTime - lastHighTime) : (thisTime - lastLowTime)) > DEGLITCH_TIME;
116 
117  if ( !high && time_hysteresis && error > relaySettings.HysteresisThresh ){
118  /* POSITIVE CROSSING DETECTED */
119 
120  float this_amplitude = 2 * sqrtf(accum_sin*accum_sin + accum_cos*accum_cos) / accumulated;
121  float this_gain = this_amplitude / relaySettings.Amplitude;
122 
123  accumulated = 0;
124  accum_sin = 0;
125  accum_cos = 0;
126 
127  if(rateRelayRunning[axis] == false) {
128  rateRelayRunning[axis] = true;
129  relay.Period[axis] = 200;
130  relay.Gain[axis] = 0;
131  } else {
132  // Low pass filter each amplitude and period
133  relay.Gain[axis] = relay.Gain[axis] * AMPLITUDE_ALPHA + this_gain * (1 - AMPLITUDE_ALPHA);
134  relay.Period[axis] = relay.Period[axis] * PERIOD_ALPHA + dT * (1 - PERIOD_ALPHA);
135  }
136  lastHighTime = thisTime;
137  high = true;
138  RelayTuningSet(&relay);
139 
140  } else if ( high && time_hysteresis && error < -relaySettings.HysteresisThresh ) {
141  /* FALLING CROSSING DETECTED */
142 
143  lastLowTime = thisTime;
144  high = false;
145 
146  }
147 
148  return 0;
149 }
