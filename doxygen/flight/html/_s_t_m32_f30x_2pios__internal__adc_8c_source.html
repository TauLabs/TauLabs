00029 /*
00030  * @note This is a stripped-down ADC driver intended primarily for sampling
00031  * voltage and current values.  Samples are averaged over the period between
00032  * fetches so that relatively accurate measurements can be obtained without
00033  * forcing higher-level logic to poll aggressively.
00034  *
00035  * @todo This module needs more work to be more generally useful.  It should
00036  * almost certainly grow callback support so that e.g. voltage and current readings
00037  * can be shipped out for coulomb counting purposes.  The F1xx interface presumes
00038  * use with analog sensors, but that implementation largely dominates the ADC
00039  * resources.  Rather than commit to a new API without a defined use case, we
00040  * should stick to our lightweight subset until we have a better idea of what's needed.
00041  */

00472 {

00475          * Accumulate sampled values.

00480                 increment = 2;

00498                                 adc_dev->channel_map[scan_index]->accumulator /= 2;
00499                                 adc_dev->channel_map[scan_index]->count /= 2;
00500                         }
