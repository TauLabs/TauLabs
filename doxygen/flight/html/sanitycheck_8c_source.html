00174 static int32_t check_stabilization_settings(int index, bool multirotor)
00175 {
00176         // Make sure the modes have identical sizes
00177         if (MANUALCONTROLSETTINGS_STABILIZATION1SETTINGS_NUMELEM != MANUALCONTROLSETTINGS_STABILIZATION2SETTINGS_NUMELEM ||
00178                 MANUALCONTROLSETTINGS_STABILIZATION1SETTINGS_NUMELEM != MANUALCONTROLSETTINGS_STABILIZATION3SETTINGS_NUMELEM)
00179                 return SYSTEMALARMS_CONFIGERROR_MULTIROTOR;
00180 
00181         uint8_t modes[MANUALCONTROLSETTINGS_STABILIZATION1SETTINGS_NUMELEM];
00182 
00183         // Get the different axis modes for this switch position
00184         switch(index) {
00185                 case 1:
00186                         ManualControlSettingsStabilization1SettingsGet(modes);
00187                         break;
00188                 case 2:
00189                         ManualControlSettingsStabilization2SettingsGet(modes);
00190                         break;
00191                 case 3:
00192                         ManualControlSettingsStabilization3SettingsGet(modes);
00193                         break;
00194                 default:
00195                         return SYSTEMALARMS_CONFIGERROR_NONE;
00196         }
00197 
00198         // For multirotors verify that nothing is set to "none"
00199         if (multirotor) {
00200                 for(uint32_t i = 0; i < NELEMENTS(modes); i++) {
00201                         if (modes[i] == MANUALCONTROLSETTINGS_STABILIZATION1SETTINGS_NONE)
00202                                 return SYSTEMALARMS_CONFIGERROR_MULTIROTOR;
00203 
00204                         // If this axis allows enabling an autotune behavior without the module
00205                         // running then set an alarm now that aututune module initializes the
00206                         // appropriate objects
00207                         if ((modes[i] == MANUALCONTROLSETTINGS_STABILIZATION1SETTINGS_RELAYRATE || 
00208                                 modes[i] == MANUALCONTROLSETTINGS_STABILIZATION1SETTINGS_RELAYATTITUDE) &&
00209                                 (!TaskMonitorQueryRunning(TASKINFO_RUNNING_AUTOTUNE)))
00210                                 return SYSTEMALARMS_CONFIGERROR_AUTOTUNE;
00211                 }
00212         }
00213 
00214         // Warning: This assumes that certain conditions in the XML file are met.  That 
00215         // MANUALCONTROLSETTINGS_STABILIZATION1SETTINGS_NONE has the same numeric value for each channel
00216         // and is the same for STABILIZATIONDESIRED_STABILIZATIONMODE_NONE
00217 
00218         return SYSTEMALARMS_CONFIGERROR_NONE;
00219 }
