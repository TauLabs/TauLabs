00410 static void updateStats()

00412         static portTickType lastTickCount = 0;
00413         SystemStatsData stats;
00414 
00415         // Get stats and update
00416         SystemStatsGet(&stats);
00417         stats.FlightTime = xTaskGetTickCount() * portTICK_RATE_MS;
00418 #if defined(ARCH_POSIX) || defined(ARCH_WIN32)
00419         // POSIX port of FreeRTOS doesn't have xPortGetFreeHeapSize()

00424

00426         stats.IRQStackRemaining = GetFreeIrqStackSize();

00428         // When idleCounterClear was not reset by the idle-task, it means the idle-task did not run
00429         if (idleCounterClear) {
00430                 idleCounter = 0;
00431         }
00432 
00433         portTickType now = xTaskGetTickCount();
00434         if (now > lastTickCount) {
00435                 uint32_t dT = (xTaskGetTickCount() - lastTickCount) * portTICK_RATE_MS; // in ms
00436 
00437                 // In the case of a slightly miscalibrated max idle count, make sure CPULoad does
00438                 // not go negative and set an alarm inappropriately.
00439                 float idleFraction = ((float)idleCounter / ((float)dT / 1000.0f)) / (float)IDLE_COUNTS_PER_SEC_AT_NO_LOAD;
00440                 if (idleFraction > 1)
00441                         stats.CPULoad = 0;
00442                 else
00443                         stats.CPULoad = 100 - roundf(100.0f * idleFraction);
00444         } //else: TickCount has wrapped, do not calc now
00445         lastTickCount = now;
00446         idleCounterClear = 1;
00447         
00448 #if defined(PIOS_INCLUDE_ADC) && defined(PIOS_ADC_USE_TEMP_SENSOR)
00449         float temp_voltage = 3.3 * PIOS_ADC_DevicePinGet(PIOS_INTERNAL_ADC, 0) / ((1 << 12) - 1);
00450         const float STM32_TEMP_V25 = 1.43; /* V */
00451         const float STM32_TEMP_AVG_SLOPE = 4.3; /* mV/C */
00452         stats.CPUTemp = (temp_voltage-STM32_TEMP_V25) * 1000 / STM32_TEMP_AVG_SLOPE + 25;
