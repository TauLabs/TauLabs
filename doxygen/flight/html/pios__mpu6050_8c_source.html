140 int32_t PIOS_MPU6050_Init(uint32_t i2c_id, uint8_t i2c_addr, const struct pios_mpu60x0_cfg *cfg)
141 {
142  pios_mpu6050_dev = PIOS_MPU6050_alloc();
143 
144  if (pios_mpu6050_dev == NULL)
145  return -1;
146 
147  pios_mpu6050_dev->i2c_id = i2c_id;
148  pios_mpu6050_dev->i2c_addr = i2c_addr;
149  pios_mpu6050_dev->cfg = cfg;
150 
151  /* Configure the MPU6050 Sensor */
152  PIOS_MPU6050_Config(cfg);
153 
154  int result = xTaskCreate(PIOS_MPU6050_Task, (const signed char *)"pios_mpu6050",
155  MPU6050_TASK_STACK, NULL, MPU6050_TASK_PRIORITY,
156  &pios_mpu6050_dev->TaskHandle);
157  PIOS_Assert(result == pdPASS);
158 
159  /* Set up EXTI line */
160  PIOS_EXTI_Init(cfg->exti_cfg);
161 
162 #if defined(PIOS_MPU6050_ACCEL)
163  PIOS_SENSORS_Register(PIOS_SENSOR_ACCEL, pios_mpu6050_dev->accel_queue);
164 #endif /* PIOS_MPU6050_ACCEL */
165 
166  PIOS_SENSORS_Register(PIOS_SENSOR_GYRO, pios_mpu6050_dev->gyro_queue);
167 
168  return 0;
169 }

177 static void PIOS_MPU6050_Config(struct pios_mpu60x0_cfg const *cfg)
178 {
179 #if defined(PIOS_MPU6050_SIMPLE_INIT_SEQUENCE)
180 
181  // Reset chip
182  PIOS_MPU6050_SetReg(PIOS_MPU60X0_PWR_MGMT_REG, PIOS_MPU60X0_PWRMGMT_IMU_RST);
183 
184  // Reset sensors signal path
185  PIOS_MPU6050_SetReg(PIOS_MPU60X0_USER_CTRL_REG, PIOS_MPU60X0_USERCTL_GYRO_RST);
186 
187  // Give chip some time to initialize
188  PIOS_DELAY_WaitmS(10);
189 
190  //Power management configuration
191  PIOS_MPU6050_SetReg(PIOS_MPU60X0_PWR_MGMT_REG, cfg->Pwr_mgmt_clk);
192 
193  // User control
194  PIOS_MPU6050_SetReg(PIOS_MPU60X0_USER_CTRL_REG, cfg->User_ctl);
195 
196  // Digital low-pass filter and scale
197  // set this before sample rate else sample rate calculation will fail
198  PIOS_MPU6050_SetLPF(cfg->default_filter);
199 
200  // Sample rate
201  PIOS_MPU6050_SetSampleRate(cfg->default_samplerate);
202 
203  // Set the gyro scale
204  PIOS_MPU6050_SetGyroRange(PIOS_MPU60X0_SCALE_500_DEG);
205 
206 #if defined(PIOS_MPU6050_ACCEL)
207  // Set the accel scale
208  PIOS_MPU6050_SetAccelRange(PIOS_MPU60X0_ACCEL_8G);
209 #endif /* PIOS_MPU6050_ACCEL */
210 
211  // Interrupt configuration
212  PIOS_MPU6050_SetReg(PIOS_MPU60X0_INT_CFG_REG, cfg->interrupt_cfg);
213 
214  // Interrupt enable
215  PIOS_MPU6050_SetReg(PIOS_MPU60X0_INT_EN_REG, cfg->interrupt_en);
216 
217 #else /* PIOS_MPU6050_SIMPLE_INIT_SEQUENCE */
218 
219  /* This init sequence should really be dropped in favor of something
220  * less redundant but it seems to be hard to get it running well
221  * on all different targets.
222  */
223 
224  // Reset chip
225  PIOS_MPU6050_SetReg(PIOS_MPU60X0_PWR_MGMT_REG, 0x80 | cfg->Pwr_mgmt_clk);
226  do {
227  PIOS_DELAY_WaitmS(5);
228  } while (PIOS_MPU6050_GetReg(PIOS_MPU60X0_PWR_MGMT_REG) & 0x80);
229 
230  PIOS_DELAY_WaitmS(25);
231 
232  // Reset chip and fifo
233  PIOS_MPU6050_SetReg(PIOS_MPU60X0_USER_CTRL_REG, 0x80 | 0x01 | 0x02 | 0x04);
234  do {
235  PIOS_DELAY_WaitmS(5);
236  } while (PIOS_MPU6050_GetReg(PIOS_MPU60X0_USER_CTRL_REG) & 0x07);
237 
238  PIOS_DELAY_WaitmS(25);
239 
240  //Power management configuration
241  PIOS_MPU6050_SetReg(PIOS_MPU60X0_PWR_MGMT_REG, cfg->Pwr_mgmt_clk);
242 
243  // Interrupt configuration
244  PIOS_MPU6050_SetReg(PIOS_MPU60X0_INT_CFG_REG, cfg->interrupt_cfg);
245 
246  // Interrupt enable
247  PIOS_MPU6050_SetReg(PIOS_MPU60X0_INT_EN_REG, cfg->interrupt_en);
248 
249 #if defined(PIOS_MPU6050_ACCEL)
250  // Set the accel scale
251  PIOS_MPU6050_SetAccelRange(PIOS_MPU60X0_ACCEL_8G);
252 #endif
253 
254  // Digital low-pass filter and scale
255  // set this before sample rate else sample rate calculation will fail
256  PIOS_MPU6050_SetLPF(cfg->default_filter);
257 
258  // Sample rate
259  PIOS_MPU6050_SetSampleRate(cfg->default_samplerate);
260 
261  // Set the gyro scale
262  PIOS_MPU6050_SetGyroRange(PIOS_MPU60X0_SCALE_500_DEG);
263 
264  // User control
265  PIOS_MPU6050_SetReg(PIOS_MPU60X0_USER_CTRL_REG, cfg->User_ctl);
266 
267  //Power management configuration
268  PIOS_MPU6050_SetReg(PIOS_MPU60X0_PWR_MGMT_REG, cfg->Pwr_mgmt_clk);
269 
270  // Interrupt configuration
271  PIOS_MPU6050_SetReg(PIOS_MPU60X0_INT_CFG_REG, cfg->interrupt_cfg);
272 
273  // Interrupt enable
274  PIOS_MPU6050_SetReg(PIOS_MPU60X0_INT_EN_REG, cfg->interrupt_en);
275 
276 #endif /* PIOS_MPU6050_SIMPLE_INIT_SEQUENCE */
277 }

305 void PIOS_MPU6050_SetSampleRate(uint16_t samplerate_hz)
306 {
307  uint16_t filter_frequency = 8000;
308 
309  if (pios_mpu6050_dev->filter != PIOS_MPU60X0_LOWPASS_256_HZ)
310  filter_frequency = 1000;
311 
312  // limit samplerate to filter frequency
313  if (samplerate_hz > filter_frequency)
314  samplerate_hz = filter_frequency;
315 
316  // calculate divisor, round to nearest integeter
317  int32_t divisor = (int32_t)(((float)filter_frequency / samplerate_hz) + 0.5f) - 1;
318 
319  // limit resulting divisor to register value range
320  if (divisor < 0)
321  divisor = 0;
322 
323  if (divisor > 0xff)
324  divisor = 0xff;
325 
326  PIOS_MPU6050_SetReg(PIOS_MPU60X0_SMPLRT_DIV_REG, (uint8_t)divisor);
327 }

515 static void PIOS_MPU6050_Task(void *parameters)
516 {
517  while (1) {
518  //Wait for data ready interrupt
519  if (xSemaphoreTake(pios_mpu6050_dev->data_ready_sema, portMAX_DELAY) != pdTRUE)
520  continue;
521 
522  enum {
523  IDX_ACCEL_XOUT_H = 0,
524  IDX_ACCEL_XOUT_L,
525  IDX_ACCEL_YOUT_H,
526  IDX_ACCEL_YOUT_L,
527  IDX_ACCEL_ZOUT_H,
528  IDX_ACCEL_ZOUT_L,
529  IDX_TEMP_OUT_H,
530  IDX_TEMP_OUT_L,
531  IDX_GYRO_XOUT_H,
532  IDX_GYRO_XOUT_L,
533  IDX_GYRO_YOUT_H,
534  IDX_GYRO_YOUT_L,
535  IDX_GYRO_ZOUT_H,
536  IDX_GYRO_ZOUT_L,
537  BUFFER_SIZE,
538  };
539 
540  uint8_t mpu6050_rec_buf[BUFFER_SIZE];
541 
542  if (PIOS_MPU6050_Read(PIOS_MPU60X0_ACCEL_X_OUT_MSB, mpu6050_rec_buf, sizeof(mpu6050_rec_buf)) < 0) {
543  continue;
544  }
545 
546  // Rotate the sensor to OP convention. The datasheet defines X as towards the right
547  // and Y as forward. OP convention transposes this. Also the Z is defined negatively
548  // to our convention
549 
550 #if defined(PIOS_MPU6050_ACCEL)
551 
552  // Currently we only support rotations on top so switch X/Y accordingly
553  struct pios_sensor_accel_data accel_data;
554  struct pios_sensor_gyro_data gyro_data;
555 
556  switch (pios_mpu6050_dev->cfg->orientation) {
557  case PIOS_MPU60X0_TOP_0DEG:
558  accel_data.y = (int16_t)(mpu6050_rec_buf[IDX_ACCEL_XOUT_H] << 8 | mpu6050_rec_buf[IDX_ACCEL_XOUT_L]);
559  accel_data.x = (int16_t)(mpu6050_rec_buf[IDX_ACCEL_YOUT_H] << 8 | mpu6050_rec_buf[IDX_ACCEL_YOUT_L]);
560  gyro_data.y = (int16_t)(mpu6050_rec_buf[IDX_GYRO_XOUT_H] << 8 | mpu6050_rec_buf[IDX_GYRO_XOUT_L]);
561  gyro_data.x = (int16_t)(mpu6050_rec_buf[IDX_GYRO_YOUT_H] << 8 | mpu6050_rec_buf[IDX_GYRO_YOUT_L]);
562  break;
563  case PIOS_MPU60X0_TOP_90DEG:
564  accel_data.y = - (int16_t)(mpu6050_rec_buf[IDX_ACCEL_YOUT_H] << 8 | mpu6050_rec_buf[IDX_ACCEL_YOUT_L]);
565  accel_data.x = (int16_t)(mpu6050_rec_buf[IDX_ACCEL_XOUT_H] << 8 | mpu6050_rec_buf[IDX_ACCEL_XOUT_L]);
566  gyro_data.y = - (int16_t)(mpu6050_rec_buf[IDX_GYRO_YOUT_H] << 8 | mpu6050_rec_buf[IDX_GYRO_YOUT_L]);
567  gyro_data.x = (int16_t)(mpu6050_rec_buf[IDX_GYRO_XOUT_H] << 8 | mpu6050_rec_buf[IDX_GYRO_XOUT_L]);
568  break;
569  case PIOS_MPU60X0_TOP_180DEG:
570  accel_data.y = - (int16_t)(mpu6050_rec_buf[IDX_ACCEL_XOUT_H] << 8 | mpu6050_rec_buf[IDX_ACCEL_XOUT_L]);
571  accel_data.x = - (int16_t)(mpu6050_rec_buf[IDX_ACCEL_YOUT_H] << 8 | mpu6050_rec_buf[IDX_ACCEL_YOUT_L]);
572  gyro_data.y = - (int16_t)(mpu6050_rec_buf[IDX_GYRO_XOUT_H] << 8 | mpu6050_rec_buf[IDX_GYRO_XOUT_L]);
573  gyro_data.x = - (int16_t)(mpu6050_rec_buf[IDX_GYRO_YOUT_H] << 8 | mpu6050_rec_buf[IDX_GYRO_YOUT_L]);
574  break;
575  case PIOS_MPU60X0_TOP_270DEG:
576  accel_data.y = (int16_t)(mpu6050_rec_buf[IDX_ACCEL_YOUT_H] << 8 | mpu6050_rec_buf[IDX_ACCEL_YOUT_L]);
577  accel_data.x = - (int16_t)(mpu6050_rec_buf[IDX_ACCEL_XOUT_H] << 8 | mpu6050_rec_buf[IDX_ACCEL_XOUT_L]);
578  gyro_data.y = (int16_t)(mpu6050_rec_buf[IDX_GYRO_YOUT_H] << 8 | mpu6050_rec_buf[IDX_GYRO_YOUT_L]);
579  gyro_data.x = - (int16_t)(mpu6050_rec_buf[IDX_GYRO_XOUT_H] << 8 | mpu6050_rec_buf[IDX_GYRO_XOUT_L]);
580  break;
581  }
582 
583  gyro_data.z = - (int16_t)(mpu6050_rec_buf[IDX_GYRO_ZOUT_H] << 8 | mpu6050_rec_buf[IDX_GYRO_ZOUT_L]);
584  accel_data.z = - (int16_t)(mpu6050_rec_buf[IDX_ACCEL_ZOUT_H] << 8 | mpu6050_rec_buf[IDX_ACCEL_ZOUT_L]);
585 
586  int16_t raw_temp = (int16_t)(mpu6050_rec_buf[IDX_TEMP_OUT_H] << 8 | mpu6050_rec_buf[IDX_TEMP_OUT_L]);
587  float temperature = 35.0f + ((float)raw_temp + 512.0f) / 340.0f;
588 
589  // Apply sensor scaling
590  float accel_scale = PIOS_MPU6050_GetAccelScale();
591  accel_data.x *= accel_scale;
592  accel_data.y *= accel_scale;
593  accel_data.z *= accel_scale;
594  accel_data.temperature = temperature;
595 
596  float gyro_scale = PIOS_MPU6050_GetGyroScale();
597  gyro_data.x *= gyro_scale;
598  gyro_data.y *= gyro_scale;
599  gyro_data.z *= gyro_scale;
600  gyro_data.temperature = temperature;
601 
602  xQueueSendToBack(pios_mpu6050_dev->accel_queue, (void *)&accel_data, 0);
603 
604  xQueueSendToBack(pios_mpu6050_dev->gyro_queue, (void *)&gyro_data, 0);
605 
606 #else
607 
608  struct pios_sensor_gyro_data gyro_data;
609 
610  switch (pios_mpu6050_dev->cfg->orientation) {
611  case PIOS_MPU60X0_TOP_0DEG:
