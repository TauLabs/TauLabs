00561                         continue;

00564                     IDX_ACCEL_XOUT_H = 0,
00565                     IDX_ACCEL_XOUT_L,

00575                     IDX_GYRO_YOUT_L,

00579                 };

00581                 uint8_t mpu6050_rec_buf[BUFFER_SIZE];
00582

00592

00595                 struct pios_sensor_gyro_data gyro_data;
00596

00598                 case PIOS_MPU60X0_TOP_0DEG:
00599                         accel_data.y = (int16_t)(mpu6050_rec_buf[IDX_ACCEL_XOUT_H] << 8 | mpu6050_rec_buf[IDX_ACCEL_XOUT_L]);

00601                         gyro_data.y  = (int16_t)(mpu6050_rec_buf[IDX_GYRO_XOUT_H] << 8 | mpu6050_rec_buf[IDX_GYRO_XOUT_L]);
00602                         gyro_data.x  = (int16_t)(mpu6050_rec_buf[IDX_GYRO_YOUT_H] << 8 | mpu6050_rec_buf[IDX_GYRO_YOUT_L]);
00603                         break;

00605                         accel_data.y = - (int16_t)(mpu6050_rec_buf[IDX_ACCEL_YOUT_H] << 8 | mpu6050_rec_buf[IDX_ACCEL_YOUT_L]);
00606                         accel_data.x = (int16_t)(mpu6050_rec_buf[IDX_ACCEL_XOUT_H] << 8 | mpu6050_rec_buf[IDX_ACCEL_XOUT_L]);
00607                         gyro_data.y  = - (int16_t)(mpu6050_rec_buf[IDX_GYRO_YOUT_H] << 8 | mpu6050_rec_buf[IDX_GYRO_YOUT_L]);
