68 void arm_fir_fast_q15(
69  const arm_fir_instance_q15 * S,
70  q15_t * pSrc,
71  q15_t * pDst,
72  uint32_t blockSize)
73 {
74  q15_t *pState = S->pState; /* State pointer */
75  q15_t *pCoeffs = S->pCoeffs; /* Coefficient pointer */
76  q15_t *pStateCurnt; /* Points to the current sample of the state */
77  q31_t acc0, acc1, acc2, acc3; /* Accumulators */
78  q15_t *pb; /* Temporary pointer for coefficient buffer */
79  q15_t *px; /* Temporary q31 pointer for SIMD state buffer accesses */
80  q31_t x0, x1, x2, c0; /* Temporary variables to hold SIMD state and coefficient values */
81  uint32_t numTaps = S->numTaps; /* Number of taps in the filter */
82  uint32_t tapCnt, blkCnt; /* Loop counters */
83 
84 
85  /* S->pState points to state array which contains previous frame (numTaps - 1) samples */
86  /* pStateCurnt points to the location where the new input data should be written */
87  pStateCurnt = &(S->pState[(numTaps - 1u)]);
88 
89  /* Apply loop unrolling and compute 4 output values simultaneously.
90  * The variables acc0 ... acc3 hold output values that are being computed:
91  *
92  * acc0 = b[numTaps-1] * x[n-numTaps-1] + b[numTaps-2] * x[n-numTaps-2] + b[numTaps-3] * x[n-numTaps-3] +...+ b[0] * x[0]
93  * acc1 = b[numTaps-1] * x[n-numTaps] + b[numTaps-2] * x[n-numTaps-1] + b[numTaps-3] * x[n-numTaps-2] +...+ b[0] * x[1]
94  * acc2 = b[numTaps-1] * x[n-numTaps+1] + b[numTaps-2] * x[n-numTaps] + b[numTaps-3] * x[n-numTaps-1] +...+ b[0] * x[2]
95  * acc3 = b[numTaps-1] * x[n-numTaps+2] + b[numTaps-2] * x[n-numTaps+1] + b[numTaps-3] * x[n-numTaps] +...+ b[0] * x[3]
96  */
97 
98  blkCnt = blockSize >> 2;
99 
100  /* First part of the processing with loop unrolling. Compute 4 outputs at a time.
101  ** a second loop below computes the remaining 1 to 3 samples. */
102  while(blkCnt > 0u)
103  {
104  /* Copy four new input samples into the state buffer.
105  ** Use 32-bit SIMD to move the 16-bit data. Only requires two copies. */
106  *pStateCurnt++ = *pSrc++;
107  *pStateCurnt++ = *pSrc++;
108  *pStateCurnt++ = *pSrc++;
109  *pStateCurnt++ = *pSrc++;
110 
111 
112  /* Set all accumulators to zero */
113  acc0 = 0;
114  acc1 = 0;
115  acc2 = 0;
116  acc3 = 0;
117 
118  /* Typecast q15_t pointer to q31_t pointer for state reading in q31_t */
119  px = pState;
120 
121  /* Typecast q15_t pointer to q31_t pointer for coefficient reading in q31_t */
122  pb = pCoeffs;
123 
124  /* Read the first two samples from the state buffer: x[n-N], x[n-N-1] */
125  x0 = *__SIMD32(px)++;
126 
127  /* Read the third and forth samples from the state buffer: x[n-N-2], x[n-N-3] */
128  x2 = *__SIMD32(px)++;
129 
130  /* Loop over the number of taps. Unroll by a factor of 4.
131  ** Repeat until we've computed numTaps-(numTaps%4) coefficients. */
132  tapCnt = numTaps >> 2;
133 
134  while(tapCnt > 0)
135  {
136  /* Read the first two coefficients using SIMD: b[N] and b[N-1] coefficients */
137  c0 = *__SIMD32(pb)++;
138 
139  /* acc0 += b[N] * x[n-N] + b[N-1] * x[n-N-1] */
140  acc0 = __SMLAD(x0, c0, acc0);
141 
142  /* acc2 += b[N] * x[n-N-2] + b[N-1] * x[n-N-3] */
143  acc2 = __SMLAD(x2, c0, acc2);
144 
145  /* pack x[n-N-1] and x[n-N-2] */
146 #ifndef ARM_MATH_BIG_ENDIAN
147  x1 = __PKHBT(x2, x0, 0);
148 #else
149  x1 = __PKHBT(x0, x2, 0);
150 #endif
151 
152  /* Read state x[n-N-4], x[n-N-5] */
153  x0 = _SIMD32_OFFSET(px);
154 
155  /* acc1 += b[N] * x[n-N-1] + b[N-1] * x[n-N-2] */
156  acc1 = __SMLADX(x1, c0, acc1);
157 
158  /* pack x[n-N-3] and x[n-N-4] */
159 #ifndef ARM_MATH_BIG_ENDIAN
160  x1 = __PKHBT(x0, x2, 0);
161 #else
162  x1 = __PKHBT(x2, x0, 0);
163 #endif
164 
165  /* acc3 += b[N] * x[n-N-3] + b[N-1] * x[n-N-4] */
166  acc3 = __SMLADX(x1, c0, acc3);
167 
168  /* Read coefficients b[N-2], b[N-3] */
169  c0 = *__SIMD32(pb)++;
170 
171  /* acc0 += b[N-2] * x[n-N-2] + b[N-3] * x[n-N-3] */
172  acc0 = __SMLAD(x2, c0, acc0);
173 
174  /* Read state x[n-N-6], x[n-N-7] with offset */
175  x2 = _SIMD32_OFFSET(px + 2u);
176 
177  /* acc2 += b[N-2] * x[n-N-4] + b[N-3] * x[n-N-5] */
178  acc2 = __SMLAD(x0, c0, acc2);
179 
180  /* acc1 += b[N-2] * x[n-N-3] + b[N-3] * x[n-N-4] */
181  acc1 = __SMLADX(x1, c0, acc1);
182 
183  /* pack x[n-N-5] and x[n-N-6] */
184 #ifndef ARM_MATH_BIG_ENDIAN
185  x1 = __PKHBT(x2, x0, 0);
186 #else
187  x1 = __PKHBT(x0, x2, 0);
188 #endif
189 
190  /* acc3 += b[N-2] * x[n-N-5] + b[N-3] * x[n-N-6] */
191  acc3 = __SMLADX(x1, c0, acc3);
192 
193  /* Update state pointer for next state reading */
194  px += 4u;
195 
196  /* Decrement tap count */
197  tapCnt--;
198 
199  }
200 
201  /* If the filter length is not a multiple of 4, compute the remaining filter taps.
202  ** This is always be 2 taps since the filter length is even. */
203  if((numTaps & 0x3u) != 0u)
204  {
205 
206  /* Read last two coefficients */
207  c0 = *__SIMD32(pb)++;
208 
209  /* Perform the multiply-accumulates */
210  acc0 = __SMLAD(x0, c0, acc0);
211  acc2 = __SMLAD(x2, c0, acc2);
212 
213  /* pack state variables */
214 #ifndef ARM_MATH_BIG_ENDIAN
215  x1 = __PKHBT(x2, x0, 0);
216 #else
217  x1 = __PKHBT(x0, x2, 0);
218 #endif
219 
220  /* Read last state variables */
221  x0 = *__SIMD32(px);
222 
223  /* Perform the multiply-accumulates */
224  acc1 = __SMLADX(x1, c0, acc1);
225 
226  /* pack state variables */
227 #ifndef ARM_MATH_BIG_ENDIAN
228  x1 = __PKHBT(x0, x2, 0);
229 #else
230  x1 = __PKHBT(x2, x0, 0);
231 #endif
232 
233  /* Perform the multiply-accumulates */
234  acc3 = __SMLADX(x1, c0, acc3);
235  }
236 
237  /* The results in the 4 accumulators are in 2.30 format. Convert to 1.15 with saturation.
238  ** Then store the 4 outputs in the destination buffer. */
239 
240 #ifndef ARM_MATH_BIG_ENDIAN
241 
242  *__SIMD32(pDst)++ =
243  __PKHBT(__SSAT((acc0 >> 15), 16), __SSAT((acc1 >> 15), 16), 16);
244 
245  *__SIMD32(pDst)++ =
246  __PKHBT(__SSAT((acc2 >> 15), 16), __SSAT((acc3 >> 15), 16), 16);
247 
248 #else
249 
250  *__SIMD32(pDst)++ =
251  __PKHBT(__SSAT((acc1 >> 15), 16), __SSAT((acc0 >> 15), 16), 16);
252 
253  *__SIMD32(pDst)++ =
254  __PKHBT(__SSAT((acc3 >> 15), 16), __SSAT((acc2 >> 15), 16), 16);
255 
256 
257 #endif /* #ifndef ARM_MATH_BIG_ENDIAN */
258 
259  /* Advance the state pointer by 4 to process the next group of 4 samples */
260  pState = pState + 4u;
261 
262  /* Decrement the loop counter */
263  blkCnt--;
264  }
265 
266  /* If the blockSize is not a multiple of 4, compute any remaining output samples here.
267  ** No loop unrolling is used. */
268  blkCnt = blockSize % 0x4u;
269  while(blkCnt > 0u)
270  {
271  /* Copy two samples into state buffer */
272  *pStateCurnt++ = *pSrc++;
273 
274  /* Set the accumulator to zero */
275  acc0 = 0;
276 
277  /* Use SIMD to hold states and coefficients */
278  px = pState;
279  pb = pCoeffs;
280 
281  tapCnt = numTaps >> 1u;
282 
283  do
284  {
285 
286  acc0 += (q31_t) * px++ * *pb++;
287  acc0 += (q31_t) * px++ * *pb++;
288 
289  tapCnt--;
290  }
291  while(tapCnt > 0u);
292 
293  /* The result is in 2.30 format. Convert to 1.15 with saturation.
294  ** Then store the output in the destination buffer. */
295  *pDst++ = (q15_t) (__SSAT((acc0 >> 15), 16));
296 
297  /* Advance state pointer by 1 for the next sample */
298  pState = pState + 1u;
299 
300  /* Decrement the loop counter */
301  blkCnt--;
302  }
303 
304  /* Processing is complete.
305  ** Now copy the last numTaps - 1 samples to the satrt of the state buffer.
306  ** This prepares the state buffer for the next function call. */
307 
308  /* Points to the start of the state buffer */
309  pStateCurnt = S->pState;
310 
311  /* Calculation of count for copying integer writes */
312  tapCnt = (numTaps - 1u) >> 2;
313 
314  while(tapCnt > 0u)
315  {
316  *pStateCurnt++ = *pState++;
317  *pStateCurnt++ = *pState++;
318  *pStateCurnt++ = *pState++;
319  *pStateCurnt++ = *pState++;
320 
321  tapCnt--;
322 
323  }
324 
325  /* Calculation of count for remaining q15_t data */
326  tapCnt = (numTaps - 1u) % 0x4u;
327 
328  /* copy remaining data */
329  while(tapCnt > 0u)
330  {
331  *pStateCurnt++ = *pState++;
332 
333  /* Decrement the loop counter */
334  tapCnt--;
335  }
336 
337 }
