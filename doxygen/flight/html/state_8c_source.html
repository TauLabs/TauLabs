00382         } else {

00384         }
00385 
00386         if (retval < 0) {       // No sensor data.  Alarm set by calling method

00390         // Rotate sensor board into body frame

00392                 float tmpVec[3];

00395                 // the original vectors.
00396                 rot_mult(glblAtt->Rbs, prelim_accels, tmpVec, false);

00398 
00399                 rot_mult(glblAtt->Rbs, prelim_gyros, tmpVec, false);
00400                 memcpy(prelim_gyros, tmpVec, sizeof(tmpVec));
00401         }
00402 
00403         // Store rotated accels
00404         accels->x = prelim_accels[0];
00405         accels->y = prelim_accels[1];
00406         accels->z = prelim_accels[2];
00407 
00408         // Store rotated gyros, optionally with bias correction
00409         if (glblAtt->bias_correct_gyro) {
00410                 // Applying integral component here so it can be seen on the gyros and correct bias
00411                 gyros->x = prelim_gyros[0] + glblAtt->gyro_correct_int[0];
00412                 gyros->y = prelim_gyros[1] + glblAtt->gyro_correct_int[1];
00413                 gyros->z = prelim_gyros[2] + glblAtt->gyro_correct_int[2];
00414         } else {
00415                 gyros->x = prelim_gyros[0];
00416                 gyros->y = prelim_gyros[1];
00417                 gyros->z = prelim_gyros[2];
00418         }
00419 
00420         // Estimate accel bias while user flies level
00421         if (glblAtt->trim_requested) {
00422                 if (glblAtt->trim_samples >= MAX_TRIM_FLIGHT_SAMPLES) {
00423                         glblAtt->trim_requested = false;

00433                                 glblAtt->trim_accels[1] += accels->y;

00435                         }
00436                 }

00446 static void updateT3(GPSVelocityData * gpsVelocityData, PositionActualData * positionActualData)
00447 {
00448         //Load UAVOs
00449         VelocityActualData velocityActualData;
00450         VelocityActualGet(&velocityActualData);
00451 
00452         // Get a subset of the GPSPosition information
00453         int32_t LL_int[2];
00454         float altitude;
00455         float geoidSeparation;
00456         GPSPositionLatitudeGet(&LL_int[0]);
00457         GPSPositionLongitudeGet(&LL_int[1]);
00458         GPSPositionAltitudeGet(&altitude);
00459         GPSPositionGeoidSeparationGet(&geoidSeparation);
00460 
00461         // Get NED coordinates from GPS lat-lon
00462         float gps_NED[3];
00463         LLA2NED(LL_int, altitude, geoidSeparation, gps_NED);
00464 
00465         // Calculate filter coefficients
00466         float dT = .100f;
00467         float tauPosNorthEast = 0.3f;
00468         float tauPosDown = 0.5f;
00469         float tauVelNorthEast = 0.01f;
00470         float tauVelDown = 0.1f;
00471         float alphaPosNorthEast = dT / (dT + tauPosNorthEast);
00472         float alphaPosDown = dT / (dT + tauPosDown);
00473         float alphaVelNorthEast = dT / (dT + tauVelNorthEast);
00474         float alphaVelDown = dT / (dT + tauVelDown);
00475 
00476         //Low pass filter for velocity
00477         velocityActualData.North = (1 - alphaVelNorthEast) * velocityActualData.North +
00478             alphaVelNorthEast * gpsVelocityData->North;
00479         velocityActualData.East = (1 - alphaVelNorthEast) * velocityActualData.East +
00480             alphaVelNorthEast * gpsVelocityData->East;
00481         velocityActualData.Down = (1 - alphaVelDown) * velocityActualData.Down +
00482             alphaVelDown * gpsVelocityData->Down;
00483 
00484         //Complementary filter for position
00485         positionActualData->North = (1 - alphaPosNorthEast) * (positionActualData->North +
00486                                        (velocityActualData.North * dT)) + alphaPosNorthEast * gps_NED[0];
00487         positionActualData->East = (1 - alphaPosNorthEast) * (positionActualData->East +
00488                                        (velocityActualData.East * dT)) + alphaPosNorthEast * gps_NED[1];
00489         positionActualData->Down = (1 - alphaPosDown) * (positionActualData->Down +
00490                                   (velocityActualData.Down * dT)) + alphaPosDown * gps_NED[2];
00491 
00492         //Very slowly use GPS heading data to converge. This is a poor way of doing things, but will work in the short term for testing.
00493         if (fabs(velocityActualData.North) > 3.0f &&    //Instead of calculating velocity norm, use a gauge approach
00494                         (fabs(velocityActualData.North) > 4.0f || fabs(velocityActualData.East) > 3.0f)) {
00495                 float heading = atan2f(velocityActualData.East, velocityActualData.North);
00496 
00497                 AttitudeActualData attitudeActual;
00498                 AttitudeActualGet(&attitudeActual);
00499 
00500                 while (heading - attitudeActual.Yaw < -180.0f) {
00501                         heading += 360.0f;
00502                 }
00503                 while (heading - attitudeActual.Yaw > 180.0f) {
00504                         heading -= 360.0f;
00505                 }
00506 
00507                 attitudeActual.Yaw = .9f * attitudeActual.Yaw + 0.1f * heading;
00508 
00509                 // Convert into quaternions (makes assumptions about quaternions and RPY order)
00510                 RPY2Quaternion(&attitudeActual.Roll, &attitudeActual.q1);
00511 
00512                 if (!AttitudeActualReadOnly()) {
00513                         AttitudeActualSet(&attitudeActual);
00514                 }
00515         }
00516 
00517         // Do not update position and velocity estimates when in simulation mode
00518         PositionActualSet(positionActualData);
00519         VelocityActualSet(&velocityActualData);
00520 }

00525 static void updateSO3(float *gyros, float dT)
00526 {
00527         {
00528                 // scoping variables to save memory
00529                 // Work out time derivative from INSAlgo writeup
00530                 // Also accounts for the fact that gyros are in deg/s
00531                 float qdot[4];
00532                 qdot[0] =
00533                     (-glblAtt->q[1] * gyros[0] - glblAtt->q[2] * gyros[1] -
00534                      glblAtt->q[3] * gyros[2]) * dT * DEG2RAD / 2;
00535                 qdot[1] =
00536                     (glblAtt->q[0] * gyros[0] - glblAtt->q[3] * gyros[1] +
00537                      glblAtt->q[2] * gyros[2]) * dT * DEG2RAD / 2;
00538                 qdot[2] =
00539                     (glblAtt->q[3] * gyros[0] + glblAtt->q[0] * gyros[1] -
00540                      glblAtt->q[1] * gyros[2]) * dT * DEG2RAD / 2;
00541                 qdot[3] =
00542                     (-glblAtt->q[2] * gyros[0] + glblAtt->q[1] * gyros[1] +
00543                      glblAtt->q[0] * gyros[2]) * dT * DEG2RAD / 2;
00544 
00545                 // Integrate a time step
00546                 glblAtt->q[0] = glblAtt->q[0] + qdot[0];
00547                 glblAtt->q[1] = glblAtt->q[1] + qdot[1];
00548                 glblAtt->q[2] = glblAtt->q[2] + qdot[2];
00549                 glblAtt->q[3] = glblAtt->q[3] + qdot[3];
00550 
00551                 if (glblAtt->q[0] < 0) {
00552                         glblAtt->q[0] = -glblAtt->q[0];
00553                         glblAtt->q[1] = -glblAtt->q[1];
00554                         glblAtt->q[2] = -glblAtt->q[2];
00555                         glblAtt->q[3] = -glblAtt->q[3];
00556                 }
00557         }
00558 
00559         // Renomalize
00560         float qmag = sqrtf(powf(glblAtt->q[0], 2.0f) + powf(glblAtt->q[1], 2.0f) +
00561                   powf(glblAtt->q[2], 2.0f) + powf(glblAtt->q[3], 2.0f));
00562         glblAtt->q[0] = glblAtt->q[0] / qmag;
00563         glblAtt->q[1] = glblAtt->q[1] / qmag;
00564         glblAtt->q[2] = glblAtt->q[2] / qmag;
00565         glblAtt->q[3] = glblAtt->q[3] / qmag;
00566 
00567         // If quaternion has become inappropriately short or is nan reinit.
00568         // THIS SHOULD NEVER ACTUALLY HAPPEN
00569         if ((fabs(qmag) < 1e-3) || (qmag != qmag)) {
00570                 glblAtt->q[0] = 1;
00571                 glblAtt->q[1] = 0;
00572                 glblAtt->q[2] = 0;
00573                 glblAtt->q[3] = 0;
00574         }
00575 
00576         AttitudeActualData attitudeActual;
00577         AttitudeActualGet(&attitudeActual);
00578 
00579         quat_copy(glblAtt->q, &attitudeActual.q1);
00580 
00581         // Convert into Euler angles (makes assumptions about quarternions and RPY order)
00582         Quaternion2RPY(&attitudeActual.q1, &attitudeActual.Roll);
00583 
00584         AttitudeActualSet(&attitudeActual);
00585 }
