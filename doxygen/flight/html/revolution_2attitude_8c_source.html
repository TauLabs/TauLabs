270 static void AttitudeTask(void *parameters)
271 {
272  bool first_run = true;
273  uint32_t last_algorithm;
274  bool last_complementary;
275  AlarmsClear(SYSTEMALARMS_ALARM_ATTITUDE);
276 
277  // Force settings update to make sure rotation loaded
278  settingsUpdatedCb(NULL);
279 
280  // Wait for all the sensors be to read
281  vTaskDelay(100);
282 
283  // Invalidate previous algorithm to trigger a first run
284  last_algorithm = 0xfffffff;
285  last_complementary = false;
286 
287  // Main task loop
288  while (1) {
289 
290  int32_t ret_val = -1;
291 
292  // When changing the attitude filter reinitialize
293  if (last_algorithm != stateEstimation.AttitudeFilter) {
294  last_algorithm = stateEstimation.AttitudeFilter;
295  first_run = true;
296  }
297 
298  // There are two options to select:
299  // Attitude filter - what sets the attitude
300  // Navigation filter - what sets the position and velocity
301  // If the INS is used for either then it should run
302  bool ins = (stateEstimation.AttitudeFilter == STATEESTIMATION_ATTITUDEFILTER_INSOUTDOOR) ||
303  (stateEstimation.AttitudeFilter == STATEESTIMATION_ATTITUDEFILTER_INSINDOOR) ||
304  (stateEstimation.NavigationFilter == STATEESTIMATION_NAVIGATIONFILTER_INS);
305 
306  // INS outdoor mode when used for navigation OR explicit outdoor mode
307  bool outdoor = (stateEstimation.AttitudeFilter == STATEESTIMATION_ATTITUDEFILTER_INSOUTDOOR) ||
308  ((stateEstimation.AttitudeFilter == STATEESTIMATION_ATTITUDEFILTER_COMPLEMENTARY) &&
309  (stateEstimation.NavigationFilter == STATEESTIMATION_NAVIGATIONFILTER_INS));
310 
311  // Complementary filter only needed when used for attitude
312  bool complementary = stateEstimation.AttitudeFilter == STATEESTIMATION_ATTITUDEFILTER_COMPLEMENTARY;
313 
314  // Update one or both filters
315  if (ins) {
316  ret_val = updateAttitudeINSGPS(first_run, outdoor);
317  if (complementary)
318  updateAttitudeComplementary(first_run || complementary != last_complementary, true);
319  } else {
320  ret_val = updateAttitudeComplementary(first_run, false);
321  }
322 
323  last_complementary = complementary;
324 
325  // Get the requested data
326  // This function blocks on data queue
327  switch (stateEstimation.AttitudeFilter ) {
328  case STATEESTIMATION_ATTITUDEFILTER_COMPLEMENTARY:
329  setAttitudeComplementary();
330  break;
331  case STATEESTIMATION_ATTITUDEFILTER_INSOUTDOOR:
332  case STATEESTIMATION_ATTITUDEFILTER_INSINDOOR:
333  setAttitudeINSGPS();
334  break;
335  }
336 
337  // Use the selected source for position and velocity
338  switch (stateEstimation.NavigationFilter) {
339  case STATEESTIMATION_NAVIGATIONFILTER_INS:
340  // TODO: When running in dual mode and the INS is not initialized set
341  // an error here
342  setNavigationINSGPS();
343  break;
344  default:
345  setNavigationRaw();
346  }
347 
348  if(ret_val == 0)
349  first_run = false;
350 
351  PIOS_WDG_UpdateFlag(PIOS_WDG_ATTITUDE);
352  }
353 }

363 static int32_t updateAttitudeComplementary(bool first_run, bool secondary)
364 {
365  UAVObjEvent ev;
366  GyrosData gyrosData;
367  AccelsData accelsData;
368  static int32_t timeval;
369  float dT;
370 
371  // Wait until the accel and gyro object is updated, if a timeout then go to failsafe
372  if (!secondary && (
373  xQueueReceive(gyroQueue, &ev, FAILSAFE_TIMEOUT_MS / portTICK_RATE_MS) != pdTRUE ||
374  xQueueReceive(accelQueue, &ev, 1 / portTICK_RATE_MS) != pdTRUE ) )
375  {
376  // When one of these is updated so should the other
377  // Do not set attitude timeout warnings in simulation mode
378  if (!AttitudeActualReadOnly()){
379  AlarmsSet(SYSTEMALARMS_ALARM_ATTITUDE,SYSTEMALARMS_ALARM_WARNING);
380  return -1;
381  }
382  }
383 
384  AccelsGet(&accelsData);
385 
386  // When this algorithm is first run force it to a known condition
387  if(first_run) {
388  MagnetometerData magData;
389  magData.x = 100;
390  magData.y = 0;
391  magData.z = 0;
392 
393  // Wait for a mag reading if a magnetometer was registered
394  if (PIOS_SENSORS_GetQueue(PIOS_SENSOR_MAG) != NULL) {
395  if ( !secondary && xQueueReceive(magQueue, &ev, 20 / portTICK_RATE_MS) != pdTRUE ) {
396  return -1;
397  }
398  MagnetometerGet(&magData);
399  }
400 
401  float RPY[3];
402  float theta = atan2f(accelsData.x, -accelsData.z);
403  RPY[1] = theta * RAD2DEG;
404  RPY[0] = atan2f(-accelsData.y, -accelsData.z / cosf(theta)) * RAD2DEG;
405  RPY[2] = atan2f(-magData.y, magData.x) * RAD2DEG;
406  RPY2Quaternion(RPY, cf_q);
407 
408  complementary_filter_state.initialization = CF_POWERON;
409  complementary_filter_state.reset_timeval = PIOS_DELAY_GetRaw();
410  timeval = PIOS_DELAY_GetRaw();
411 
412  complementary_filter_state.arming_count = 0;
413 
414  return 0;
415  }
416 
417  FlightStatusData flightStatus;
418  FlightStatusGet(&flightStatus);
419 
420  uint32_t ms_since_reset = PIOS_DELAY_DiffuS(complementary_filter_state.reset_timeval) / 1000;
421  if (complementary_filter_state.initialization == CF_POWERON) {
422  // Wait one second before starting to initialize
423  complementary_filter_state.initialization =
424  (ms_since_reset > 1000) ?
425  CF_INITIALIZING :
426  CF_POWERON;
427  } else if(complementary_filter_state.initialization == CF_INITIALIZING &&
428  (ms_since_reset < 7000) &&
429  (ms_since_reset > 1000)) {
430 
431  // For first 7 seconds use accels to get gyro bias
432  attitudeSettings.AccelKp = 0.1f + 0.1f * (xTaskGetTickCount() < 4000);
433  attitudeSettings.AccelKi = 0.1f;
434  attitudeSettings.YawBiasRate = 0.1f;
435  attitudeSettings.MagKp = 0.1f;
436  } else if ((attitudeSettings.ZeroDuringArming == ATTITUDESETTINGS_ZERODURINGARMING_TRUE) &&
437  (flightStatus.Armed == FLIGHTSTATUS_ARMED_ARMING)) {
438 
439  // Use a rapidly decrease accelKp to force the attitude to snap back
440  // to level and then converge more smoothly
441  if (complementary_filter_state.arming_count < 20)
442  attitudeSettings.AccelKp = 1.0f;
443  else if (attitudeSettings.AccelKp > 0.1f)
444  attitudeSettings.AccelKp -= 0.01f;
445  complementary_filter_state.arming_count++;
446 
447  // Set the other parameters to drive faster convergence
448  attitudeSettings.AccelKi = 0.1f;
449  attitudeSettings.YawBiasRate = 0.1f;
450  attitudeSettings.MagKp = 0.1f;
451 
452  // Don't apply LPF to the accels during arming
453  complementary_filter_state.accel_filter_enabled = false;
454 
455  // Indicate arming so that after arming it reloads
456  // the normal settings
457  if (complementary_filter_state.initialization != CF_ARMING) {
458  accumulate_gyro_zero();
459  complementary_filter_state.initialization = CF_ARMING;
460  complementary_filter_state.accumulating_gyro = true;
461  }
462 
463  } else if (complementary_filter_state.initialization == CF_ARMING ||
464  complementary_filter_state.initialization == CF_INITIALIZING) {
465 
466  AttitudeSettingsGet(&attitudeSettings);
467  if(complementary_filter_state.accel_alpha > 0.0f)
468  complementary_filter_state.accel_filter_enabled = true;
469 
470  // If arming that means we were accumulating gyro
471  // samples. Compute new bias.
472  if (complementary_filter_state.initialization == CF_ARMING) {
473  accumulate_gyro_compute();
474  complementary_filter_state.accumulating_gyro = false;
475  complementary_filter_state.arming_count = 0;
476  }
477 
478  // Indicate normal mode to prevent rerunning this code
479  complementary_filter_state.initialization = CF_NORMAL;
480  }
481 
482  GyrosGet(&gyrosData);
483  accumulate_gyro(&gyrosData);
484 
485  // Compute the dT using the cpu clock
486  dT = PIOS_DELAY_DiffuS(timeval) / 1000000.0f;
487  timeval = PIOS_DELAY_GetRaw();
488 
489  float grot[3];
490  float accel_err[3];
491  float *grot_filtered = complementary_filter_state.grot_filtered;
492  float *accels_filtered = complementary_filter_state.accels_filtered;
493 
494  // Apply smoothing to accel values, to reduce vibration noise before main calculations.
495  apply_accel_filter(&accelsData.x,accels_filtered);
496 
497  // Rotate gravity to body frame and cross with accels
498  grot[0] = -(2 * (cf_q[1] * cf_q[3] - cf_q[0] * cf_q[2]));
499  grot[1] = -(2 * (cf_q[2] * cf_q[3] + cf_q[0] * cf_q[1]));
500  grot[2] = -(cf_q[0]*cf_q[0] - cf_q[1]*cf_q[1] - cf_q[2]*cf_q[2] + cf_q[3]*cf_q[3]);
501  CrossProduct((const float *) &accelsData.x, (const float *) grot, accel_err);
502 
503  // Apply same filtering to the rotated attitude to match delays
504  apply_accel_filter(grot,grot_filtered);
505 
506  // Compute the error between the predicted direction of gravity and smoothed acceleration
507  CrossProduct((const float *) accels_filtered, (const float *) grot_filtered, accel_err);
508 
509  float grot_mag;
510  if (complementary_filter_state.accel_filter_enabled)
511  grot_mag = sqrtf(grot_filtered[0]*grot_filtered[0] + grot_filtered[1]*grot_filtered[1] + grot_filtered[2]*grot_filtered[2]);
512  else
513  grot_mag = 1.0f;
514 
515  // Account for accel magnitude
516  float accel_mag;
517  accel_mag = accels_filtered[0]*accels_filtered[0] + accels_filtered[1]*accels_filtered[1] + accels_filtered[2]*accels_filtered[2];
518  accel_mag = sqrtf(accel_mag);
519  if (grot_mag > 1.0e-3f && accel_mag > 1.0e-3f) {
520  accel_err[0] /= (accel_mag * grot_mag);
521  accel_err[1] /= (accel_mag * grot_mag);
522  accel_err[2] /= (accel_mag * grot_mag);
523  } else {
524  accel_err[0] = 0;
525  accel_err[1] = 0;
526  accel_err[2] = 0;
527  }
528 
529  float mag_err[3];
530  if ( secondary || xQueueReceive(magQueue, &ev, 0) == pdTRUE )
531  {
532  // Rotate gravity to body frame and cross with accels
533  float brot[3];
534  float Rbe[3][3];
535  MagnetometerData mag;
536 
537  Quaternion2R(cf_q, Rbe);
538  MagnetometerGet(&mag);
539 
540  // If the mag is producing bad data don't use it (normally bad calibration)
541  if (mag.x == mag.x && mag.y == mag.y && mag.z == mag.z &&
542  homeLocation.Set == HOMELOCATION_SET_TRUE) {
543  rot_mult(Rbe, homeLocation.Be, brot, false);
544 
545  float mag_len = sqrtf(mag.x * mag.x + mag.y * mag.y + mag.z * mag.z);
546  mag.x /= mag_len;
547  mag.y /= mag_len;
548  mag.z /= mag_len;
549 
550  float bmag = sqrtf(brot[0] * brot[0] + brot[1] * brot[1] + brot[2] * brot[2]);
551  brot[0] /= bmag;
552  brot[1] /= bmag;
553  brot[2] /= bmag;
554 
555  // Only compute if neither vector is null
556  if (bmag < 1 || mag_len < 1)
557  mag_err[0] = mag_err[1] = mag_err[2] = 0;
558  else
559  CrossProduct((const float *) &mag.x, (const float *) brot, mag_err);
560 
561  if (mag_err[2] != mag_err[2])
562  mag_err[2] = 0;
563  } else
564  mag_err[2] = 0;
565  } else {
566  mag_err[2] = 0;
567  }
568 
569  // Accumulate integral of error. Scale here so that units are (deg/s) but Ki has units of s
570  GyrosBiasData gyrosBias;
571  GyrosBiasGet(&gyrosBias);
572  gyrosBias.x -= accel_err[0] * attitudeSettings.AccelKi;
573  gyrosBias.y -= accel_err[1] * attitudeSettings.AccelKi;
574  gyrosBias.z -= mag_err[2] * attitudeSettings.MagKi;
575  GyrosBiasSet(&gyrosBias);
576 
577  // Correct rates based on error, integral component dealt with in updateSensors
578  gyrosData.x += accel_err[0] * attitudeSettings.AccelKp / dT;
579  gyrosData.y += accel_err[1] * attitudeSettings.AccelKp / dT;
580  gyrosData.z += mag_err[2] * attitudeSettings.MagKp / dT;
581 
582  // Work out time derivative from INSAlgo writeup
583  // Also accounts for the fact that gyros are in deg/s
584  float qdot[4];
585  qdot[0] = (-cf_q[1] * gyrosData.x - cf_q[2] * gyrosData.y - cf_q[3] * gyrosData.z) * dT * DEG2RAD / 2;
586  qdot[1] = (cf_q[0] * gyrosData.x - cf_q[3] * gyrosData.y + cf_q[2] * gyrosData.z) * dT * DEG2RAD / 2;
587  qdot[2] = (cf_q[3] * gyrosData.x + cf_q[0] * gyrosData.y - cf_q[1] * gyrosData.z) * dT * DEG2RAD / 2;
588  qdot[3] = (-cf_q[2] * gyrosData.x + cf_q[1] * gyrosData.y + cf_q[0] * gyrosData.z) * dT * DEG2RAD / 2;
589 
590  // Take a time step
591  cf_q[0] = cf_q[0] + qdot[0];
592  cf_q[1] = cf_q[1] + qdot[1];
593  cf_q[2] = cf_q[2] + qdot[2];
594  cf_q[3] = cf_q[3] + qdot[3];
595 
596  if(cf_q[0] < 0) {
597  cf_q[0] = -cf_q[0];
598  cf_q[1] = -cf_q[1];
599  cf_q[2] = -cf_q[2];
600  cf_q[3] = -cf_q[3];
601  }
602 
603  // Renomalize
604  float qmag;
605  qmag = sqrtf(cf_q[0]*cf_q[0] + cf_q[1]*cf_q[1] + cf_q[2]*cf_q[2] + cf_q[3]*cf_q[3]);
606  cf_q[0] = cf_q[0] / qmag;
607  cf_q[1] = cf_q[1] / qmag;
608  cf_q[2] = cf_q[2] / qmag;
609  cf_q[3] = cf_q[3] / qmag;
610 
611  // If quaternion has become inappropriately short or is nan reinit.
612  // THIS SHOULD NEVER ACTUALLY HAPPEN
613  if((fabs(qmag) < 1.0e-3f) || (qmag != qmag)) {
614  cf_q[0] = 1;
615  cf_q[1] = 0;
616  cf_q[2] = 0;
617  cf_q[3] = 0;
618  }
619 
620  if (!secondary)
621  AlarmsClear(SYSTEMALARMS_ALARM_ATTITUDE);
622 
623  return 0;
624 }
