69  * Setup a com port based on the passed cfg, driver and buffer sizes. rx or tx size of 0 disables rx or tx
70  */
71 #if defined(PIOS_INCLUDE_USART) && defined(PIOS_INCLUDE_COM)
72 static void PIOS_Board_configure_com (const struct pios_usart_cfg *usart_port_cfg, size_t rx_buf_len, size_t tx_buf_len,
73  const struct pios_com_driver *com_driver, uintptr_t *pios_com_id)
74 {
75  uint32_t pios_usart_id;
76  if (PIOS_USART_Init(&pios_usart_id, usart_port_cfg)) {
77  PIOS_Assert(0);
78  }
79 
80  uint8_t * rx_buffer;
81  if (rx_buf_len > 0) {
82  rx_buffer = (uint8_t *) pvPortMalloc(rx_buf_len);
83  PIOS_Assert(rx_buffer);
84  } else {
85  rx_buffer = NULL;
86  }
87 
88  uint8_t * tx_buffer;
89  if (tx_buf_len > 0) {
90  tx_buffer = (uint8_t *) pvPortMalloc(tx_buf_len);
91  PIOS_Assert(tx_buffer);
92  } else {
93  tx_buffer = NULL;
94  }
95 
96  if (PIOS_COM_Init(pios_com_id, com_driver, pios_usart_id,
97  rx_buffer, rx_buf_len,
98  tx_buffer, tx_buf_len)) {
99  PIOS_Assert(0);
100  }
101 }
102 #endif /* PIOS_INCLUDE_USART && PIOS_INCLUDE_COM */
