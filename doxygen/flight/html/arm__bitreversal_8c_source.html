25  * @brief In-place bit reversal function.
26  * @param[in, out] *pSrc points to the in-place buffer of floating-point data type.
27  * @param[in] fftSize length of the FFT.
28  * @param[in] bitRevFactor bit reversal modifier that supports different size FFTs with the same bit reversal table.
29  * @param[in] *pBitRevTab points to the bit reversal table.
30  * @return none.
31  */
32 
33 void arm_bitreversal_f32(
34  float32_t * pSrc,
35  uint16_t fftSize,
36  uint16_t bitRevFactor,
37  uint16_t * pBitRevTab)
38 {
39  uint16_t fftLenBy2, fftLenBy2p1;
40  uint16_t i, j;
41  float32_t in;
42 
43  /* Initializations */
44  j = 0u;
45  fftLenBy2 = fftSize >> 1u;
46  fftLenBy2p1 = (fftSize >> 1u) + 1u;
47 
48  /* Bit Reversal Implementation */
49  for (i = 0u; i <= (fftLenBy2 - 2u); i += 2u)
50  {
51  if(i < j)
52  {
53  /* pSrc[i] <-> pSrc[j]; */
54  in = pSrc[2u * i];
55  pSrc[2u * i] = pSrc[2u * j];
56  pSrc[2u * j] = in;
57 
58  /* pSrc[i+1u] <-> pSrc[j+1u] */
59  in = pSrc[(2u * i) + 1u];
60  pSrc[(2u * i) + 1u] = pSrc[(2u * j) + 1u];
61  pSrc[(2u * j) + 1u] = in;
62 
63  /* pSrc[i+fftLenBy2p1] <-> pSrc[j+fftLenBy2p1] */
64  in = pSrc[2u * (i + fftLenBy2p1)];
65  pSrc[2u * (i + fftLenBy2p1)] = pSrc[2u * (j + fftLenBy2p1)];
66  pSrc[2u * (j + fftLenBy2p1)] = in;
67 
68  /* pSrc[i+fftLenBy2p1+1u] <-> pSrc[j+fftLenBy2p1+1u] */
69  in = pSrc[(2u * (i + fftLenBy2p1)) + 1u];
70  pSrc[(2u * (i + fftLenBy2p1)) + 1u] =
71  pSrc[(2u * (j + fftLenBy2p1)) + 1u];
72  pSrc[(2u * (j + fftLenBy2p1)) + 1u] = in;
73 
74  }
75 
76  /* pSrc[i+1u] <-> pSrc[j+1u] */
77  in = pSrc[2u * (i + 1u)];
78  pSrc[2u * (i + 1u)] = pSrc[2u * (j + fftLenBy2)];
79  pSrc[2u * (j + fftLenBy2)] = in;
80 
81  /* pSrc[i+2u] <-> pSrc[j+2u] */
82  in = pSrc[(2u * (i + 1u)) + 1u];
83  pSrc[(2u * (i + 1u)) + 1u] = pSrc[(2u * (j + fftLenBy2)) + 1u];
84  pSrc[(2u * (j + fftLenBy2)) + 1u] = in;
85 
86  /* Reading the index for the bit reversal */
87  j = *pBitRevTab;
88 
89  /* Updating the bit reversal index depending on the fft length */
90  pBitRevTab += bitRevFactor;
91  }
92 }
