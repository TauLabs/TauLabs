48 {

52 #ifdef PIOS_INCLUDE_FREERTOS

54 #else
55  PIOS_DELAY_WaitmS(20);

62 static enum opahrs_result opahrs_msg_v1_send_req(const struct opahrs_msg_v1 *req)
63 {
64  int32_t rc;
65  struct opahrs_msg_v1 link_rx;
66 
67  for (uint8_t retries = 0; retries < 20; retries++) {
68  struct opahrs_msg_v1 *rsp = &link_rx;
69 
70  if ((rc = opahrs_msg_txrx((const uint8_t *)req, (uint8_t *) rsp, sizeof(*rsp))) < 0) {
71  return OPAHRS_RESULT_FAILED;
72  }
73 
74  /* Make sure we got a sane response by checking the magic */
75  if ((rsp->head.magic != OPAHRS_MSG_MAGIC_HEAD) || (rsp->tail.magic != OPAHRS_MSG_MAGIC_TAIL)) {
76  return OPAHRS_RESULT_FAILED;
77  }
78 
79  switch (rsp->head.type) {
80  case OPAHRS_MSG_TYPE_LINK:
81  switch (rsp->payload.link.state) {
82  case OPAHRS_MSG_LINK_STATE_BUSY:
83  case OPAHRS_MSG_LINK_STATE_INACTIVE:
84  /* Wait for a small delay and retry */
85 #ifdef PIOS_INCLUDE_FREERTOS
86  vTaskDelay(1 / portTICK_RATE_MS);
87 #else
88  PIOS_DELAY_WaitmS(20);
89 #endif
90  continue;
91  case OPAHRS_MSG_LINK_STATE_READY:
92  /* Peer was ready when we Tx'd so they have now Rx'd our message */
93  return OPAHRS_RESULT_OK;
94  }
95  break;
96  case OPAHRS_MSG_TYPE_USER_V0:
97  case OPAHRS_MSG_TYPE_USER_V1:
98  /* Wait for a small delay and retry */
99 #ifdef PIOS_INCLUDE_FREERTOS
100  vTaskDelay(1 / portTICK_RATE_MS);
101 #else
102  PIOS_DELAY_WaitmS(50);
103 #endif
104  continue;
105  }
106  }
107 
108  return OPAHRS_RESULT_TIMEOUT;
109 }

111 static enum opahrs_result opahrs_msg_v1_recv_rsp(enum opahrs_msg_v1_tag tag, struct opahrs_msg_v1 *rsp)
112 {
113  struct opahrs_msg_v1 link_tx;
114 
115  opahrs_msg_v1_init_link_tx(&link_tx, OPAHRS_MSG_LINK_TAG_NOP);
116 
117  for (uint8_t retries = 0; retries < 20; retries++) {
118  if (opahrs_msg_txrx((const uint8_t *)&link_tx, (uint8_t *) rsp, sizeof(*rsp)) < 0) {
119  return OPAHRS_RESULT_FAILED;
120  }
121 
122  /* Make sure we got a sane response by checking the magic */
123  if ((rsp->head.magic != OPAHRS_MSG_MAGIC_HEAD) || (rsp->tail.magic != OPAHRS_MSG_MAGIC_TAIL)) {
124  return OPAHRS_RESULT_FAILED;
125  }
126 
127  switch (rsp->head.type) {
128  case OPAHRS_MSG_TYPE_LINK:
129  switch (rsp->payload.link.state) {
130  case OPAHRS_MSG_LINK_STATE_BUSY:
131  /* Wait for a small delay and retry */
132 #ifdef PIOS_INCLUDE_FREERTOS
133  vTaskDelay(1 / portTICK_RATE_MS);
134 #else
135  PIOS_DELAY_WaitmS(20);
136 #endif
137  continue;
138  case OPAHRS_MSG_LINK_STATE_INACTIVE:
139  case OPAHRS_MSG_LINK_STATE_READY:
140  /* somehow, we've missed our response */
141  return OPAHRS_RESULT_FAILED;
142  }
143  break;
144  case OPAHRS_MSG_TYPE_USER_V0:
145  /* This isn't the type we expected */
146  return OPAHRS_RESULT_FAILED;
147  break;
148  case OPAHRS_MSG_TYPE_USER_V1:
149  if (rsp->payload.user.t == tag) {
150  return OPAHRS_RESULT_OK;
151  } else {
152  return OPAHRS_RESULT_FAILED;
153  }
154  break;
155  }
156  }
157 
158  return OPAHRS_RESULT_TIMEOUT;
159 }

161 static enum opahrs_result PIOS_OPAHRS_v1_simple_req(enum opahrs_msg_v1_tag req_type, struct opahrs_msg_v1 *rsp, enum opahrs_msg_v1_tag rsp_type)
162 {
163  struct opahrs_msg_v1 req;
164  enum opahrs_result rc;
165 
166  /* Make up an empty request */
167  opahrs_msg_v1_init_user_tx(&req, req_type);
168 
169  /* Send the message until it is received */
170  rc = opahrs_msg_v1_send_req(&req);
171  if ((rc == OPAHRS_RESULT_OK) && rsp) {
172  /* We need a specific kind of reply, go get it */
173  return opahrs_msg_v1_recv_rsp(rsp_type, rsp);
174  }
175 
176  return rc;
177 }

187 enum opahrs_result PIOS_OPAHRS_resync(void)
188 {
189  struct opahrs_msg_v1 req;
190  struct opahrs_msg_v1 rsp;
191 
192  enum opahrs_result rc = OPAHRS_RESULT_FAILED;
193 
194  opahrs_msg_v1_init_link_tx(&req, OPAHRS_MSG_LINK_TAG_NOP);
195 
196  PIOS_SPI_RC_PinSet(PIOS_OPAHRS_SPI, 0);
197 #ifdef PIOS_INCLUDE_FREERTOS
198  vTaskDelay(1 / portTICK_RATE_MS);
199 #else
200  PIOS_DELAY_WaitmS(20);
201 #endif
202 
203  for (uint32_t i = 0; i < sizeof(req); i++) {
204  /* Tx a shortened (by one byte) message to walk through all byte positions */
205  opahrs_msg_v1_init_rx(&rsp);
206  PIOS_SPI_TransferBlock(PIOS_OPAHRS_SPI, (uint8_t *) & req, (uint8_t *) & rsp, sizeof(req) - 1, NULL);
207 
208  /* Good magic means we're sync'd */
209  if ((rsp.head.magic == OPAHRS_MSG_MAGIC_HEAD) && (rsp.tail.magic == OPAHRS_MSG_MAGIC_TAIL)) {
210  /* We need to shift out one more byte to compensate for the short tx */
211  PIOS_SPI_TransferByte(PIOS_OPAHRS_SPI, 0x00);
212  rc = OPAHRS_RESULT_OK;
213  break;
214  }
215 #ifdef PIOS_INCLUDE_FREERTOS
216  vTaskDelay(1 / portTICK_RATE_MS);
217 #else
218  PIOS_DELAY_WaitmS(10);
219 #endif
220  }
221 
222  PIOS_SPI_RC_PinSet(PIOS_OPAHRS_SPI, 1);
223  //vTaskDelay(5 / portTICK_RATE_MS);
224 
225  return rc;
226 }
