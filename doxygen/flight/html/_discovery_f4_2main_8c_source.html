00194 uint32_t LedPWM(uint32_t pwm_period, uint32_t pwm_sweep_steps, uint32_t count) {
00195         uint32_t curr_step = (count / pwm_period) % pwm_sweep_steps; /* 0 - pwm_sweep_steps */
00196         uint32_t pwm_duty = pwm_period * curr_step / pwm_sweep_steps; /* fraction of pwm_period */
00197 
00198         uint32_t curr_sweep = (count / (pwm_period * pwm_sweep_steps)); /* ticks once per full sweep */
00199         if (curr_sweep & 1) {
00200                 pwm_duty = pwm_period - pwm_duty; /* reverse direction in odd sweeps */
00201         }
00202         return ((count % pwm_period) > pwm_duty) ? 1 : 0;
00203 }
