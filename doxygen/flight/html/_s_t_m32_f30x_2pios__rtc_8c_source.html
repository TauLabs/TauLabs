00052 {
00053         RCC_BackupResetCmd(ENABLE);
00054         RCC_BackupResetCmd(DISABLE);
00055         RCC_APB1PeriphClockCmd(RCC_APB1Periph_PWR, ENABLE);
00056         PWR_BackupAccessCmd(ENABLE);
00057         // Divide external 8 MHz clock by 32 to 250kHz
00058         RCC_RTCCLKConfig(cfg->clksrc);
00059         RCC_RTCCLKCmd(ENABLE);

00061         RTC_WakeUpCmd(DISABLE);
00062         // Divide 250kHz Mhz clock by 16 to 15625Hz
00063         RTC_WakeUpClockConfig(RTC_WakeUpClock_RTCCLK_Div16);
00064         // Divide 15625Hz by 25 to get 625 Hz
00065         RTC_SetWakeUpCounter(cfg->prescaler); //cfg->prescaler);
00066         RTC_WakeUpCmd(ENABLE);
00067         
00068         /* Configure and enable the RTC Second interrupt */
00069         EXTI_InitTypeDef ExtiInit = {
00070                 .EXTI_Line = EXTI_Line20, // matches above GPIO pin
00071                 .EXTI_Mode = EXTI_Mode_Interrupt,
00072                 .EXTI_Trigger = EXTI_Trigger_Rising,
00073                 .EXTI_LineCmd = ENABLE,
00074         };
00075         EXTI_Init((EXTI_InitTypeDef*)&ExtiInit);
00076         NVIC_Init((NVIC_InitTypeDef*)&cfg->irq.init);
