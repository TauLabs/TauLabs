114 int32_t PIOS_EXTI_Init(const struct pios_exti_cfg * cfg)
115 {
116  PIOS_Assert(cfg);
117  PIOS_Assert(&__start__exti);
118  PIOS_Assert(cfg >= &__start__exti);
119  PIOS_Assert(cfg < &__stop__exti);
120 
121  uint8_t cfg_index = cfg - &__start__exti;
122 
123  /* Connect this config to the requested vector */
124  uint8_t line_index = PIOS_EXTI_line_to_index(cfg->line);
125 
126  if (pios_exti_line_to_cfg_map[line_index] != PIOS_EXTI_INVALID) {
127  /* Someone else already has this mapped */
128  goto out_fail;
129  }
130 
131  /* Bind the config to the exti line */
132  pios_exti_line_to_cfg_map[line_index] = cfg_index;
133 
134  /* Initialize the GPIO pin */
135  GPIO_Init(cfg->pin.gpio, (GPIO_InitTypeDef*)&cfg->pin.init);
136 
137  /* Set up the EXTI interrupt source */
138  uint8_t exti_source_port = PIOS_EXTI_gpio_port_to_exti_source_port(cfg->pin.gpio);
139  uint8_t exti_source_pin = PIOS_EXTI_gpio_pin_to_exti_source_pin(cfg->pin.init.GPIO_Pin);
140  SYSCFG_EXTILineConfig(exti_source_port, exti_source_pin);
141  EXTI_Init((EXTI_InitTypeDef*)&cfg->exti.init);
142 
143  /* Enable the interrupt channel */
144  NVIC_Init((NVIC_InitTypeDef*)&cfg->irq.init);
145 
146  return 0;
147 
148 out_fail:
149  return -1;
150 }

152 static bool PIOS_EXTI_generic_irq_handler(uint8_t line_index)
153 {
154  uint8_t cfg_index = pios_exti_line_to_cfg_map[line_index];
155 
156  PIOS_Assert(&__start__exti);
157 
158  if (cfg_index > NELEMENTS(pios_exti_line_to_cfg_map) ||
159  cfg_index == PIOS_EXTI_INVALID) {
160  /* Unconfigured interrupt just fired! */
161  return false;
162  }
163 
164  struct pios_exti_cfg * cfg = &__start__exti + cfg_index;
165  return cfg->vector();
166 }

172  if (EXTI_GetITStatus(EXTI_Line##line) != RESET) { \
173  EXTI_ClearITPendingBit(EXTI_Line##line); \

180  PIOS_EXTI_generic_irq_handler(line); \
181  }
