00245 static int32_t objectTransaction(UAVTalkConnectionData *connection, UAVObjHandle obj, uint16_t instId, uint8_t type, int32_t timeoutMs)
00246 {
00247         int32_t respReceived;
00248         
00249         // Send object depending on if a response is needed
00250         if (type == UAVTALK_TYPE_OBJ_ACK || type == UAVTALK_TYPE_OBJ_ACK_TS || type == UAVTALK_TYPE_OBJ_REQ)
00251         {
00252                 // Get transaction lock (will block if a transaction is pending)
00253                 xSemaphoreTakeRecursive(connection->transLock, portMAX_DELAY);
00254                 // Send object
00255                 xSemaphoreTakeRecursive(connection->lock, portMAX_DELAY);
00256                 connection->respObj = obj;
00257                 connection->respInstId = instId;
00258                 sendObject(connection, obj, instId, type);
00259                 xSemaphoreGiveRecursive(connection->lock);
00260                 // Wait for response (or timeout)
00261                 respReceived = xSemaphoreTake(connection->respSema, timeoutMs/portTICK_RATE_MS);
00262                 // Check if a response was received
00263                 if (respReceived == pdFALSE)
00264                 {
00265                         // Cancel transaction
00266                         xSemaphoreTakeRecursive(connection->lock, portMAX_DELAY);
00267                         xSemaphoreTake(connection->respSema, 0); // non blocking call to make sure the value is reset to zero (binary sema)
00268                         connection->respObj = 0;
00269                         xSemaphoreGiveRecursive(connection->lock);
00270                         xSemaphoreGiveRecursive(connection->transLock);
00271                         return -1;
00272                 }
00273                 else
00274                 {
00275                         xSemaphoreGiveRecursive(connection->transLock);
00276                         return 0;
00277                 }
00278         }
00279         else if (type == UAVTALK_TYPE_OBJ || type == UAVTALK_TYPE_OBJ_TS)
00280         {
00281                 xSemaphoreTakeRecursive(connection->lock, portMAX_DELAY);
00282                 sendObject(connection, obj, instId, type);
00283                 xSemaphoreGiveRecursive(connection->lock);
00284                 return 0;
