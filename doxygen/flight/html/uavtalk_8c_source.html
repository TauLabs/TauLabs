242 static int32_t objectTransaction(UAVTalkConnectionData *connection, UAVObjHandle obj, uint16_t instId, uint8_t type, int32_t timeoutMs)
243 {
244  int32_t respReceived;
245 
246  // Send object depending on if a response is needed
247  if (type == UAVTALK_TYPE_OBJ_ACK || type == UAVTALK_TYPE_OBJ_ACK_TS || type == UAVTALK_TYPE_OBJ_REQ)
248  {
249  // Get transaction lock (will block if a transaction is pending)
250  xSemaphoreTakeRecursive(connection->transLock, portMAX_DELAY);
251  // Send object
252  xSemaphoreTakeRecursive(connection->lock, portMAX_DELAY);
253  connection->respObj = obj;
254  connection->respInstId = instId;
255  sendObject(connection, obj, instId, type);
256  xSemaphoreGiveRecursive(connection->lock);
257  // Wait for response (or timeout)
258  respReceived = xSemaphoreTake(connection->respSema, timeoutMs/portTICK_RATE_MS);
259  // Check if a response was received
260  if (respReceived == pdFALSE)
261  {
262  // Cancel transaction
263  xSemaphoreTakeRecursive(connection->lock, portMAX_DELAY);
264  xSemaphoreTake(connection->respSema, 0); // non blocking call to make sure the value is reset to zero (binary sema)
265  connection->respObj = 0;
266  xSemaphoreGiveRecursive(connection->lock);
267  xSemaphoreGiveRecursive(connection->transLock);
268  return -1;
269  }
270  else
271  {
272  xSemaphoreGiveRecursive(connection->transLock);
273  return 0;
274  }
275  }
276  else if (type == UAVTALK_TYPE_OBJ || type == UAVTALK_TYPE_OBJ_TS)
277  {
278  xSemaphoreTakeRecursive(connection->lock, portMAX_DELAY);
279  sendObject(connection, obj, instId, type);
280  xSemaphoreGiveRecursive(connection->lock);
281  return 0;
282  }
283  else
284  {
285  return -1;
286  }
287 }

295 UAVTalkRxState UAVTalkProcessInputStreamQuiet(UAVTalkConnection connectionHandle, uint8_t rxbyte)
296 {
297  UAVTalkConnectionData *connection;
298  CHECKCONHANDLE(connectionHandle,connection,return -1);
299 
300  UAVTalkInputProcessor *iproc = &connection->iproc;
301  ++connection->stats.rxBytes;
302 
303  if (iproc->state == UAVTALK_STATE_ERROR || iproc->state == UAVTALK_STATE_COMPLETE)
304  iproc->state = UAVTALK_STATE_SYNC;
305 
306  if (iproc->rxPacketLength < 0xffff)
307  iproc->rxPacketLength++; // update packet byte count
308 
309  // Receive state machine
310  switch (iproc->state)
311  {
312  case UAVTALK_STATE_SYNC:
313  if (rxbyte != UAVTALK_SYNC_VAL)
314  break;
315 
316  // Initialize and update the CRC
317  iproc->cs = PIOS_CRC_updateByte(0, rxbyte);
318 
319  iproc->rxPacketLength = 1;
320 
321  iproc->state = UAVTALK_STATE_TYPE;
322  break;
323 
324  case UAVTALK_STATE_TYPE:
325 
326  // update the CRC
327  iproc->cs = PIOS_CRC_updateByte(iproc->cs, rxbyte);
328 
329  if ((rxbyte & UAVTALK_TYPE_MASK) != UAVTALK_TYPE_VER)
330  {
331  iproc->state = UAVTALK_STATE_ERROR;
332  break;
333  }
334 
335  iproc->type = rxbyte;
336 
337  iproc->packet_size = 0;
338 
339  iproc->state = UAVTALK_STATE_SIZE;
340  iproc->rxCount = 0;
341  break;
342 
343  case UAVTALK_STATE_SIZE:
344 
345  // update the CRC
346  iproc->cs = PIOS_CRC_updateByte(iproc->cs, rxbyte);
347 
348  if (iproc->rxCount == 0)
349  {
350  iproc->packet_size += rxbyte;
351  iproc->rxCount++;
352  break;
353  }
354 
355  iproc->packet_size += rxbyte << 8;
356 
357  if (iproc->packet_size < UAVTALK_MIN_HEADER_LENGTH || iproc->packet_size > UAVTALK_MAX_HEADER_LENGTH + UAVTALK_MAX_PAYLOAD_LENGTH)
358  { // incorrect packet size
359  iproc->state = UAVTALK_STATE_ERROR;
360  break;
361  }
362 
363  iproc->rxCount = 0;
364  iproc->objId = 0;
365  iproc->state = UAVTALK_STATE_OBJID;
366  break;
367 
368  case UAVTALK_STATE_OBJID:
369 
370  // update the CRC
371  iproc->cs = PIOS_CRC_updateByte(iproc->cs, rxbyte);
372 
373  iproc->objId += rxbyte << (8*(iproc->rxCount++));
374 
375  if (iproc->rxCount < 4)
376  break;
377 
378  // Search for object.
379  iproc->obj = UAVObjGetByID(iproc->objId);
380 
381  // Determine data length
382  if (iproc->type == UAVTALK_TYPE_OBJ_REQ || iproc->type == UAVTALK_TYPE_ACK || iproc->type == UAVTALK_TYPE_NACK)
383  {
384  iproc->length = 0;
385  iproc->instanceLength = 0;
386  }
387  else
388  {
389  if (iproc->obj)
390  {
391  iproc->length = UAVObjGetNumBytes(iproc->obj);
392  iproc->instanceLength = (UAVObjIsSingleInstance(iproc->obj) ? 0 : 2);
393  iproc->timestampLength = (iproc->type & UAVTALK_TIMESTAMPED) ? 2 : 0;
394  }
395  else
396  {
397  // We don't know if it's a multi-instance object, so just assume it's 0.
398  iproc->instanceLength = 0;
399  iproc->length = iproc->packet_size - iproc->rxPacketLength;
400  }
401  }
402 
403  // Check length and determine next state
404  if (iproc->length >= UAVTALK_MAX_PAYLOAD_LENGTH)
405  {
406  connection->stats.rxErrors++;
407  iproc->state = UAVTALK_STATE_ERROR;
408  break;
409  }
410 
411  // Check the lengths match
412  if ((iproc->rxPacketLength + iproc->instanceLength + iproc->timestampLength + iproc->length) != iproc->packet_size)
413  { // packet error - mismatched packet size
414  connection->stats.rxErrors++;
415  iproc->state = UAVTALK_STATE_ERROR;
416  break;
417  }
418 
419  iproc->instId = 0;
420  if (iproc->type == UAVTALK_TYPE_NACK)
421  {
422  // If this is a NACK, we skip to Checksum
423  iproc->state = UAVTALK_STATE_CS;
424  }
425  // Check if this is a single instance object (i.e. if the instance ID field is coming next)
426  else if ((iproc->obj != 0) && !UAVObjIsSingleInstance(iproc->obj))
427  {
428  iproc->state = UAVTALK_STATE_INSTID;
429  }
430  // Check if this is a single instance and has a timestamp in it
431  else if ((iproc->obj != 0) && (iproc->type & UAVTALK_TIMESTAMPED))
432  {
433  iproc->timestamp = 0;
434  iproc->state = UAVTALK_STATE_TIMESTAMP;
435  }
436  else
437  {
438  // If there is a payload get it, otherwise receive checksum
439  if (iproc->length > 0)
440  iproc->state = UAVTALK_STATE_DATA;
441  else
442  iproc->state = UAVTALK_STATE_CS;
443  }
444  iproc->rxCount = 0;
445 
446  break;
447 
448  case UAVTALK_STATE_INSTID:
449 
450  // update the CRC
451  iproc->cs = PIOS_CRC_updateByte(iproc->cs, rxbyte);
452 
453  iproc->instId += rxbyte << (8*(iproc->rxCount++));
454 
455  if (iproc->rxCount < 2)
456  break;
457 
458  iproc->rxCount = 0;
459 
460  // If there is a timestamp, get it
461  if ((iproc->length > 0) && (iproc->type & UAVTALK_TIMESTAMPED))
462  {
463  iproc->timestamp = 0;
464  iproc->state = UAVTALK_STATE_TIMESTAMP;
465  }
466  // If there is a payload get it, otherwise receive checksum
467  else if (iproc->length > 0)
468  iproc->state = UAVTALK_STATE_DATA;
469  else
470  iproc->state = UAVTALK_STATE_CS;
471 
472  break;
473 
474  case UAVTALK_STATE_TIMESTAMP:
475  // update the CRC
476  iproc->cs = PIOS_CRC_updateByte(iproc->cs, rxbyte);
477 
478  iproc->timestamp += rxbyte << (8*(iproc->rxCount++));
479 
480  if (iproc->rxCount < 2)
481  break;
482 
483  iproc->rxCount = 0;
484 
485  // If there is a payload get it, otherwise receive checksum
486  if (iproc->length > 0)
487  iproc->state = UAVTALK_STATE_DATA;
488  else
489  iproc->state = UAVTALK_STATE_CS;
490  break;
491 
492  case UAVTALK_STATE_DATA:
493 
494  // update the CRC
495  iproc->cs = PIOS_CRC_updateByte(iproc->cs, rxbyte);
496 
497  connection->rxBuffer[iproc->rxCount++] = rxbyte;
498  if (iproc->rxCount < iproc->length)
499  break;
500 
501  iproc->state = UAVTALK_STATE_CS;
502  iproc->rxCount = 0;
503  break;
504 
505  case UAVTALK_STATE_CS:
506 
507  // the CRC byte
508  if (rxbyte != iproc->cs)
509  { // packet error - faulty CRC
510  connection->stats.rxErrors++;
511  iproc->state = UAVTALK_STATE_ERROR;
512  break;
513  }
514 
515  if (iproc->rxPacketLength != (iproc->packet_size + 1))
516  { // packet error - mismatched packet size
517  connection->stats.rxErrors++;
518  iproc->state = UAVTALK_STATE_ERROR;
519  break;
520  }
521 
522  connection->stats.rxObjectBytes += iproc->length;
523  connection->stats.rxObjects++;
524 
525  iproc->state = UAVTALK_STATE_COMPLETE;
526  break;
527 
528  default:
529  connection->stats.rxErrors++;
530  iproc->state = UAVTALK_STATE_ERROR;
531  }
532 
533  // Done
534  return iproc->state;
535 }

571 UAVTalkRxState UAVTalkRelayInputStream(UAVTalkConnection connectionHandle, uint8_t rxbyte)
572 {
573  UAVTalkRxState state = UAVTalkProcessInputStreamQuiet(connectionHandle, rxbyte);
574 
575  if (state == UAVTALK_STATE_COMPLETE)
576  {
577  UAVTalkConnectionData *connection;
578  CHECKCONHANDLE(connectionHandle,connection,return -1);
579  UAVTalkInputProcessor *iproc = &connection->iproc;
580 
581  if (!connection->outStream) return -1;
582 
583  // Setup type and object id fields
584  connection->txBuffer[0] = UAVTALK_SYNC_VAL; // sync byte
585  connection->txBuffer[1] = iproc->type;
586  // data length inserted here below
587  connection->txBuffer[4] = (uint8_t)(iproc->objId & 0xFF);
588  connection->txBuffer[5] = (uint8_t)((iproc->objId >> 8) & 0xFF);
589  connection->txBuffer[6] = (uint8_t)((iproc->objId >> 16) & 0xFF);
590  connection->txBuffer[7] = (uint8_t)((iproc->objId >> 24) & 0xFF);
591 
592  // Setup instance ID if one is required
593  int32_t dataOffset = 8;
594  if (iproc->instanceLength > 0)
595  {
596  connection->txBuffer[8] = (uint8_t)(iproc->instId & 0xFF);
597  connection->txBuffer[9] = (uint8_t)((iproc->instId >> 8) & 0xFF);
598  dataOffset = 10;
599  }
600 
601  // Add timestamp when the transaction type is appropriate
602  if (iproc->type & UAVTALK_TIMESTAMPED)
603  {
604  portTickType time = xTaskGetTickCount();
605  connection->txBuffer[dataOffset] = (uint8_t)(time & 0xFF);
606  connection->txBuffer[dataOffset + 1] = (uint8_t)((time >> 8) & 0xFF);
607  dataOffset += 2;
608  }
609 
610  // Copy data (if any)
611  memcpy(&connection->txBuffer[dataOffset], connection->rxBuffer, iproc->length);
612 
613  // Store the packet length
614  connection->txBuffer[2] = (uint8_t)((dataOffset + iproc->length) & 0xFF);
615  connection->txBuffer[3] = (uint8_t)(((dataOffset + iproc->length) >> 8) & 0xFF);
616 
617  // Copy the checksum
618  connection->txBuffer[dataOffset + iproc->length] = iproc->cs;
619 
620  // Send the buffer.
621  if (UAVTalkSendBuf(connectionHandle, connection->txBuffer, iproc->rxPacketLength) < 0)
622  return UAVTALK_STATE_ERROR;
623  }
624 
625  return state;
626 }

717 static int32_t receiveObject(UAVTalkConnectionData *connection, uint8_t type, uint32_t objId, uint16_t instId, uint8_t* data, int32_t length)
718 {
719  UAVObjHandle obj;
720  int32_t ret = 0;
721 
722  // Get the handle to the Object. Will be zero
723  // if object does not exist.
724  obj = UAVObjGetByID(objId);
725 
726  // Process message type
727  switch (type) {
728  case UAVTALK_TYPE_OBJ:
729  case UAVTALK_TYPE_OBJ_TS:
730  // All instances, not allowed for OBJ messages
731  if (obj && (instId != UAVOBJ_ALL_INSTANCES))
732  {
733  // Unpack object, if the instance does not exist it will be created!
734  UAVObjUnpack(obj, instId, data);
735  // Check if an ack is pending
736  updateAck(connection, obj, instId);
737  }
738  else
739  {
740  ret = -1;
741  }
742  break;
743  case UAVTALK_TYPE_OBJ_ACK:
744  case UAVTALK_TYPE_OBJ_ACK_TS:
745  // All instances, not allowed for OBJ_ACK messages
746  if (obj && (instId != UAVOBJ_ALL_INSTANCES))
747  {
748  // Unpack object, if the instance does not exist it will be created!
749  if ( UAVObjUnpack(obj, instId, data) == 0 )
750  {
751  // Transmit ACK
752  sendObject(connection, obj, instId, UAVTALK_TYPE_ACK);
753  }
754  else
755  {
756  ret = -1;
757  }
758  }
759  else
760  {
761  ret = -1;
762  }
763  break;
764  case UAVTALK_TYPE_OBJ_REQ:
765  // Send requested object if message is of type OBJ_REQ
766  if (obj == 0)
767  sendNack(connection, objId);
768  else
769  sendObject(connection, obj, instId, UAVTALK_TYPE_OBJ);
770  break;
771  case UAVTALK_TYPE_NACK:
772  // Do nothing on flight side, let it time out.
773  break;
774  case UAVTALK_TYPE_ACK:
775  // All instances, not allowed for ACK messages
776  if (obj && (instId != UAVOBJ_ALL_INSTANCES))
777  {
778  // Check if an ack is pending
779  updateAck(connection, obj, instId);
780  }
781  else
782  {
783  ret = -1;
784  }
785  break;
786  default:
787  ret = -1;
788  }
789  // Done
790  return ret;
791 }

817 static int32_t sendObject(UAVTalkConnectionData *connection, UAVObjHandle obj, uint16_t instId, uint8_t type)
818 {
819  uint32_t numInst;
820  uint32_t n;
821 
822  // If all instances are requested and this is a single instance object, force instance ID to zero
823  if ( instId == UAVOBJ_ALL_INSTANCES && UAVObjIsSingleInstance(obj) )
824  {
825  instId = 0;
826  }

877 static int32_t sendSingleObject(UAVTalkConnectionData *connection, UAVObjHandle obj, uint16_t instId, uint8_t type)
878 {
879  int32_t length;
880  int32_t dataOffset;
881  uint32_t objId;
882 
883  if (!connection->outStream) return -1;
884 
885  // Setup type and object id fields
886  objId = UAVObjGetID(obj);
887  connection->txBuffer[0] = UAVTALK_SYNC_VAL; // sync byte
888  connection->txBuffer[1] = type;
889  // data length inserted here below
890  connection->txBuffer[4] = (uint8_t)(objId & 0xFF);
891  connection->txBuffer[5] = (uint8_t)((objId >> 8) & 0xFF);
892  connection->txBuffer[6] = (uint8_t)((objId >> 16) & 0xFF);
893  connection->txBuffer[7] = (uint8_t)((objId >> 24) & 0xFF);
894 
895  // Setup instance ID if one is required
896  if (UAVObjIsSingleInstance(obj))
897  {
898  dataOffset = 8;
899  }
900  else
901  {
902  connection->txBuffer[8] = (uint8_t)(instId & 0xFF);
903  connection->txBuffer[9] = (uint8_t)((instId >> 8) & 0xFF);
904  dataOffset = 10;
905  }
906 
907  // Add timestamp when the transaction type is appropriate
908  if (type & UAVTALK_TIMESTAMPED)
909  {
910  portTickType time = xTaskGetTickCount();
911  connection->txBuffer[dataOffset] = (uint8_t)(time & 0xFF);
912  connection->txBuffer[dataOffset + 1] = (uint8_t)((time >> 8) & 0xFF);
913  dataOffset += 2;
914  }
915 
916  // Determine data length
917  if (type == UAVTALK_TYPE_OBJ_REQ || type == UAVTALK_TYPE_ACK)
918  {
919  length = 0;
920  }
921  else
922  {
923  length = UAVObjGetNumBytes(obj);
924  }
925 
926  // Check length
927  if (length >= UAVTALK_MAX_PAYLOAD_LENGTH)
928  {
929  return -1;
930  }
931 
932  // Copy data (if any)
933  if (length > 0)
934  {
935  if ( UAVObjPack(obj, instId, &connection->txBuffer[dataOffset]) < 0 )
936  {
937  return -1;
938  }
939  }
940 
941  // Store the packet length
942  connection->txBuffer[2] = (uint8_t)((dataOffset+length) & 0xFF);
943  connection->txBuffer[3] = (uint8_t)(((dataOffset+length) >> 8) & 0xFF);
944 
945  // Calculate checksum
946  connection->txBuffer[dataOffset+length] = PIOS_CRC_updateCRC(0, connection->txBuffer, dataOffset+length);
947 
948  uint16_t tx_msg_len = dataOffset+length+UAVTALK_CHECKSUM_LENGTH;
949  int32_t rc = (*connection->outStream)(connection->txBuffer, tx_msg_len);
950 
951  if (rc == tx_msg_len) {
952  // Update stats
953  ++connection->stats.txObjects;
954  connection->stats.txBytes += tx_msg_len;
955  connection->stats.txObjectBytes += length;
956  }
957 
958  // Done
959  return 0;
960 }
