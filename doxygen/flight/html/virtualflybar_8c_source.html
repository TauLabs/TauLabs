00044 int stabilization_virtual_flybar(float gyro, float command, float *output, float dT, bool reinit, uint32_t axis, StabilizationSettingsData *settings)
00045 {
00046         float gyro_gain = 1.0f;
00047         float kp = 0, ki = 0;
00048 
00049         if(reinit)
00050                 vbar_integral[axis] = 0;
00051 
00052         // Track the angle of the virtual flybar which includes a slow decay
00053         vbar_integral[axis] = vbar_integral[axis] * vbar_decay + gyro * dT;
00054         vbar_integral[axis] = bound(vbar_integral[axis], settings->VbarMaxAngle);
00055 
00056         // Command signal can indicate how much to disregard the gyro feedback (fast flips)
00057         if (settings->VbarGyroSuppress > 0) {
00058                 gyro_gain = (1.0f - fabs(command) * settings->VbarGyroSuppress / 100.0f);
00059                 gyro_gain = (gyro_gain < 0) ? 0 : gyro_gain;
00060         }
00061 
00062         // Get the settings for the correct axis
00063         switch(axis)
00064         {
00065                 case ROLL:
00066                         kp = settings->VbarRollPI[STABILIZATIONSETTINGS_VBARROLLPI_KP];
00067                         ki = settings->VbarRollPI[STABILIZATIONSETTINGS_VBARROLLPI_KI];
00068                         break;
00069                 case PITCH:
00070                         kp = settings->VbarPitchPI[STABILIZATIONSETTINGS_VBARROLLPI_KP];
00071                         ki = settings->VbarPitchPI[STABILIZATIONSETTINGS_VBARROLLPI_KI];
00072                         break;
00073                 case YAW:
00074                         kp = settings->VbarYawPI[STABILIZATIONSETTINGS_VBARROLLPI_KP];
00075                         ki = settings->VbarYawPI[STABILIZATIONSETTINGS_VBARROLLPI_KI];
00076                         break;
00077                 default:
00078                         PIOS_DEBUG_Assert(0);
00079         }
00080 
00081         // Command signal is composed of stick input added to the gyro and virtual flybar
00082         *output = command * settings->VbarSensitivity[axis] - 
00083             gyro_gain * (vbar_integral[axis] * ki + gyro * kp);
00084 
00085         return 0;
00086 }
