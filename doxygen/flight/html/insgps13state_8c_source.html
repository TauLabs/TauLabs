00380 //  *************  CovariancePrediction *************
00381 //  Does the prediction step of the Kalman filter for the covariance matrix
00382 //  Output, Pnew, overwrites P, the input covariance
00383 //  Pnew = (I+F*T)*P*(I+F*T)' + T^2*G*Q*G'
00384 //  Q is the discrete time covariance of process noise
00385 //  Q is vector of the diagonal for a square matrix with
00386 //    dimensions equal to the number of disturbance noise variables
00387 //  The General Method is very inefficient,not taking advantage of the sparse F and G
00388 //  The first Method is very specific to this implementation
00389 //  ************************************************

00393 static void CovariancePrediction(float F[NUMX][NUMX], float G[NUMX][NUMW],
00394                           float Q[NUMW], float dT, float P[NUMX][NUMX])
00395 {
00396         float Dummy[NUMX][NUMX], dTsq;
00397         uint8_t i, j, k;
00398 
00399         //  Pnew = (I+F*T)*P*(I+F*T)' + T^2*G*Q*G' = T^2[(P/T + F*P)*(I/T + F') + G*Q*G')]
00400 
00401         dTsq = dT * dT;
00402 
00403         for (i = 0; i < NUMX; i++)      // Calculate Dummy = (P/T +F*P)
00404                 for (j = 0; j < NUMX; j++) {
00405                         Dummy[i][j] = P[i][j] / dT;
00406                         for (k = 0; k < NUMX; k++)
00407                                 Dummy[i][j] += F[i][k] * P[k][j];
00408                 }
00409         for (i = 0; i < NUMX; i++)      // Calculate Pnew = Dummy/T + Dummy*F' + G*Qw*G'
00410                 for (j = i; j < NUMX; j++) {    // Use symmetry, ie only find upper triangular
00411                         P[i][j] = Dummy[i][j] / dT;
00412                         for (k = 0; k < NUMX; k++)
00413                                 P[i][j] += Dummy[i][k] * F[j][k];       // P = Dummy/T + Dummy*F'
00414                         for (k = 0; k < NUMW; k++)
00415                                 P[i][j] += Q[k] * G[i][k] * G[j][k];    // P = Dummy/T + Dummy*F' + G*Q*G'
00416                         P[j][i] = P[i][j] = P[i][j] * dTsq;     // Pnew = T^2*P and fill in lower triangular;
00417                 }
00418 }

01358 //  *************  SerialUpdate *******************
01359 //  Does the update step of the Kalman filter for the covariance and estimate
01360 //  Outputs are Xnew & Pnew, and are written over P and X
01361 //  Z is actual measurement, Y is predicted measurement
01362 //  Xnew = X + K*(Z-Y), Pnew=(I-K*H)*P,
01363 //    where K=P*H'*inv[H*P*H'+R]
01364 //  NOTE the algorithm assumes R (measurement covariance matrix) is diagonal
01365 //    i.e. the measurment noises are uncorrelated.
01366 //  It therefore uses a serial update that requires no matrix inversion by
01367 //    processing the measurements one at a time.
01368 //  Algorithm - see Grewal and Andrews, "Kalman Filtering,2nd Ed" p.121 & p.253
01369 //            - or see Simon, "Optimal State Estimation," 1st Ed, p.150
01370 //  The SensorsUsed variable is a bitwise mask indicating which sensors
01371 //     should be used in the update.
01372 //  ************************************************
01373 
01374 static void SerialUpdate(float H[NUMV][NUMX], float R[NUMV], float Z[NUMV],
01375                   float Y[NUMV], float P[NUMX][NUMX], float X[NUMX],
01376                   uint16_t SensorsUsed)
01377 {
01378         float HP[NUMX], HPHR, Error;
01379         uint8_t i, j, k, m;
01380 
01381         for (m = 0; m < NUMV; m++) {
01382 
01383                 if (SensorsUsed & (0x01 << m)) {        // use this sensor for update
01384 
01385                         for (j = 0; j < NUMX; j++) {    // Find Hp = H*P
01386                                 HP[j] = 0;
01387                                 for (k = 0; k < NUMX; k++)
01388                                         HP[j] += H[m][k] * P[k][j];
01389                         }
01390                         HPHR = R[m];    // Find  HPHR = H*P*H' + R
01391                         for (k = 0; k < NUMX; k++)
01392                                 HPHR += HP[k] * H[m][k];
01393 
01394                         for (k = 0; k < NUMX; k++)
01395                                 K[k][m] = HP[k] / HPHR; // find K = HP/HPHR
01396 
01397                         for (i = 0; i < NUMX; i++) {    // Find P(m)= P(m-1) + K*HP
01398                                 for (j = i; j < NUMX; j++)
01399                                         P[i][j] = P[j][i] =
01400                                             P[i][j] - K[i][m] * HP[j];
01401                         }
01402 
01403                         Error = Z[m] - Y[m];
01404                         for (i = 0; i < NUMX; i++)      // Find X(m)= X(m-1) + K*Error
01405                                 X[i] = X[i] + K[i][m] * Error;
01406 
01407                 }
01408         }
01409 }

01411 //  *************  RungeKutta **********************
01412 //  Does a 4th order Runge Kutta numerical integration step
01413 //  Output, Xnew, is written over X
01414 //  NOTE the algorithm assumes time invariant state equations and
01415 //    constant inputs over integration step
01416 //  ************************************************
01417 
01418 static void RungeKutta(float X[NUMX], float U[NUMU], float dT)
01419 {
01420 
01421         float dT2 =
01422             dT / 2.0f, K1[NUMX], K2[NUMX], K3[NUMX], K4[NUMX], Xlast[NUMX];
01423         uint8_t i;
01424 
01425         for (i = 0; i < NUMX; i++)
01426                 Xlast[i] = X[i];        // make a working copy
01427 
01428         StateEq(X, U, K1);      // k1 = f(x,u)
01429         for (i = 0; i < NUMX; i++)
01430                 X[i] = Xlast[i] + dT2 * K1[i];
01431         StateEq(X, U, K2);      // k2 = f(x+0.5*dT*k1,u)
01432         for (i = 0; i < NUMX; i++)
01433                 X[i] = Xlast[i] + dT2 * K2[i];
01434         StateEq(X, U, K3);      // k3 = f(x+0.5*dT*k2,u)
01435         for (i = 0; i < NUMX; i++)
01436                 X[i] = Xlast[i] + dT * K3[i];
01437         StateEq(X, U, K4);      // k4 = f(x+dT*k3,u)
01438 
01439         // Xnew  = X + dT*(k1+2*k2+2*k3+k4)/6
01440         for (i = 0; i < NUMX; i++)
01441                 X[i] =
01442                     Xlast[i] + dT * (K1[i] + 2.0f * K2[i] + 2.0f * K3[i] +
01443                                      K4[i]) / 6.0f;
01444 }
