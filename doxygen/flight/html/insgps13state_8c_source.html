375 // ************* CovariancePrediction *************
376 // Does the prediction step of the Kalman filter for the covariance matrix
377 // Output, Pnew, overwrites P, the input covariance
378 // Pnew = (I+F*T)*P*(I+F*T)' + T^2*G*Q*G'
379 // Q is the discrete time covariance of process noise
380 // Q is vector of the diagonal for a square matrix with
381 // dimensions equal to the number of disturbance noise variables
382 // The General Method is very inefficient,not taking advantage of the sparse F and G
383 // The first Method is very specific to this implementation
384 // ************************************************
385

388 void CovariancePrediction(float F[NUMX][NUMX], float G[NUMX][NUMW],
389  float Q[NUMW], float dT, float P[NUMX][NUMX])
390 {
391  float Dummy[NUMX][NUMX], dTsq;
392  uint8_t i, j, k;
393 
394  // Pnew = (I+F*T)*P*(I+F*T)' + T^2*G*Q*G' = T^2[(P/T + F*P)*(I/T + F') + G*Q*G')]
395 
396  dTsq = dT * dT;
397 
398  for (i = 0; i < NUMX; i++) // Calculate Dummy = (P/T +F*P)
399  for (j = 0; j < NUMX; j++) {
400  Dummy[i][j] = P[i][j] / dT;
401  for (k = 0; k < NUMX; k++)
402  Dummy[i][j] += F[i][k] * P[k][j];
403  }
404  for (i = 0; i < NUMX; i++) // Calculate Pnew = Dummy/T + Dummy*F' + G*Qw*G'
405  for (j = i; j < NUMX; j++) { // Use symmetry, ie only find upper triangular
406  P[i][j] = Dummy[i][j] / dT;
407  for (k = 0; k < NUMX; k++)
408  P[i][j] += Dummy[i][k] * F[j][k]; // P = Dummy/T + Dummy*F'
409  for (k = 0; k < NUMW; k++)
410  P[i][j] += Q[k] * G[i][k] * G[j][k]; // P = Dummy/T + Dummy*F' + G*Q*G'
411  P[j][i] = P[i][j] = P[i][j] * dTsq; // Pnew = T^2*P and fill in lower triangular;
412  }
413 }

417 void CovariancePrediction(float F[NUMX][NUMX], float G[NUMX][NUMW],
418  float Q[NUMW], float dT, float P[NUMX][NUMX])
419 {
420  float D[NUMX][NUMX], T, Tsq;
421  uint8_t i, j;
422 
423  // Pnew = (I+F*T)*P*(I+F*T)' + T^2*G*Q*G' = scalar expansion from symbolic manipulator
424 
425  T = dT;
426  Tsq = dT * dT;
427 
428  for (i = 0; i < NUMX; i++) // Create a copy of the upper triangular of P
429  for (j = i; j < NUMX; j++)
430  D[i][j] = P[i][j];
431 
432  // Brute force calculation of the elements of P
433  P[0][0] = D[3][3] * Tsq + (2 * D[0][3]) * T + D[0][0];
434  P[0][1] = P[1][0] =
435  D[3][4] * Tsq + (D[0][4] + D[1][3]) * T + D[0][1];
436  P[0][2] = P[2][0] =
437  D[3][5] * Tsq + (D[0][5] + D[2][3]) * T + D[0][2];
438  P[0][3] = P[3][0] =
439  (F[3][6] * D[3][6] + F[3][7] * D[3][7] + F[3][8] * D[3][8] +
440  F[3][9] * D[3][9]) * Tsq + (D[3][3] + F[3][6] * D[0][6] +
441  F[3][7] * D[0][7] +
442  F[3][8] * D[0][8] +
443  F[3][9] * D[0][9]) * T + D[0][3];
444  P[0][4] = P[4][0] =
445  (F[4][6] * D[3][6] + F[4][7] * D[3][7] + F[4][8] * D[3][8] +
446  F[4][9] * D[3][9]) * Tsq + (D[3][4] + F[4][6] * D[0][6] +
447  F[4][7] * D[0][7] +
448  F[4][8] * D[0][8] +
449  F[4][9] * D[0][9]) * T + D[0][4];
450  P[0][5] = P[5][0] =
451  (F[5][6] * D[3][6] + F[5][7] * D[3][7] + F[5][8] * D[3][8] +

453  F[5][7] * D[0][7] +

455  F[5][9] * D[0][9]) * T + D[0][5];

458  F[6][10] * D[3][10] + F[6][11] * D[3][11] +

461  F[6][9] * D[0][9] +
462  F[6][10] * D[0][10] +

1353 // ************* SerialUpdate *******************
1354 // Does the update step of the Kalman filter for the covariance and estimate
1355 // Outputs are Xnew & Pnew, and are written over P and X
1356 // Z is actual measurement, Y is predicted measurement
1357 // Xnew = X + K*(Z-Y), Pnew=(I-K*H)*P,
1358 // where K=P*H'*inv[H*P*H'+R]
1359 // NOTE the algorithm assumes R (measurement covariance matrix) is diagonal
1360 // i.e. the measurment noises are uncorrelated.
1361 // It therefore uses a serial update that requires no matrix inversion by
1362 // processing the measurements one at a time.
1363 // Algorithm - see Grewal and Andrews, "Kalman Filtering,2nd Ed" p.121 & p.253
1364 // - or see Simon, "Optimal State Estimation," 1st Ed, p.150
1365 // The SensorsUsed variable is a bitwise mask indicating which sensors
1366 // should be used in the update.
1367 // ************************************************
1368 
1369 void SerialUpdate(float H[NUMV][NUMX], float R[NUMV], float Z[NUMV],
1370  float Y[NUMV], float P[NUMX][NUMX], float X[NUMX],
1371  uint16_t SensorsUsed)
1372 {
1373  float HP[NUMX], HPHR, Error;
1374  uint8_t i, j, k, m;
1375 
1376  for (m = 0; m < NUMV; m++) {
1377 
1378  if (SensorsUsed & (0x01 << m)) { // use this sensor for update
1379 
1380  for (j = 0; j < NUMX; j++) { // Find Hp = H*P
1381  HP[j] = 0;
1382  for (k = 0; k < NUMX; k++)
1383  HP[j] += H[m][k] * P[k][j];
1384  }
1385  HPHR = R[m]; // Find HPHR = H*P*H' + R
1386  for (k = 0; k < NUMX; k++)
1387  HPHR += HP[k] * H[m][k];
1388 
1389  for (k = 0; k < NUMX; k++)
1390  K[k][m] = HP[k] / HPHR; // find K = HP/HPHR
1391 
1392  for (i = 0; i < NUMX; i++) { // Find P(m)= P(m-1) + K*HP
1393  for (j = i; j < NUMX; j++)
1394  P[i][j] = P[j][i] =
1395  P[i][j] - K[i][m] * HP[j];
1396  }
1397 
1398  Error = Z[m] - Y[m];
1399  for (i = 0; i < NUMX; i++) // Find X(m)= X(m-1) + K*Error
1400  X[i] = X[i] + K[i][m] * Error;
1401 
1402  }
1403  }
1404 }
