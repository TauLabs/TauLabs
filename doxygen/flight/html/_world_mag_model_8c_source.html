00320 int WMM_ComputeSphericalHarmonicVariables(WMMtype_CoordSpherical *CoordSpherical, uint16_t nMax, WMMtype_SphericalHarmonicVariables *SphVariables)
00321 
00322    /* Computes Spherical variables
00323       Variables computed are (a/r)^(n+2), cos_m(lamda) and sin_m(lambda) for spherical harmonic
00324       summations. (Equations 10-12 in the WMM Technical Report)
00325       INPUT   Ellip  data  structure with the following elements
00326       float a; semi-major axis of the ellipsoid
00327       float b; semi-minor axis of the ellipsoid
00328       float fla;  flattening
00329       float epssq; first eccentricity squared
00330       float eps;  first eccentricity
00331       float re; mean radius of  ellipsoid
00332       CoordSpherical    A data structure with the following elements
00333       float lambda; ( longitude)
00334       float phig; ( geocentric latitude )
00335       float r;            ( distance from the center of the ellipsoid)
00336       nMax   integer     ( Maxumum degree of spherical harmonic secular model)\
00337 
00338       OUTPUT  SphVariables  Pointer to the   data structure with the following elements
00339       float RelativeRadiusPower[WMM_MAX_MODEL_DEGREES+1];   [earth_reference_radius_km  sph. radius ]^n
00340       float cos_mlambda[WMM_MAX_MODEL_DEGREES+1]; cp(m)  - cosine of (mspherical coord. longitude)
00341       float sin_mlambda[WMM_MAX_MODEL_DEGREES+1];  sp(m)  - sine of (mspherical coord. longitude)
00342       CALLS : none
00343     */
00344 {
00345         float cos_lambda, sin_lambda;
00346         uint16_t m, n;
00347 
00348         cos_lambda = cosf(CoordSpherical->lambda * DEG2RAD);
00349         sin_lambda = sinf(CoordSpherical->lambda * DEG2RAD);
00350 
00351         /* for n = 0 ... model_order, compute (Radius of Earth / Spherica radius r)^(n+2)
00352            for n  1..nMax-1 (this is much faster than calling pow MAX_N+1 times).      */
00353 
00354         SphVariables->RelativeRadiusPower[0] = (Ellip->re / CoordSpherical->r) * (Ellip->re / CoordSpherical->r);
00355         for (n = 1; n <= nMax; n++)
00356                 SphVariables->RelativeRadiusPower[n] = SphVariables->RelativeRadiusPower[n - 1] * (Ellip->re / CoordSpherical->r);
00357 
00358         /*
00359            Compute cos(m*lambda), sin(m*lambda) for m = 0 ... nMax
00360            cos(a + b) = cos(a)*cos(b) - sin(a)*sin(b)
00361            sin(a + b) = cos(a)*sin(b) + sin(a)*cos(b)
00362          */
00363         SphVariables->cos_mlambda[0] = 1.0;
00364         SphVariables->sin_mlambda[0] = 0.0;
00365 
00366         SphVariables->cos_mlambda[1] = cos_lambda;
00367         SphVariables->sin_mlambda[1] = sin_lambda;
00368         for (m = 2; m <= nMax; m++)
00369         {
00370                 SphVariables->cos_mlambda[m] = SphVariables->cos_mlambda[m - 1] * cos_lambda - SphVariables->sin_mlambda[m - 1] * sin_lambda;
00371                 SphVariables->sin_mlambda[m] = SphVariables->cos_mlambda[m - 1] * sin_lambda + SphVariables->sin_mlambda[m - 1] * cos_lambda;
00372         }
00373 
00374         return 0;   // OK
00375 }

00411 int WMM_Summation(WMMtype_LegendreFunction * LegendreFunction,
00412                        WMMtype_SphericalHarmonicVariables * SphVariables,
00413                        WMMtype_CoordSpherical * CoordSpherical, WMMtype_MagneticResults * MagneticResults)
00414 {
00415         /* Computes Geomagnetic Field Elements X, Y and Z in Spherical coordinate system using
00416            spherical harmonic summation.
00417 
00418            The vector Magnetic field is given by -grad V, where V is Geomagnetic scalar potential
00419            The gradient in spherical coordinates is given by:
00420 
00421            dV ^     1 dV ^        1     dV ^
00422            grad V = -- r  +  - -- t  +  -------- -- p
00423            dr       r dt       r sin(t) dp
00424 
00425            INPUT :  LegendreFunction
00426            MagneticModel
00427            SphVariables
00428            CoordSpherical
00429            OUTPUT : MagneticResults
00430 
00431            CALLS : WMM_SummationSpecial
00432 
00433            Manoj Nair, June, 2009 Manoj.C.Nair@Noaa.Gov
00434          */
00435 
00436     uint16_t m, n, index;
00437         float cos_phi;
00438 
00439         MagneticResults->Bz = 0.0;
00440         MagneticResults->By = 0.0;
00441         MagneticResults->Bx = 0.0;
00442 
00443         for (n = 1; n <= MagneticModel->nMax; n++)
00444         {
00445                 for (m = 0; m <= n; m++)
00446                 {
00447                         index = (n * (n + 1) / 2 + m);
00448 
00449 /*                  nMax        (n+2)     n     m            m           m
00450         Bz =   -SUM (a/r)   (n+1) SUM  [g cos(m p) + h sin(m p)] P (sin(phi))
00451                         n=1                   m=0   n            n           n  */
00452 /* Equation 12 in the WMM Technical report.  Derivative with respect to radius.*/
00453                         MagneticResults->Bz -=
00454                             SphVariables->RelativeRadiusPower[n] *
00455                             (WMM_get_main_field_coeff_g(index) *
00456                              SphVariables->cos_mlambda[m] + WMM_get_main_field_coeff_h(index) * SphVariables->sin_mlambda[m])
00457                             * (float)(n + 1) * LegendreFunction->Pcup[index];
00458 
00459 /*                1 nMax  (n+2)    n     m            m           m
00460         By =    SUM (a/r) (m)  SUM  [g cos(m p) + h sin(m p)] dP (sin(phi))
00461                    n=1             m=0   n            n           n  */
00462 /* Equation 11 in the WMM Technical report. Derivative with respect to longitude, divided by radius. */
00463                         MagneticResults->By +=
00464                             SphVariables->RelativeRadiusPower[n] *
00465                             (WMM_get_main_field_coeff_g(index) *
00466                              SphVariables->sin_mlambda[m] - WMM_get_main_field_coeff_h(index) * SphVariables->cos_mlambda[m])
00467                             * (float)(m) * LegendreFunction->Pcup[index];
00468 /*                 nMax  (n+2) n     m            m           m
00469         Bx = - SUM (a/r)   SUM  [g cos(m p) + h sin(m p)] dP (sin(phi))
00470                    n=1         m=0   n            n           n  */
00471 /* Equation 10  in the WMM Technical report. Derivative with respect to latitude, divided by radius. */
00472 
00473                         MagneticResults->Bx -=
00474                             SphVariables->RelativeRadiusPower[n] *
00475                             (WMM_get_main_field_coeff_g(index) *
00476                              SphVariables->cos_mlambda[m] + WMM_get_main_field_coeff_h(index) * SphVariables->sin_mlambda[m])
00477                             * LegendreFunction->dPcup[index];
00478 
00479                 }
00480         }
00481 
00482         cos_phi = cosf(CoordSpherical->phig * DEG2RAD);
00483         if (fabs(cos_phi) > 1.0e-10)
00484         {
00485                 MagneticResults->By = MagneticResults->By / cos_phi;
00486         }
00487         else
00488         {
00489         /* Special calculation for component - By - at Geographic poles.
00490          * If the user wants to avoid using this function,  please make sure that
00491          * the latitude is not exactly +/-90. An option is to make use the function
00492          * WMM_CheckGeographicPoles.
00493          */
00494                 if (WMM_SummationSpecial(SphVariables, CoordSpherical, MagneticResults) < 0)
00495                     return -1;  // error
00496         }
00497 
00498         return 0;   // OK
00499 }

00551         Bx = - SUM (a/r)   SUM  [g cos(m p) + h sin(m p)] dP (sin(phi))
00552                    n=1         m=0   n            n           n  */
00553 /* Derivative with respect to latitude, divided by radius. */
00554 
00555                         MagneticResults->Bx -=
00556                             SphVariables->RelativeRadiusPower[n] *
00557                             (WMM_get_secular_var_coeff_g(index) *
00558                              SphVariables->cos_mlambda[m] + WMM_get_secular_var_coeff_h(index) * SphVariables->sin_mlambda[m])
00559                             * LegendreFunction->dPcup[index];
00560                 }
00561         }
00562         cos_phi = cosf(CoordSpherical->phig * DEG2RAD);
00563         if (fabs(cos_phi) > 1.0e-10)
00564         {
00565                 MagneticResults->By = MagneticResults->By / cos_phi;
00566         }
00567         else
00568                 /* Special calculation for component By at Geographic poles */
00569         {
00570                 if (WMM_SecVarSummationSpecial(SphVariables, CoordSpherical, MagneticResults) < 0)
00571                     return -1;  // error
00572         }
00573 
00574     return 0;   // OK
00575 }

00653                                 float Bx;    ( North )
00654                                 float By;         ( East )

00687 int WMM_PcupHigh(float *Pcup, float *dPcup, float x, uint16_t nMax)
00688 
00689 /*      This function evaluates all of the Schmidt-semi normalized associated Legendre
00690         functions up to degree nMax. The functions are initially scaled by
00691         10^280 sin^m in order to minimize the effects of underflow at large m
00692         near the poles (see Holmes and Featherstone 2002, J. Geodesy, 76, 279-299).
00693         Note that this function performs the same operation as WMM_PcupLow.
00694         However this function also can be used for high degree (large nMax) models.
00695 
00696         Calling Parameters:
00697                 INPUT
00698                         nMax:    Maximum spherical harmonic degree to compute.
00699                         x:              cos(colatitude) or sin(latitude).
00700 
00701                 OUTPUT
00702                         Pcup:   A vector of all associated Legendgre polynomials evaluated at
00703                                         x up to nMax. The lenght must by greater or equal to (nMax+1)*(nMax+2)/2.
00704                   dPcup:   Derivative of Pcup(x) with respect to latitude
00705 
00706                 CALLS : none
00707         Notes:
00708 
00709   Adopted from the FORTRAN code written by Mark Wieczorek September 25, 2005.
00710 
00711   Manoj Nair, Nov, 2009 Manoj.C.Nair@Noaa.Gov
00712 
00713   Change from the previous version
00714   The prevous version computes the derivatives as
00715   dP(n,m)(x)/dx, where x = sin(latitude) (or cos(colatitude) ).
00716   However, the WMM Geomagnetic routines requires dP(n,m)(x)/dlatitude.
00717   Hence the derivatives are multiplied by sin(latitude).
00718   Removed the options for CS phase and normalizations.
00719 
00720   Note: In geomagnetism, the derivatives of ALF are usually found with
00721   respect to the colatitudes. Here the derivatives are found with respect
00722   to the latitude. The difference is a sign reversal for the derivative of
00723   the Associated Legendre Functions.
00724 
00725   The derivates can't be computed for latitude = |90| degrees.
00726         */
00727 {
00728     uint16_t    k, kstart, m, n;
00729     float       pm2, pm1, pmm, plm, rescalem, z, scalef;
00730 
00731         float       *f1 = (float *) MALLOC(sizeof(float) * NUMPCUP);
00732         float       *f2 = (float *) MALLOC(sizeof(float) * NUMPCUP);
00733         float       *PreSqr = (float *) MALLOC(sizeof(float) * NUMPCUP);
00734 
00735     if (!PreSqr || !f2 || !f1)
00736     {   // memory allocation error
00737         if (PreSqr) FREE(PreSqr);
00738         if (f2) FREE(f2);
00739         if (f1) FREE(f1);
00740 
00741         return -1;
00742     }
00743 
00744         if (fabs(x) == 1.0)
00745         {
00746             FREE(PreSqr);
00747             FREE(f2);
00748             FREE(f1);
00749 
00750                 // printf("Error in PcupHigh: derivative cannot be calculated at poles\n");
00751                 return -2;
00752         }
00753 
00754         scalef = 1.0e-280;
00755 
00756         for (n = 0; n <= 2 * nMax + 1; ++n)
00757                 PreSqr[n] = sqrtf((float)(n));
00758 
00759         k = 2;
00760 
00761         for (n = 2; n <= nMax; n++)
00762         {
00763                 k = k + 1;
00764                 f1[k] = (float)(2 * n - 1) / (float)(n);
00765                 f2[k] = (float)(n - 1) / (float)(n);
00766                 for (m = 1; m <= n - 2; m++)
00767                 {
00768                         k = k + 1;
00769                         f1[k] = (float)(2 * n - 1) / PreSqr[n + m] / PreSqr[n - m];
00770                         f2[k] = PreSqr[n - m - 1] * PreSqr[n + m - 1] / PreSqr[n + m] / PreSqr[n - m];
00771                 }
00772                 k = k + 2;
00773         }
00774 
00775         /*z = sin (geocentric latitude) */
00776         z = sqrtf((1.0 - x) * (1.0 + x));
00777         pm2 = 1.0;
00778         Pcup[0] = 1.0;
00779         dPcup[0] = 0.0;
00780         if (nMax == 0)
00781     {
00782         FREE(PreSqr);
00783         FREE(f2);
00784         FREE(f1);
00785         return -3;
00786     }
00787         pm1 = x;
00788         Pcup[1] = pm1;
00789         dPcup[1] = z;
00790         k = 1;
00791 
00792         for (n = 2; n <= nMax; n++)
00793         {
00794                 k = k + n;
00795                 plm = f1[k] * x * pm1 - f2[k] * pm2;
00796                 Pcup[k] = plm;
00797                 dPcup[k] = (float)(n) * (pm1 - x * plm) / z;
00798                 pm2 = pm1;
00799                 pm1 = plm;
00800         }
00801 
00802         pmm = PreSqr[2] * scalef;
00803         rescalem = 1.0 / scalef;
00804         kstart = 0;
00805 
00806         for (m = 1; m <= nMax - 1; ++m)
00807         {
00808                 rescalem = rescalem * z;
00809 
00810                 /* Calculate Pcup(m,m) */
00811                 kstart = kstart + m + 1;
00812                 pmm = pmm * PreSqr[2 * m + 1] / PreSqr[2 * m];
00813                 Pcup[kstart] = pmm * rescalem / PreSqr[2 * m + 1];
00814                 dPcup[kstart] = -((float)(m) * x * Pcup[kstart] / z);
00815                 pm2 = pmm / PreSqr[2 * m + 1];
00816                 /* Calculate Pcup(m+1,m) */
00817                 k = kstart + m + 1;
00818                 pm1 = x * PreSqr[2 * m + 1] * pm2;
00819                 Pcup[k] = pm1 * rescalem;
00820                 dPcup[k] = ((pm2 * rescalem) * PreSqr[2 * m + 1] - x * (float)(m + 1) * Pcup[k]) / z;
00821                 /* Calculate Pcup(n,m) */
00822                 for (n = m + 2; n <= nMax; ++n)
00823                 {
00824                         k = k + n;
00825                         plm = x * f1[k] * pm1 - f2[k] * pm2;
00826                         Pcup[k] = plm * rescalem;
00827                         dPcup[k] = (PreSqr[n + m] * PreSqr[n - m] * (pm1 * rescalem) - (float)(n) * x * Pcup[k]) / z;
00828                         pm2 = pm1;
00829                         pm1 = plm;
00830                 }
00831         }
00832 
00833         /* Calculate Pcup(nMax,nMax) */
00834         rescalem = rescalem * z;
00835         kstart = kstart + m + 1;
00836         pmm = pmm / PreSqr[2 * nMax];
00837         Pcup[kstart] = pmm * rescalem;
00838         dPcup[kstart] = -(float)(nMax) * x * Pcup[kstart] / z;
00839 
00840         // *********
00841         // free allocated memory
00842 
00843         FREE(PreSqr);
00844     FREE(f2);
00845         FREE(f1);
00846 
00847     // *********
00848 
00849         return 0;   // OK
00850 }

00852 int WMM_PcupLow(float *Pcup, float *dPcup, float x, uint16_t nMax)
00853 
00854 /*   This function evaluates all of the Schmidt-semi normalized associated Legendre
00855         functions up to degree nMax.
00856 
00857         Calling Parameters:
00858                 INPUT
00859                         nMax:    Maximum spherical harmonic degree to compute.
00860                         x:              cos(colatitude) or sin(latitude).
00861 
00862                 OUTPUT
00863                         Pcup:   A vector of all associated Legendgre polynomials evaluated at
00864                                         x up to nMax.
00865                    dPcup: Derivative of Pcup(x) with respect to latitude
00866 
00867         Notes: Overflow may occur if nMax > 20 , especially for high-latitudes.
00868         Use WMM_PcupHigh for large nMax.
00869 
00870    Writted by Manoj Nair, June, 2009 . Manoj.C.Nair@Noaa.Gov.
00871 
00872   Note: In geomagnetism, the derivatives of ALF are usually found with
00873   respect to the colatitudes. Here the derivatives are found with respect
00874   to the latitude. The difference is a sign reversal for the derivative of
00875   the Associated Legendre Functions.
00876 */
00877 {
00878     uint16_t    n, m, index, index1, index2;
00879     float       k, z;
00880 
00881     float       *schmidtQuasiNorm = (float *) MALLOC(sizeof(float) * NUMPCUP);
00882     if (!schmidtQuasiNorm)
00883     {   // memory allocation error
00884         return -1;
00885     }
00886 
00887         Pcup[0] = 1.0;
00888         dPcup[0] = 0.0;
00889 
00890         /*sin (geocentric latitude) - sin_phi */
00891         z = sqrtf((1.0 - x) * (1.0 + x));
00892 
00893         /*       First, Compute the Gauss-normalized associated Legendre  functions */
00894         for (n = 1; n <= nMax; n++)
00895         {
00896                 for (m = 0; m <= n; m++)
00897                 {
00898                         index = (n * (n + 1) / 2 + m);
00899                         if (n == m)
00900                         {
00901                                 index1 = (n - 1) * n / 2 + m - 1;
00902                                 Pcup[index] = z * Pcup[index1];
00903                                 dPcup[index] = z * dPcup[index1] + x * Pcup[index1];
00904                         }
00905                         else
00906                         if (n == 1 && m == 0)
00907                         {
00908                                 index1 = (n - 1) * n / 2 + m;
00909                                 Pcup[index] = x * Pcup[index1];
00910                                 dPcup[index] = x * dPcup[index1] - z * Pcup[index1];
00911                         }
00912                         else
00913                         if (n > 1 && n != m)
00914                         {
00915                                 index1 = (n - 2) * (n - 1) / 2 + m;
00916                                 index2 = (n - 1) * n / 2 + m;
00917                                 if (m > n - 2)
00918                                 {
00919                                         Pcup[index] = x * Pcup[index2];
00920                                         dPcup[index] = x * dPcup[index2] - z * Pcup[index2];
00921                                 }
00922                                 else
00923                                 {
00924                                         k = (float)(((n - 1) * (n - 1)) - (m * m)) / (float)((2 * n - 1)
00925                                                                                              * (2 * n - 3));
00926                                         Pcup[index] = x * Pcup[index2] - k * Pcup[index1];
00927                                         dPcup[index] = x * dPcup[index2] - z * Pcup[index2] - k * dPcup[index1];
00928                                 }
00929                         }
00930                 }
00931         }
00932 /*Compute the ration between the Gauss-normalized associated Legendre
00933   functions and the Schmidt quasi-normalized version. This is equivalent to
00934   sqrt((m==0?1:2)*(n-m)!/(n+m!))*(2n-1)!!/(n-m)!  */
00935 
00936         schmidtQuasiNorm[0] = 1.0;
00937         for (n = 1; n <= nMax; n++)
00938         {
00939                 index = (n * (n + 1) / 2);
00940                 index1 = (n - 1) * n / 2;
00941                 /* for m = 0 */
00942                 schmidtQuasiNorm[index] = schmidtQuasiNorm[index1] * (float)(2 * n - 1) / (float)n;
00943 
00944                 for (m = 1; m <= n; m++)
00945                 {
00946                         index = (n * (n + 1) / 2 + m);
00947                         index1 = (n * (n + 1) / 2 + m - 1);
00948                         schmidtQuasiNorm[index] = schmidtQuasiNorm[index1] * sqrtf((float)((n - m + 1) * (m == 1 ? 2 : 1)) / (float)(n + m));
00949                 }
00950 
00951         }
00952 
00953 /* Converts the  Gauss-normalized associated Legendre
00954           functions to the Schmidt quasi-normalized version using pre-computed
00955           relation stored in the variable schmidtQuasiNorm */
00956 
00957         for (n = 1; n <= nMax; n++)
00958         {
00959                 for (m = 0; m <= n; m++)
00960                 {
00961                         index = (n * (n + 1) / 2 + m);
00962                         Pcup[index] = Pcup[index] * schmidtQuasiNorm[index];
00963                         dPcup[index] = -dPcup[index] * schmidtQuasiNorm[index];
00964                         /* The sign is changed since the new WMM routines use derivative with respect to latitude
00965                            insted of co-latitude */
00966                 }
00967         }
00968 
00969         FREE(schmidtQuasiNorm);
00970 
00971         return 0;   // OK
00972 }

00974 int WMM_SummationSpecial(WMMtype_SphericalHarmonicVariables *
00975                               SphVariables, WMMtype_CoordSpherical * CoordSpherical, WMMtype_MagneticResults * MagneticResults)
00976         /* Special calculation for the component By at Geographic poles.
00977            Manoj Nair, June, 2009 manoj.c.nair@noaa.gov
00978            INPUT: MagneticModel
00979            SphVariables
00980            CoordSpherical
00981            OUTPUT: MagneticResults
00982            CALLS : none
00983            See Section 1.4, "SINGULARITIES AT THE GEOGRAPHIC POLES", WMM Technical report
00984 
00985          */
00986 {
00987     uint16_t    n, index;
00988     float       k, sin_phi;
00989     float       schmidtQuasiNorm1;
00990     float       schmidtQuasiNorm2;
00991     float       schmidtQuasiNorm3;
00992 
00993     float       *PcupS = (float *) MALLOC(sizeof(float) * NUMPCUPS);
00994     if (!PcupS)
00995         return -1;  // memory allocation error
00996 
00997         PcupS[0] = 1;
00998         schmidtQuasiNorm1 = 1.0;
00999 
01000         MagneticResults->By = 0.0;
01001         sin_phi = sinf(CoordSpherical->phig * DEG2RAD);
01002 
01003         for (n = 1; n <= MagneticModel->nMax; n++)
01004         {
01005 
01006                 /*Compute the ration between the Gauss-normalized associated Legendre
01007                    functions and the Schmidt quasi-normalized version. This is equivalent to
01008                    sqrt((m==0?1:2)*(n-m)!/(n+m!))*(2n-1)!!/(n-m)!  */
01009 
01010                 index = (n * (n + 1) / 2 + 1);
01011                 schmidtQuasiNorm2 = schmidtQuasiNorm1 * (float)(2 * n - 1) / (float)n;
01012                 schmidtQuasiNorm3 = schmidtQuasiNorm2 * sqrtf((float)(n * 2) / (float)(n + 1));
01013                 schmidtQuasiNorm1 = schmidtQuasiNorm2;
01014                 if (n == 1)
01015                 {
01016                         PcupS[n] = PcupS[n - 1];
01017                 }
01018                 else
01019                 {
01020                         k = (float)(((n - 1) * (n - 1)) - 1) / (float)((2 * n - 1) * (2 * n - 3));
01021                         PcupS[n] = sin_phi * PcupS[n - 1] - k * PcupS[n - 2];
01022                 }
01023 
01024 /*                1 nMax  (n+2)    n     m            m           m
01025         By =    SUM (a/r) (m)  SUM  [g cos(m p) + h sin(m p)] dP (sin(phi))
01026                    n=1             m=0   n            n           n  */
01027 /* Equation 11 in the WMM Technical report. Derivative with respect to longitude, divided by radius. */
01028 
01029                 MagneticResults->By +=
01030                     SphVariables->RelativeRadiusPower[n] *
01031                     (WMM_get_main_field_coeff_g(index) *
01032                      SphVariables->sin_mlambda[1] - WMM_get_main_field_coeff_h(index) * SphVariables->cos_mlambda[1])
01033                     * PcupS[n] * schmidtQuasiNorm3;
01034         }
01035 
01036         FREE(PcupS);
01037 
01038         return 0;   // OK
01039 }

01041 int WMM_SecVarSummationSpecial(WMMtype_SphericalHarmonicVariables *
01042                                     SphVariables, WMMtype_CoordSpherical * CoordSpherical, WMMtype_MagneticResults * MagneticResults)

01045 
01046            INPUT: MagneticModel
01047            SphVariables
01048            CoordSpherical
01049            OUTPUT: MagneticResults
01050            CALLS : none
01051 
01052          */

01054     float       k, sin_phi;
01055     float       schmidtQuasiNorm1;
01056     float       schmidtQuasiNorm2;
01057     float       schmidtQuasiNorm3;
01058 
01059     float       *PcupS = (float *) MALLOC(sizeof(float) * NUMPCUPS);
01060     if (!PcupS)
01061         return -1;  // memory allocation error
01062 
01063         PcupS[0] = 1;
01064         schmidtQuasiNorm1 = 1.0;

01077                         PcupS[n] = PcupS[n - 1];

01079                 else

01181 int WMM_DateToYear(uint16_t month, uint16_t day, uint16_t year)
01182 // Converts a given calendar date into a decimal year
01183 {
01184         uint16_t temp = 0;      // Total number of days
01185         uint16_t MonthDays[13] = { 0, 31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31 };
01186         uint16_t ExtraDay = 0;
01187         uint16_t i;
01188 
01189         if ((year % 4 == 0 && year % 100 != 0) || (year % 400 == 0))
01190                 ExtraDay = 1;
01191         MonthDays[2] += ExtraDay;
01192 
01193         /******************Validation********************************/
01194 
01195         if (month <= 0 || month > 12)
01196                 return -1;  // error
01197 
01198         if (day <= 0 || day > MonthDays[month])
01199                 return -2;  // error
01200 
01201         /****************Calculation of t***************************/
01202         for (i = 1; i <= month; i++)
01203                 temp += MonthDays[i - 1];
01204         temp += day;
01205         
01206         decimal_date = year + (temp - 1) / (365.0 + ExtraDay);
01207 
01208         return 0;   // OK
01209 }

01211 int WMM_GeodeticToSpherical(WMMtype_CoordGeodetic * CoordGeodetic, WMMtype_CoordSpherical * CoordSpherical)
01212 // Converts Geodetic coordinates to Spherical coordinates
01213 // Convert geodetic coordinates, (defined by the WGS-84
01214 // reference ellipsoid), to Earth Centered Earth Fixed Cartesian
01215 // coordinates, and then to spherical coordinates.
01216 {
01217         float CosLat, SinLat, rc, xp, zp;       // all local variables
01218 
01219         CosLat = cosf(CoordGeodetic->phi * DEG2RAD);
01220         SinLat = sinf(CoordGeodetic->phi * DEG2RAD);
01221 
01222         // compute the local radius of curvature on the WGS-84 reference ellipsoid
01223         rc = Ellip->a / sqrtf(1.0 - Ellip->epssq * SinLat * SinLat);
01224 
01225         // compute ECEF Cartesian coordinates of specified point (for longitude=0)
01226 
01227         xp = (rc + CoordGeodetic->HeightAboveEllipsoid) * CosLat;
01228         zp = (rc * (1.0 - Ellip->epssq) + CoordGeodetic->HeightAboveEllipsoid) * SinLat;
01229 
01230         // compute spherical radius and angle lambda and phi of specified point
01231 
01232         CoordSpherical->r = sqrtf(xp * xp + zp * zp);
01233         CoordSpherical->phig = asinf(zp / CoordSpherical->r) * RAD2DEG; // geocentric latitude
01234         CoordSpherical->lambda = CoordGeodetic->lambda; // longitude
01235 
01236         return 0;   // OK
01237 }
