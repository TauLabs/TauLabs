00217 {
00218         PIOS_Assert(data);
00219         PIOS_Assert((addr & 0x0001) == 0)
00220

00222

00224                 return -1;

00228

00230         if (!PIOS_Flash_Internal_GetSectorInfo(addr,
00231                                                 &sector_start,
00232                                                 &sector_size)) {
00233                 /* We're asking for an invalid flash address */
00234                 return -2;
00235         }
00236 
00237         /* Ensure that the entire write occurs within the same sector */
00238         if ((uintptr_t)addr + len > sector_start + sector_size) {
00239                 /* Write crosses the end of the sector */
00240                 return -3;
00241         }
00242 
00243         /* Write the data */
00244         for (uint16_t i = 0; i < (len & ~1); i += 2) {
00245                 /* Check if content has been changed prios to write.
00246                  * This should enhance performance and make this compatible with the F3 chip.
00247                  */
00248                 uint8_t temp[2];
00249                 if (PIOS_Flash_Internal_ReadData(flash_id, addr + i, temp, sizeof(temp)) != 0)
00250                         return -4;
00251 
00252                 if (temp[0] == data[i] && temp[1] == data[i + 1])
00253                         continue;
00254 
00255                 FLASH_Status status;
00256                 status = FLASH_ProgramHalfWord(addr + i, data[i + 1] << 8 | data[i]);
00257                 PIOS_Assert(status == FLASH_COMPLETE);
00258         }
00259         /* Handle uneven writes by filling up with 0xff*/
00260         if ((len & 1) != 0) {
00261                 FLASH_Status status;
00262                 status = FLASH_ProgramHalfWord(addr + len - 1, 0xff << 8 | data[len - 1]);
00263                 PIOS_Assert(status == FLASH_COMPLETE);
00264         }
00265 
00266         return 0;
