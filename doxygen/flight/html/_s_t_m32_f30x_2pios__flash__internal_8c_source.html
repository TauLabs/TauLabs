39 static bool PIOS_Flash_Internal_GetSectorInfo(uint32_t address, uint32_t *sector_start, uint32_t *sector_size)
40 {
41  if (address < FLASH_BASE ||
42  address >= FLASH_BASE + PIOS_SYS_getCPUFlashSize())
43  return false;
44 
45  *sector_start = address & (~(STM32F30X_FLASH_SECTOR_SIZE - 1));
46  *sector_size = STM32F30X_FLASH_SECTOR_SIZE;
47 
48  return true;
49 }

122 static int32_t PIOS_Flash_Internal_StartTransaction(uintptr_t flash_id)
123 {
124  struct pios_internal_flash_dev * flash_dev = (struct pios_internal_flash_dev *)flash_id;
125 
126  if (!PIOS_Flash_Internal_Validate(flash_dev))
127  return -1;
128 
129 #if defined(PIOS_INCLUDE_FREERTOS)
130  if (xSemaphoreTake(flash_dev->transaction_lock, portMAX_DELAY) != pdTRUE)
131  return -2;
132 #endif /* defined(PIOS_INCLUDE_FREERTOS) */
133 
134  /* Unlock the internal flash so we can write to it */
135  FLASH_Unlock();
136  return 0;
137 }

157 static int32_t PIOS_Flash_Internal_EraseSector(uintptr_t flash_id, uint32_t addr)
158 {
159  struct pios_internal_flash_dev * flash_dev = (struct pios_internal_flash_dev *)flash_id;
160 
161  if (!PIOS_Flash_Internal_Validate(flash_dev))
162  return -1;

164 #if defined(PIOS_INCLUDE_WDG)
165  PIOS_WDG_Clear();
166 #endif
167 
168  uint32_t sector_start;
169  uint32_t sector_size;
170  if (!PIOS_Flash_Internal_GetSectorInfo(addr,
171  &sector_start,
172  &sector_size)) {
173  /* We're asking for an invalid flash address */
174  return -2;
175  }
176 
177  FLASH_Status ret = FLASH_ErasePage(sector_start);
178  if (ret != FLASH_COMPLETE)

184 static int32_t PIOS_Flash_Internal_ReadData(uintptr_t flash_id, uint32_t addr, uint8_t * data, uint16_t len)
185 {
186  PIOS_Assert(data);
187 
188  struct pios_internal_flash_dev * flash_dev = (struct pios_internal_flash_dev *)flash_id;
189 
190  if (!PIOS_Flash_Internal_Validate(flash_dev))
191  return -1;
192 
193  uint32_t sector_start;
194  uint32_t sector_size;
195 
196  /* Ensure that the base address is in a valid sector */
197  if (!PIOS_Flash_Internal_GetSectorInfo(addr,
198  &sector_start,
199  &sector_size)) {
200  /* We're asking for an invalid flash address */
201  return -2;
202  }
203 
204  /* Ensure that the entire read occurs within the same sector */
205  if ((uintptr_t)addr + len > sector_start + sector_size) {
206  /* Read crosses the end of the sector */
207  return -3;
208  }
209 
210  /* Read the data into the buffer directly */
211  memcpy(data, (void *)addr, len);
212 
213  return 0;
214 }

216 static int32_t PIOS_Flash_Internal_WriteData(uintptr_t flash_id, uint32_t addr, uint8_t * data, uint16_t len)
217 {
218  PIOS_Assert(data);
219  PIOS_Assert((addr & 0x0001) == 0)
220 
221  struct pios_internal_flash_dev * flash_dev = (struct pios_internal_flash_dev *)flash_id;
222 
223  if (!PIOS_Flash_Internal_Validate(flash_dev))
224  return -1;
225 
226  uint32_t sector_start;
227  uint32_t sector_size;
228 
229  /* Ensure that the base address is in a valid sector */
230  if (!PIOS_Flash_Internal_GetSectorInfo(addr,
231  &sector_start,
232  &sector_size)) {
233  /* We're asking for an invalid flash address */
234  return -2;
235  }
236 
237  /* Ensure that the entire write occurs within the same sector */
238  if ((uintptr_t)addr + len > sector_start + sector_size) {
239  /* Write crosses the end of the sector */
240  return -3;
241  }
242 
243  /* Write the data */
244  for (uint16_t i = 0; i < (len & ~1); i += 2) {
245  /* Check if content has been changed prios to write.
246  * This should enhance performance and make this compatible with the F3 chip.
247  */
248  uint8_t temp[2];
249  if (PIOS_Flash_Internal_ReadData(flash_id, addr + i, temp, sizeof(temp)) != 0)
250  return -4;
251 
252  if (temp[0] == data[i] && temp[1] == data[i + 1])
253  continue;
254 
255  FLASH_Status status;
256  status = FLASH_ProgramHalfWord(addr + i, data[i + 1] << 8 | data[i]);
257  PIOS_Assert(status == FLASH_COMPLETE);
258  }
259  /* Handle uneven writes by filling up with 0xff*/
260  if ((len & 1) != 0) {
261  FLASH_Status status;
262  status = FLASH_ProgramHalfWord(addr + len - 1, 0xff << 8 | data[len - 1]);
263  PIOS_Assert(status == FLASH_COMPLETE);
264  }
265 
266  return 0;
267 }
