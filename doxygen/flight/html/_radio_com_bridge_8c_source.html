00320 
00321         if (state == UAVTALK_STATE_COMPLETE)
00322         {
00323                 // Is this a local UAVObject?
00324                 // We only generate GcsReceiver ojects, we don't consume them.
00325                 if ((iproc->obj != NULL) && (iproc->objId != GCSRECEIVER_OBJID))
00326                 {
00327                         // We treat the ObjectPersistence object differently
00328                         if(iproc->objId == OBJECTPERSISTENCE_OBJID)
00329                         {
00330                                 // Unpack object, if the instance does not exist it will be created!
00331                                 UAVObjUnpack(iproc->obj, iproc->instId, connection->rxBuffer);
00332 
00333                                 // Get the ObjectPersistence object.
00334                                 ObjectPersistenceData obj_per;
00335                                 ObjectPersistenceGet(&obj_per);
00336 
00337                                 // Is this concerning or setting object?
00338                                 if (obj_per.ObjectID == OPLINKSETTINGS_OBJID)
00339                                 {
00340                                         // Queue up the ACK.
00341                                         queueEvent(data->uavtalkEventQueue, (void*)iproc->obj, iproc->instId, EV_SEND_ACK);
00342 
00343                                         // Is this a save, load, or delete?
00344                                         bool success = true;
00345                                         switch (obj_per.Operation)
00346                                         {
00347                                         case OBJECTPERSISTENCE_OPERATION_LOAD:
