178 static void telemetryTxTask(void *parameters)
179 {
180  UAVObjEvent ev;
181 
182  // Loop forever
183  while (1) {
184 #ifdef PIOS_INCLUDE_WDG
185  PIOS_WDG_UpdateFlag(PIOS_WDG_TELEMETRY);
186 #endif
187  // Wait for queue message
188  if (xQueueReceive(data->uavtalkEventQueue, &ev, MAX_PORT_DELAY) == pdTRUE) {
189  if ((ev.event == EV_UPDATED) || (ev.event == EV_UPDATE_REQ))
190  {
191  // Send update (with retries)
192  uint32_t retries = 0;
193  int32_t success = -1;
194  while (retries < MAX_RETRIES && success == -1) {
195  success = UAVTalkSendObject(data->outUAVTalkCon, ev.obj, 0, 0, RETRY_TIMEOUT_MS) == 0;
196  if (!success)
197  ++retries;
198  }
199  data->comTxRetries += retries;
200  }
201  else if(ev.event == EV_SEND_ACK)
202  {
203  // Send the ACK
204  uint32_t retries = 0;
205  int32_t success = -1;
206  while (retries < MAX_RETRIES && success == -1) {
207  success = UAVTalkSendAck(data->outUAVTalkCon, ev.obj, ev.instId) == 0;
208  if (!success)
209  ++retries;
210  }
211  data->comTxRetries += retries;
212  }
213  else if(ev.event == EV_SEND_NACK)
214  {
215  // Send the NACK
216  uint32_t retries = 0;
217  int32_t success = -1;
218  while (retries < MAX_RETRIES && success == -1) {
219  success = UAVTalkSendNack(data->outUAVTalkCon, UAVObjGetID(ev.obj)) == 0;
220  if (!success)
221  ++retries;
222  }
223  data->comTxRetries += retries;
224  }
225  }
226  }
227 }

232 static void radioRxTask(void *parameters)
233 {
234  // Task loop
235  while (1) {
236 #ifdef PIOS_INCLUDE_WDG
237  PIOS_WDG_UpdateFlag(PIOS_WDG_RADIORX);
238 #endif
239  uint8_t serial_data[1];
240  uint16_t bytes_to_process = PIOS_COM_ReceiveBuffer(PIOS_COM_RADIO, serial_data, sizeof(serial_data), MAX_PORT_DELAY);
241  if (bytes_to_process > 0)
242  for (uint8_t i = 0; i < bytes_to_process; i++)
243  if (UAVTalkRelayInputStream(data->outUAVTalkCon, serial_data[i]) == UAVTALK_STATE_ERROR)
244  data->UAVTalkErrors++;
245  }
246 }

251 static void radioTxTask(void *parameters)
252 {
253  // Task loop
254  while (1) {
255  uint32_t inputPort = PIOS_COM_TELEMETRY;
256 #ifdef PIOS_INCLUDE_WDG
257  PIOS_WDG_UpdateFlag(PIOS_WDG_RADIOTX);

263 #endif /* PIOS_INCLUDE_USB */
264  if(inputPort)
265  {
266  uint8_t serial_data[1];
267  uint16_t bytes_to_process = PIOS_COM_ReceiveBuffer(inputPort, serial_data, sizeof(serial_data), MAX_PORT_DELAY);
268  if (bytes_to_process > 0)
269  for (uint8_t i = 0; i < bytes_to_process; i++)
270  ProcessInputStream(data->inUAVTalkCon, serial_data[i]);
271  }
272  }

314 static void ProcessInputStream(UAVTalkConnection connectionHandle, uint8_t rxbyte)
315 {
316  // Keep reading until we receive a completed packet.
317  UAVTalkRxState state = UAVTalkRelayInputStream(connectionHandle, rxbyte);
318  UAVTalkConnectionData *connection = (UAVTalkConnectionData*)(connectionHandle);
319  UAVTalkInputProcessor *iproc = &(connection->iproc);
320 
321  if (state == UAVTALK_STATE_COMPLETE)
322  {
323  // Is this a local UAVObject?
324  // We only generate GcsReceiver ojects, we don't consume them.
325  if ((iproc->obj != NULL) && (iproc->objId != GCSRECEIVER_OBJID))
326  {
327  // We treat the ObjectPersistence object differently
328  if(iproc->objId == OBJECTPERSISTENCE_OBJID)
329  {
330  // Unpack object, if the instance does not exist it will be created!
331  UAVObjUnpack(iproc->obj, iproc->instId, connection->rxBuffer);
332 
333  // Get the ObjectPersistence object.
334  ObjectPersistenceData obj_per;
335  ObjectPersistenceGet(&obj_per);
336 
337  // Is this concerning or setting object?
338  if (obj_per.ObjectID == OPLINKSETTINGS_OBJID)
339  {
340  // Queue up the ACK.
341  queueEvent(data->uavtalkEventQueue, (void*)iproc->obj, iproc->instId, EV_SEND_ACK);
342 
343  // Is this a save, load, or delete?
344  bool success = true;
345  switch (obj_per.Operation)
346  {
347  case OBJECTPERSISTENCE_OPERATION_LOAD:
348  {
349 #if defined(PIOS_INCLUDE_FLASH_EEPROM)
350  // Load the settings.
351  OPLinkSettingsData oplinkSettings;
352  if (PIOS_EEPROM_Load((uint8_t*)&oplinkSettings, sizeof(OPLinkSettingsData)) == 0)
353  OPLinkSettingsSet(&oplinkSettings);
354  else
355  success = false;
356 #endif
357  break;
358  }
359  case OBJECTPERSISTENCE_OPERATION_SAVE:
360  {
361 #if defined(PIOS_INCLUDE_FLASH_EEPROM)
362  // Save the settings.
363  OPLinkSettingsData oplinkSettings;
364  OPLinkSettingsGet(&oplinkSettings);
365  int32_t ret = PIOS_EEPROM_Save((uint8_t*)&oplinkSettings, sizeof(OPLinkSettingsData));
366  if (ret != 0)
367  success = false;
368 #endif
369  break;
370  }
371  case OBJECTPERSISTENCE_OPERATION_DELETE:
372  {
373 #if defined(PIOS_INCLUDE_FLASH_EEPROM)
374  // Erase the settings.
375  OPLinkSettingsData oplinkSettings;
376  uint8_t *ptr = (uint8_t*)&oplinkSettings;
377  memset(ptr, 0, sizeof(OPLinkSettingsData));
378  int32_t ret = PIOS_EEPROM_Save(ptr, sizeof(OPLinkSettingsData));
379  if (ret != 0)
380  success = false;
381 #endif
382  break;
383  }
384  default:
385  break;
386  }
387  if (success == true)
388  {
389  obj_per.Operation = OBJECTPERSISTENCE_OPERATION_COMPLETED;
390  ObjectPersistenceSet(&obj_per);
391  }
392  }
393  }
394  else
395  {
396  switch (iproc->type)
397  {
398  case UAVTALK_TYPE_OBJ:
399  // Unpack object, if the instance does not exist it will be created!
400  UAVObjUnpack(iproc->obj, iproc->instId, connection->rxBuffer);
401  break;
402  case UAVTALK_TYPE_OBJ_REQ:
403  // Queue up an object send request.
404  queueEvent(data->uavtalkEventQueue, (void*)iproc->obj, iproc->instId, EV_UPDATE_REQ);
405  break;
406  case UAVTALK_TYPE_OBJ_ACK:
407  if (UAVObjUnpack(iproc->obj, iproc->instId, connection->rxBuffer) == 0)
408  // Queue up an ACK
409  queueEvent(data->uavtalkEventQueue, (void*)iproc->obj, iproc->instId, EV_SEND_ACK);
410  break;
411  }
412  }
413  }
414 
415  } else if(state == UAVTALK_STATE_ERROR) {
416  data->UAVTalkErrors++;
417 
418  // Send a NACK if required.
419  if((iproc->obj) && (iproc->type == UAVTALK_TYPE_OBJ_ACK))
420  // Queue up a NACK
421  queueEvent(data->uavtalkEventQueue, iproc->obj, iproc->instId, EV_SEND_NACK);
422  }
423 }

445 static void configureComCallback(OPLinkSettingsOutputConnectionOptions com_port, OPLinkSettingsComSpeedOptions com_speed)
446 {
447 
448  // Get the settings.
449  OPLinkSettingsData oplinkSettings;
450  OPLinkSettingsGet(&oplinkSettings);
451 
452  // Set the output telemetry port and speed.
453  switch (com_port)
454  {
455  case OPLINKSETTINGS_OUTPUTCONNECTION_REMOTEHID:
456  oplinkSettings.InputConnection = OPLINKSETTINGS_INPUTCONNECTION_HID;
457  break;
458  case OPLINKSETTINGS_OUTPUTCONNECTION_REMOTEVCP:
459  oplinkSettings.InputConnection = OPLINKSETTINGS_INPUTCONNECTION_VCP;
460  break;
461  case OPLINKSETTINGS_OUTPUTCONNECTION_REMOTETELEMETRY:
462  oplinkSettings.InputConnection = OPLINKSETTINGS_INPUTCONNECTION_TELEMETRY;
463  break;
464  case OPLINKSETTINGS_OUTPUTCONNECTION_REMOTEFLEXI:
465  oplinkSettings.InputConnection = OPLINKSETTINGS_INPUTCONNECTION_FLEXI;
466  break;
467  case OPLINKSETTINGS_OUTPUTCONNECTION_TELEMETRY:
468  oplinkSettings.InputConnection = OPLINKSETTINGS_INPUTCONNECTION_HID;
469  break;
470  case OPLINKSETTINGS_OUTPUTCONNECTION_FLEXI:
471  oplinkSettings.InputConnection = OPLINKSETTINGS_INPUTCONNECTION_HID;
472  break;
473  }
474  oplinkSettings.ComSpeed = com_speed;
475 
476  // A non-coordinator modem should not set a remote telemetry connection.
477  oplinkSettings.OutputConnection = OPLINKSETTINGS_OUTPUTCONNECTION_REMOTEHID;
478 
479  // Update the OPLinkSettings object.
480  OPLinkSettingsSet(&oplinkSettings);
481 
482  // Perform the update.
483  updateSettings();
484 }

489 static void updateSettings()
490 {
491 
492  // Get the settings.
493  OPLinkSettingsData oplinkSettings;
494  OPLinkSettingsGet(&oplinkSettings);
495 
496  bool is_coordinator = (oplinkSettings.Coordinator == OPLINKSETTINGS_COORDINATOR_TRUE);
497  if (is_coordinator)
498  {
499  // Set the remote com configuration parameters
500  PIOS_RFM22B_SetRemoteComConfig(pios_rfm22b_id, oplinkSettings.OutputConnection, oplinkSettings.ComSpeed);
501 
502  // Configure the RFM22B device as coordinator or not
503  PIOS_RFM22B_SetCoordinator(pios_rfm22b_id, true);
504 
505  // Set the frequencies.
506  PIOS_RFM22B_SetFrequencyRange(pios_rfm22b_id, oplinkSettings.MinFrequency, oplinkSettings.MaxFrequency);
507 
508  // Set the maximum radio RF power.
509  switch (oplinkSettings.MaxRFPower)
510  {
511  case OPLINKSETTINGS_MAXRFPOWER_125:
512  PIOS_RFM22B_SetTxPower(pios_rfm22b_id, RFM22_tx_pwr_txpow_0);
513  break;
514  case OPLINKSETTINGS_MAXRFPOWER_16:
515  PIOS_RFM22B_SetTxPower(pios_rfm22b_id, RFM22_tx_pwr_txpow_1);
516  break;
517  case OPLINKSETTINGS_MAXRFPOWER_316:
518  PIOS_RFM22B_SetTxPower(pios_rfm22b_id, RFM22_tx_pwr_txpow_2);
519  break;
520  case OPLINKSETTINGS_MAXRFPOWER_63:
521  PIOS_RFM22B_SetTxPower(pios_rfm22b_id, RFM22_tx_pwr_txpow_3);
522  break;
523  case OPLINKSETTINGS_MAXRFPOWER_126:
524  PIOS_RFM22B_SetTxPower(pios_rfm22b_id, RFM22_tx_pwr_txpow_4);
525  break;
526  case OPLINKSETTINGS_MAXRFPOWER_25:
527  PIOS_RFM22B_SetTxPower(pios_rfm22b_id, RFM22_tx_pwr_txpow_5);
528  break;
529  case OPLINKSETTINGS_MAXRFPOWER_50:
530  PIOS_RFM22B_SetTxPower(pios_rfm22b_id, RFM22_tx_pwr_txpow_6);
531  break;
532  case OPLINKSETTINGS_MAXRFPOWER_100:
533  PIOS_RFM22B_SetTxPower(pios_rfm22b_id, RFM22_tx_pwr_txpow_7);
534  break;
535  }
536 
537  // Set the radio destination ID.
538  PIOS_RFM22B_SetDestinationId(pios_rfm22b_id, oplinkSettings.PairID);
539  }
540 
541  // Determine what com ports we're using.
542  switch (oplinkSettings.InputConnection)
543  {
544  case OPLINKSETTINGS_INPUTCONNECTION_VCP:
545  PIOS_COM_TELEMETRY = PIOS_COM_TELEM_VCP;
546  break;
547  case OPLINKSETTINGS_INPUTCONNECTION_TELEMETRY:
548  PIOS_COM_TELEMETRY = PIOS_COM_TELEM_UART_TELEM;
549  break;
550  case OPLINKSETTINGS_INPUTCONNECTION_FLEXI:
551  PIOS_COM_TELEMETRY = PIOS_COM_TELEM_UART_FLEXI;
552  break;
553  default:
554  PIOS_COM_TELEMETRY = 0;
555  break;
556  }
557 
558  switch (oplinkSettings.OutputConnection)
559  {
560  case OPLINKSETTINGS_OUTPUTCONNECTION_FLEXI:
561  PIOS_COM_RADIO = PIOS_COM_TELEM_UART_FLEXI;
562  break;
563  case OPLINKSETTINGS_OUTPUTCONNECTION_TELEMETRY:
564  PIOS_COM_RADIO = PIOS_COM_TELEM_UART_TELEM;
565  break;
566  default:
567  PIOS_COM_RADIO = PIOS_COM_RFM22B;
568  break;
569  }
570 
571  // Configure the com port speeds.
572  switch (oplinkSettings.ComSpeed) {
573  case OPLINKSETTINGS_COMSPEED_2400:
574  if (is_coordinator && PIOS_COM_RADIO) PIOS_COM_ChangeBaud(PIOS_COM_RADIO, 2400);
575  if (PIOS_COM_TELEMETRY) PIOS_COM_ChangeBaud(PIOS_COM_TELEMETRY, 2400);
576  break;
577  case OPLINKSETTINGS_COMSPEED_4800:
578  if (is_coordinator && PIOS_COM_RADIO) PIOS_COM_ChangeBaud(PIOS_COM_RADIO, 4800);
579  if (PIOS_COM_TELEMETRY) PIOS_COM_ChangeBaud(PIOS_COM_TELEMETRY, 4800);
580  break;
581  case OPLINKSETTINGS_COMSPEED_9600:
582  if (is_coordinator && PIOS_COM_RADIO) PIOS_COM_ChangeBaud(PIOS_COM_RADIO, 9600);
583  if (PIOS_COM_TELEMETRY) PIOS_COM_ChangeBaud(PIOS_COM_TELEMETRY, 9600);
584  break;
585  case OPLINKSETTINGS_COMSPEED_19200:
586  if (is_coordinator && PIOS_COM_RADIO) PIOS_COM_ChangeBaud(PIOS_COM_RADIO, 19200);
587  if (PIOS_COM_TELEMETRY) PIOS_COM_ChangeBaud(PIOS_COM_TELEMETRY, 19200);
588  break;
589  case OPLINKSETTINGS_COMSPEED_38400:
590  if (is_coordinator && PIOS_COM_RADIO) PIOS_COM_ChangeBaud(PIOS_COM_RADIO, 38400);
591  if (PIOS_COM_TELEMETRY) PIOS_COM_ChangeBaud(PIOS_COM_TELEMETRY, 38400);
592  break;
