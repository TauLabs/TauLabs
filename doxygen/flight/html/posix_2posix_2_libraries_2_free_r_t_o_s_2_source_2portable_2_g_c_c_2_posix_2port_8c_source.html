210 portSTACK_TYPE *pxPortInitialiseStack( portSTACK_TYPE *pxTopOfStack, pdTASK_CODE pxCode, void *pvParameters )
211 {
212  /* Should actually keep this struct on the stack. */
213  xParams *pxThisThreadParams = pvPortMalloc( sizeof( xParams ) );
214 
215  /* Initialize scheduler during first call */
216  (void)pthread_once( &hSigSetupThread, prvSetupSignalsAndSchedulerPolicy );
217 
222  PORT_ENTER();
223 
224  /* No need to join the threads. */
225  pthread_attr_init( &xThreadAttributes );
226  pthread_attr_setdetachstate( &xThreadAttributes, PTHREAD_CREATE_DETACHED );
227 
228  /* Add the task parameters. */
229  pxThisThreadParams->pxCode = pxCode;
230  pxThisThreadParams->pvParams = pvParameters;
231 
232  lIndexOfLastAddedTask = prvGetFreeThreadState();
233 
234  PORT_LOCK( xYieldingThreadMutex );
235 
236  pxThreads[ lIndexOfLastAddedTask ].threadStatus = THREAD_STARTING;
237  pxThreads[ lIndexOfLastAddedTask ].uxCriticalNesting = 0;
238 
239  /* create the thead */
240  PORT_ASSERT( 0 == pthread_create( &( pxThreads[ lIndexOfLastAddedTask ].hThread ), &xThreadAttributes, prvWaitForStart, (void *)pxThisThreadParams ) );
241 
242  /* Let the task run a bit and wait until it suspends. */
243  while ( pxThreads[ lIndexOfLastAddedTask ].threadStatus == THREAD_STARTING ) sched_yield();
244 
245  /* this ensures the sleeping thread reached deep sleep (and not more) */
246  PORT_UNLOCK( xYieldingThreadMutex );
247 
248  PORT_LEAVE();
249 
250  return pxTopOfStack;
251 }

275 portBASE_TYPE xPortStartScheduler( void )
276 {
277  sigset_t xSignalToBlock;
278 
284  /* do not respond to SUSPEND signal (but all others) */
285  sigemptyset( &xSignalToBlock );
286  (void)pthread_sigmask( SIG_SETMASK, &xSignalToBlock, NULL );
287  sigemptyset(&xSignalToBlock);
288  sigaddset(&xSignalToBlock,SIG_SUSPEND);
289  (void)pthread_sigmask(SIG_BLOCK, &xSignalToBlock, NULL);
290 
291  /* Start the first task. This gives up the RunningThreadMutex*/
292  vPortStartFirstTask();
293 
298  portLONG sleepTimeUS = portTICK_RATE_MICROSECONDS;
299  portLONG actualSleepTime;
300  struct timeval lastTime,currentTime;
301  gettimeofday( &lastTime, NULL );
302  struct timespec wait;
303 
304  while ( pdTRUE != xSchedulerEnd )
305  {
306  /* wait for the specified wait time */
307  wait.tv_sec = sleepTimeUS / 1000000;
308  wait.tv_nsec = 1000 * ( sleepTimeUS % 1000000 );
309  nanosleep( &wait, NULL );
310 
311  /* check the time */
312  gettimeofday( &currentTime, NULL);
313  actualSleepTime = 1000000 * ( currentTime.tv_sec - lastTime.tv_sec ) + ( currentTime.tv_usec - lastTime.tv_usec );
314 
315  /* only hit the tick if we slept at least half the period */
316  if ( actualSleepTime >= sleepTimeUS/2 ) {
317 
318  vPortSystemTickHandler();
319 
320  /* check the time again */
321  gettimeofday( &currentTime, NULL);
322  actualSleepTime = 1000000 * ( currentTime.tv_sec - lastTime.tv_sec ) + ( currentTime.tv_usec - lastTime.tv_usec );
323 
324  /* sleep until the next tick is due */
325  sleepTimeUS += portTICK_RATE_MICROSECONDS;
326  }
327 
328  /* reduce remaining sleep time by the slept time */
329  sleepTimeUS -= actualSleepTime;
330  lastTime = currentTime;
331 
332  /* safety checks */
333  if (sleepTimeUS <=0 || sleepTimeUS >= 3 * portTICK_RATE_MICROSECONDS) sleepTimeUS = portTICK_RATE_MICROSECONDS;
334 
335  }
336 
337  PORT_PRINT( "Cleaning Up, Exiting.\n" );
338  /* Cleanup the mutexes */
339  pthread_mutex_destroy( &xRunningThreadMutex );
340  pthread_mutex_destroy( &xYieldingThreadMutex );
341  pthread_mutex_destroy( &xGuardMutex );
342  vPortFree( (void *)pxThreads );
343 
344  /* Should not get here! */
345  return 0;
346 }

395 void vPortExitCritical( void )
396 {
397  PORT_ENTER();
398 
399  /* Check for unmatched exits. */
400  PORT_ASSERT( prvGetThreadHandleByThread(pthread_self())->uxCriticalNesting > 0 );
401  if ( prvGetThreadHandleByThread(pthread_self())->uxCriticalNesting > 0 )
402  {
403  prvGetThreadHandleByThread(pthread_self())->uxCriticalNesting--;
404  }
405 
406  /* If we have reached 0 then re-enable the interrupts. */
407  if( prvGetThreadHandleByThread(pthread_self())->uxCriticalNesting == 0 )
408  {
409  /* Have we missed ticks? This is the equivalent of pending an interrupt. */
410  if ( pdTRUE == xPendYield )
411  {
412  xPendYield = pdFALSE;
413  prvPortYield();
414  }
415 
416  xInterruptsEnabled = pdTRUE;
417 
418  }
419 
420  PORT_LEAVE();
421 }

429 void prvPortYield()
430 {
431  xThreadState *xTaskToSuspend, *xTaskToResume;
432 
433  /* timer handler should NOT get in our way (just in case) */
434  xInterruptsEnabled = pdFALSE;
435 
436  /* suspend the current task */
437  xTaskToSuspend = prvGetThreadHandleByThread( pthread_self() );
438 
442  PORT_ASSERT( xTaskToSuspend->threadStatus == THREAD_RUNNING );
443 
447  vTaskSwitchContext();
448 
452  xTaskToResume = prvGetThreadHandle( xTaskGetCurrentTaskHandle() );
453  if ( xTaskToSuspend != xTaskToResume )
454  {
455  /* Resume the other thread first */
456  prvResumeThread( xTaskToResume );
457 
458  /* prepare the current task for yielding */
459  xTaskToSuspend->threadStatus = THREAD_YIELDING;
460 
468  while ( xTaskToSuspend->threadStatus == THREAD_YIELDING ) {
469  pthread_kill( xTaskToSuspend->hThread, SIG_SUSPEND );
470  sched_yield();
471  }
472 
477  } else {
481  if (xTaskToResume->uxCriticalNesting==0) {
482  xInterruptsEnabled = pdTRUE;
483  }
484  }
485 
486 }

587 void vPortSystemTickHandler()
588 {
594  PORT_LOCK( xGuardMutex );
595 
596  /* thread MUST be running */
597  if ( prvGetThreadHandle(xTaskGetCurrentTaskHandle())->threadStatus!=THREAD_RUNNING ) {
598  xPendYield = pdTRUE;
599  PORT_UNLOCK( xGuardMutex );
600  return;
601  }
602 
603  /* interrupts MUST be enabled */
604  if ( xInterruptsEnabled != pdTRUE ) {
605  xPendYield = pdTRUE;
606  PORT_UNLOCK( xGuardMutex );
607  return;
608  }
609 
610  /* this should always be true, but it can't harm to check */
611  PORT_ASSERT( prvGetThreadHandle(xTaskGetCurrentTaskHandle())->uxCriticalNesting==0 );
612 
613  /* acquire switching mutex for synchronization */
614  PORT_LOCK(xYieldingThreadMutex);
615 
616  xThreadState *xTaskToSuspend = prvGetThreadHandle( xTaskGetCurrentTaskHandle() );
617 
618 
627  xTaskToSuspend->threadStatus = THREAD_PREEMPTING;
628  while ( xTaskToSuspend->threadStatus != THREAD_SLEEPING ) {
629  pthread_kill( xTaskToSuspend->hThread, SIG_SUSPEND );
630  sched_yield();
631  }
632 
636  PORT_UNLOCK( xYieldingThreadMutex );
637  PORT_LOCK( xRunningThreadMutex );
638 
649  vTaskIncrementTick();
650 
651 
652 #if ( configUSE_PREEMPTION == 1 )
653 
656  vTaskSwitchContext();
657 
658  xTaskToSuspend = prvGetThreadHandle( xTaskGetCurrentTaskHandle() );
659 #endif
660 
664  prvResumeThread( xTaskToSuspend );
665 
669  PORT_UNLOCK( xRunningThreadMutex );
670 
671  /* finish up */
672  PORT_UNLOCK( xGuardMutex );
673 }

679 void vPortForciblyEndThread( void *pxTaskToDelete )
680 {
681 xTaskHandle hTaskToDelete = ( xTaskHandle )pxTaskToDelete;
682 xThreadState* xTaskToDelete;
683 xThreadState* xTaskToResume;
684 
685  PORT_ENTER();
686 
687  xTaskToDelete = prvGetThreadHandle( hTaskToDelete );
688  xTaskToResume = prvGetThreadHandle( xTaskGetCurrentTaskHandle() );
689 
690  PORT_ASSERT( xTaskToDelete );
691  PORT_ASSERT( xTaskToResume );
692 
693  if ( xTaskToResume == xTaskToDelete )
694  {
695  /* This is a suicidal thread, need to select a different task to run. */
696  vTaskSwitchContext();
697  xTaskToResume = prvGetThreadHandle( xTaskGetCurrentTaskHandle() );
698  }
699 
700  if ( pthread_self() != xTaskToDelete->hThread )
701  {
702  /* Cancelling a thread that is not me. */
703 
704  /* Send a signal to wake the task so that it definitely cancels. */
705  pthread_testcancel();
706  pthread_cancel( xTaskToDelete->hThread );
707 
708  }
709  else
710  {
711  /* Resume the other thread. */
712  prvResumeThread( xTaskToResume );
713  /* Pthread Clean-up function will note the cancellation. */
714  /* Release the execution. */
715 
716  PORT_UNLOCK( xRunningThreadMutex );
717 
718  //PORT_LEAVE();
719  PORT_UNLOCK( xGuardMutex );
720  /* Commit suicide */
721  pthread_exit( (void *)1 );
722  }
723  PORT_LEAVE();
724 }

732 {

773 void prvSuspendSignalHandler(int sig)
774 {
775 
776  portBASE_TYPE hangover;
777 
778  /* make sure who we are */
779  xThreadState* myself = prvGetThreadHandleByThread(pthread_self());
780  PORT_ASSERT( myself );
781 
782  /* make sure we are actually supposed to sleep */
783  if (myself->threadStatus != THREAD_YIELDING && myself->threadStatus != THREAD_STARTING && myself->threadStatus != THREAD_PREEMPTING ) {
784  /* Spurious signal has arrived, we are not really supposed to halt.
785  * Not a real problem, we can safely ignore that. */
786  return;
787  }
788 
789  /* we need that to wake up later (cond_wait needs a mutex locked) */
790  PORT_LOCK(myself->threadSleepMutex);
791 
792  /* even waking up is a bit different depending on how we went to sleep */
793  hangover = myself->threadStatus;
794 
795  myself->threadStatus = THREAD_SLEEPING;
796 
797  if ( hangover == THREAD_STARTING ) {
804  PORT_LOCK(xYieldingThreadMutex);
805  PORT_UNLOCK(xYieldingThreadMutex);
806  } else if ( hangover == THREAD_YIELDING) {
813  PORT_UNLOCK(xRunningThreadMutex );
814  PORT_UNLOCK(xGuardMutex );
815  } else if ( hangover == THREAD_PREEMPTING) {
822  PORT_LOCK(xYieldingThreadMutex);
823  PORT_UNLOCK(xRunningThreadMutex );
824  PORT_UNLOCK(xYieldingThreadMutex);
825  }
826 
827  /* deep sleep until wake condition is met*/
828  pthread_cond_wait( &myself->threadSleepCond, &myself->threadSleepMutex );
829 
830  /* waking */
831  myself->threadStatus = THREAD_WAKING;
832 
833  /* synchronize with waker - quick assertion if the right thread got the condition sent to*/
834  PORT_LOCK(xResumingThreadMutex);
835  PORT_ASSERT(prvGetThreadHandle( xTaskGetCurrentTaskHandle())==myself);
836  PORT_UNLOCK(xResumingThreadMutex);
837 
838  /* we don't need that condition mutex anymore */
839  PORT_UNLOCK(myself->threadSleepMutex);
840 
841  /* we ARE the running thread now (the one and only) */
842  PORT_LOCK(xRunningThreadMutex);
843 
848  PORT_LOCK( xGuardMutex );
849  if ( myself->uxCriticalNesting == 0 )
850  {
851  xInterruptsEnabled = pdTRUE;
852  }
853  else
854  {
855  xInterruptsEnabled = pdFALSE;
856  }
857 
858  myself->threadStatus = THREAD_RUNNING;
859 
864  if (hangover!=THREAD_YIELDING) {
865  PORT_UNLOCK( xGuardMutex );
866  }
867 
868 }

901 void prvSetupSignalsAndSchedulerPolicy( void )
902 {
903 /* The following code would allow for configuring the scheduling of this task as a Real-time task.
904  * The process would then need to be run with higher privileges for it to take affect.
905 int iPolicy;
906 int iResult;
907 int iSchedulerPriority;
908  iResult = pthread_getschedparam( pthread_self(), &iPolicy, &iSchedulerPriority );
909  iResult = pthread_attr_setschedpolicy( &xThreadAttributes, SCHED_FIFO );
910  iPolicy = SCHED_FIFO;
911  iResult = pthread_setschedparam( pthread_self(), iPolicy, &iSchedulerPriority ); */
912 
913 struct sigaction sigsuspendself;
914 portLONG lIndex;
