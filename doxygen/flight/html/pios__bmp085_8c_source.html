00189 void PIOS_BMP085_ReadADC(void)
00190 {
00191         uint8_t Data[3];
00192         Data[0] = 0;
00193         Data[1] = 0;
00194         Data[2] = 0;
00195 
00196         /* Read and store the 16bit result */
00197         if (CurrentRead == TemperatureConv) {
00198                 /* Read the temperature conversion */
00199                 while (PIOS_BMP085_Read(BMP085_ADC_MSB, Data, 2) != 0)
00200                         continue;
00201                 RawTemperature = ((Data[0] << 8) | Data[1]);
00202 
00203                 X1 = (RawTemperature - CalibData.AC6) * CalibData.AC5 >> 15;
00204                 X2 = ((int32_t) CalibData.MC << 11) / (X1 + CalibData.MD);
00205                 B5 = X1 + X2;
00206                 Temperature = (B5 + 8) >> 4;
00207         } else {
00208                 /* Read the pressure conversion */
00209                 while (PIOS_BMP085_Read(BMP085_ADC_MSB, Data, 3) != 0)
00210                         continue;
00211                 RawPressure = ((Data[0] << 16) | (Data[1] << 8) | Data[2]) >> (8 - BMP085_OVERSAMPLING);
00212 
00213                 B6 = B5 - 4000;
00214                 X1 = (CalibData.B2 * (B6 * B6 >> 12)) >> 11;
00215                 X2 = CalibData.AC2 * B6 >> 11;
00216                 X3 = X1 + X2;
00217                 B3 = ((((int32_t) CalibData.AC1 * 4 + X3) << BMP085_OVERSAMPLING) + 2) >> 2;
00218                 X1 = CalibData.AC3 * B6 >> 13;
00219                 X2 = (CalibData.B1 * (B6 * B6 >> 12)) >> 16;
00220                 X3 = ((X1 + X2) + 2) >> 2;
00221                 B4 = (CalibData.AC4 * (uint32_t) (X3 + 32768)) >> 15;
00222                 B7 = ((uint32_t) RawPressure - B3) * (50000 >> BMP085_OVERSAMPLING);
00223                 P = B7 < 0x80000000 ? (B7 * 2) / B4 : (B7 / B4) * 2;
00224 
00225                 X1 = (P >> 8) * (P >> 8);
00226                 X1 = (X1 * 3038) >> 16;
00227                 X2 = (-7357 * P) >> 16;
00228                 Pressure = P + ((X1 + X2 + 3791) >> 4);
00229         }
00230 }
