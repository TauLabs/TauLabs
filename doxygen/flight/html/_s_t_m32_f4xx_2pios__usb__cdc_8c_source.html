00208 {

00488         bool valid = PIOS_USB_CDC_validate(usb_cdc_dev);
00489         PIOS_Assert(valid);
00490 
00491         /* Give back UART State Bitmap */
00492         /* UART State Bitmap
00493          *   15-7: reserved
00494          *      6:  bOverRun    overrun error
00495          *      5:  bParity     parity error
00496          *      4:  bFraming    framing error
00497          *      3:  bRingSignal RI
00498          *      2:  bBreak      break reception
00499          *      1:  bTxCarrier  DSR
00500          *      0:  bRxCarrier  DCD
00501          */
00502 
00503         /* Currently, we only handle TxCarrier and RxCarrier reporting */
00504         uint16_t new_uart_state = 0;
00505         if (usb_cdc_dev->tx_out_cb) {
00506                 /* Someone is going to providing FC->PC data, advertise an RxCarrier to the host */
00507                 new_uart_state |= 0x1;
00508         }
00509         if (usb_cdc_dev->rx_in_cb) {
00510                 /* Someone is consuming PC->FC data, advertise a TxCarrier to the host */
00511                 new_uart_state |= 0x2;
00512         }
00513 
00514         /* Has anything changed since we last sent a notification? */
00515         if ((new_uart_state ^ usb_cdc_dev->prev_uart_state) & 0x3) {
00516                 usb_cdc_dev->prev_uart_state = new_uart_state;
00517 
00518                 uart_state.bmUartState = htousbs(new_uart_state);
00519 
00520                 PIOS_USBHOOK_EndpointTx(usb_cdc_dev->cfg->ctrl_tx_ep,
00521                                         (uint8_t *)&uart_state,
00522                                         sizeof(uart_state));
00523         }
