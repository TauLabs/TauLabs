161 /*******************************************************************************
162 * Function Name : PIOS_USBHOOK_Reset.
163 * Description : Custom HID reset routine.
164 * Input : None.
165 * Output : None.
166 * Return : None.
167 *******************************************************************************/
168 static void PIOS_USBHOOK_Reset(void)
169 {
170  /* Set DEVICE as not configured */
171  pInformation->Current_Configuration = 0;
172  pInformation->Current_Interface = 0; /*the default Interface */
173 
174  /* Current Feature initialization */
175  pInformation->Current_Feature = 0;
176 
177 #ifdef STM32F10X_CL
178  /* EP0 is already configured in DFU_Init() by USB_SIL_Init() function */
179 
180  /* Init EP1 IN as Interrupt endpoint */
181  OTG_DEV_EP_Init(EP1_IN, OTG_DEV_EP_TYPE_INT, 2);
182 
183  /* Init EP1 OUT as Interrupt endpoint */
184  OTG_DEV_EP_Init(EP1_OUT, OTG_DEV_EP_TYPE_INT, 2);
185 #else
186  SetBTABLE(BTABLE_ADDRESS);
187 
188  /* Initialize Endpoint 0 (Control) */
189  SetEPType(ENDP0, EP_CONTROL);
190  SetEPTxAddr(ENDP0, ENDP0_TXADDR);
191  SetEPTxStatus(ENDP0, EP_TX_STALL);
192  Clear_Status_Out(ENDP0);
193 
194  SetEPRxAddr(ENDP0, ENDP0_RXADDR);
195  SetEPRxCount(ENDP0, Device_Property.MaxPacketSize);
196  SetEPRxValid(ENDP0);
197 
198 #if defined(PIOS_INCLUDE_USB_HID)
199  /* Initialize Endpoint 1 (HID) */
200  SetEPType(ENDP1, EP_INTERRUPT);
201  SetEPTxAddr(ENDP1, ENDP1_TXADDR);
202  SetEPTxCount(ENDP1, PIOS_USB_BOARD_HID_DATA_LENGTH);
203  SetEPTxStatus(ENDP1, EP_TX_NAK);
204 
205  SetEPRxAddr(ENDP1, ENDP1_RXADDR);
206  SetEPRxCount(ENDP1, PIOS_USB_BOARD_HID_DATA_LENGTH);
207  SetEPRxStatus(ENDP1, EP_RX_VALID);
208 #endif /* PIOS_INCLUDE_USB_HID */
209 
210 #if defined(PIOS_INCLUDE_USB_CDC)
211  /* Initialize Endpoint 2 (CDC Call Control) */
212  SetEPType(ENDP2, EP_INTERRUPT);
213  SetEPTxAddr(ENDP2, ENDP2_TXADDR);
214  SetEPTxStatus(ENDP2, EP_TX_NAK);
215 
216  SetEPRxAddr(ENDP2, ENDP2_RXADDR);
217  SetEPRxCount(ENDP2, PIOS_USB_BOARD_CDC_MGMT_LENGTH);
218  SetEPRxStatus(ENDP2, EP_RX_DIS);
219 
220  /* Initialize Endpoint 3 (CDC Data) */
221  SetEPType(ENDP3, EP_BULK);
222  SetEPTxAddr(ENDP3, ENDP3_TXADDR);
223  SetEPTxStatus(ENDP3, EP_TX_NAK);
224 
225  SetEPRxAddr(ENDP3, ENDP3_RXADDR);
226  SetEPRxCount(ENDP3, PIOS_USB_BOARD_CDC_DATA_LENGTH);
227  SetEPRxStatus(ENDP3, EP_RX_VALID);
228 
229 #endif /* PIOS_INCLUDE_USB_CDC */
230 
231  /* Set this device to response on default address */
232  SetDeviceAddress(0);
233 #endif /* STM32F10X_CL */
234 
235  bDeviceState = ATTACHED;
236 }

300 static RESULT PIOS_USBHOOK_Data_Setup(uint8_t RequestNo)
301 {
302  uint8_t *(*CopyOutRoutine) (uint16_t);
303  const uint8_t *(*CopyInRoutine) (uint16_t);
304 
305  CopyInRoutine = NULL;
306  CopyOutRoutine = NULL;
307 
308  switch (Type_Recipient) {
309  case (STANDARD_REQUEST | INTERFACE_RECIPIENT):
310  switch (pInformation->USBwIndex0) {
311 #if defined(PIOS_INCLUDE_USB_CDC)
312  case 2: /* HID Interface */
313 #else
314  case 0: /* HID Interface */
315 #endif
316  switch (RequestNo) {
317  case GET_DESCRIPTOR:
318  switch (pInformation->USBwValue1) {
319  case USB_DESC_TYPE_REPORT:
320  CopyInRoutine = PIOS_USBHOOK_GetReportDescriptor;
321  break;
322  case USB_DESC_TYPE_HID:
323  CopyInRoutine = PIOS_USBHOOK_GetHIDDescriptor;
324  break;
325  }
326  }
327  }
328  break;
329 
330  case (CLASS_REQUEST | INTERFACE_RECIPIENT):
331  switch (pInformation->USBwIndex0) {
332 #if defined(PIOS_INCLUDE_USB_CDC)
333  case 2: /* HID Interface */
334 #else
335  case 0: /* HID Interface */
336 #endif
337  switch (RequestNo) {
338  case USB_HID_REQ_GET_PROTOCOL:
339  CopyInRoutine = PIOS_USBHOOK_GetProtocolValue;
340  break;
341  }
342 
343  break;
344 #if defined(PIOS_INCLUDE_USB_CDC)
345  case 0: /* CDC Call Control Interface */
346  switch (RequestNo) {
347  case USB_CDC_REQ_SET_LINE_CODING:
348  CopyOutRoutine = PIOS_USB_CDC_SetLineCoding;
349  break;
350  case USB_CDC_REQ_GET_LINE_CODING:
351  CopyInRoutine = PIOS_USB_CDC_GetLineCoding;
352  break;
353  }
354 
355  break;
356 
357  case 1: /* CDC Data Interface */
358  switch (RequestNo) {
359  case 0:
360  break;
361  }
362 
363  break;
364 #endif /* PIOS_INCLUDE_USB_CDC */
365  }
366  break;
367  }
368 
369  /* No registered copy routine */
370  if ((CopyInRoutine == NULL) && (CopyOutRoutine == NULL)) {
371  return USB_UNSUPPORT;
372  }
373 
374  /* Registered copy in AND copy out routine */
375  if ((CopyInRoutine != NULL) && (CopyOutRoutine != NULL)) {
376  /* This should never happen */
377  return USB_UNSUPPORT;
378  }
379 
380  if (CopyInRoutine != NULL) {
381  pInformation->Ctrl_Info.CopyDataIn = CopyInRoutine;
382  pInformation->Ctrl_Info.Usb_wOffset = 0;
383  (*CopyInRoutine) (0);
384  } else if (CopyOutRoutine != NULL) {
385  pInformation->Ctrl_Info.CopyDataOut = CopyOutRoutine;
386  pInformation->Ctrl_Info.Usb_rOffset = 0;
387  (*CopyOutRoutine) (0);
388  }
389 
390  return USB_SUCCESS;
391 }

402 static RESULT PIOS_USBHOOK_NoData_Setup(uint8_t RequestNo)
403 {
404  switch (Type_Recipient) {
405  case (CLASS_REQUEST | INTERFACE_RECIPIENT):
406  switch (pInformation->USBwIndex0) {
407 #if defined(PIOS_INCLUDE_USB_CDC)
408  case 2: /* HID */
409 #else
410  case 0: /* HID */
411 #endif
412  switch (RequestNo) {
413  case USB_HID_REQ_SET_PROTOCOL:
414  return PIOS_USBHOOK_SetProtocol();

417 
418  break;
419 
420 #if defined(PIOS_INCLUDE_USB_CDC)

423  case USB_CDC_REQ_SET_CONTROL_LINE_STATE:
424  return PIOS_USB_CDC_SetControlLineState();
425  break;
426  }
427 
428  break;
429 #endif /* PIOS_INCLUDE_USB_CDC */
430  }
431 
432  break;
433  }
434 
435  return USB_UNSUPPORT;
436 }
