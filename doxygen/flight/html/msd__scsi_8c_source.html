46 {

55  else
56  {
57

64  Inquiry_Data = MSD_Standard_Inquiry_Data2;
65  }
66 
67  if (MSD_CBW.CB[4] <= STANDARD_INQUIRY_DATA_LEN)
68  Inquiry_Data_Length = MSD_CBW.CB[4];
69  else
70  Inquiry_Data_Length = STANDARD_INQUIRY_DATA_LEN;
71 
72  }
73  MSD_Transfer_Data_Request(Inquiry_Data, Inquiry_Data_Length);
74 }

206 /*******************************************************************************
207 * Function Name : MSD_SCSI_Read10_Cmd
208 * Description : SCSI Read10 Command routine.
209 * Input : None.
210 * Output : None.
211 * Return : None.
212 *******************************************************************************/
213 void MSD_SCSI_Read10_Cmd(uint8_t lun , uint32_t LBA , uint32_t BlockNbr)
214 {
215 
216  if (MSD_Bot_State == BOT_IDLE)
217  {
218  if (!(MSD_SCSI_Address_Management(MSD_CBW.bLUN, SCSI_READ10, LBA, BlockNbr)))/*address out of range*/
219  {
220  return;
221  }
222 
223  if ((MSD_CBW.bmFlags & 0x80) != 0)
224  {
225  MSD_Bot_State = BOT_DATA_IN;
226  MSD_Read_Memory(lun, LBA , BlockNbr);
227  }
228  else
229  {
230  MSD_Bot_Abort(BOTH_DIR);
231  MSD_Set_Scsi_Sense_Data(MSD_CBW.bLUN, ILLEGAL_REQUEST, INVALID_FIELED_IN_COMMAND);
232  MSD_Set_CSW (CSW_CMD_FAILED, SEND_CSW_ENABLE);
233  }
234  return;
235  }
236  else if (MSD_Bot_State == BOT_DATA_IN)
237  {
238  MSD_Read_Memory(lun , LBA , BlockNbr);
239  }
240 }

390 /*******************************************************************************
391 * Function Name : MSD_SCSI_Address_Management
392 * Description : Test the received address.
393 * Input : uint8_t Cmd : the command can be SCSI_READ10 or SCSI_WRITE10.
394 * Output : None.
395 * Return : Read\Write status (bool).
396 *******************************************************************************/
397 bool MSD_SCSI_Address_Management(uint8_t lun , uint8_t Cmd , uint32_t LBA , uint32_t BlockNbr)
398 {
399 
400  if ((LBA + BlockNbr) > MSD_Mass_Block_Count[lun] )
401  {
402  if (Cmd == SCSI_WRITE10)
403  {
404  MSD_Bot_Abort(BOTH_DIR);
405  }
406  MSD_Bot_Abort(DIR_IN);
407  MSD_Set_Scsi_Sense_Data(lun, ILLEGAL_REQUEST, ADDRESS_OUT_OF_RANGE);
408  MSD_Set_CSW (CSW_CMD_FAILED, SEND_CSW_DISABLE);
409  return (FALSE);
410  }
411 
412 
413  if (MSD_CBW.dDataLength != BlockNbr * MSD_Mass_Block_Size[lun])
414  {
415  if (Cmd == SCSI_WRITE10)
416  {
417  MSD_Bot_Abort(BOTH_DIR);
418  }
419  else
420  {
421  MSD_Bot_Abort(DIR_IN);
422  }
423  MSD_Set_Scsi_Sense_Data(MSD_CBW.bLUN, ILLEGAL_REQUEST, INVALID_FIELED_IN_COMMAND);
424  MSD_Set_CSW (CSW_CMD_FAILED, SEND_CSW_DISABLE);
425  return (FALSE);
426  }
427  return (TRUE);
428 }
