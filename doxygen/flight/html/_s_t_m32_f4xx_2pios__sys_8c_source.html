186 int32_t PIOS_SYS_Reset(void)
187 {
188  /* Disable all RTOS tasks */
189 #if defined(PIOS_INCLUDE_FREERTOS)
190  /* port specific FreeRTOS function to disable tasks (nested) */
191  portENTER_CRITICAL();

193

195  PIOS_IRQ_Disable();
196 
197  // turn off all board LEDs
198 #if defined(PIOS_LED_HEARTBEAT)
199  PIOS_LED_Off(PIOS_LED_HEARTBEAT);
200 #endif /* PIOS_LED_HEARTBEAT */
201 #if defined(PIOS_LED_ALARM)
202  PIOS_LED_Off(PIOS_LED_ALARM);
203 #endif /* PIOS_LED_ALARM */
204 
205  /* XXX F10x port resets most (but not all) peripherals ... do we care? */
206 
207  /* Reset STM32 */
208  NVIC_SystemReset();
209 
210  while (1) ;
211 
212  /* We will never reach this point */
213  return -1;
214 }

251 int32_t PIOS_SYS_SerialNumberGet(char *str)
252 {
253  int i;
254 
255  /* Stored in the so called "electronic signature" */
256  for (i = 0; i < PIOS_SYS_SERIAL_NUM_ASCII_LEN; ++i) {
257  uint8_t b = MEM8(0x1fff7a10 + (i / 2));
258  if (!(i & 1))
259  b >>= 4;
260  b &= 0x0f;
261 
262  str[i] = ((b > 9) ? ('A' - 10) : '0') + b;
263  }
264  str[i] = '\0';
265 
266  /* No error */
267  return 0;
268 }
