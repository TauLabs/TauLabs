00167 static void overoSyncTask(void *parameters)
00168 {
00169         UAVObjEvent ev;
00170 
00171         // Kick off SPI transfers (once one is completed another will automatically transmit)
00172         overosync->sent_objects = 0;
00173         overosync->failed_objects = 0;
00174         overosync->received_objects = 0;
00175         
00176         portTickType lastUpdateTime = xTaskGetTickCount();
00177         portTickType updateTime;
00178 
00179         bool initialized = false;
00180         uint8_t last_connected = OVEROSYNCSTATS_CONNECTED_FALSE;
00181 
00182         // Loop forever
00183         while (1) {
00184                 // Wait for queue message
00185                 if (xQueueReceive(queue, &ev, portMAX_DELAY) == pdTRUE) {
00186 
00187                         // For the first seconds do not send updates to allow the
00188                         // overo to boot.  Then enable it and act normally.
00189                         if (!initialized && xTaskGetTickCount() < 5000) {
00190                                 continue;
00191                         } else if (!initialized) {
00192                                 initialized = true;
00193                                 PIOS_OVERO_Enable(pios_overo_id);
00194                         }
00195 
00196                         // Process event.  This calls transmitData
00197                         UAVTalkSendObjectTimestamped(uavTalkCon, ev.obj, ev.instId, false, 0);
00198                         
00199                         updateTime = xTaskGetTickCount();
00200                         if(((portTickType) (updateTime - lastUpdateTime)) > 1000) {
00201                                 // Update stats.  This will trigger a local send event too
00202                                 OveroSyncStatsData syncStats;
00203                                 syncStats.Send = overosync->sent_bytes;
00204                                 syncStats.Connected = syncStats.Send > 500 ? OVEROSYNCSTATS_CONNECTED_TRUE : OVEROSYNCSTATS_CONNECTED_FALSE;
00205                                 syncStats.DroppedUpdates = overosync->failed_objects;
00206                                 syncStats.Packets = PIOS_OVERO_GetPacketCount(pios_overo_id);
00207                                 OveroSyncStatsSet(&syncStats);
00208                                 overosync->failed_objects = 0;
00209                                 overosync->sent_bytes = 0;
00210                                 lastUpdateTime = updateTime;
00211 
00212                                 // When first connected, send all the settings.  Right now this
00213                                 // will fail since all the settings will overfill the buffer and
00214                                 if (last_connected == OVEROSYNCSTATS_CONNECTED_FALSE &&
00215                                         syncStats.Connected == OVEROSYNCSTATS_CONNECTED_TRUE) {
00216                                         UAVObjIterate(&send_settings);
00217                                 }
00218 
00219                                 // Because the previous code only happens on connection and the
00220                                 // remote logging program doesn't send the settings to the log
00221                                 // when arming starts we send all settings every thirty seconds
00222                                 static uint32_t second_count = 0;
00223                                 if (second_count ++ > 30) {
00224                                         UAVObjIterate(&send_settings);
00225                                         second_count = 0;
00226                                 }
00227                                 last_connected = syncStats.Connected;
00228                         }
00229 
00230                         // TODO: Check the receive buffer
00231                 }
00232         }
