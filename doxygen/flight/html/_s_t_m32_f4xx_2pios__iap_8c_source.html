69 void PIOS_IAP_Init( void )
70 {
71  /* Enable CRC clock */
72  RCC_AHB1PeriphClockCmd(RCC_AHB1Periph_CRC, ENABLE);
73 
74  /* Enable PWR and BKP clock */
75  RCC_APB1PeriphClockCmd(RCC_APB1Periph_PWR | RCC_AHB1Periph_BKPSRAM, ENABLE);
76 
77  /* Enable write access to Backup domain */
78  PWR_BackupAccessCmd(ENABLE);
79 
80  /* Clear Tamper pin Event(TE) pending flag */
81  RTC_ClearFlag(RTC_FLAG_TAMP1F);
82 }

92 uint32_t PIOS_IAP_CheckRequest( void )
93 {
94  uint32_t retval = false;
95  uint16_t reg1;
96  uint16_t reg2;
97 
98  reg1 = RTC_ReadBackupRegister( MAGIC_REG_1 );
99  reg2 = RTC_ReadBackupRegister( MAGIC_REG_2 );
100 
101  if( reg1 == IAP_MAGIC_WORD_1 && reg2 == IAP_MAGIC_WORD_2 ) {
102  // We have a match.
103  retval = true;
104  } else {
105  retval = false;
106  }
107  return retval;
108 }
