00057 {
00058         struct pios_adxl345_data accel_data;
00059         float gyro[4];
00060 
00061         // Only wait the time for two nominal updates before setting an alarm
00062         if (xQueueReceive(*gyro_queue, (void *const)gyro, LOOP_RATE_MS * 2) ==
00063             errQUEUE_EMPTY) {
00064                 AlarmsSet(SYSTEMALARMS_ALARM_ATTITUDE,
00065                           SYSTEMALARMS_ALARM_ERROR);
00066                 return -1;
00067         }
00068         // Do not read raw sensor data in simulation mode
00069         if (GyrosReadOnly() || AccelsReadOnly())
00070                 return 0;
00071 
00072         // No accel data available
00073         if (PIOS_ADXL345_FifoElements() == 0)
00074                 return -1;
00075 
00076         // Scale ADC data into deg/s. First sample is temperature, so ignore.
00077         // Rotate data from internal gryoscope sensor frame into board sensor frame
00078         prelim_gyros[0] = -(gyro[1] - GYRO_NEUTRAL_BIAS) * 0.42f * inertialSensorSettings->GyroScale[SENSORSETTINGS_GYROSCALE_X];
00079         prelim_gyros[1] =  (gyro[2] - GYRO_NEUTRAL_BIAS) * 0.42f * inertialSensorSettings->GyroScale[SENSORSETTINGS_GYROSCALE_Y];
00080         prelim_gyros[2] = -(gyro[3] - GYRO_NEUTRAL_BIAS) * 0.42f * inertialSensorSettings->GyroScale[SENSORSETTINGS_GYROSCALE_Z];
00081 
00082         // When this is enabled remove estimate of bias
00083         if (glblAtt->bias_correct_gyro) {
00084                 prelim_gyros[0] -= gyrosBias->x;
00085                 prelim_gyros[1] -= gyrosBias->y;
00086                 prelim_gyros[2] -= gyrosBias->z;
00087         }
00088 
00089         // Process accelerometer sensor data. In this case, average the data
00090         int32_t x = 0;
00091         int32_t y = 0;
00092         int32_t z = 0;
00093         uint8_t i = 0;
00094         uint8_t samples_remaining;
00095         do {
00096                 i++;
00097                 samples_remaining = PIOS_ADXL345_Read(&accel_data);
00098 
00099                 //Assign data, rotating from internal accelerometer frame into board sensor frame
00100                 x +=  accel_data.x;
00101                 y += -accel_data.y;
00102                 z += -accel_data.z;
00103         } while ((i < 32) && (samples_remaining > 0));  //<-- i=32 being hardcoded means that if the accelerometer ADC sample rate is increased, we could wind up never being able to empty the buffer
00104 
00105         // Apply scaling and bias correction in sensor frame
00106         prelim_accels[0] = (float)x / i * ACCEL_SCALE * inertialSensorSettings->AccelScale[0] - inertialSensorSettings->AccelBias[0];
00107         prelim_accels[1] = (float)y / i * ACCEL_SCALE * inertialSensorSettings->AccelScale[1] - inertialSensorSettings->AccelBias[1];
00108         prelim_accels[2] = (float)z / i * ACCEL_SCALE * inertialSensorSettings->AccelScale[2] - inertialSensorSettings->AccelBias[2];
00109 
00110         return 0;
00111 }
