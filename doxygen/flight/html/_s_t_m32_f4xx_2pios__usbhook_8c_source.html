00170         /*

00346 static uint8_t PIOS_USBHOOK_CLASS_Setup(void *pdev, USB_SETUP_REQ *req)
00347 {
00348         switch (req->bmRequest & (USB_REQ_TYPE_MASK | USB_REQ_RECIPIENT_MASK)) {
00349         case (USB_REQ_TYPE_STANDARD | USB_REQ_RECIPIENT_INTERFACE):
00350         case (USB_REQ_TYPE_CLASS    | USB_REQ_RECIPIENT_INTERFACE):
00351         {
00352                 uint8_t ifnum = LOBYTE(req->wIndex);
00353                 if ((ifnum < NELEMENTS(usb_if_table)) &&
00354                     (usb_if_table[ifnum].ifops && usb_if_table[ifnum].ifops->setup)) {
00355                         usb_if_table[ifnum].ifops->setup(usb_if_table[ifnum].context, 
00356                                                          (struct usb_setup_request *)req);
00357                         if (!(req->bmRequest & 0x80) && req->wLength > 0) {
00358                                 /* Request is a host-to-device data setup packet, keep track of the request details for the EP0_RxReady call */
00359                                 usb_ep0_active_req.bmRequestType = req->bmRequest;
00360                                 usb_ep0_active_req.bRequest      = req->bRequest;
00361                                 usb_ep0_active_req.wValue        = req->wValue;
00362                                 usb_ep0_active_req.wIndex        = req->wIndex;
00363                                 usb_ep0_active_req.wLength       = req->wLength;
00364                         }
00365                 } else {
00366                         /* No Setup handler or Setup handler failed */
00367                         USBD_CtlError (&pios_usb_otg_core_handle, req);
00368                 }
00369                 break;
00370         }
00371         default:
00372                 /* Unhandled Setup */
00373                 USBD_CtlError (&pios_usb_otg_core_handle, req);
