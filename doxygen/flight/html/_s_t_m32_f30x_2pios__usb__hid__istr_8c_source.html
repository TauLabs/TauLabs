148 /*******************************************************************************
149 * Function Name : STM32_PCD_OTG_ISR_Handler
150 * Description : Handles all USB Device Interrupts
151 * Input : None
152 * Output : None
153 * Return : status
154 *******************************************************************************/
155 u32 STM32_PCD_OTG_ISR_Handler(void)
156 {
157  USB_OTG_GINTSTS_TypeDef gintr_status;
158  u32 retval = 0;
159 
160  if (USBD_FS_IsDeviceMode()) { /* ensure that we are in device mode */
161  gintr_status.d32 = OTGD_FS_ReadCoreItr();
162 
163  /*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*/
164 
165  /* If there is no interrupt pending exit the interrupt routine */
166  if (!gintr_status.d32) {
167  return 0;
168  }
169 
170  /*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*/
171  /* Early Suspend interrupt */
172 #ifdef INTR_ERLYSUSPEND
173  if (gintr_status.b.erlysuspend) {
174  retval |= OTGD_FS_Handle_EarlySuspend_ISR();
175  }
176 #endif /* INTR_ERLYSUSPEND */
177 
178  /*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*/
179  /* End of Periodic Frame interrupt */
180 #ifdef INTR_EOPFRAME
181  if (gintr_status.b.eopframe) {
182  retval |= OTGD_FS_Handle_EOPF_ISR();
183  }
184 #endif /* INTR_EOPFRAME */
185 
186  /*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*/
187  /* Non Periodic Tx FIFO Emty interrupt */
188 #ifdef INTR_NPTXFEMPTY
189  if (gintr_status.b.nptxfempty) {
190  retval |= OTGD_FS_Handle_NPTxFE_ISR();
191  }
192 #endif /* INTR_NPTXFEMPTY */
193 
194  /*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*/
195  /* Wakeup or RemoteWakeup interrupt */
196 #ifdef INTR_WKUPINTR
197  if (gintr_status.b.wkupintr) {
198  retval |= OTGD_FS_Handle_Wakeup_ISR();
199  }
200 #endif /* INTR_WKUPINTR */
201 
202  /*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*/
203  /* Suspend interrupt */
204 #ifdef INTR_USBSUSPEND
205  if (gintr_status.b.usbsuspend) {
206  /* check if SUSPEND is possible */
207  if (fSuspendEnabled) {
208  Suspend();
209  } else {
210  /* if not possible then resume after xx ms */
211  Resume(RESUME_LATER); /* This case shouldn't happen in OTG Device mode because
212  there's no ESOF interrupt to increment the ResumeS.bESOFcnt in the Resume state machine */
213  }
214 
215  retval |= OTGD_FS_Handle_USBSuspend_ISR();
216  }
217 #endif /* INTR_USBSUSPEND */
218 
219  /*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*/
220  /* Start of Frame interrupt */
221 #ifdef INTR_SOFINTR
222  if (gintr_status.b.sofintr) {
223  /* Update the frame number variable */
224  bIntPackSOF++;
225 
226  retval |= OTGD_FS_Handle_Sof_ISR();
227  }
228 #endif /* INTR_SOFINTR */
229 
230  /*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*/
231  /* Receive FIFO Queue Status Level interrupt */
232 #ifdef INTR_RXSTSQLVL
233  if (gintr_status.b.rxstsqlvl) {
234  retval |= OTGD_FS_Handle_RxStatusQueueLevel_ISR();
235  }
236 #endif /* INTR_RXSTSQLVL */
237 
238  /*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*/
239  /* Enumeration Done interrupt */
240 #ifdef INTR_ENUMDONE
241  if (gintr_status.b.enumdone) {
242  retval |= OTGD_FS_Handle_EnumDone_ISR();
243  }
244 #endif /* INTR_ENUMDONE */
