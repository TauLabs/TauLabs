132 static void pathPlannerTask(void *parameters)
133 {
134  // If the PathStatus isn't available no follower is running and we should abort
135  while (PathStatusHandle() == NULL || !TaskMonitorQueryRunning(TASKINFO_RUNNING_PATHFOLLOWER)) {
136  AlarmsSet(SYSTEMALARMS_ALARM_PATHPLANNER, SYSTEMALARMS_ALARM_CRITICAL);
137  vTaskDelay(1000);
138  }
139  AlarmsClear(SYSTEMALARMS_ALARM_PATHPLANNER);
140 
141  PathPlannerSettingsConnectCallback(settingsUpdated);
142  settingsUpdated(PathPlannerSettingsHandle());
143 
144  WaypointConnectCallback(waypointsUpdated);
145  WaypointActiveConnectCallback(waypointsUpdated);
146 
147  PathStatusConnectCallback(pathStatusUpdated);
148 
149  FlightStatusData flightStatus;
150 
151  // Main thread loop
152  bool pathplanner_active = false;
153  path_status_updated = false;
154 
155  while (1)
156  {
157 
158  vTaskDelay(UPDATE_RATE_MS / portTICK_RATE_MS);
159 
160  // When not running the path planner short circuit and wait
161  FlightStatusGet(&flightStatus);
162  if (flightStatus.FlightMode != FLIGHTSTATUS_FLIGHTMODE_PATHPLANNER) {
163  pathplanner_active = false;
164  continue;
165  }
166 
167  if(pathplanner_active == false) {
168  // Reset the state. Active waypoint should be set to an invalid
169  // value to force waypoint 0 to become activated when starting
170  // Note: this needs to be done before the callback is triggered!
171  active_waypoint = -1;
172  previous_waypoint = -1;
173 
174  // This triggers callback to update variable
175  WaypointActiveGet(&waypointActive);
176  waypointActive.Index = 0;
177  WaypointActiveSet(&waypointActive);
178 
179  pathplanner_active = true;
180  continue;
181  }
182 
183  /* This method determines if we have achieved the goal of the active */
184  /* waypoint */
185  if (path_status_updated)
186  checkTerminationCondition();
187 
188  /* If advance waypoint takes a long time to calculate then it should */
189  /* be called from here when the active_waypoints does not equal the */
190  /* WaypointActive.Index */
191  /* if (active_waypoint != WaypointActive.Index) */
192  /* advanceWaypoint(WaypointActive.Index) */
193  }
194 }

255 static void advanceWaypoint()
256 {
257  WaypointActiveGet(&waypointActive);
258 
259  // Store the currently active waypoint. This is used in activeWaypoint to plot
260  // a waypoint from this (previous) waypoint to the newly selected one
261  previous_waypoint = waypointActive.Index;
262 
263  // Default implementation simply jumps to the next possible waypoint. Insert any
264  // conditional logic desired here.
265  // Note: In the case of conditional logic it is the responsibilty of the implementer
266  // to ensure all possible paths are valid.
267  waypointActive.Index++;
268 
269  if (waypointActive.Index >= UAVObjGetNumInstances(WaypointHandle())) {
270  holdCurrentPosition();
271 
272  // Do not reset path_status_updated here to avoid this method constantly being called
273  return;
274  } else {
275  WaypointActiveSet(&waypointActive);
276  }
277 
278  // Invalidate any pending path status updates
279  path_status_updated = false;
280 }

290 static void activateWaypoint(int idx)
291 {
292  active_waypoint = idx;
293 
294  if (idx >= UAVObjGetNumInstances(WaypointHandle())) {
295  // Attempting to access invalid waypoint. Fall back to position hold at current location
296  AlarmsSet(SYSTEMALARMS_ALARM_PATHPLANNER, SYSTEMALARMS_ALARM_ERROR);
297  holdCurrentPosition();
298  return;
299  }
300 
301  // Get the activated waypoint
302  WaypointInstGet(idx, &waypoint);
303 
304  PathDesiredData pathDesired;
305 
306  pathDesired.End[PATHDESIRED_END_NORTH] = waypoint.Position[WAYPOINT_POSITION_NORTH];
307  pathDesired.End[PATHDESIRED_END_EAST] = waypoint.Position[WAYPOINT_POSITION_EAST];
308  pathDesired.End[PATHDESIRED_END_DOWN] = waypoint.Position[WAYPOINT_POSITION_DOWN];
309  pathDesired.ModeParameters = waypoint.ModeParameters;
310 
311  // Use this to ensure the cases match up (catastrophic if not) and to cover any cases
312  // that don't make sense to come from the path planner
313  switch(waypoint.Mode) {
314  case WAYPOINT_MODE_FLYVECTOR:
315  pathDesired.Mode = PATHDESIRED_MODE_FLYVECTOR;
316  break;
317  case WAYPOINT_MODE_FLYENDPOINT:
318  pathDesired.Mode = PATHDESIRED_MODE_FLYENDPOINT;
319  break;
320  case WAYPOINT_MODE_FLYCIRCLELEFT:
321  pathDesired.Mode = PATHDESIRED_MODE_FLYCIRCLELEFT;
322  break;
323  case WAYPOINT_MODE_FLYCIRCLERIGHT:
324  pathDesired.Mode = PATHDESIRED_MODE_FLYCIRCLERIGHT;
325  break;
326  default:
327  holdCurrentPosition();
328  AlarmsSet(SYSTEMALARMS_ALARM_PATHPLANNER, SYSTEMALARMS_ALARM_ERROR);
329  return;
330  }
331 
332  pathDesired.EndingVelocity = waypoint.Velocity;
333 
334  if(previous_waypoint < 0) {
335  // For first waypoint, get current position as start point
336  PositionActualData positionActual;
337  PositionActualGet(&positionActual);
338 
339  pathDesired.Start[PATHDESIRED_START_NORTH] = positionActual.North;
340  pathDesired.Start[PATHDESIRED_START_EAST] = positionActual.East;
341  pathDesired.Start[PATHDESIRED_START_DOWN] = positionActual.Down - 1;
342  pathDesired.StartingVelocity = waypoint.Velocity;
343  } else {
344  // Get previous waypoint as start point
345  WaypointData waypointPrev;
346  WaypointInstGet(previous_waypoint, &waypointPrev);
347 
348  pathDesired.Start[PATHDESIRED_END_NORTH] = waypointPrev.Position[WAYPOINT_POSITION_NORTH];
349  pathDesired.Start[PATHDESIRED_END_EAST] = waypointPrev.Position[WAYPOINT_POSITION_EAST];
350  pathDesired.Start[PATHDESIRED_END_DOWN] = waypointPrev.Position[WAYPOINT_POSITION_DOWN];
351  pathDesired.StartingVelocity = waypointPrev.Velocity;
352  }
353 
354  PathDesiredSet(&pathDesired);
355 
356  // Invalidate any pending path status updates
357  path_status_updated = false;
358 
359  AlarmsClear(SYSTEMALARMS_ALARM_PATHPLANNER);
360 }

406  retval = pathplanner_save_path(5);
407  operation = true;
408  break;
409  }
410 
411  if (pathPlannerSettings.PreprogrammedPath != preprogrammedPath &&
412  pathPlannerSettings.FlashOperation == PATHPLANNERSETTINGS_FLASHOPERATION_NONE) {
413  switch(pathPlannerSettings.PreprogrammedPath) {
414  case PATHPLANNERSETTINGS_PREPROGRAMMEDPATH_NONE:
415  break;
416  case PATHPLANNERSETTINGS_PREPROGRAMMEDPATH_10M_BOX:
417  createPathBox();
418  break;
419  case PATHPLANNERSETTINGS_PREPROGRAMMEDPATH_LOGO:
420  createPathLogo();
421  break;
422 
423  }
424  }
425 
426  if (operation && (retval == 0)) {
427  pathPlannerSettings.FlashOperation = PATHPLANNERSETTINGS_FLASHOPERATION_COMPLETED;
428  PathPlannerSettingsSet(&pathPlannerSettings);
429  } else if (retval != 0) {
430  pathPlannerSettings.FlashOperation = PATHPLANNERSETTINGS_FLASHOPERATION_FAILED;
431  PathPlannerSettingsSet(&pathPlannerSettings);
432  }
