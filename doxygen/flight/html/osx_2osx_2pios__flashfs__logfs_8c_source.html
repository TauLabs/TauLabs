84  * The bits within these enum values must progress ONLY
85  * from 1 -> 0 so that we can write later ones on top

88 enum arena_state {
89  /*
90  * The STM32F30X flash subsystem is only capable of
91  * writing words or halfwords. In this case we use halfwords.
92  * In addition to that it is only capable to write to erased
93  * cells (0xffff) or write a cell from anything to (0x0000).
94  * To cope with this, the F3 needs carefully crafted enum values.
95  * For this to work the underlying flash driver has to
96  * check each halfword if it has changed before writing.
97  */
98  ARENA_STATE_ERASED = 0xFFFFFFFF,
99  ARENA_STATE_RESERVED = 0xE6E6FFFF,
100  ARENA_STATE_ACTIVE = 0xE6E66666,
101  ARENA_STATE_OBSOLETE = 0x00000000,
102 };

119 static int32_t logfs_erase_arena(const struct logfs_state * logfs, uint8_t arena_id)
120 {
121  uintptr_t arena_addr = logfs_get_addr (logfs, arena_id, 0);
122 
123  /* Erase all of the sectors in the arena */
124  for (uint8_t sector_id = 0;
125  sector_id < (logfs->cfg->arena_size / logfs->cfg->sector_size);
126  sector_id++) {
127  if (logfs->driver->erase_sector(logfs->flash_id,
128  arena_addr + (sector_id * logfs->cfg->sector_size))) {
129  return -1;
130  }
131  }
132 
133  /* Mark this arena as fully erased */
134  struct arena_header arena_hdr = {
135  .magic = logfs->cfg->fs_magic,
136  .state = ARENA_STATE_ERASED,
137  };
138 
139  if (logfs->driver->write_data(logfs->flash_id,
140  arena_addr,
141  (uint8_t *)&arena_hdr,
142  sizeof(arena_hdr)) != 0) {
143  return -2;
144  }
145 
146  /* Arena is ready to be activated */
147  return 0;
148 }

156 static int32_t logfs_reserve_arena (const struct logfs_state * logfs, uint8_t arena_id)
157 {
158  uintptr_t arena_addr = logfs_get_addr (logfs, arena_id, 0);
159 
160  /* Read in the current arena header */
161  struct arena_header arena_hdr;
162  if (logfs->driver->read_data(logfs->flash_id,
163  arena_addr,
164  (uint8_t *)&arena_hdr,
165  sizeof(arena_hdr)) != 0) {
166  return -1;
167  }
168  if (arena_hdr.state != ARENA_STATE_ERASED) {
169  /* Arena was not erased, can't reserve it */
170  return -2;
171  }
172 
173  /* Set the arena state to reserved */
174  arena_hdr.state = ARENA_STATE_RESERVED;
175 
176  /* Write the arena header back to flash */
177  if (logfs->driver->write_data(logfs->flash_id,
178  arena_addr,
179  (uint8_t *)&arena_hdr,
180  sizeof(arena_hdr)) != 0) {
181  return -3;
182  }
183 
184  /* Arena is ready to be filled */
185  return 0;
186 }

211 static int32_t logfs_activate_arena(const struct logfs_state * logfs, uint8_t arena_id)
212 {
213  uintptr_t arena_addr = logfs_get_addr(logfs, arena_id, 0);
214 
215  /* Make sure this arena has been previously erased */
216  struct arena_header arena_hdr;
217  if (logfs->driver->read_data(logfs->flash_id,
218  arena_addr,
219  (uint8_t *)&arena_hdr,
220  sizeof (arena_hdr)) != 0) {
221  /* Failed to read arena header */
222  return -1;
223  }
224  if ((arena_hdr.state != ARENA_STATE_RESERVED) &&
225  (arena_hdr.state != ARENA_STATE_ERASED)) {
226  /* Arena was not erased or reserved, can't activate it */
227  return -2;
228  }
229 
230  /* Mark this arena as active */
231  arena_hdr.state = ARENA_STATE_ACTIVE;
232  if (logfs->driver->write_data(logfs->flash_id,
233  arena_addr,
234  (uint8_t *)&arena_hdr,
235  sizeof(arena_hdr)) != 0) {
236  return -3;
237  }
238 
239  /* The arena is now activated and the log may be mounted */
240  return 0;
241 }

291 static int32_t logfs_find_active_arena(const struct logfs_state * logfs)
292 {
293  /* Search for the lowest numbered active arena */
294  for (uint8_t arena_id = 0;
295  arena_id < logfs->cfg->total_fs_size / logfs->cfg->arena_size;
296  arena_id++) {
297  uintptr_t arena_addr = logfs_get_addr (logfs, arena_id, 0);
298  /* Load the arena header */
299  struct arena_header arena_hdr;
300  if (logfs->driver->read_data(logfs->flash_id,
301  arena_addr,
302  (uint8_t *)&arena_hdr,
303  sizeof (arena_hdr)) != 0) {
304  return -2;
305  }
306  if ((arena_hdr.state == ARENA_STATE_ACTIVE) &&
307  (arena_hdr.magic == logfs->cfg->fs_magic)) {
308  /* This is the first active arena */
309  return arena_id;
310  }
311  }
312 
313  /* Didn't find an active arena */
314  return -1;

345 /* NOTE: Must be called while holding the flash transaction lock */
346 static int32_t logfs_raw_copy_bytes (const struct logfs_state * logfs, uintptr_t src_addr, uint16_t src_size, uintptr_t dst_addr)
347 {
348 #define RAW_COPY_BLOCK_SIZE 16
349  uint8_t data_block[RAW_COPY_BLOCK_SIZE];
350 
351  while (src_size) {
352  uint16_t blk_size;
353  if (src_size >= RAW_COPY_BLOCK_SIZE) {
354  /* Copy a full block */
355  blk_size = RAW_COPY_BLOCK_SIZE;
356  } else {
357  /* Copy the remainder */
358  blk_size = src_size;
359  }
360 
361  /* Read a block of data from source */
362  if (logfs->driver->read_data(logfs->flash_id,
363  src_addr,
364  data_block,
365  blk_size) != 0) {
366  /* Failed to read next chunk from source */
367  return -1;
368  }
369 
370  /* Write a block of data to destination */
371  if (logfs->driver->write_data(logfs->flash_id,
372  dst_addr,
373  data_block,
374  blk_size) != 0) {
375  /* Failed to write chunk to destination */
376  return -2;
377  }
378 
379  /* Update the src/dst pointers */
380  src_size -= blk_size;
381  src_addr += blk_size;
382  dst_addr += blk_size;
383  }
384 
385  return 0;
386 }

419 static int32_t logfs_mount_log(struct logfs_state * logfs, uint8_t arena_id)
420 {
421  PIOS_Assert (!logfs->mounted);
422 
423  logfs->num_active_slots = 0;
424  logfs->num_free_slots = 0;
425  logfs->active_arena_id = arena_id;
426 
427  /* Scan the log to find out how full it is */
428  for (uint16_t slot_id = 1;
429  slot_id < (logfs->cfg->arena_size / logfs->cfg->slot_size);
430  slot_id++) {
431  struct slot_header slot_hdr;
432  uintptr_t slot_addr = logfs_get_addr (logfs, logfs->active_arena_id, slot_id);
433  if (logfs->driver->read_data(logfs->flash_id,
434  slot_addr,
435  (uint8_t *)&slot_hdr,
436  sizeof (slot_hdr)) != 0) {
437  return -1;
438  }
439 
440  /*
441  * Empty slots must be in a continguous block at the
442  * end of the arena.
443  */
444  PIOS_Assert (slot_hdr.state == SLOT_STATE_EMPTY ||
445  logfs->num_free_slots == 0);
446 
447  switch (slot_hdr.state) {
448  case SLOT_STATE_EMPTY:
449  logfs->num_free_slots++;
450  break;
451  case SLOT_STATE_ACTIVE:
452  logfs->num_active_slots++;
453  break;
454  case SLOT_STATE_RESERVED:
455  case SLOT_STATE_OBSOLETE:
456  break;
457  }
458  }
459 
460  /* Scan is complete, mark the arena mounted */
461  logfs->active_arena_id = arena_id;
462  logfs->mounted = true;
463 
464  return 0;
465 }

518 int32_t PIOS_FLASHFS_Logfs_Init(uintptr_t * fs_id, const struct flashfs_logfs_cfg * cfg, const struct pios_flash_driver * driver, uintptr_t flash_id)
519 {
520  PIOS_Assert(cfg);
521  PIOS_Assert(fs_id);
522  PIOS_Assert(driver);
523 
524  /* We must have at least 2 arenas for garbage collection to work */
525  PIOS_Assert((cfg->total_fs_size / cfg->arena_size > 1));
526 
527  /* Make sure the underlying flash driver provides the minimal set of required methods */
528  PIOS_Assert(driver->start_transaction);
529  PIOS_Assert(driver->end_transaction);
530  PIOS_Assert(driver->erase_sector);
531  PIOS_Assert(driver->write_data);
532  PIOS_Assert(driver->read_data);
533 
534  int8_t rc;
535 
536  struct logfs_state * logfs;
537 
538  logfs = (struct logfs_state *) PIOS_FLASHFS_Logfs_alloc();
539  if (!logfs) {
540  rc = -1;
541  goto out_exit;
542  }
543 
544  /* Bind configuration parameters to this filesystem instance */
545  logfs->cfg = cfg; /* filesystem configuration */
546  logfs->driver = driver; /* lower-level flash driver */
547  logfs->flash_id = flash_id; /* lower-level flash device id */
548  logfs->mounted = false;
549 
550  if (logfs->driver->start_transaction(logfs->flash_id) != 0) {
551  rc = -1;
552  goto out_exit;
553  }
554 
555  bool found = false;
556  int32_t arena_id;
557  for (uint8_t try = 0; !found && try < 2; try++) {
558  /* Find the active arena */
559  arena_id = logfs_find_active_arena(logfs);
560  if (arena_id >= 0) {
561  /* Found the active arena */
562  found = true;
563  break;
564  } else {
565  /* No active arena found, erase and activate arena 0 */
566  if (logfs_erase_arena(logfs, 0) != 0)
567  break;
568  if (logfs_activate_arena(logfs, 0) != 0)
569  break;
570  }
571  }
572 
573  if (!found) {
574  /* Still no active arena, something is broken */
575  rc = -2;
576  goto out_end_trans;
577  }
578 
579  /* We've found an active arena, mount it */
580  if (logfs_mount_log(logfs, arena_id) != 0) {
581  /* Failed to mount the log, something is broken */
582  rc = -3;
583  goto out_end_trans;
584  }
585 
586  /* Log has been mounted */
587  rc = 0;
588 
589  *fs_id = (uintptr_t) logfs;
590 
591 out_end_trans:
592  logfs->driver->end_transaction(logfs->flash_id);
593 
594 out_exit:
595  return rc;
596 }

616 /* NOTE: Must be called while holding the flash transaction lock */
617 static int32_t logfs_garbage_collect (struct logfs_state * logfs) {
618  PIOS_Assert (logfs->mounted);
619 
620  /* Source arena is the active arena */
621  uint8_t src_arena_id = logfs->active_arena_id;
622 
623  /* Compute destination arena */
624  uint8_t dst_arena_id = (logfs->active_arena_id + 1) % (logfs->cfg->total_fs_size / logfs->cfg->arena_size);
625 
626  /* Erase destination arena */
627  if (logfs_erase_arena (logfs, dst_arena_id) != 0) {
628  return -1;
629  }
630 
631  /* Reserve the destination arena so we can start filling it */
632  if (logfs_reserve_arena (logfs, dst_arena_id) != 0) {
633  /* Unable to reserve the arena */
634  return -2;
635  }
636 
637  /* Copy active slots from active arena to destination arena */
638  uint16_t dst_slot_id = 1;
639  for (uint16_t src_slot_id = 1;
640  src_slot_id < (logfs->cfg->arena_size / logfs->cfg->slot_size);
641  src_slot_id++) {
642  struct slot_header slot_hdr;
643  uintptr_t src_addr = logfs_get_addr (logfs, src_arena_id, src_slot_id);
644  if (logfs->driver->read_data(logfs->flash_id,
645  src_addr,
646  (uint8_t *)&slot_hdr,
647  sizeof (slot_hdr)) != 0) {
648  return -3;
649  }
650 
651  if (slot_hdr.state == SLOT_STATE_ACTIVE) {
652  uintptr_t dst_addr = logfs_get_addr (logfs, dst_arena_id, dst_slot_id);
653  if (logfs_raw_copy_bytes(logfs,
654  src_addr,
655  sizeof(slot_hdr) + slot_hdr.obj_size,
656  dst_addr) != 0) {
657  /* Failed to copy all bytes */
658  return -4;
659  }
660  dst_slot_id++;
661  }
662  }
663 
664  /* Activate the destination arena */
665  if (logfs_activate_arena (logfs, dst_arena_id) != 0) {
666  return -5;
667  }
668 
669  /* Unmount the source arena */
670  if (logfs_unmount_log (logfs) != 0) {
671  return -6;
672  }
673 
674  /* Obsolete the source arena */
675  if (logfs_obsolete_arena (logfs, src_arena_id) != 0) {
676  return -7;
677  }
678 
679  /* Mount the new arena */
680  if (logfs_mount_log (logfs, dst_arena_id) != 0) {
681  return -8;
682  }
683 
684  return 0;
685 }

687 /* NOTE: Must be called while holding the flash transaction lock */
688 static int16_t logfs_object_find_next (const struct logfs_state * logfs, struct slot_header * slot_hdr, uint16_t * curr_slot, uint32_t obj_id, uint16_t obj_inst_id)
689 {
690  PIOS_Assert(slot_hdr);
691  PIOS_Assert(curr_slot);
692 
693  /* First slot in the arena is reserved for arena header, skip it. */
694  if (*curr_slot == 0) *curr_slot = 1;
695 
696  for (uint16_t slot_id = *curr_slot;
697  slot_id < (logfs->cfg->arena_size / logfs->cfg->slot_size);
698  slot_id++) {
699  uintptr_t slot_addr = logfs_get_addr (logfs, logfs->active_arena_id, slot_id);
700 
701  if (logfs->driver->read_data(logfs->flash_id,
702  slot_addr,
703  (uint8_t *)slot_hdr,
704  sizeof (*slot_hdr)) != 0) {
705  return -2;
706  }
707  if (slot_hdr->state == SLOT_STATE_EMPTY) {
708  /* We hit the end of the log */
709  break;
710  }
711  if (slot_hdr->state == SLOT_STATE_ACTIVE &&
712  slot_hdr->obj_id == obj_id &&
713  slot_hdr->obj_inst_id == obj_inst_id) {
714  /* Found what we were looking for */

724 /* NOTE: Must be called while holding the flash transaction lock */
725 /* OPTIMIZE: could trust that there is at most one active version of every object and terminate the search when we find one */
726 static int8_t logfs_delete_object (struct logfs_state * logfs, uint32_t obj_id, uint16_t obj_inst_id)
727 {
728  int8_t rc;
729 
730  bool more = true;
731  uint16_t curr_slot_id = 0;
732  do {
733  struct slot_header slot_hdr;
734  switch (logfs_object_find_next (logfs, &slot_hdr, &curr_slot_id, obj_id, obj_inst_id)) {
735  case 0:
736  /* Found a matching slot. Obsolete it. */
737  slot_hdr.state = SLOT_STATE_OBSOLETE;
738  uintptr_t slot_addr = logfs_get_addr (logfs, logfs->active_arena_id, curr_slot_id);
739 
740  if (logfs->driver->write_data(logfs->flash_id,
741  slot_addr,
742  (uint8_t *)&slot_hdr,
743  sizeof(slot_hdr)) != 0) {
744  rc = -2;
745  goto out_exit;
746  }
747  /* Object has been successfully obsoleted and is no longer active */
748  logfs->num_active_slots--;
749  break;
750  case -1:
751  /* Search completed, object not found */
752  more = false;
753  rc = 0;
754  break;
755  default:
756  /* Error occurred during search */
757  rc = -1;
758  goto out_exit;
759  }
760  } while (more);
761 
762 out_exit:
763  return rc;
764 }

766 /* NOTE: Must be called while holding the flash transaction lock */
767 static int8_t logfs_reserve_free_slot (struct logfs_state * logfs, uint16_t * slot_id, struct slot_header * slot_hdr, uint32_t obj_id, uint16_t obj_inst_id, uint16_t obj_size)
768 {
769  PIOS_Assert(slot_id);
770  PIOS_Assert(slot_hdr);
771 
772  if (logfs->num_free_slots < 1) {
773  /* No free slots to allocate */
774  return -1;
775  }
776 
777  if (obj_size > (logfs->cfg->slot_size - sizeof (slot_hdr))) {
778  /* This object is too big for the slot */
779  return -2;
780  }
781 
782  uint16_t candidate_slot_id = (logfs->cfg->arena_size / logfs->cfg->slot_size) - logfs->num_free_slots;
783  PIOS_Assert(candidate_slot_id > 0);
784 
785  uintptr_t slot_addr = logfs_get_addr (logfs, logfs->active_arena_id, candidate_slot_id);
786 
787  if (logfs->driver->read_data(logfs->flash_id,
788  slot_addr,
789  (uint8_t *)slot_hdr,
790  sizeof (*slot_hdr)) != 0) {
791  /* Failed to read slot header for candidate slot */
792  return -3;
793  }
794 
795  if (slot_hdr->state != SLOT_STATE_EMPTY) {
796  /* Candidate slot isn't empty! Something is broken. */
797  PIOS_DEBUG_Assert(0);
798  return -4;
799  }
800 
801  /* Mark this slot as RESERVED */
802  slot_hdr->state = SLOT_STATE_RESERVED;
803  slot_hdr->obj_id = obj_id;
804  slot_hdr->obj_inst_id = obj_inst_id;
805  slot_hdr->obj_size = obj_size;
806 
807  if (logfs->driver->write_data(logfs->flash_id,
808  slot_addr,
809  (uint8_t *)slot_hdr,
810  sizeof(*slot_hdr)) != 0) {
811  /* Failed to write the slot header */
812  return -5;
813  }
814 
815  /* FIXME: If the header write (above) failed, may have partially written data, thus corrupting that slot but we would have missed decrementing this counter */
816  logfs->num_free_slots--;
817 
818  *slot_id = candidate_slot_id;
819  return 0;
820 }

822 /* NOTE: Must be called while holding the flash transaction lock */
823 static int8_t logfs_append_to_log (struct logfs_state * logfs, uint32_t obj_id, uint16_t obj_inst_id, uint8_t * obj_data, uint16_t obj_size)
824 {
825  /* Reserve a free slot for our new object */
826  uint16_t free_slot_id;
827  struct slot_header slot_hdr;
828  if (logfs_reserve_free_slot (logfs, &free_slot_id, &slot_hdr, obj_id, obj_inst_id, obj_size) != 0) {
829  /* Failed to reserve a free slot */
830  return -1;
831  }
832 
833  /* Compute slot address */
834  uintptr_t slot_addr = logfs_get_addr (logfs, logfs->active_arena_id, free_slot_id);
835 
836  /* Write the data into the reserved slot, starting after the slot header */
837  uintptr_t slot_offset = sizeof(slot_hdr);
838  while (obj_size > 0) {
839  /* Individual writes must fit entirely within a single page buffer. */
840  uint16_t page_remaining = logfs->cfg->page_size - (slot_offset % logfs->cfg->page_size);
841  uint16_t write_size = MIN(obj_size, page_remaining);
842  if (logfs->driver->write_data (logfs->flash_id,
843  slot_addr + slot_offset,
844  obj_data,
845  write_size) != 0) {
846  /* Failed to write the object data to the slot */
847  return -2;
848  }
849 
850  /* Update our accounting */
851  obj_data += write_size;
852  slot_offset += write_size;
853  obj_size -= write_size;
854  }
855 
856  /* Mark this slot active in one atomic step */
857  slot_hdr.state = SLOT_STATE_ACTIVE;
858  if (logfs->driver->write_data (logfs->flash_id,
859  slot_addr,
860  (uint8_t *)&slot_hdr,
861  sizeof(slot_hdr)) != 0) {
862  /* Failed to mark the slot active */
863  return -4;
864  }
865 
866  /* Object has been successfully written to the slot */
867  logfs->num_active_slots++;
868  return 0;
869 }

895 int32_t PIOS_FLASHFS_ObjSave(uintptr_t fs_id, uint32_t obj_id, uint16_t obj_inst_id, uint8_t * obj_data, uint16_t obj_size)
896 {
897  int8_t rc;
898 
899  struct logfs_state * logfs = (struct logfs_state *)fs_id;
900 
901  if (!PIOS_FLASHFS_Logfs_validate(logfs)) {
902  rc = -1;
903  goto out_exit;
904  }
905 
906  PIOS_Assert(obj_size <= (logfs->cfg->slot_size - sizeof(struct slot_header)));
907 
908  if (logfs->driver->start_transaction(logfs->flash_id) != 0) {
909  rc = -2;
910  goto out_exit;
911  }
912 
913  if (logfs_delete_object (logfs, obj_id, obj_inst_id) != 0) {
914  rc = -3;
915  goto out_end_trans;
916  }
917 
918  /*
919  * All old versions of this object + instance have been invalidated.
920  * Write the new object.
921  */
922 
923  /* Check if the arena is entirely full. */
924  if (logfs_fs_is_full(logfs)) {
925  /* Note: Filesystem Full means we're full of *active* records so gc won't help at all. */
926  rc = -4;
927  goto out_end_trans;
928  }
929 
930  /* Is garbage collection required? */
931  if (logfs_log_is_full(logfs)) {
932  /* Note: Log Full means the log is full but may contain obsolete slots so gc may free some space */
933  if (logfs_garbage_collect(logfs) != 0) {
934  rc = -5;
935  goto out_end_trans;
936  }
937  /* Check one more time just to be sure we actually free'd some space */
938  if (logfs_log_is_full(logfs)) {
939  /*
940  * Log is still full even after gc!
941  * NOTE: This should not happen since the filesystem wasn't full
942  * when we checked above so gc should have helped.
943  */
944  PIOS_DEBUG_Assert(0);
945  rc = -6;
946  goto out_end_trans;
947  }
948  }
949 
950  /* We have room for our new object. Append it to the log. */
951  if (logfs_append_to_log(logfs, obj_id, obj_inst_id, obj_data, obj_size) != 0) {
952  /* Error during append */
953  rc = -7;
954  goto out_end_trans;
955  }
956 
957  /* Object successfully written to the log */
958  rc = 0;
959 
960 out_end_trans:
961  logfs->driver->end_transaction(logfs->flash_id);
962 
963 out_exit:
964  return rc;
965 }
