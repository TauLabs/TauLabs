00061 /* From  Cain, Clark, "Error-Correction Coding For Digital Communications", pp. 216. */
00062 void
00063 Modified_Berlekamp_Massey (void)
00064 {       
00065   int n, L, L2, k, d, i;
00066   int psi[MAXDEG], psi2[MAXDEG], D[MAXDEG];
00067   int gamma[MAXDEG];
00068         
00069   /* initialize Gamma, the erasure locator polynomial */
00070   init_gamma(gamma);
00071 
00072   /* initialize to z */
00073   copy_poly(D, gamma);
00074   mul_z_poly(D);
00075         
00076   copy_poly(psi, gamma);        
00077   k = -1; L = NErasures;
00078         
00079   for (n = NErasures; n < RS_ECC_NPARITY; n++) {
00080         
00081     d = compute_discrepancy(psi, synBytes, L, n);
00082                 
00083     if (d != 0) {
00084                 
00085       /* psi2 = psi - d*D */
00086       for (i = 0; i < MAXDEG; i++) psi2[i] = psi[i] ^ gmult(d, D[i]);
00087                 
00088                 
00089       if (L < (n-k)) {
00090         L2 = n-k;
00091         k = n-L;
00092         /* D = scale_poly(ginv(d), psi); */
00093         for (i = 0; i < MAXDEG; i++) D[i] = gmult(psi[i], ginv(d));
00094         L = L2;
00095       }
00096                         
00097       /* psi = psi2 */
00098       for (i = 0; i < MAXDEG; i++) psi[i] = psi2[i];
00099     }
00100                 
00101     mul_z_poly(D);
00102   }
00103         
00104   for(i = 0; i < MAXDEG; i++) Lambda[i] = psi[i];
00105   compute_modified_omega();
00106 
00107         
00108 }

00128 mult_polys (int dst[], int p1[], int p2[])
00129 {
00130   int i, j;
00131   int tmp1[MAXDEG*2];
00132         
00133   for (i=0; i < (MAXDEG*2); i++) dst[i] = 0;

00136     for(j=MAXDEG; j<(MAXDEG*2); j++) tmp1[j]=0;
00137

00139     for(j=0; j<MAXDEG; j++) tmp1[j]=gmult(p2[j], p1[i]);
00140     /* and mult (shift) tmp1 right by i */
00141     for (j = (MAXDEG*2)-1; j >= i; j--) tmp1[j] = tmp1[j-i];
00142     for (j = 0; j < i; j++) tmp1[j] = 0;
00143                 
00144     /* add into partial product */
00145     for(j=0; j < (MAXDEG*2); j++) dst[j] ^= tmp1[j];
