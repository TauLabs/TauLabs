00198 static void gpsTask(void *parameters)
00199 {
00200         portTickType xDelay = 100 / portTICK_RATE_MS;
00201         uint32_t timeNowMs = xTaskGetTickCount() * portTICK_RATE_MS;
00202 
00203         GPSPositionData gpsposition;
00204         uint8_t gpsProtocol;
00205 
00206         ModuleSettingsGPSDataProtocolGet(&gpsProtocol);
00207 
00208 #if defined(PIOS_GPS_PROVIDES_AIRSPEED)
00209         gps_airspeed_initialize();
00210 #endif
00211 
00212         timeOfLastUpdateMs = timeNowMs;
00213         timeOfLastCommandMs = timeNowMs;
00214 
00215         GPSPositionGet(&gpsposition);
00216         // Loop forever
00217         while (1)
00218         {
00219                 uint8_t c;
00220 
00221                 // This blocks the task until there is something on the buffer
00222                 while (PIOS_COM_ReceiveBuffer(gpsPort, &c, 1, xDelay) > 0)
00223                 {
00224                         int res;
00225                         switch (gpsProtocol) {
00226 #if defined(PIOS_INCLUDE_GPS_NMEA_PARSER)
00227                                 case MODULESETTINGS_GPSDATAPROTOCOL_NMEA:
00228                                         res = parse_nmea_stream (c,gps_rx_buffer, &gpsposition, &gpsRxStats);
00229                                         break;
00230 #endif
00231 #if defined(PIOS_INCLUDE_GPS_UBX_PARSER)
00232                                 case MODULESETTINGS_GPSDATAPROTOCOL_UBX:
00233                                         res = parse_ubx_stream (c,gps_rx_buffer, &gpsposition, &gpsRxStats);
00234                                         break;
00235 #endif
00236                                 default:
00237                                         res = NO_PARSER; // this should not happen
00238                                         break;
00239                         }
00240 
00241                         if (res == PARSER_COMPLETE) {
00242                                 timeNowMs = xTaskGetTickCount() * portTICK_RATE_MS;
00243                                 timeOfLastUpdateMs = timeNowMs;
00244                                 timeOfLastCommandMs = timeNowMs;
00245                         }
00246                 }
00247 
00248                 // Check for GPS timeout
00249                 timeNowMs = xTaskGetTickCount() * portTICK_RATE_MS;
00250                 if ((timeNowMs - timeOfLastUpdateMs) >= GPS_TIMEOUT_MS) {
00251                         // we have not received any valid GPS sentences for a while.
00252                         // either the GPS is not plugged in or a hardware problem or the GPS has locked up.
00253                         uint8_t status = GPSPOSITION_STATUS_NOGPS;
00254                         GPSPositionStatusSet(&status);
00255                         AlarmsSet(SYSTEMALARMS_ALARM_GPS, SYSTEMALARMS_ALARM_ERROR);
00256                 } else {
00257                         // we appear to be receiving GPS sentences OK, we've had an update
00258                         //criteria for GPS-OK taken from this post...
00259                         //http://forums.openpilot.org/topic/1523-professors-insgps-in-svn/page__view__findpost__p__5220
00260                         if ((gpsposition.PDOP < 3.5) && (gpsposition.Satellites >= 7) &&
00261                                         (gpsposition.Status == GPSPOSITION_STATUS_FIX3D)) {
00262                                 AlarmsClear(SYSTEMALARMS_ALARM_GPS);
00263 #ifdef PIOS_GPS_SETS_HOMELOCATION
00264                                 HomeLocationData home;
00265                                 HomeLocationGet(&home);
00266 
00267                                 if (home.Set == HOMELOCATION_SET_FALSE)
00268                                         setHomeLocation(&gpsposition);
00269 #endif
00270                         } else if (gpsposition.Status == GPSPOSITION_STATUS_FIX3D)
00271                                                 AlarmsSet(SYSTEMALARMS_ALARM_GPS, SYSTEMALARMS_ALARM_WARNING);
00272                                         else
00273                                                 AlarmsSet(SYSTEMALARMS_ALARM_GPS, SYSTEMALARMS_ALARM_CRITICAL);
00274                 }
00275 
00276         }

00287         GPSTimeData gps;

00292                 // Store LLA

00294                 home.Longitude = gpsData->Longitude;
00295                 home.Altitude = gpsData->Altitude + gpsData->GeoidSeparation;
00296 
00297                 // Compute home ECEF coordinates and the rotation matrix into NED
00298                 double LLA[3] = { ((double)home.Latitude) / 10e6, ((double)home.Longitude) / 10e6, ((double)home.Altitude) };
00299 
00300                 // Compute magnetic flux direction at home location
00301                 if (WMM_GetMagVector(LLA[0], LLA[1], LLA[2], gps.Month, gps.Day, gps.Year, &home.Be[0]) >= 0)
