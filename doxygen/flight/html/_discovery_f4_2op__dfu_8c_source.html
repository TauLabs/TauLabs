78 void DataDownload(DownloadAction action) {
79  if ((DeviceState == downloading)) {
80 
81  uint8_t packetSize;
82  uint32_t offset;
83  uint32_t partoffset;
84  SendBuffer[0] = 0x01;
85  SendBuffer[1] = Download;
86  SendBuffer[2] = downPacketCurrent >> 24;
87  SendBuffer[3] = downPacketCurrent >> 16;
88  SendBuffer[4] = downPacketCurrent >> 8;
89  SendBuffer[5] = downPacketCurrent;
90  if (downPacketCurrent == downPacketTotal - 1) {
91  packetSize = downSizeOfLastPacket;
92  } else {
93  packetSize = 14;
94  }
95  for (uint8_t x = 0; x < packetSize; ++x) {
96  partoffset = (downPacketCurrent * 14 * 4) + (x * 4);
97  offset = baseOfAdressType(downType) + partoffset;
98  if (!flash_read(SendBuffer + (6 + x * 4), offset,
99  currentProgrammingDestination)) {
100  DeviceState = Last_operation_failed;
101  }
102  }
103  downPacketCurrent = downPacketCurrent + 1;
104  if (downPacketCurrent > downPacketTotal - 1) {
105  DeviceState = Last_operation_Success;
106  Aditionals = (uint32_t) Download;
107  }
108  sendData(SendBuffer + 1, 63);
109  }

111 void processComand(uint8_t *xReceive_Buffer) {
112 
113  Command = xReceive_Buffer[COMMAND];
114 #ifdef DEBUG_SSP
115  char str[63]= {0};
116  sprintf(str,"Received COMMAND:%d|",Command);
117  PIOS_COM_SendString(PIOS_COM_TELEM_USB,str);
118 #endif
119  EchoReqFlag = (Command >> 7);
120  EchoAnsFlag = (Command >> 6) & 0x01;
121  StartFlag = (Command >> 5) & 0x01;
122  Count = xReceive_Buffer[COUNT] << 24;
123  Count += xReceive_Buffer[COUNT + 1] << 16;
124  Count += xReceive_Buffer[COUNT + 2] << 8;
125  Count += xReceive_Buffer[COUNT + 3];
126 
127  Data = xReceive_Buffer[DATA] << 24;
128  Data += xReceive_Buffer[DATA + 1] << 16;
129  Data += xReceive_Buffer[DATA + 2] << 8;
130  Data += xReceive_Buffer[DATA + 3];
131  Data0 = xReceive_Buffer[DATA];
132  Data1 = xReceive_Buffer[DATA + 1];
133  Data2 = xReceive_Buffer[DATA + 2];
134  Data3 = xReceive_Buffer[DATA + 3];
135  Command = Command & 0b00011111;
136 
137  if (EchoReqFlag == 1) {
138  memcpy(echoBuffer, xReceive_Buffer, 64);
139  }
140  switch (Command) {
141  case EnterDFU:
142  if (((DeviceState == BLidle) && (Data0 < numberOfDevices))
143  || (DeviceState == DFUidle)) {
144  if (Data0 > 0)
145  OPDfuIni(true);
146  DeviceState = DFUidle;
147  currentProgrammingDestination = devicesTable[Data0].programmingType;
148  currentDeviceCanRead = devicesTable[Data0].readWriteFlags & 0x01;
149  currentDeviceCanWrite = devicesTable[Data0].readWriteFlags >> 1
150  & 0x01;
151  currentDevice = devicesTable[Data0];
152  uint8_t result = 0;
153  switch (currentProgrammingDestination) {
154  case Self_flash:
155  result = PIOS_BL_HELPER_FLASH_Ini();
156  break;
157  case Remote_flash_via_spi:
158  result = true;
159  break;
160  default:
161  DeviceState = Last_operation_failed;
162  Aditionals = (uint16_t) Command;
163  }
164  if (result != 1) {
165  DeviceState = Last_operation_failed;
166  Aditionals = (uint32_t) Command;
167  }
168  }
169  break;
170  case Upload:
171  if ((DeviceState == DFUidle) || (DeviceState == uploading)) {
172  if ((StartFlag == 1) && (Next_Packet == 0)) {
173  TransferType = Data0;
174  SizeOfTransfer = Count;
175  Next_Packet = 1;
176  Expected_CRC = Data2 << 24;
177  Expected_CRC += Data3 << 16;
178  Expected_CRC += xReceive_Buffer[DATA + 4] << 8;
179  Expected_CRC += xReceive_Buffer[DATA + 5];
180  SizeOfLastPacket = Data1;
181 
182  if (isBiggerThanAvailable(TransferType, (SizeOfTransfer - 1)
183  * 14 * 4 + SizeOfLastPacket * 4) == true) {
184  DeviceState = outsideDevCapabilities;
185  Aditionals = (uint32_t) Command;
186  } else {
187  uint8_t result = 1;
188  if (TransferType == FW) {
189  switch (currentProgrammingDestination) {
190  case Self_flash:
191  result = PIOS_BL_HELPER_FLASH_Start();
192  break;
193  case Remote_flash_via_spi:
194  result = false;
195  break;
196  default:
197  break;
198  }
199  }
200  if (result != 1) {
201  DeviceState = Last_operation_failed;
202  Aditionals = (uint32_t) Command;
203  } else {
204 
205  DeviceState = uploading;
206  }
207  }
208  } else if ((StartFlag != 1) && (Next_Packet != 0)) {
209  if (Count > SizeOfTransfer) {
210  DeviceState = too_many_packets;
211  Aditionals = Count;
212  } else if (Count == Next_Packet - 1) {
213  uint8_t numberOfWords = 14;
214  if (Count == SizeOfTransfer - 1)//is this the last packet?
215  {
216  numberOfWords = SizeOfLastPacket;
217  }
218  uint8_t result = 0;
219  switch (currentProgrammingDestination) {
220  case Self_flash:
221  for (uint8_t x = 0; x < numberOfWords; ++x) {
222  offset = 4 * x;
223  Data = xReceive_Buffer[DATA + offset] << 24;
224  Data += xReceive_Buffer[DATA + 1 + offset] << 16;
225  Data += xReceive_Buffer[DATA + 2 + offset] << 8;
226  Data += xReceive_Buffer[DATA + 3 + offset];
227  aux = baseOfAdressType(TransferType) + (uint32_t)(
228  Count * 14 * 4 + x * 4);
229  result = 0;
230  for (int retry = 0; retry < MAX_WRI_RETRYS; ++retry) {
231  if (result == 0) {
232  result = (FLASH_ProgramWord(aux, Data)
233  == FLASH_COMPLETE) ? 1 : 0;
234  }
235  }
236  }
237  break;
238  case Remote_flash_via_spi:
239  result = false; // No support for this for the PipX
240  break;
241  default:
242  result = 0;
243  break;
244  }
245  if (result != 1) {
246  DeviceState = Last_operation_failed;
247  Aditionals = (uint32_t) Command;
248  }
249 
250  ++Next_Packet;
251  } else {
252  DeviceState = wrong_packet_received;
253  Aditionals = Count;
254  }
255  } else {
256  DeviceState = Last_operation_failed;
257  Aditionals = (uint32_t) Command;
258  }
259  }
260  break;
261  case Req_Capabilities:
262  OPDfuIni(true);
263  Buffer[0] = 0x01;
264  Buffer[1] = Rep_Capabilities;
265  if (Data0 == 0) {
266  Buffer[2] = 0;
267  Buffer[3] = 0;
268  Buffer[4] = 0;
269  Buffer[5] = 0;
270  Buffer[6] = 0;
271  Buffer[7] = numberOfDevices;
272  uint16_t WRFlags = 0;
273  for (int x = 0; x < numberOfDevices; ++x) {
274  WRFlags = ((devicesTable[x].readWriteFlags << (x * 2))
275  | WRFlags);
276  }
277  Buffer[8] = WRFlags >> 8;
278  Buffer[9] = WRFlags;
279  } else {
280  Buffer[2] = devicesTable[Data0 - 1].sizeOfCode >> 24;
281  Buffer[3] = devicesTable[Data0 - 1].sizeOfCode >> 16;
282  Buffer[4] = devicesTable[Data0 - 1].sizeOfCode >> 8;
283  Buffer[5] = devicesTable[Data0 - 1].sizeOfCode;
284  Buffer[6] = Data0;
285  Buffer[7] = devicesTable[Data0 - 1].BL_Version;
286  Buffer[8] = devicesTable[Data0 - 1].sizeOfDescription;
287  Buffer[9] = devicesTable[Data0 - 1].devID;
288  Buffer[10] = devicesTable[Data0 - 1].FW_Crc >> 24;
289  Buffer[11] = devicesTable[Data0 - 1].FW_Crc >> 16;
290  Buffer[12] = devicesTable[Data0 - 1].FW_Crc >> 8;
291  Buffer[13] = devicesTable[Data0 - 1].FW_Crc;
292  Buffer[14] = devicesTable[Data0 - 1].devID >> 8;
293  Buffer[15] = devicesTable[Data0 - 1].devID;
294  }
295  sendData(Buffer + 1, 63);
296  break;
297  case JumpFW:
298  if (Data == 0x5AFE) {
299  /* Force board into safe mode */
300  PIOS_IAP_WriteBootCount(0xFFFF);
301  }
302  FLASH_Lock();
303  JumpToApp = 1;
304  break;
305  case Reset:
306  PIOS_SYS_Reset();
307  break;
308  case Abort_Operation:
309  Next_Packet = 0;
310  DeviceState = DFUidle;
311  JumpToApp = 0;
312  break;
313 
314  case Op_END:
315  if (DeviceState == uploading) {
316  if (Next_Packet - 1 == SizeOfTransfer) {
317  Next_Packet = 0;
318  if ((TransferType != FW) || (Expected_CRC == CalcFirmCRC())) {
319  DeviceState = Last_operation_Success;
320  } else {
321  DeviceState = CRC_Fail;
322  }
323  }
324  if (Next_Packet - 1 < SizeOfTransfer) {
325  Next_Packet = 0;
326  DeviceState = too_few_packets;
327  }
328  }
329  break;
330  case Download_Req:
331 #ifdef DEBUG_SSP
332  sprintf(str,"COMMAND:DOWNLOAD_REQ 1 Status=%d|",DeviceState);
333  PIOS_COM_SendString(PIOS_COM_TELEM_USB,str);
334 #endif
335  if (DeviceState == DFUidle) {
336 #ifdef DEBUG_SSP
337  PIOS_COM_SendString(PIOS_COM_TELEM_USB,"COMMAND:DOWNLOAD_REQ 1|");
338 #endif
339  downType = Data0;
340  downPacketTotal = Count;
341  downSizeOfLastPacket = Data1;
342  if (isBiggerThanAvailable(downType, (downPacketTotal - 1) * 14 * 4
343  + downSizeOfLastPacket * 4) == 1) {
344  DeviceState = outsideDevCapabilities;
345  Aditionals = (uint32_t) Command;
346 
347  } else {
348  downPacketCurrent = 0;
349  DeviceState = downloading;
350  }
351  } else {
352  DeviceState = Last_operation_failed;
353  Aditionals = (uint32_t) Command;
354  }
355  break;
356 
357  case Status_Request:
358  Buffer[0] = 0x01;
359  Buffer[1] = Status_Rep;
360  if (DeviceState == wrong_packet_received) {
361  Buffer[2] = Aditionals >> 24;
362  Buffer[3] = Aditionals >> 16;
363  Buffer[4] = Aditionals >> 8;
364  Buffer[5] = Aditionals;
365  } else {
366  Buffer[2] = 0;
367  Buffer[3] = ((uint16_t) Aditionals) >> 8;
368  Buffer[4] = ((uint16_t) Aditionals);
369  Buffer[5] = 0;
370  }
371  Buffer[6] = DeviceState;
372  Buffer[7] = 0;
373  Buffer[8] = 0;
374  Buffer[9] = 0;
375  sendData(Buffer + 1, 63);
376  if (DeviceState == Last_operation_Success) {
377  DeviceState = DFUidle;
378  }
379  break;
380  case Status_Rep:
381 
382  break;
383 
384  }
385  if (EchoReqFlag == 1) {
386  echoBuffer[0] = echoBuffer[0] | (1 << 6);
387  sendData(echoBuffer, 63);
388  }
389  return;
390 }
391 void OPDfuIni(uint8_t discover) {
392  const struct pios_board_info * bdinfo = &pios_board_info_blob;
393  Device dev;
394 
395  dev.programmingType = Self_flash;
396  dev.readWriteFlags = (BOARD_READABLE | (BOARD_WRITABLE << 1));
397  dev.startOfUserCode = bdinfo->fw_base;
398  dev.sizeOfCode = bdinfo->fw_size;
399  dev.sizeOfDescription = bdinfo->desc_size;
400  dev.BL_Version = bdinfo->bl_rev;
401  dev.FW_Crc = CalcFirmCRC();
402  dev.devID = (bdinfo->board_type << 8) | (bdinfo->board_rev);
403  dev.devType = bdinfo->hw_type;
404  numberOfDevices = 1;

410 uint32_t baseOfAdressType(DFUTransfer type) {

423 uint8_t isBiggerThanAvailable(DFUTransfer type, uint32_t size) {
424  switch (type) {
425  case FW:
426  return (size > currentDevice.sizeOfCode) ? 1 : 0;
427  break;
428  case Descript:
429  return (size > currentDevice.sizeOfDescription) ? 1 : 0;
430  break;
431  default:
432  return true;
