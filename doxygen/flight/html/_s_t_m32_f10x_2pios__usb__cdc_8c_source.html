132 int32_t PIOS_USB_CDC_Init(uint32_t * usbcdc_id, const struct pios_usb_cdc_cfg * cfg, uint32_t lower_id)
133 {
134  PIOS_Assert(usbcdc_id);
135  PIOS_Assert(cfg);
136 
137  struct pios_usb_cdc_dev * usb_cdc_dev;
138 
139  usb_cdc_dev = (struct pios_usb_cdc_dev *) PIOS_USB_CDC_alloc();
140  if (!usb_cdc_dev) goto out_fail;
141 
142  /* Bind the configuration to the device instance */
143  usb_cdc_dev->cfg = cfg;
144  usb_cdc_dev->lower_id = lower_id;
145 
146  pios_usb_cdc_id = (uint32_t) usb_cdc_dev;
147 
148  /* Bind lower level callbacks into the USB infrastructure */
149  pEpInt_OUT[cfg->ctrl_tx_ep - 1] = PIOS_USB_CDC_CTRL_EP_IN_Callback;
150  pEpInt_IN[cfg->data_tx_ep - 1] = PIOS_USB_CDC_DATA_EP_IN_Callback;
151  pEpInt_OUT[cfg->data_rx_ep - 1] = PIOS_USB_CDC_DATA_EP_OUT_Callback;
152 
153  *usbcdc_id = (uint32_t) usb_cdc_dev;
154 
155  return 0;
156 
157 out_fail:
158  return -1;
159 }

193 static void PIOS_USB_CDC_RxStart(uint32_t usbcdc_id, uint16_t rx_bytes_avail) {
194  struct pios_usb_cdc_dev * usb_cdc_dev = (struct pios_usb_cdc_dev *)usbcdc_id;
195 
196  bool valid = PIOS_USB_CDC_validate(usb_cdc_dev);
197  PIOS_Assert(valid);
198 
199  if (!PIOS_USB_CheckAvailable(usb_cdc_dev->lower_id)) {
200  return;
201  }
202 
203  // If endpoint was stalled and there is now space make it valid
204  PIOS_IRQ_Disable();
205  if ((GetEPRxStatus(usb_cdc_dev->cfg->data_rx_ep) != EP_RX_VALID) &&
206  (rx_bytes_avail >= sizeof(usb_cdc_dev->rx_packet_buffer))) {
207  SetEPRxStatus(usb_cdc_dev->cfg->data_rx_ep, EP_RX_VALID);
208  }
209  PIOS_IRQ_Enable();
210 }

212 static void PIOS_USB_CDC_SendData(struct pios_usb_cdc_dev * usb_cdc_dev)
213 {
214  uint16_t bytes_to_tx;
215 
216  if (!usb_cdc_dev->tx_out_cb) {
217  return;
218  }
219 
220  bool need_yield = false;
221  bytes_to_tx = (usb_cdc_dev->tx_out_cb)(usb_cdc_dev->tx_out_context,
222  usb_cdc_dev->tx_packet_buffer,
223  sizeof(usb_cdc_dev->tx_packet_buffer),
224  NULL,
225  &need_yield);
226  if (bytes_to_tx == 0) {
227  return;
228  }
229 
230  UserToPMABufferCopy(usb_cdc_dev->tx_packet_buffer,
231  GetEPTxAddr(usb_cdc_dev->cfg->data_tx_ep),
232  bytes_to_tx);
233  SetEPTxCount(usb_cdc_dev->cfg->data_tx_ep, bytes_to_tx);
234  SetEPTxValid(usb_cdc_dev->cfg->data_tx_ep);
235 
236 #if defined(PIOS_INCLUDE_FREERTOS)
237  if (need_yield) {
238  vPortYieldFromISR();
239  }
240 #endif /* PIOS_INCLUDE_FREERTOS */
241 }

272 static void PIOS_USB_CDC_DATA_EP_OUT_Callback(void)
273 {
274  struct pios_usb_cdc_dev * usb_cdc_dev = (struct pios_usb_cdc_dev *)pios_usb_cdc_id;
275 
276  bool valid = PIOS_USB_CDC_validate(usb_cdc_dev);
277  PIOS_Assert(valid);
278 
279  uint32_t DataLength;
280 
281  /* Get the number of received data on the selected Endpoint */
282  DataLength = GetEPRxCount(usb_cdc_dev->cfg->data_rx_ep);
283  if (DataLength > sizeof(usb_cdc_dev->rx_packet_buffer)) {
284  usb_cdc_dev->rx_oversize++;
285  DataLength = sizeof(usb_cdc_dev->rx_packet_buffer);
286  }
287 
288  /* Use the memory interface function to read from the selected endpoint */
289  PMAToUserBufferCopy((uint8_t *) usb_cdc_dev->rx_packet_buffer,
290  GetEPRxAddr(usb_cdc_dev->cfg->data_rx_ep),
291  DataLength);
292 
293  if (!usb_cdc_dev->rx_in_cb) {
294  /* No Rx call back registered, disable the receiver */
295  SetEPRxStatus(usb_cdc_dev->cfg->data_rx_ep, EP_RX_NAK);
296  return;
297  }
298 
299  uint16_t headroom;
300  bool need_yield = false;
301  uint16_t rc;
302  rc = (usb_cdc_dev->rx_in_cb)(usb_cdc_dev->rx_in_context,
303  usb_cdc_dev->rx_packet_buffer,
304  DataLength,
305  &headroom,
306  &need_yield);
307 
308  if (rc < DataLength) {
309  /* Lost bytes on rx */
310  usb_cdc_dev->rx_dropped += (DataLength - rc);
311  }
312 
313  if (headroom >= sizeof(usb_cdc_dev->rx_packet_buffer)) {
314  /* We have room for a maximum length message */
315  SetEPRxStatus(usb_cdc_dev->cfg->data_rx_ep, EP_RX_VALID);
316  } else {
317  /* Not enough room left for a message, apply backpressure */
318  SetEPRxStatus(usb_cdc_dev->cfg->data_rx_ep, EP_RX_NAK);
319  }
320 
321 #if defined(PIOS_INCLUDE_FREERTOS)
322  if (need_yield) {
323  vPortYieldFromISR();
324  }
325 #endif /* PIOS_INCLUDE_FREERTOS */
326 }

329 RESULT PIOS_USB_CDC_SetControlLineState(void)
330 {
331  struct pios_usb_cdc_dev * usb_cdc_dev = (struct pios_usb_cdc_dev *)pios_usb_cdc_id;
332 
333  bool valid = PIOS_USB_CDC_validate(usb_cdc_dev);
334  if (!valid) {
335  /* No CDC interface is configured */
336  return USB_UNSUPPORT;
337  }
338 
339  uint8_t wValue0 = pInformation->USBwValue0;
340  uint8_t wValue1 = pInformation->USBwValue1;
341 
342  control_line_state = wValue1 << 8 | wValue0;
343 
344  return USB_SUCCESS;

413 static void PIOS_USB_CDC_CTRL_EP_IN_Callback(void)
414 {
415  struct pios_usb_cdc_dev * usb_cdc_dev = (struct pios_usb_cdc_dev *)pios_usb_cdc_id;
416 
417  bool valid = PIOS_USB_CDC_validate(usb_cdc_dev);
418  PIOS_Assert(valid);
419 
420  /* Give back UART State Bitmap */
421  /* UART State Bitmap
422  * 15-7: reserved
423  * 6: bOverRun overrun error
424  * 5: bParity parity error
425  * 4: bFraming framing error
426  * 3: bRingSignal RI
427  * 2: bBreak break reception
428  * 1: bTxCarrier DSR
429  * 0: bRxCarrier DCD
430  */
431  uart_state.bmUartState = htousbs(0x0003);
432 
433  UserToPMABufferCopy((uint8_t *) &uart_state,
434  GetEPTxAddr(usb_cdc_dev->cfg->ctrl_tx_ep),
435  sizeof(uart_state));
436  SetEPTxCount(usb_cdc_dev->cfg->ctrl_tx_ep, PIOS_USB_BOARD_CDC_MGMT_LENGTH);
437  SetEPTxValid(usb_cdc_dev->cfg->ctrl_tx_ep);
438 }
