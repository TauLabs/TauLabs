00271 static void PIOS_USB_CDC_DATA_EP_OUT_Callback(void)
00272 {
00273         struct pios_usb_cdc_dev * usb_cdc_dev = (struct pios_usb_cdc_dev *)pios_usb_cdc_id;
00274 
00275         bool valid = PIOS_USB_CDC_validate(usb_cdc_dev);
00276         PIOS_Assert(valid);
00277 
00278         uint32_t DataLength;
00279 
00280         /* Get the number of received data on the selected Endpoint */
00281         DataLength = GetEPRxCount(usb_cdc_dev->cfg->data_rx_ep);
00282         if (DataLength > sizeof(usb_cdc_dev->rx_packet_buffer)) {
00283                 usb_cdc_dev->rx_oversize++;
00284                 DataLength = sizeof(usb_cdc_dev->rx_packet_buffer);
00285         }
00286 
00287         /* Use the memory interface function to read from the selected endpoint */
00288         PMAToUserBufferCopy((uint8_t *) usb_cdc_dev->rx_packet_buffer,
00289                         GetEPRxAddr(usb_cdc_dev->cfg->data_rx_ep),
00290                         DataLength);
00291 
00292         if (!usb_cdc_dev->rx_in_cb) {
00293                 /* No Rx call back registered, disable the receiver */
00294                 SetEPRxStatus(usb_cdc_dev->cfg->data_rx_ep, EP_RX_NAK);
00295                 return;
00296         }
00297 
00298         uint16_t headroom;
00299         bool need_yield = false;
00300         uint16_t rc;
00301         rc = (usb_cdc_dev->rx_in_cb)(usb_cdc_dev->rx_in_context,
00302                                 usb_cdc_dev->rx_packet_buffer,
00303                                 DataLength,
00304                                 &headroom,
00305                                 &need_yield);
00306 
00307         if (rc < DataLength) {
00308                 /* Lost bytes on rx */
00309                 usb_cdc_dev->rx_dropped += (DataLength - rc);
00310         }
00311 
00312         if (headroom >= sizeof(usb_cdc_dev->rx_packet_buffer)) {
00313                 /* We have room for a maximum length message */
00314                 SetEPRxStatus(usb_cdc_dev->cfg->data_rx_ep, EP_RX_VALID);
00315         } else {
00316                 /* Not enough room left for a message, apply backpressure */
00317                 SetEPRxStatus(usb_cdc_dev->cfg->data_rx_ep, EP_RX_NAK);
00318         }
00319 
00320 #if defined(PIOS_INCLUDE_FREERTOS)
00321         portEND_SWITCHING_ISR(need_yield);
00322 #endif  /* PIOS_INCLUDE_FREERTOS */
00323 }
