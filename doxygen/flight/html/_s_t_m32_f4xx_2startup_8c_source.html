00063 void
00064 _main(void)
00065 {
00066         // load the stack base for the current stack before we attempt to branch to any function
00067         // that might bounds-check the stack
00068         asm volatile ("mov r10, %0" : : "r" (&irq_stack[0]) : );
00069 
00070         /* enable usage, bus and memory faults */
00071         SCB->SHCSR |= SCB_SHCSR_USGFAULTENA_Msk | SCB_SHCSR_BUSFAULTENA_Msk | SCB_SHCSR_MEMFAULTENA_Msk;
00072 
00073         /* configure FP state save behaviour - automatic, lazy save */
00074         FPU->FPCCR |= FPU_FPCCR_ASPEN_Msk | FPU_FPCCR_LSPEN_Msk;
00075 
00076         /* configure default FPU state */
00077         FPU->FPDSCR |= FPU_FPDSCR_DN_Msk;       /* enable Default NaN */
00078 
00079         /* enable the FPU */
00080         SCB->CPACR |= (0xf << 20);      // turn on CP10/11 for FP support on cores that implement it
00081 
00082         /* copy initialised data from flash to RAM */
00083         memcpy(&_sdata, &_sidata, &_edata - &_sdata);
00084 
00085         /* zero the BSS */
00086         memset(&_sbss, 0, &_ebss - &_sbss);
00087 
00088         /* zero any 'fast' RAM that's been used */
00089         memset(&_sfast, 0, &_efast - &_sfast);
00090 
00091         /* fill most of the IRQ/bootstrap stack with a watermark pattern so we can measure how much is used */
00092         /* leave a little space at the top in case memset() isn't a leaf with no locals */
00093         memset(&irq_stack, 0xa5, sizeof(irq_stack) - 64);
00094 
00095         /* call main */
00096         (void)main();
00097 }
