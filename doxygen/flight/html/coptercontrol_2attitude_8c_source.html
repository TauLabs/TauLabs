197 static void AttitudeTask(void *parameters)
198 {
199  AlarmsClear(SYSTEMALARMS_ALARM_ATTITUDE);
200 
201  // Set critical error and wait until the accel is producing data
202  while(PIOS_ADXL345_FifoElements() == 0) {
203  AlarmsSet(SYSTEMALARMS_ALARM_ATTITUDE, SYSTEMALARMS_ALARM_CRITICAL);
204  PIOS_WDG_UpdateFlag(PIOS_WDG_ATTITUDE);
205  }
206 
207  const struct pios_board_info * bdinfo = &pios_board_info_blob;
208 
209  bool cc3d = bdinfo->board_rev == 0x02;
210 
211  if (!cc3d) {
212 #if defined(PIOS_INCLUDE_ADC)
213  // Create queue for passing gyro data, allow 2 back samples in case
214  gyro_queue = xQueueCreate(1, sizeof(float) * 4);
215  PIOS_Assert(gyro_queue != NULL);
216  PIOS_ADC_SetQueue(gyro_queue);
217  PIOS_ADC_Config((PIOS_ADC_RATE / 1000.0f) * UPDATE_RATE);
218 #endif
219  }
220 
221  // Force settings update to make sure rotation loaded
222  settingsUpdatedCb(AttitudeSettingsHandle());
223 
224  enum complimentary_filter_status complimentary_filter_status;
225  complimentary_filter_status = CF_POWERON;
226 
227  uint32_t arming_count = 0;
228 
229  // Main task loop
230  while (1) {
231 
232  FlightStatusData flightStatus;
233  FlightStatusGet(&flightStatus);
234 
235  if (complimentary_filter_status == CF_POWERON) {
236 
237  complimentary_filter_status = (xTaskGetTickCount() > 1000) ?
238  CF_INITIALIZING : CF_POWERON;
239 
240  } else if(complimentary_filter_status == CF_INITIALIZING &&
241  (xTaskGetTickCount() < 7000) &&
242  (xTaskGetTickCount() > 1000)) {
243 
244  // For first 7 seconds use accels to get gyro bias
245  accelKp = 0.1f + 0.1f * (xTaskGetTickCount() < 4000);
246  accelKi = 0.1;
247  yawBiasRate = 0.1;
248  accel_filter_enabled = false;
249 
250  } else if (zero_during_arming &&
251  (flightStatus.Armed == FLIGHTSTATUS_ARMED_ARMING)) {
252 
253  // Use a rapidly decrease accelKp to force the attitude to snap back
254  // to level and then converge more smoothly
255  if (arming_count < 20)
256  accelKp = 1.0f;
257  else if (accelKp > 0.1f)
258  accelKp -= 0.01f;
259  arming_count++;
260 
261  accelKi = 0.1f;
262  yawBiasRate = 0.1f;
263  accel_filter_enabled = false;
264 
265  // Indicate arming so that after arming it reloads
266  // the normal settings
267  if (complimentary_filter_status != CF_ARMING) {
268  accumulate_gyro_zero();
269  complimentary_filter_status = CF_ARMING;
270  accumulating_gyro = true;
271  }
272 
273  } else if (complimentary_filter_status == CF_ARMING ||
274  complimentary_filter_status == CF_INITIALIZING) {
275 
276  // Reload settings (all the rates)
277  AttitudeSettingsAccelKiGet(&accelKi);
278  AttitudeSettingsAccelKpGet(&accelKp);
279  AttitudeSettingsYawBiasRateGet(&yawBiasRate);
280  if(accel_alpha > 0.0f)
281  accel_filter_enabled = true;
282 
283  // If arming that means we were accumulating gyro
284  // samples. Compute new bias.
285  if (complimentary_filter_status == CF_ARMING) {
286  accumulate_gyro_compute();
287  accumulating_gyro = false;
288  arming_count = 0;
289  }
290 
291  // Indicate normal mode to prevent rerunning this code
292  complimentary_filter_status = CF_NORMAL;
293  }
294 
295  PIOS_WDG_UpdateFlag(PIOS_WDG_ATTITUDE);
296 
297  AccelsData accels;
298  GyrosData gyros;
299  int32_t retval = 0;
300 
301  if (cc3d)
302  retval = updateSensorsCC3D(&accels, &gyros);
303  else
304  retval = updateSensors(&accels, &gyros);
305 
306  // Only update attitude when sensor data is good
307  if (retval != 0)
308  AlarmsSet(SYSTEMALARMS_ALARM_ATTITUDE, SYSTEMALARMS_ALARM_ERROR);
309  else {
310  // Do not update attitude data in simulation mode
311  if (!AttitudeActualReadOnly())
312  updateAttitude(&accels, &gyros);
313 
314  AlarmsClear(SYSTEMALARMS_ALARM_ATTITUDE);
315  }
316  }
317 }

324 static int32_t updateSensors(AccelsData * accelsData, GyrosData * gyrosData)
325 {
326  struct pios_adxl345_data accel_data;
327  float gyro[4];
328 
329  // Only wait the time for two nominal updates before setting an alarm
330  if(xQueueReceive(gyro_queue, (void * const) gyro, UPDATE_RATE * 2) == errQUEUE_EMPTY) {
331  AlarmsSet(SYSTEMALARMS_ALARM_ATTITUDE, SYSTEMALARMS_ALARM_ERROR);
332  return -1;
333  }
334 
335  // Do not read raw sensor data in simulation mode
336  if (GyrosReadOnly() || AccelsReadOnly())
337  return 0;
338 
339  // No accel data available
340  if(PIOS_ADXL345_FifoElements() == 0)
341  return -1;
342 
343  // Convert the ADC data into the standard normalized format
344  struct pios_sensor_gyro_data gyros;
345  gyros.temperature = gyro[0];
346  gyros.x = -(gyro[1] - GYRO_NEUTRAL) * IDG_GYRO_GAIN;
347  gyros.y = (gyro[2] - GYRO_NEUTRAL) * IDG_GYRO_GAIN;
348  gyros.z = -(gyro[3] - GYRO_NEUTRAL) * IDG_GYRO_GAIN;
349 
350  // Convert the ADXL345 data into the standard normalized format
351  int32_t x = 0;
352  int32_t y = 0;
353  int32_t z = 0;
354  uint8_t i = 0;
355  uint8_t samples_remaining;
356  do {
357  i++;
358  samples_remaining = PIOS_ADXL345_Read(&accel_data);
359  x += accel_data.x;
360  y += -accel_data.y;
361  z += -accel_data.z;
362  } while ( (i < 32) && (samples_remaining > 0) );
363 
364  struct pios_sensor_accel_data accels;
365 
366  accels.x = ((float) x / i) * ADXL345_ACCEL_SCALE;
367  accels.y = ((float) y / i) * ADXL345_ACCEL_SCALE;
368  accels.z = ((float) z / i) * ADXL345_ACCEL_SCALE;
369 
370  // Apply rotation / calibration and assign to the UAVO
371  update_gyros(&gyros, gyrosData);
372  update_accels(&accels, accelsData);
373 
374  update_trimming(accelsData);
375 
376 
377 
378  GyrosSet(gyrosData);
379  AccelsSet(accelsData);
380 
381  return 0;
382 }

425 static void update_accels(struct pios_sensor_accel_data *accels, AccelsData * accelsData)
426 {
427  // Average and scale the accels before rotation
428  float accels_out[3] = {accels->x * sensorSettings.AccelScale[0] - sensorSettings.AccelBias[0],
429  accels->y * sensorSettings.AccelScale[1] - sensorSettings.AccelBias[1],
430  accels->z * sensorSettings.AccelScale[2] - sensorSettings.AccelBias[2]};
431 
432  if (rotate) {
433  float accel_rotated[3];
434  rot_mult(R, accels_out, accel_rotated, false);
435  accelsData->x = accel_rotated[0];
436  accelsData->y = accel_rotated[1];
437  accelsData->z = accel_rotated[2];
438  } else {
439  accelsData->x = accels_out[0];
440  accelsData->y = accels_out[1];
441  accelsData->z = accels_out[2];
442  }
443 
444  accelsData->temperature = accels->temperature;
445 }

451 static void update_gyros(struct pios_sensor_gyro_data *gyros, GyrosData * gyrosData)
452 {
453  static float gyro_temp_bias[3] = {0,0,0};
454 
455  // Scale the gyros
456  float gyros_out[3] = {gyros->x * sensorSettings.GyroScale[0],
457  gyros->y * sensorSettings.GyroScale[1],
458  gyros->z * sensorSettings.GyroScale[2]};
459 
460  if (rotate) {
461  float gyros[3];
462  rot_mult(R, gyros_out, gyros, false);
463  gyrosData->x = gyros[0];
464  gyrosData->y = gyros[1];
465  gyrosData->z = gyros[2];
466  } else {
467  gyrosData->x = gyros_out[0];
468  gyrosData->y = gyros_out[1];
469  gyrosData->z = gyros_out[2];
470  }
471 
472  // When computing the bias accumulate samples
473  accumulate_gyro(gyrosData, gyro_temp_bias);
474 
475  // Update the bias due to the temperature
476  updateTemperatureComp(gyrosData->temperature, gyro_temp_bias);
477 
478  if(bias_correct_gyro) {
479  // Applying integral component here so it can be seen on the gyros and correct bias
480  gyrosData->x -= gyro_temp_bias[0] + gyro_correct_int[0];
481  gyrosData->y -= gyro_temp_bias[1] + gyro_correct_int[1];
482  gyrosData->z -= gyro_temp_bias[2] + gyro_correct_int[2];
483  }
484 
485  // Because most crafts wont get enough information from gravity to zero yaw gyro, we try
486  // and make it average zero (weakly)
487  gyro_correct_int[2] += gyrosData->z * yawBiasRate;
488 
489  gyrosData->temperature = gyros->temperature;
490 }

564 static inline void apply_accel_filter(const float * raw, float * filtered)
565 {
566  if(accel_filter_enabled) {
567  filtered[0] = filtered[0] * accel_alpha + raw[0] * (1 - accel_alpha);
568  filtered[1] = filtered[1] * accel_alpha + raw[1] * (1 - accel_alpha);
569  filtered[2] = filtered[2] * accel_alpha + raw[2] * (1 - accel_alpha);
570  } else {
571  filtered[0] = raw[0];
572  filtered[1] = raw[1];
573  filtered[2] = raw[2];
574  }

577 static void updateAttitude(AccelsData * accelsData, GyrosData * gyrosData)
578 {
579  float dT;
580  portTickType thisSysTime = xTaskGetTickCount();
581  static portTickType lastSysTime = 0;
582  static float accels_filtered[3] = {0,0,0};
583  static float grot_filtered[3] = {0,0,0};
584 
585  dT = (thisSysTime == lastSysTime) ? 0.001 : (portMAX_DELAY & (thisSysTime - lastSysTime)) / portTICK_RATE_MS / 1000.0f;
586  lastSysTime = thisSysTime;
587 
588  // Bad practice to assume structure order, but saves memory
589  float * gyros = &gyrosData->x;
590  float * accels = &accelsData->x;
591 
592  float grot[3];
593  float accel_err[3];
594 
595  // Apply smoothing to accel values, to reduce vibration noise before main calculations.
596  apply_accel_filter(accels,accels_filtered);
597 
598  // Rotate gravity to body frame, filter and cross with accels
599  grot[0] = -(2 * (q[1] * q[3] - q[0] * q[2]));
600  grot[1] = -(2 * (q[2] * q[3] + q[0] * q[1]));
601  grot[2] = -(q[0] * q[0] - q[1]*q[1] - q[2]*q[2] + q[3]*q[3]);
602 
603  // Apply same filtering to the rotated attitude to match delays
604  apply_accel_filter(grot,grot_filtered);
605 
606  // Compute the error between the predicted direction of gravity and smoothed acceleration
607  CrossProduct((const float *) accels_filtered, (const float *) grot_filtered, accel_err);
608 
609  // Account for accel magnitude
610  float accel_mag = sqrtf(accels_filtered[0]*accels_filtered[0] + accels_filtered[1]*accels_filtered[1] + accels_filtered[2]*accels_filtered[2]);
611 
612  // Account for filtered gravity vector magnitude
613  float grot_mag;
614 
615  if (accel_filter_enabled)
616  grot_mag = sqrtf(grot_filtered[0]*grot_filtered[0] + grot_filtered[1]*grot_filtered[1] + grot_filtered[2]*grot_filtered[2]);
617  else
618  grot_mag = 1.0f;
619 
620  if (grot_mag > 1.0e-3f && accel_mag > 1.0e-3f) {
621  accel_err[0] /= (accel_mag*grot_mag);
622  accel_err[1] /= (accel_mag*grot_mag);
623  accel_err[2] /= (accel_mag*grot_mag);
624 
625  // Accumulate integral of error. Scale here so that units are (deg/s) but Ki has units of s
626  gyro_correct_int[0] -= accel_err[0] * accelKi;
627  gyro_correct_int[1] -= accel_err[1] * accelKi;
628 
629  // Correct rates based on error, integral component dealt with in updateSensors
630  gyros[0] += accel_err[0] * accelKp / dT;
631  gyros[1] += accel_err[1] * accelKp / dT;
632  gyros[2] += accel_err[2] * accelKp / dT;
633  }
634 
635  { // scoping variables to save memory
636  // Work out time derivative from INSAlgo writeup
637  // Also accounts for the fact that gyros are in deg/s
638  float qdot[4];
639  qdot[0] = (-q[1] * gyros[0] - q[2] * gyros[1] - q[3] * gyros[2]) * dT * DEG2RAD / 2;
640  qdot[1] = (q[0] * gyros[0] - q[3] * gyros[1] + q[2] * gyros[2]) * dT * DEG2RAD / 2;
641  qdot[2] = (q[3] * gyros[0] + q[0] * gyros[1] - q[1] * gyros[2]) * dT * DEG2RAD / 2;
642  qdot[3] = (-q[2] * gyros[0] + q[1] * gyros[1] + q[0] * gyros[2]) * dT * DEG2RAD / 2;
643 
644  // Take a time step
645  q[0] = q[0] + qdot[0];
646  q[1] = q[1] + qdot[1];
647  q[2] = q[2] + qdot[2];
648  q[3] = q[3] + qdot[3];
649 
650  if(q[0] < 0) {
651  q[0] = -q[0];
652  q[1] = -q[1];
653  q[2] = -q[2];
654  q[3] = -q[3];
655  }
656  }
657 
658  // Renomalize
659  float qmag = sqrtf(q[0]*q[0] + q[1]*q[1] + q[2]*q[2] + q[3]*q[3]);
660  q[0] = q[0] / qmag;
661  q[1] = q[1] / qmag;
662  q[2] = q[2] / qmag;
663  q[3] = q[3] / qmag;
664 
665  // If quaternion has become inappropriately short or is nan reinit.
666  // THIS SHOULD NEVER ACTUALLY HAPPEN
667  if((fabs(qmag) < 1e-3) || (qmag != qmag)) {
668  q[0] = 1;
669  q[1] = 0;
670  q[2] = 0;
671  q[3] = 0;
672  }
673 
674  AttitudeActualData attitudeActual;
675  AttitudeActualGet(&attitudeActual);
676 
677  quat_copy(q, &attitudeActual.q1);
678 
679  // Convert into eueler degrees (makes assumptions about RPY order)
680  Quaternion2RPY(&attitudeActual.q1,&attitudeActual.Roll);
681 
682  AttitudeActualSet(&attitudeActual);
683 }

689 static void updateTemperatureComp(float temperature, float *temp_bias)
690 {
691  static int temp_counter = 0;
692  static float temp_accum = 0;
693 
694  static const float TEMP_MIN = -10;
695  static const float TEMP_MAX = 60;
696 
697  if (temperature < TEMP_MIN)
698  temperature = TEMP_MIN;
699  if (temperature > TEMP_MAX)
700  temperature = TEMP_MAX;
701 
702  if (temp_counter < 500) {
703  temp_accum += temperature;
704  temp_counter ++;
705  } else {
706  float t = temp_accum / temp_counter;
707  temp_accum = 0;
708  temp_counter = 0;
709 
710  // Compute a third order polynomial for each chanel after each 500 samples
711  temp_bias[0] = sensorSettings.XGyroTempCoeff[0] +
712  sensorSettings.XGyroTempCoeff[1] * t +
713  sensorSettings.XGyroTempCoeff[2] * powf(t,2) +
714  sensorSettings.XGyroTempCoeff[3] * powf(t,3);
715  temp_bias[1] = sensorSettings.YGyroTempCoeff[0] +
716  sensorSettings.YGyroTempCoeff[1] * t +
717  sensorSettings.YGyroTempCoeff[2] * powf(t,2) +
718  sensorSettings.YGyroTempCoeff[3] * powf(t,3);
719  temp_bias[2] = sensorSettings.ZGyroTempCoeff[0] +
720  sensorSettings.ZGyroTempCoeff[1] * t +
721  sensorSettings.ZGyroTempCoeff[2] * powf(t,2) +
722  sensorSettings.ZGyroTempCoeff[3] * powf(t,3);
723  }

726 static void settingsUpdatedCb(UAVObjEvent * objEv) {
727  AttitudeSettingsData attitudeSettings;
728  AttitudeSettingsGet(&attitudeSettings);
729  SensorSettingsGet(&sensorSettings);
730 
731 
732  accelKp = attitudeSettings.AccelKp;
733  accelKi = attitudeSettings.AccelKi;
734  yawBiasRate = attitudeSettings.YawBiasRate;
735 
736  // Calculate accel filter alpha, in the same way as for gyro data in stabilization module.
737  const float fakeDt = 0.0025;
738  if(attitudeSettings.AccelTau < 0.0001) {
739  accel_alpha = 0; // not trusting this to resolve to 0
740  accel_filter_enabled = false;
741  } else {
742  accel_alpha = expf(-fakeDt / attitudeSettings.AccelTau);
743  accel_filter_enabled = true;
744  }
745 
746  zero_during_arming = attitudeSettings.ZeroDuringArming == ATTITUDESETTINGS_ZERODURINGARMING_TRUE;
747  bias_correct_gyro = attitudeSettings.BiasCorrectGyro == ATTITUDESETTINGS_BIASCORRECTGYRO_TRUE;
748 
749  gyro_correct_int[0] = 0;
750  gyro_correct_int[1] = 0;
751  gyro_correct_int[2] = 0;
752 
753  // Indicates not to expend cycles on rotation
754  if(attitudeSettings.BoardRotation[0] == 0 && attitudeSettings.BoardRotation[1] == 0 &&
755  attitudeSettings.BoardRotation[2] == 0) {
756  rotate = 0;
757 
758  // Shouldn't be used but to be safe
759  float rotationQuat[4] = {1,0,0,0};
760  Quaternion2R(rotationQuat, R);
761  } else {
762  float rotationQuat[4];
763  const float rpy[3] = {attitudeSettings.BoardRotation[ATTITUDESETTINGS_BOARDROTATION_ROLL] / 100.0f,
764  attitudeSettings.BoardRotation[ATTITUDESETTINGS_BOARDROTATION_PITCH] / 100.0f,
765  attitudeSettings.BoardRotation[ATTITUDESETTINGS_BOARDROTATION_YAW] / 100.0f};
766  RPY2Quaternion(rpy, rotationQuat);
767  Quaternion2R(rotationQuat, R);
768  rotate = 1;
769  }
770 
771  if (attitudeSettings.TrimFlight == ATTITUDESETTINGS_TRIMFLIGHT_START) {
772  trim_accels[0] = 0;
773  trim_accels[1] = 0;
774  trim_accels[2] = 0;
775  trim_samples = 0;
776  trim_requested = true;
777  } else if (attitudeSettings.TrimFlight == ATTITUDESETTINGS_TRIMFLIGHT_LOAD) {
778  trim_requested = false;
779 
780  // Get sensor data mean
781  float a_body[3] = { trim_accels[0] / trim_samples,
782  trim_accels[1] / trim_samples,
783  trim_accels[2] / trim_samples
784  };
785 
786  // Inverse rotation of sensor data, from body frame into sensor frame
787  float a_sensor[3];
788  rot_mult(R, a_body, a_sensor, true);
789 
790  // Temporary variables
791  float psi, theta, phi;
792 
793  psi = attitudeSettings.BoardRotation[ATTITUDESETTINGS_BOARDROTATION_YAW] * DEG2RAD / 100.0f;
794 
795  float cP = cosf(psi);
796  float sP = sinf(psi);
797 
798  // In case psi is too small, we have to use a different equation to solve for theta
799  if (fabs(psi) > PI / 2)
800  theta = atanf((a_sensor[1] + cP * (sP * a_sensor[0] -
801  cP * a_sensor[1])) / (sP * a_sensor[2]));
802  else
803  theta = atanf((a_sensor[0] - sP * (sP * a_sensor[0] -
804  cP * a_sensor[1])) / (cP * a_sensor[2]));
805 
806  phi = atan2f((sP * a_sensor[0] - cP * a_sensor[1]) / GRAVITY,
807  (a_sensor[2] / cosf(theta) / GRAVITY));
808 
809  attitudeSettings.BoardRotation[ATTITUDESETTINGS_BOARDROTATION_ROLL] = phi * RAD2DEG * 100.0f;
810  attitudeSettings.BoardRotation[ATTITUDESETTINGS_BOARDROTATION_PITCH] = theta * RAD2DEG * 100.0f;
811 
812  attitudeSettings.TrimFlight = ATTITUDESETTINGS_TRIMFLIGHT_NORMAL;
813  AttitudeSettingsSet(&attitudeSettings);
814 
815  }
