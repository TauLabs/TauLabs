00308 static int32_t updateSensors(AccelsData * accelsData, GyrosData * gyrosData)
00309 {
00310         struct pios_adxl345_data accel_data;
00311         float gyro[4];
00312         
00313         // Only wait the time for two nominal updates before setting an alarm
00314         if(xQueueReceive(gyro_queue, (void * const) gyro, PIOS_INTERNAL_ADC_UPDATE_RATE * 2) == errQUEUE_EMPTY) {
00315                 AlarmsSet(SYSTEMALARMS_ALARM_ATTITUDE, SYSTEMALARMS_ALARM_ERROR);
00316                 return -1;
00317         }
00318 
00319         // Do not read raw sensor data in simulation mode
00320         if (GyrosReadOnly() || AccelsReadOnly())
00321                 return 0;
00322 
00323         // No accel data available
00324         if(PIOS_ADXL345_FifoElements() == 0)
00325                 return -1;
00326         
00327         // Convert the ADC data into the standard normalized format
00328         struct pios_sensor_gyro_data gyros;
00329         gyros.temperature = gyro[0];
00330         gyros.x = -(gyro[1] - GYRO_NEUTRAL) * IDG_GYRO_GAIN;
00331         gyros.y = (gyro[2] - GYRO_NEUTRAL) * IDG_GYRO_GAIN;
00332         gyros.z = -(gyro[3] - GYRO_NEUTRAL) * IDG_GYRO_GAIN;
00333         
00334         // Convert the ADXL345 data into the standard normalized format
00335         int32_t x = 0;
00336         int32_t y = 0;
00337         int32_t z = 0;
00338         uint8_t i = 0;
00339         uint8_t samples_remaining;
00340         do {
00341                 i++;
00342                 samples_remaining = PIOS_ADXL345_Read(&accel_data);
00343                 x +=  accel_data.x;
00344                 y += -accel_data.y;
00345                 z += -accel_data.z;
00346         } while ( (i < 32) && (samples_remaining > 0) );
00347 
00348         struct pios_sensor_accel_data accels;
00349         
00350         accels.x = ((float) x / i) * ADXL345_ACCEL_SCALE;
00351         accels.y = ((float) y / i) * ADXL345_ACCEL_SCALE;
00352         accels.z = ((float) z / i) * ADXL345_ACCEL_SCALE;
00353 
00354         // Apply rotation / calibration and assign to the UAVO
00355         update_gyros(&gyros, gyrosData);
00356         update_accels(&accels, accelsData);
00357         
00358         update_trimming(accelsData);
00359 
00360                 
00361 
00362         GyrosSet(gyrosData);
00363         AccelsSet(accelsData);
00364 
00365         return 0;
00366 }

00561 static void updateAttitude(AccelsData * accelsData, GyrosData * gyrosData)
00562 {
00563         float dT;
00564         portTickType thisSysTime = xTaskGetTickCount();
00565         static portTickType lastSysTime = 0;
00566         static float accels_filtered[3] = {0,0,0};
00567         static float grot_filtered[3] = {0,0,0};
00568 
00569         dT = (thisSysTime == lastSysTime) ? 0.001 : (portMAX_DELAY & (thisSysTime - lastSysTime)) / portTICK_RATE_MS / 1000.0f;
00570         lastSysTime = thisSysTime;
00571         
00572         // Bad practice to assume structure order, but saves memory
00573         float * gyros = &gyrosData->x;
00574         float * accels = &accelsData->x;
00575         
00576         float grot[3];
00577         float accel_err[3];
00578 
00579         // Apply smoothing to accel values, to reduce vibration noise before main calculations.
00580         apply_accel_filter(accels,accels_filtered);
00581         
00582         // Rotate gravity to body frame, filter and cross with accels
00583         grot[0] = -(2 * (q[1] * q[3] - q[0] * q[2]));
00584         grot[1] = -(2 * (q[2] * q[3] + q[0] * q[1]));
00585         grot[2] = -(q[0] * q[0] - q[1]*q[1] - q[2]*q[2] + q[3]*q[3]);
00586 
00587         // Apply same filtering to the rotated attitude to match delays
00588         apply_accel_filter(grot,grot_filtered);
00589         
00590         // Compute the error between the predicted direction of gravity and smoothed acceleration
00591         CrossProduct((const float *) accels_filtered, (const float *) grot_filtered, accel_err);
00592         
00593         // Account for accel magnitude
00594         float accel_mag = sqrtf(accels_filtered[0]*accels_filtered[0] + accels_filtered[1]*accels_filtered[1] + accels_filtered[2]*accels_filtered[2]);
00595 
00596         // Account for filtered gravity vector magnitude
00597         float grot_mag;
00598 
00599         if (accel_filter_enabled)
00600                 grot_mag = sqrtf(grot_filtered[0]*grot_filtered[0] + grot_filtered[1]*grot_filtered[1] + grot_filtered[2]*grot_filtered[2]);
00601         else
00602                 grot_mag = 1.0f;
00603 
00604         if (grot_mag > 1.0e-3f && accel_mag > 1.0e-3f) {
00605                 accel_err[0] /= (accel_mag*grot_mag);
00606                 accel_err[1] /= (accel_mag*grot_mag);
00607                 accel_err[2] /= (accel_mag*grot_mag);
00608                 
00609                 // Accumulate integral of error.  Scale here so that units are (deg/s) but Ki has units of s
00610                 gyro_correct_int[0] -= accel_err[0] * accelKi;
00611                 gyro_correct_int[1] -= accel_err[1] * accelKi;
00612                 
00613                 // Correct rates based on error, integral component dealt with in updateSensors
00614                 gyros[0] += accel_err[0] * accelKp / dT;
00615                 gyros[1] += accel_err[1] * accelKp / dT;
00616                 gyros[2] += accel_err[2] * accelKp / dT;
00617         }
00618         
00619         { // scoping variables to save memory
00620                 // Work out time derivative from INSAlgo writeup
00621                 // Also accounts for the fact that gyros are in deg/s
00622                 float qdot[4];
00623                 qdot[0] = (-q[1] * gyros[0] - q[2] * gyros[1] - q[3] * gyros[2]) * dT * DEG2RAD / 2;
00624                 qdot[1] = (q[0] * gyros[0] - q[3] * gyros[1] + q[2] * gyros[2]) * dT * DEG2RAD / 2;
00625                 qdot[2] = (q[3] * gyros[0] + q[0] * gyros[1] - q[1] * gyros[2]) * dT * DEG2RAD / 2;
00626                 qdot[3] = (-q[2] * gyros[0] + q[1] * gyros[1] + q[0] * gyros[2]) * dT * DEG2RAD / 2;
00627                 
00628                 // Take a time step
00629                 q[0] = q[0] + qdot[0];
00630                 q[1] = q[1] + qdot[1];
00631                 q[2] = q[2] + qdot[2];
00632                 q[3] = q[3] + qdot[3];
00633                 
00634                 if(q[0] < 0) {
00635                         q[0] = -q[0];
00636                         q[1] = -q[1];
00637                         q[2] = -q[2];
00638                         q[3] = -q[3];
00639                 }
00640         }
00641         
00642         // Renomalize
00643         float qmag = sqrtf(q[0]*q[0] + q[1]*q[1] + q[2]*q[2] + q[3]*q[3]);
00644         q[0] = q[0] / qmag;
00645         q[1] = q[1] / qmag;
00646         q[2] = q[2] / qmag;
00647         q[3] = q[3] / qmag;
00648         
00649         // If quaternion has become inappropriately short or is nan reinit.
00650         // THIS SHOULD NEVER ACTUALLY HAPPEN
00651         if((fabs(qmag) < 1e-3) || (qmag != qmag)) {
00652                 q[0] = 1;
00653                 q[1] = 0;
00654                 q[2] = 0;
00655                 q[3] = 0;
00656         }
00657         
00658         AttitudeActualData attitudeActual;
00659         AttitudeActualGet(&attitudeActual);
00660         
00661         quat_copy(q, &attitudeActual.q1);
00662         
00663         // Convert into eueler degrees (makes assumptions about RPY order)
00664         Quaternion2RPY(&attitudeActual.q1,&attitudeActual.Roll);
00665         
00666         AttitudeActualSet(&attitudeActual);
