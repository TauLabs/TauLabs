00094         while (1) {

00097                 // the module could listen to multiple objects.
00098                 // Since this object only executes on object updates, the task
00099                 // will block until an object event is pushed in the queue.
00100                 while (xQueueReceive(queue, &ev, portMAX_DELAY) != pdTRUE) ;

00106                 // object connected in the queue.
00107                 if (ev.obj == ExampleObject1Handle()) {
00108                         // Update settings with latest value
00109                         ExampleSettingsGet(&settings);
00110 
00111                         // Get the input object
00112                         ExampleObject1Get(&data1);
00113 
00114                         // Determine how to update the output object
00115                         if (settings.StepDirection == EXAMPLESETTINGS_STEPDIRECTION_UP) {
00116                                 step = settings.StepSize;
00117                         } else {
00118                                 step = -settings.StepSize;
00119                         }
00120 
00121                         // Update data
00122                         data2.Field1 = data1.Field1 + step;
00123                         data2.Field2 = data1.Field2 + step;
00124                         data2.Field3 = data1.Field3 + step;
