44 int parse_ubx_stream (uint8_t c, char *gps_rx_buffer, GPSPositionData *GpsData, struct GPS_RX_STATS *gpsRxStats)
45 {
46  enum proto_states {
47  START,
48  UBX_SY2,
49  UBX_CLASS,
50  UBX_ID,
51  UBX_LEN1,
52  UBX_LEN2,
53  UBX_PAYLOAD,
54  UBX_CHK1,
55  UBX_CHK2,
56  FINISHED
57  };
58 
59  static enum proto_states proto_state = START;
60  static uint8_t rx_count = 0;
61  struct UBXPacket *ubx = (struct UBXPacket *)gps_rx_buffer;
62 
63  switch (proto_state) {
64  case START: // detect protocol
65  if (c == UBX_SYNC1) // first UBX sync char found
66  proto_state = UBX_SY2;
67  break;
68  case UBX_SY2:
69  if (c == UBX_SYNC2) // second UBX sync char found
70  proto_state = UBX_CLASS;
71  else
72  proto_state = START; // reset state
73  break;
74  case UBX_CLASS:
75  ubx->header.class = c;
76  proto_state = UBX_ID;
77  break;
78  case UBX_ID:
79  ubx->header.id = c;
80  proto_state = UBX_LEN1;
81  break;
82  case UBX_LEN1:
83  ubx->header.len = c;
84  proto_state = UBX_LEN2;
85  break;
86  case UBX_LEN2:
87  ubx->header.len += (c << 8);
88  if (ubx->header.len > sizeof(UBXPayload)) {
89  gpsRxStats->gpsRxOverflow++;
90  proto_state = START;
91  } else {
92  rx_count = 0;
93  proto_state = UBX_PAYLOAD;
94  }
95  break;
96  case UBX_PAYLOAD:
97  if (rx_count < ubx->header.len) {
98  ubx->payload.payload[rx_count] = c;
99  if (++rx_count == ubx->header.len)
100  proto_state = UBX_CHK1;
101  } else {
102  gpsRxStats->gpsRxOverflow++;
103  proto_state = START;
104  }
105  break;
106  case UBX_CHK1:
107  ubx->header.ck_a = c;
108  proto_state = UBX_CHK2;
109  break;
110  case UBX_CHK2:
111  ubx->header.ck_b = c;
112  if (checksum_ubx_message(ubx)) { // message complete and valid
113  parse_ubx_message(ubx, GpsData);
114  proto_state = FINISHED;
115  } else {
116  gpsRxStats->gpsRxChkSumError++;
117  proto_state = START;
118  }
119  break;
120  default: break;
121  }
122 
123  if (proto_state == START)
124  return PARSER_ERROR; // parser couldn't use this byte
125  else if (proto_state == FINISHED) {
126  gpsRxStats->gpsRxReceived++;
127  proto_state = START;
128  return PARSER_COMPLETE; // message complete & processed
129  }
130 
131  return PARSER_INCOMPLETE; // message not (yet) complete
132 }

135 // Keep track of various GPS messages needed to make up a single UAVO update
136 // time-of-week timestamp is used to correlate matching messages

138 #define POSLLH_RECEIVED (1 << 0)
139 #define STATUS_RECEIVED (1 << 1)
140 #define DOP_RECEIVED (1 << 2)
141 #define VELNED_RECEIVED (1 << 3)
142 #define SOL_RECEIVED (1 << 4)
143 #define ALL_RECEIVED (SOL_RECEIVED | VELNED_RECEIVED | DOP_RECEIVED | POSLLH_RECEIVED)

146 static struct msgtracker{
147  uint32_t currentTOW; // TOW of the message set currently in progress
148  uint8_t msg_received; // keep track of received message types
149  } msgtracker;
150 
151 // Check if a message belongs to the current data set and register it as 'received'
152 bool check_msgtracker (uint32_t tow, uint8_t msg_flag)
153 {
154 
155  if (tow > msgtracker.currentTOW ? true // start of a new message set
156  : (msgtracker.currentTOW - tow > 6*24*3600*1000)) { // 6 days, TOW wrap around occured
157  msgtracker.currentTOW = tow;
158  msgtracker.msg_received = NONE_RECEIVED;
159  } else if (tow < msgtracker.currentTOW) // message outdated (don't process)
160  return false;
161 
162  msgtracker.msg_received |= msg_flag; // register reception of this msg type
163  return true;
164 }

166 static bool checksum_ubx_message (const struct UBXPacket *ubx)
167 {
168  int i;
169  uint8_t ck_a, ck_b;
170 
171  ck_a = ubx->header.class;
172  ck_b = ck_a;
173 
174  ck_a += ubx->header.id;
175  ck_b += ck_a;
176 
177  ck_a += ubx->header.len & 0xff;
178  ck_b += ck_a;
179 
180  ck_a += ubx->header.len >> 8;
181  ck_b += ck_a;
182 
183  for (i = 0; i < ubx->header.len; i++) {
184  ck_a += ubx->payload.payload[i];
185  ck_b += ck_a;
186  }
187 
188  if (ubx->header.ck_a == ck_a &&
189  ubx->header.ck_b == ck_b)
190  return true;
191  else
192  return false;
193 
194 }

197 {
198  if (check_msgtracker(posllh->iTOW, POSLLH_RECEIVED)) {
199  if (GpsPosition->Status != GPSPOSITION_STATUS_NOFIX) {
200  GpsPosition->Altitude = (float)posllh->hMSL*0.001f;

202  GpsPosition->Latitude = posllh->lat;
203  GpsPosition->Longitude = posllh->lon;

208 static void parse_ubx_nav_sol (const struct UBX_NAV_SOL *sol, GPSPositionData *GpsPosition)
209 {
210  if (check_msgtracker(sol->iTOW, SOL_RECEIVED)) {
211  GpsPosition->Satellites = sol->numSV;
212 
213  if (sol->flags & STATUS_FLAGS_GPSFIX_OK) {
214  switch (sol->gpsFix) {
215  case STATUS_GPSFIX_2DFIX:
216  GpsPosition->Status = GPSPOSITION_STATUS_FIX2D;
217  break;
218  case STATUS_GPSFIX_3DFIX:
219  GpsPosition->Status = GPSPOSITION_STATUS_FIX3D;
220  break;
221  default: GpsPosition->Status = GPSPOSITION_STATUS_NOFIX;
222  }
223  }
224  else // fix is not valid so we make sure to treat is as NOFIX
225  GpsPosition->Status = GPSPOSITION_STATUS_NOFIX;
226  }
227 }

305 static uint32_t parse_ubx_message (const struct UBXPacket *ubx, GPSPositionData *GpsPosition)
306 {
307  uint32_t id = 0;
308 
309  switch (ubx->header.class) {
310  case UBX_CLASS_NAV:
311  switch (ubx->header.id) {
312  case UBX_ID_POSLLH:
313  parse_ubx_nav_posllh (&ubx->payload.nav_posllh, GpsPosition);
314  break;
315  case UBX_ID_DOP:
316  parse_ubx_nav_dop (&ubx->payload.nav_dop, GpsPosition);
317  break;
318  case UBX_ID_SOL:
319  parse_ubx_nav_sol (&ubx->payload.nav_sol, GpsPosition);
320  break;
321  case UBX_ID_VELNED:
322  parse_ubx_nav_velned (&ubx->payload.nav_velned, GpsPosition);
323  break;
324 #if !defined(PIOS_GPS_MINIMAL)
325  case UBX_ID_TIMEUTC:
326  parse_ubx_nav_timeutc (&ubx->payload.nav_timeutc);
327  break;
328  case UBX_ID_SVINFO:
329  parse_ubx_nav_svinfo (&ubx->payload.nav_svinfo);
330  break;
331 #endif
332  }
333  break;
334  }
335  if (msgtracker.msg_received == ALL_RECEIVED) {
336  GPSPositionSet(GpsPosition);
337  msgtracker.msg_received = NONE_RECEIVED;
338  id = GPSPOSITION_OBJID;
339  }
340  return id;
341 }
