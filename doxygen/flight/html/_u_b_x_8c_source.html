00047                 START,
00048                 UBX_SY2,
00049                 UBX_CLASS,
00050                 UBX_ID,
00051                 UBX_LEN1,
00052                 UBX_LEN2,
00053                 UBX_PAYLOAD,
00054                 UBX_CHK1,
00055                 UBX_CHK2,

00059         static enum proto_states proto_state = START;
00060         static uint8_t rx_count = 0;
00061         struct UBXPacket *ubx = (struct UBXPacket *)gps_rx_buffer;

00063         switch (proto_state) {
00064                 case START: // detect protocol
00065                         if (c ==  UBX_SYNC1) // first UBX sync char found
00066                                 proto_state = UBX_SY2;
00067                         break;
00068                 case UBX_SY2:
00069                         if (c == UBX_SYNC2) // second UBX sync char found
00070                                 proto_state = UBX_CLASS;
00071                         else
00072                                 proto_state = START; // reset state
00073                         break;
00074                 case UBX_CLASS:
00075                         ubx->header.class = c;
00076                         proto_state = UBX_ID;
00077                         break;
00078                 case UBX_ID:
00079                         ubx->header.id = c;
00080                         proto_state = UBX_LEN1;
00081                         break;
00082                 case UBX_LEN1:
00083                         ubx->header.len = c;
00084                         proto_state = UBX_LEN2;
00085                         break;
00086                 case UBX_LEN2:
00087                         ubx->header.len += (c << 8);
00088                         if (ubx->header.len > sizeof(UBXPayload)) {
00089                                 gpsRxStats->gpsRxOverflow++;
00090                                 proto_state = START;
00091                         } else {
00092                                 rx_count = 0;
00093                                 proto_state = UBX_PAYLOAD;
00094                         }
00095                         break;
00096                 case UBX_PAYLOAD:
00097                         if (rx_count < ubx->header.len) {
00098                                 ubx->payload.payload[rx_count] = c;
00099                                 if (++rx_count == ubx->header.len)
00100                                         proto_state = UBX_CHK1;
00101                         } else {
00102                                 gpsRxStats->gpsRxOverflow++;
00103                                 proto_state = START;
00104                         }
00105                         break;
00106                 case UBX_CHK1:
00107                         ubx->header.ck_a = c;
00108                         proto_state = UBX_CHK2;
00109                         break;
00110                 case UBX_CHK2:
00111                         ubx->header.ck_b = c;
00112                         if (checksum_ubx_message(ubx)) { // message complete and valid
00113                                 parse_ubx_message(ubx, GpsData);
00114                                 proto_state = FINISHED;
00115                         } else {
00116                                 gpsRxStats->gpsRxChkSumError++;
00117                                 proto_state = START;
00118                         }
00119                         break;
00120                 default: break;
00121         }
00122 
00123         if (proto_state == START)
00124                 return PARSER_ERROR;    // parser couldn't use this byte
00125         else if (proto_state == FINISHED) {
00126                 gpsRxStats->gpsRxReceived++;
00127                 proto_state = START;
00128                 return PARSER_COMPLETE; // message complete & processed
00129         }
00130 
00131         return PARSER_INCOMPLETE; // message not (yet) complete
00132 }

00151 // Check if a message belongs to the current data set and register it as 'received'
00152 bool check_msgtracker (uint32_t tow, uint8_t msg_flag)
00153 {
00154 
00155         if (tow > msgtracker.currentTOW ? true                  // start of a new message set
00156                 : (msgtracker.currentTOW - tow > 6*24*3600*1000)) { // 6 days, TOW wrap around occured
00157                 msgtracker.currentTOW = tow;
00158                 msgtracker.msg_received = NONE_RECEIVED;
00159         } else if (tow < msgtracker.currentTOW) // message outdated (don't process)
00160                                 return false;
00161 
00162         msgtracker.msg_received |= msg_flag; // register reception of this msg type
00163         return true;
00164 }

00166 static bool checksum_ubx_message (const struct UBXPacket *ubx)
00167 {
00168         int i;
00169         uint8_t ck_a, ck_b;
00170 
00171         ck_a = ubx->header.class;
00172         ck_b = ck_a;
00173 
00174         ck_a += ubx->header.id;
00175         ck_b += ck_a;
00176 
00177         ck_a += ubx->header.len & 0xff;
00178         ck_b += ck_a;
00179 
00180         ck_a += ubx->header.len >> 8;
00181         ck_b += ck_a;
00182 
00183         for (i = 0; i < ubx->header.len; i++) {
00184                 ck_a += ubx->payload.payload[i];
00185                 ck_b += ck_a;
00186         }
00187 
00188         if (ubx->header.ck_a == ck_a &&
00189                         ubx->header.ck_b == ck_b)
00190                 return true;
00191         else
00192                 return false;
