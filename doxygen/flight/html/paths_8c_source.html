00154         // Distance to go
00155         path_north = end_point[0] - start_point[0];
00156         path_east = end_point[1] - start_point[1];
00157 
00158         // Current progress location relative to start
00159         diff_north = cur_point[0] - start_point[0];
00160         diff_east = cur_point[1] - start_point[1];
00161

00164 
00165         if(dist_path < 1e-6) {
00166                 // if the path is too short, we cannot determine vector direction.
00167                 // Fly towards the endpoint to prevent flying away,
00168                 // but assume progress=1 either way.
00169                 path_endpoint( start_point, end_point, cur_point, status );
00170                 status->fractional_progress = 1;
00171                 return;
00172         }
00173 
00174         // Compute the normal to the path
00175         normal[0] = -path_east / dist_path;
00176         normal[1] = path_north / dist_path;
00177 
00178         status->fractional_progress = dot / (dist_path * dist_path);
00179         status->error = normal[0] * diff_north + normal[1] * diff_east;
00180 
00181         // Compute direction to correct error
00182         status->correction_direction[0] = (status->error > 0) ? -normal[0] : normal[0];
00183         status->correction_direction[1] = (status->error > 0) ? -normal[1] : normal[1];
00184         
00185         // Now just want magnitude of error
00186         status->error = fabs(status->error);
00187 
00188         // Compute direction to travel
00189         status->path_direction[0] = path_north / dist_path;
