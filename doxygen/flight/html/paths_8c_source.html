48 void path_progress(PathDesiredData *pathDesired,
49  float *cur_point,
50  struct path_status *status)
51 {
52  uint8_t mode = pathDesired->Mode;
53  float start_point[2] = {pathDesired->Start[0],pathDesired->Start[1]};
54  float end_point[2] = {pathDesired->End[0],pathDesired->End[1]};
55 
56  switch(mode) {
57  case PATHDESIRED_MODE_FLYVECTOR:
58  case PATHDESIRED_MODE_DRIVEVECTOR:
59  return path_vector(start_point, end_point, cur_point, status);
60  break;
61  case PATHDESIRED_MODE_FLYCIRCLERIGHT:
62  case PATHDESIRED_MODE_DRIVECIRCLERIGHT:
63  return path_curve(start_point, end_point, pathDesired->ModeParameters, cur_point, status, 1);
64  break;
65  case PATHDESIRED_MODE_FLYCIRCLELEFT:
66  case PATHDESIRED_MODE_DRIVECIRCLELEFT:
67  return path_curve(start_point, end_point, pathDesired->ModeParameters, cur_point, status, 0);
68  break;
69  case PATHDESIRED_MODE_CIRCLEPOSITIONLEFT:
70  return path_circle(end_point, pathDesired->ModeParameters, cur_point, status, 0);
71  break;
72  case PATHDESIRED_MODE_CIRCLEPOSITIONRIGHT:
73  return path_circle(end_point, pathDesired->ModeParameters, cur_point, status, 1);
74  break;
75  case PATHDESIRED_MODE_FLYENDPOINT:
76  case PATHDESIRED_MODE_DRIVEENDPOINT:
77  default:
78  // use the endpoint as default failsafe if called in unknown modes
79  return path_endpoint(start_point, end_point, cur_point, status);
80  break;
81  }

91 static void path_endpoint(float *start_point,
92  float *end_point,
93  float *cur_point,
94  struct path_status *status)
95 {
96  float path_north, path_east, diff_north, diff_east;
97  float dist_path, dist_diff;
98 
99  // we do not correct in this mode
100  status->correction_direction[0] = status->correction_direction[1] = 0;
101 
102  // Distance to go
103  path_north = end_point[0] - start_point[0];
104  path_east = end_point[1] - start_point[1];
105 
106  // Current progress location relative to end
107  diff_north = end_point[0] - cur_point[0];
108  diff_east = end_point[1] - cur_point[1];
109 
110  dist_diff = sqrtf( diff_north * diff_north + diff_east * diff_east );
111  dist_path = sqrtf( path_north * path_north + path_east * path_east );
112 
113  if(dist_diff < 1e-6 ) {
114  status->fractional_progress = 1;
115  status->error = 0;
116  status->path_direction[0] = status->path_direction[1] = 0;
117  return;
118  }
119 
120  status->fractional_progress = 1 - dist_diff / (1 + dist_path);
121  status->error = dist_diff;
122 
123  // Compute direction to travel
124  status->path_direction[0] = diff_north / dist_diff;
125  status->path_direction[1] = diff_east / dist_diff;
126 
127 }

136 static void path_vector(float *start_point,
137  float *end_point,
138  float *cur_point,
139  struct path_status *status)
140 {
141  float path_north, path_east, diff_north, diff_east;
142  float dist_path;
143  float dot;
144  float normal[2];
145 
146  // Distance to go
147  path_north = end_point[0] - start_point[0];
148  path_east = end_point[1] - start_point[1];
149 
150  // Current progress location relative to start
151  diff_north = cur_point[0] - start_point[0];
152  diff_east = cur_point[1] - start_point[1];
153 
154  dot = path_north * diff_north + path_east * diff_east;
155  dist_path = sqrtf( path_north * path_north + path_east * path_east );
156 
157  if(dist_path < 1e-6) {
158  // if the path is too short, we cannot determine vector direction.
159  // Fly towards the endpoint to prevent flying away,
160  // but assume progress=1 either way.
161  path_endpoint( start_point, end_point, cur_point, status );
162  status->fractional_progress = 1;
163  return;
164  }
165 
166  // Compute the normal to the path
167  normal[0] = -path_east / dist_path;
168  normal[1] = path_north / dist_path;
169 
170  status->fractional_progress = dot / (dist_path * dist_path);
171  status->error = normal[0] * diff_north + normal[1] * diff_east;
172 
173  // Compute direction to correct error
174  status->correction_direction[0] = (status->error > 0) ? -normal[0] : normal[0];
175  status->correction_direction[1] = (status->error > 0) ? -normal[1] : normal[1];
176 
177  // Now just want magnitude of error
178  status->error = fabs(status->error);
179 
180  // Compute direction to travel
181  status->path_direction[0] = path_north / dist_path;
182  status->path_direction[1] = path_east / dist_path;
183 
184 }

193 static void path_circle(float * center_point,
194  float radius,
195  float * cur_point,
196  struct path_status * status,
197  bool clockwise)
198 {
199  float diff_north, diff_east;
200  float cradius;
201  float normal[2];
202 
203  // Current location relative to center
204  diff_north = cur_point[0] - center_point[0];
205  diff_east = cur_point[1] - center_point[1];
206 
207  cradius = sqrtf( diff_north * diff_north + diff_east * diff_east );
208 
209  if (cradius < 1e-6) {
210  // cradius is zero, just fly somewhere and make sure correction is still a normal
211  status->fractional_progress = 1;
212  status->error = radius;
213  status->correction_direction[0] = 0;
214  status->correction_direction[1] = 1;
215  status->path_direction[0] = 1;
216  status->path_direction[1] = 0;
217  return;
218  }
219 
220  if (clockwise) {
221  // Compute the normal to the radius clockwise
222  normal[0] = -diff_east / cradius;
223  normal[1] = diff_north / cradius;
224  } else {
225  // Compute the normal to the radius counter clockwise
226  normal[0] = diff_east / cradius;
227  normal[1] = -diff_north / cradius;
228  }
229 
230  status->fractional_progress = 0;
231 
232  // error is current radius minus wanted radius - positive if too close
233  status->error = radius - cradius;
234 
235  // Compute direction to correct error
236  status->correction_direction[0] = (status->error>0?1:-1) * diff_north / cradius;
237  status->correction_direction[1] = (status->error>0?1:-1) * diff_east / cradius;
238 
239  // Compute direction to travel
240  status->path_direction[0] = normal[0];
241  status->path_direction[1] = normal[1];
242 
243  status->error = fabs(status->error);
244 }

254 static void path_curve(float * start_point,
255  float * end_point,
256  float radius,
257  float * cur_point,
258  struct path_status *status,
259  bool clockwise)
260 {
261  float diff_north, diff_east;
262  float path_north, path_east;
263  float cradius;
264  float normal[2];
265 
266  // Compute the center of the circle connecting the two points as the intersection of two circles
267  // around the two points from
268  // http://www.mathworks.com/matlabcentral/newsreader/view_thread/255121
269  float m_n, m_e, p_n, p_e, d, center[2];
270 
271  // Center between start and end
272  m_n = (start_point[0] + end_point[0]) / 2;
273  m_e = (start_point[1] + end_point[1]) / 2;
274 
275  // Normal vector the line between start and end.
276  if (clockwise) {
277  p_n = -(end_point[1] - start_point[1]);
278  p_e = (end_point[0] - start_point[0]);
279  } else {
280  p_n = (end_point[1] - start_point[1]);
281  p_e = -(end_point[0] - start_point[0]);
282  }
283 
284  // Work out how far to go along the perpendicular bisector
285  d = sqrtf(radius * radius / (p_n * p_n + p_e * p_e) - 0.25f);
286 
287  float radius_sign = (radius > 0) ? 1 : -1;
288  radius = fabs(radius);
289 
290  if (fabs(p_n) < 1e-3 && fabs(p_e) < 1e-3) {
291  center[0] = m_n;
292  center[1] = m_e;
293  } else {
294  center[0] = m_n + p_n * d * radius_sign;
295  center[1] = m_e + p_e * d * radius_sign;
296  }
297 
298  // Current location relative to center
299  diff_north = cur_point[0] - center[0];
300  diff_east = cur_point[1] - center[1];
301 
302  // Compute current radius from the center
303  cradius = sqrtf( diff_north * diff_north + diff_east * diff_east );
304 
305  // Compute error in terms of meters from the curve (the distance projected
306  // normal onto the path i.e. cross-track distance)
307  status->error = radius - cradius;
308 
309  if (cradius < 1e-6) {
310  // cradius is zero, just fly somewhere and make sure correction is still a normal
311  status->fractional_progress = 1;
312  status->error = radius;
313  status->correction_direction[0] = 0;
314  status->correction_direction[1] = 1;
315  status->path_direction[0] = 1;
316  status->path_direction[1] = 0;
317  return;
318  }
319 
320  if (clockwise) {
321  // Compute the normal to the radius clockwise
322  normal[0] = -diff_east / cradius;
323  normal[1] = diff_north / cradius;
324  } else {
325  // Compute the normal to the radius counter clockwise
326  normal[0] = diff_east / cradius;
327  normal[1] = -diff_north / cradius;
328  }
329 
330  // Compute direction to correct error
331  status->correction_direction[0] = (status->error>0?1:-1) * diff_north / cradius;
332  status->correction_direction[1] = (status->error>0?1:-1) * diff_east / cradius;
333 
334  // Compute direction to travel
335  status->path_direction[0] = normal[0];
336  status->path_direction[1] = normal[1];
337 
338  path_north = end_point[0] - start_point[0];
339  path_east = end_point[1] - start_point[1];
340  diff_north = cur_point[0] - start_point[0];
341  diff_east = cur_point[1] - start_point[1];
342  float dist_path = sqrtf( path_north * path_north + path_east * path_east );
343  float dot = path_north * diff_north + path_east * diff_east;
344 
345  status->fractional_progress = dot / (dist_path * dist_path);
346 
347  status->error = fabs(status->error);
348 }
