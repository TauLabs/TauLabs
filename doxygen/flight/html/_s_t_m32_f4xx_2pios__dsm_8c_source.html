115 /* Try to bind DSMx satellite using specified number of pulses */
116 static void PIOS_DSM_Bind(struct pios_dsm_dev *dsm_dev, uint8_t bind)
117 {
118  const struct pios_dsm_cfg *cfg = dsm_dev->cfg;
119 
120  GPIO_InitTypeDef GPIO_InitStructure = cfg->bind.init;
121  GPIO_InitStructure.GPIO_PuPd = GPIO_PuPd_UP;
122 
123  /* just to limit bind pulses */
124  if (bind > 10)
125  bind = 10;
126 
127  GPIO_Init(cfg->bind.gpio, (GPIO_InitTypeDef*)&cfg->bind.init);
128 
129  /* RX line, set high */
130  GPIO_SetBits(cfg->bind.gpio, cfg->bind.init.GPIO_Pin);
131 
132  /* on CC works up to 140ms, guess bind window is around 20-140ms after power up */
133  PIOS_DELAY_WaitmS(60);
134 
135  for (int i = 0; i < bind ; i++) {
136  /* RX line, drive low for 120us */
137  GPIO_ResetBits(cfg->bind.gpio, cfg->bind.init.GPIO_Pin);
138  PIOS_DELAY_WaituS(120);
139  /* RX line, drive high for 120us */
140  GPIO_SetBits(cfg->bind.gpio, cfg->bind.init.GPIO_Pin);
141  PIOS_DELAY_WaituS(120);
142  }
143  /* RX line, set input and wait for data */
144  GPIO_Init(cfg->bind.gpio, &GPIO_InitStructure);
145 }

175 static int PIOS_DSM_UnrollChannels(struct pios_dsm_dev *dsm_dev)
176 {
177  struct pios_dsm_state *state = &(dsm_dev->state);
178  uint8_t resolution;
179 
180 #ifdef DSM_LOST_FRAME_COUNTER
181  /* increment the lost frame counter */
182  uint8_t frames_lost = state->received_data[0];
183  state->frames_lost += (frames_lost - state->frames_lost_last);
184  state->frames_lost_last = frames_lost;
185 #endif
186 
187  /* check the frame type assuming master satellite stream */
188  uint8_t type = state->received_data[1];
189  switch (type) {
190  case 0x01:
191  case 0x02:
192  case 0x12:
193  /* DSM2, DSMJ stream */
194  if (dsm_dev->proto == PIOS_DSM_PROTO_DSM2) {
195  /* DSM2/DSMJ resolution is known from the header */
196  resolution = (type & DSM_DSM2_RES_MASK) ? 11 : 10;
197  } else {
198  /* DSMX resolution should explicitly be selected */
199  goto stream_error;
200  }
201  break;
202  case 0xA2:
203  case 0xB2:
204  /* DSMX stream */
205  if (dsm_dev->proto == PIOS_DSM_PROTO_DSMX10BIT) {
206  resolution = 10;
207  } else if (dsm_dev->proto == PIOS_DSM_PROTO_DSMX11BIT) {
208  resolution = 11;
209  } else {
210  /* DSMX resolution should explicitly be selected */
211  goto stream_error;
212  }
213  break;
214  default:
215  /* unknown yet data stream */
216  goto stream_error;
217  }
218 
219  /* unroll channels */
220  uint8_t *s = &(state->received_data[2]);
221  uint16_t mask = (resolution == 10) ? 0x03ff : 0x07ff;
222 
223  for (int i = 0; i < DSM_CHANNELS_PER_FRAME; i++) {
224  uint16_t word = ((uint16_t)s[0] << 8) | s[1];
225  s += 2;
226 
227  /* skip empty channel slot */
228  if (word == 0xffff)
229  continue;
230 
231  /* minimal data validation */
232  if ((i > 0) && (word & DSM_2ND_FRAME_MASK)) {
233  /* invalid frame data, ignore rest of the frame */
234  goto stream_error;
235  }
236 
237  /* extract and save the channel value */
238  uint8_t channel_num = (word >> resolution) & 0x0f;
239  if (channel_num < PIOS_DSM_NUM_INPUTS)
240  state->channel_data[channel_num] = (word & mask);
241  }
242 
243 #ifdef DSM_LOST_FRAME_COUNTER
244  /* put lost frames counter into the last channel for debugging */
245  state->channel_data[PIOS_DSM_NUM_INPUTS-1] = state->frames_lost;
246 #endif
247 
248  /* all channels processed */
249  return 0;
250 
251 stream_error:
252  /* either DSM2 selected with DSMX stream found, or vice-versa */
253  return -1;
254 }

256 /* Update decoder state processing input byte from the DSMx stream */
257 static void PIOS_DSM_UpdateState(struct pios_dsm_dev *dsm_dev, uint8_t byte)
258 {
259  struct pios_dsm_state *state = &(dsm_dev->state);
260  if (state->frame_found) {
261  /* receiving the data frame */
262  if (state->byte_count < DSM_FRAME_LENGTH) {
263  /* store next byte */
264  state->received_data[state->byte_count++] = byte;
265  if (state->byte_count == DSM_FRAME_LENGTH) {
266  /* full frame received - process and wait for new one */
267  if (!PIOS_DSM_UnrollChannels(dsm_dev))
268  /* data looking good */
269  state->failsafe_timer = 0;
270 
271  /* prepare for the next frame */
272  state->frame_found = 0;
273  }
274  }
275  }
276 }

278 /* Initialise DSM receiver interface */
279 int32_t PIOS_DSM_Init(uint32_t *dsm_id,
280  const struct pios_dsm_cfg *cfg,
281  const struct pios_com_driver *driver,
282  uint32_t lower_id,
283  enum pios_dsm_proto proto,
284  uint8_t bind)
285 {
286  PIOS_DEBUG_Assert(dsm_id);
287  PIOS_DEBUG_Assert(cfg);
288  PIOS_DEBUG_Assert(driver);
289 
290  struct pios_dsm_dev *dsm_dev;
291 
292  dsm_dev = (struct pios_dsm_dev *)PIOS_DSM_Alloc();
293  if (!dsm_dev)
294  return -1;
295 
296  /* Bind the configuration to the device instance */
297  dsm_dev->cfg = cfg;
298  dsm_dev->proto = proto;
299 
300  /* Bind the receiver if requested */
301  if (bind)
302  PIOS_DSM_Bind(dsm_dev, bind);
303 
304  PIOS_DSM_ResetState(dsm_dev);
305 
306  *dsm_id = (uint32_t)dsm_dev;
307 
308  /* Set comm driver callback */
309  (driver->bind_rx_cb)(lower_id, PIOS_DSM_RxInCallback, *dsm_id);
310 
311  if (!PIOS_RTC_RegisterTickCallback(PIOS_DSM_Supervisor, *dsm_id)) {
312  PIOS_DEBUG_Assert(0);
313  }
314 
315  return 0;
316 }

318 /* Comm byte received callback */
319 static uint16_t PIOS_DSM_RxInCallback(uint32_t context,
320  uint8_t *buf,
321  uint16_t buf_len,
322  uint16_t *headroom,
323  bool *need_yield)
324 {
325  struct pios_dsm_dev *dsm_dev = (struct pios_dsm_dev *)context;
326 
327  bool valid = PIOS_DSM_Validate(dsm_dev);
328  PIOS_Assert(valid);
329 
330  /* process byte(s) and clear receive timer */
331  for (uint8_t i = 0; i < buf_len; i++) {
332  PIOS_DSM_UpdateState(dsm_dev, buf[i]);
333  dsm_dev->state.receive_timer = 0;
334  }
335 
336  /* Always signal that we can accept another byte */
337  if (headroom)
338  *headroom = DSM_FRAME_LENGTH;
339 
340  /* We never need a yield */
341  *need_yield = false;
342 
343  /* Always indicate that all bytes were consumed */
344  return buf_len;
345 }

354 static int32_t PIOS_DSM_Get(uint32_t rcvr_id, uint8_t channel)
355 {
356  struct pios_dsm_dev *dsm_dev = (struct pios_dsm_dev *)rcvr_id;
357 
358  if (!PIOS_DSM_Validate(dsm_dev))
359  return PIOS_RCVR_INVALID;
360 
361  /* return error if channel is not available */
362  if (channel >= PIOS_DSM_NUM_INPUTS)
363  return PIOS_RCVR_INVALID;
364 
365  /* may also be PIOS_RCVR_TIMEOUT set by other function */
366  return dsm_dev->state.channel_data[channel];
367 }

381 static void PIOS_DSM_Supervisor(uint32_t dsm_id)
382 {
383  struct pios_dsm_dev *dsm_dev = (struct pios_dsm_dev *)dsm_id;
384 
385  bool valid = PIOS_DSM_Validate(dsm_dev);
386  PIOS_Assert(valid);
387 
388  struct pios_dsm_state *state = &(dsm_dev->state);
389 
390  /* waiting for new frame if no bytes were received in 8ms */
391  if (++state->receive_timer > 4) {
392  state->frame_found = 1;
393  state->byte_count = 0;
394  state->receive_timer = 0;
395  }
396 
397  /* activate failsafe if no frames have arrived in 102.4ms */
398  if (++state->failsafe_timer > 64) {
399  PIOS_DSM_ResetChannels(dsm_dev);
400  state->failsafe_timer = 0;
401  }
402 }
