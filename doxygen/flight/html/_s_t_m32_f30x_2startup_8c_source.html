64 void
65 _main(void)
66 {
67  // load the stack base for the current stack before we attempt to branch to any function
68  // that might bounds-check the stack
69  asm volatile ("mov r10, %0" : : "r" (&irq_stack[0]) : );
70 
71  /* enable usage, bus and memory faults */
72  SCB->SHCSR |= SCB_SHCSR_USGFAULTENA_Msk | SCB_SHCSR_BUSFAULTENA_Msk | SCB_SHCSR_MEMFAULTENA_Msk;
73 
74  /* configure FP state save behaviour - automatic, lazy save */
75  FPU->FPCCR |= FPU_FPCCR_ASPEN_Msk | FPU_FPCCR_LSPEN_Msk;
76 
77  /* configure default FPU state */
78  FPU->FPDSCR |= FPU_FPDSCR_DN_Msk; /* enable Default NaN */
79 
80  /* enable the FPU */
81  SCB->CPACR |= (0xf << 20); // turn on CP10/11 for FP support on cores that implement it
82 
83  /* copy initialised data from flash to RAM */
84  memcpy(&_sdata, &_sidata, &_edata - &_sdata);
85 
86  /* zero the BSS */
87  memset(&_sbss, 0, &_ebss - &_sbss);
88 
89  /* zero any 'fast' RAM that's been used */
90  memset(&_sfast, 0, &_efast - &_sfast);
91 
92  /* fill most of the IRQ/bootstrap stack with a watermark pattern so we can measure how much is used */
93  /* leave a little space at the top in case memset() isn't a leaf with no locals */
94  memset(&irq_stack, 0xa5, sizeof(irq_stack) - 64);
95 
96  /* call main */
97  (void)main();
98 }
